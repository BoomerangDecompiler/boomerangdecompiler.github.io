43 bool SPARCFrontEnd::canOptimizeDelayCopy(Address src, Address dest, ptrdiff_t delta,
44  Interval<Address> textLimit) const
45 {
46  // Check that the destination is within the main test section; may not be when we speculatively
47  // decode junk
48  if (!textLimit.contains(dest - 4)) {
49  return false;
50  }
51 
52  const DWord delay_inst = *reinterpret_cast<const DWord *>(src.value() + 4 + delta);
53  const DWord inst_before_dest = *reinterpret_cast<const DWord *>(dest.value() - 4 + delta);
54 
55  return delay_inst == inst_before_dest;
56 }

59 BasicBlock *SPARCFrontEnd::optimizeCallReturn(CallStatement *call, const RTL *rtl, const RTL *delay,
60  UserProc *proc)
61 {
62  if (call->isReturnAfterCall()) {
63  // The only RTL in the basic block is a ReturnStatement
64  std::list<Statement *> ls;
65 
66  // If the delay slot is a single assignment to %o7, we want to see the semantics for it, so
67  // that preservation or otherwise of %o7 is correct
68  if (delay && (delay->size() == 1) && delay->front()->isAssign() &&
69  static_cast<Assign *>(delay->front())->getLeft()->isRegN(REG_SPARC_O7)) {
70  ls.push_back(delay->front()->clone());
71  }
72 
73  ls.push_back(new ReturnStatement);
74 
75  // Constuct the RTLs for the new basic block
76  std::unique_ptr<RTLList> rtls(new RTLList);
77  BasicBlock *returnBB = createReturnBlock(
78  proc, std::move(rtls), std::unique_ptr<RTL>(new RTL(rtl->getAddress() + 1, &ls)));
79  return returnBB;
80  }
81  else {
82  // May want to put code here that checks whether or not the delay instruction redefines %o7
83  return nullptr;
84  }
85 }

89  TargetQueue &tq, Interval<Address> textLimit)
90 {
91  if (!textLimit.contains(dest)) {
92  LOG_ERROR("Branch to address %1 is beyond section limits", dest);
93  return;
94  }
95  else if (newBB == nullptr) {
96  return;
97  }
98 
99  tq.visit(cfg, dest, newBB);
100  cfg->addEdge(newBB, dest);

104 void SPARCFrontEnd::createCallToAddress(Address dest, Address address, BasicBlock *callBB,
105  ProcCFG *cfg, int offset /* = 0*/)
106 {
107  if (callBB == nullptr) {
108  return;
109  }
110 
111  const Prog *prog = cfg->getProc()->getProg();
112 
113  // If the destination address is the same as this very instruction,
114  // we have a call with iDisp30 == 0. Don't treat this as the start of a real procedure.
115  if (dest != address && prog->getFunctionByAddr(dest) == nullptr) {
116  // We don't want to call prog.visitProc just yet, in case this is a speculative decode that
117  // failed. Instead, we use the set of CallStatements (not in this procedure) that is needed
118  // by CSR
119  if (m_program->getProject()->getSettings()->traceDecoder) {
120  LOG_VERBOSE("p%1", dest);
121  }
122  }
123 
124  // Add the out edge if required
125  if (offset != 0) {
126  cfg->addEdge(callBB, address + offset);
127  }
128 }

137 bool SPARCFrontEnd::case_CALL(Address &address, DecodeResult &inst, DecodeResult &delayInst,
138  std::unique_ptr<RTLList> &BB_rtls, UserProc *proc,
139  std::list<CallStatement *> &callList, bool isPattern /* = false*/)
140 {
141  // Aliases for the call and delay RTLs
142  CallStatement *callStmt = static_cast<CallStatement *>(inst.rtl->back());
143  RTL *delayRTL = delayInst.rtl.get();
144 
145  // Emit the delay instruction, unless the delay instruction is a nop, or we have a pattern, or
146  // are followed by a restore
147  if (delayInst.type != NOP && !callStmt->isReturnAfterCall()) {
148  delayRTL->setAddress(address);
149  BB_rtls->push_back(std::move(delayInst.rtl));
150  }
151 
152  // Get the new return basic block for the special case where the delay instruction is a restore
153  BasicBlock *returnBB = optimizeCallReturn(callStmt, inst.rtl.get(), delayRTL, proc);
154 
155  int disp30 = (callStmt->getFixedDest().value() - address.value()) >> 2;
156 
157  // Don't test for small offsets if part of a move_call_move pattern.
158  // These patterns assign to %o7 in the delay slot, and so can't possibly be used to copy %pc to
159  // %o7 Similarly if followed by a restore
160  if (!isPattern && (returnBB == nullptr) && ((disp30 == 2) || (disp30 == 3))) {
161  // This is the idiomatic case where the destination is 1 or 2 instructions after the delayed
162  // instruction. Only emit the side effect of the call (%o7 := %pc) in this case. Note that
163  // the side effect occurs before the delay slot instruction (which may use the value of %o7)
164  emitCopyPC(*BB_rtls, address);
165  address += disp30 << 2;
166  return true;
167  }
168  else {
169  assert(disp30 != 1);
170 
171  // First check for helper functions
172  Address dest = callStmt->getFixedDest();
173  const BinarySymbol *symb = m_program->getBinaryFile()->getSymbols()->findSymbolByAddress(
174  dest);
175 
176  // Special check for calls to weird PLT entries which don't have symbols
177  if ((symb && symb->isImportedFunction()) && (m_program->getSymbolNameByAddr(dest) == "")) {
178  // This is one of those. Flag this as an invalid instruction
179  inst.valid = false;
180  }
181 
182  if (isHelperFunc(dest, address, *BB_rtls)) {
183  address += 8; // Skip call, delay slot
184  return true;
185  }
186 
187  // Emit the call
188  RTL *rtl = inst.rtl.get();
189  BB_rtls->push_back(std::move(inst.rtl));
190 
191  // End the current basic block
192  ProcCFG *cfg = proc->getCFG();
193  BasicBlock *callBB = cfg->createBB(BBType::Call, std::move(BB_rtls));
194 
195  if (callBB == nullptr) {
196  return false;
197  }
198 
199  // Add this call site to the set of call sites which need to be analysed later.
200  // This set will be used later to call prog.visitProc (so the proc will get decoded)
201  callList.push_back(static_cast<CallStatement *>(rtl->back()));
202 
203  if (returnBB) {
204  // Handle the call but don't add any outedges from it just yet.
205  createCallToAddress(callStmt->getFixedDest(), address, callBB, cfg);
206 
207  // Now add the out edge
208  cfg->addEdge(callBB, returnBB);
209 
210  address += inst.numBytes; // For coverage
211  // This is a CTI block that doesn't fall through and so must
212  // stop sequentially decoding
213  return false;
214  }
215  else {
216  // Else no restore after this call. An outedge may be added to the lexical successor of
217  // the call which will be 8 bytes ahead or in the case where the callee returns a
218  // struct, 12 bytes ahead. But the problem is: how do you know this function returns a
219  // struct at decode time? If forceOutEdge is set, set offset to 0 and no out-edge will
220  // be added yet
221  // MVE: FIXME!
222  int offset = !inst.forceOutEdge.isZero() ? 0 : /*call_stmt->returnsStruct()?12:8*/ 8;
223  bool ret = true;
224 
225  // Check for _exit; probably should check for other "never return" functions
226  QString name = m_program->getSymbolNameByAddr(dest);
227 
228  if (name == "_exit") {
229  // Don't keep decoding after this call
230  ret = false;
231  // Also don't add an out-edge; setting offset to 0 will do this
232  offset = 0;
233  // But we have already set the number of out-edges to 1
234  callBB->setType(BBType::Call);
235  }
236 
237  // Handle the call (register the destination as a proc) and possibly set the outedge.
238  createCallToAddress(dest, address, callBB, cfg, offset);
239 
240  if (!inst.forceOutEdge.isZero()) {
241  // There is no need to force a goto to the new out-edge, since we will continue
242  // decoding from there. If other edges exist to the outedge, they will generate the
243  // required label
244  cfg->addEdge(callBB, inst.forceOutEdge);
245  address = inst.forceOutEdge;
246  }
247  else {
248  // Continue decoding from the lexical successor
249  address += offset;
250  }
251 
252  return ret;
253  }
254  }
255 }

258 void SPARCFrontEnd::case_SD(Address &pc, ptrdiff_t delta, Interval<Address> textLimit,
259  DecodeResult &inst, DecodeResult &delay_inst,
260  std::unique_ptr<RTLList> BB_rtls, ProcCFG *cfg, TargetQueue &tq)
261 {
262  // Aliases for the SD and delay RTLs
263  GotoStatement *SD_stmt = static_cast<GotoStatement *>(inst.rtl->back());
264  RTL *delay_rtl = delay_inst.rtl.get();
265 
266  // Try the "delay instruction has been copied" optimisation,
267  // emitting the delay instruction now if the optimisation won't apply
268  if (delay_inst.type != NOP) {
269  if (canOptimizeDelayCopy(pc, SD_stmt->getFixedDest(), delta, textLimit)) {
270  SD_stmt->adjustFixedDest(-4);
271  }
272  else {
273  // Move the delay instruction before the SD. Must update the address in case there is a
274  // branch to the SD
275  delay_rtl->setAddress(pc);
276  BB_rtls->push_back(std::move(delay_inst.rtl));
277  }
278  }
279 
280  // Update the address (for coverage)
281  pc += 2 * inst.numBytes;
282 
283  // Add the SD
284  BB_rtls->push_back(std::move(inst.rtl));
285 
286  // Add the one-way branch BB
287  BasicBlock *newBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
288 
289  if (newBB != nullptr) {
290  // Visit the destination, and add the out-edge
291  Address jumpDest = SD_stmt->getFixedDest();
292  createJumpToAddress(jumpDest, newBB, cfg, tq, textLimit);
293  }
294 }

297 bool SPARCFrontEnd::case_DD(Address &address, ptrdiff_t, DecodeResult &inst,
298  DecodeResult &delay_inst, std::unique_ptr<RTLList> BB_rtls,
299  TargetQueue &, UserProc *proc, std::list<CallStatement *> &callList)
300 {
301  ProcCFG *cfg = proc->getCFG();
302  RTL *rtl = inst.rtl.get();
303  RTL *delayRTL = delay_inst.rtl.get();
304 
305  if (delay_inst.type != NOP) {
306  // Emit the delayed instruction, unless a NOP
307  delayRTL->setAddress(address);
308  BB_rtls->push_back(std::move(delay_inst.rtl));
309  }
310 
311  // Set address past this instruction and delay slot (may be changed later). This in so that we
312  // cover the jmp/call and delay slot instruction, in case we return false
313  address += 8;
314 
315  BasicBlock *newBB;
316  bool isRetOrCase = false;
317  Statement *lastStmt = rtl->back();
318 
319  switch (lastStmt->getKind()) {
320  case StmtType::Call:
321  // Will be a computed call
322  BB_rtls->push_back(std::move(inst.rtl));
323  newBB = cfg->createBB(BBType::CompCall, std::move(BB_rtls));
324  break;
325 
326  case StmtType::Ret:
327  newBB = createReturnBlock(proc, std::move(BB_rtls), std::move(inst.rtl));
328  isRetOrCase = true;
329  break;
330 
331  case StmtType::Case: {
332  BB_rtls->push_back(std::move(inst.rtl));
333  newBB = cfg->createBB(BBType::CompJump, std::move(BB_rtls));
334  BB_rtls = nullptr;
335  isRetOrCase = true;
336  SharedExp jumpDest = static_cast<CaseStatement *>(lastStmt)->getDest();
337 
338  if (jumpDest == nullptr) { // Happens if already analysed (we are now redecoding)
339  // processSwitch will update the BB type and number of outedges, decode arms, set out
340  // edges, etc.
341  IndirectJumpAnalyzer().processSwitch(newBB, proc);
342  }
343 
344  break;
345  }
346 
347  default: newBB = nullptr; break;
348  }
349 
350  if (newBB == nullptr) {
351  return false;
352  }
353 
354  Statement *last = newBB->getLastRTL()->back();
355 
356  // Do extra processing for for special types of DD
357  if (last->getKind() == StmtType::Call) {
358  // Attempt to add a return BB if the delay instruction is a RESTORE
359  CallStatement *call_stmt = static_cast<CallStatement *>(last);
360  BasicBlock *returnBB = optimizeCallReturn(call_stmt, rtl, delayRTL, proc);
361 
362  if (returnBB != nullptr) {
363  cfg->addEdge(newBB, returnBB);
364 
365  // We have to set the epilogue for the enclosing procedure (all proc's must have an
366  // epilogue) and remove the RESTORE in the delay slot that has just been pushed to the
367  // list of RTLs proc->setEpilogue(new CalleeEpilogue("__dummy",std::list<QString>()));
368  // Set the return location; this is now always %o0
369  // setReturnLocations(proc->getEpilogue(), 8 /* %o0 */);
370  newBB->removeRTL(delayRTL);
371 
372  // Add this call to the list of calls to analyse. We won't be able to analyse its
373  // callee(s), of course.
374  callList.push_back(call_stmt);
375 
376  return false;
377  }
378  else {
379  // Instead, add the standard out edge to original address+8 (now just address)
380  cfg->addEdge(newBB, address);
381  }
382 
383  // Add this call to the list of calls to analyse. We won't be able to analyse its callee(s),
384  // of course.
385  callList.push_back(call_stmt);
386  }
387 
388  // Set the address of the lexical successor of the call that is to be decoded next and create a
389  // new list of RTLs for the next basic block.
390  assert(BB_rtls == nullptr);
391  return !isRetOrCase;
392 }

395 bool SPARCFrontEnd::case_SCD(Address &address, ptrdiff_t delta, Interval<Address> textLimit,
396  DecodeResult &inst, DecodeResult &delay_inst,
397  std::unique_ptr<RTLList> BB_rtls, ProcCFG *cfg, TargetQueue &tq)
398 {
399  GotoStatement *jumpStmt = static_cast<GotoStatement *>(inst.rtl->back());
400  Address jumpDest = jumpStmt->getFixedDest();
401 
402  // Assume that if we find a call in the delay slot, it's actually a pattern such as
403  // move/call/move MVE: Check this! Only needed for HP PA/RISC
404  bool delayPattern = delay_inst.rtl->isCall();
405 
406  if (delayPattern) {
407  // Just emit the branch, and decode the instruction immediately following next.
408  // Assumes the first instruction of the pattern is not used in the true leg
409  BB_rtls->push_back(std::move(inst.rtl));
410  BasicBlock *newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
411 
412  if (newBB == nullptr) {
413  return false;
414  }
415 
416  createJumpToAddress(jumpDest, newBB, cfg, tq, textLimit);
417  // Add the "false" leg
418  cfg->addEdge(newBB, address + inst.numBytes);
419  address += inst.numBytes; // Skip the SCD only
420  // Start a new list of RTLs for the next BB
421  BB_rtls = nullptr;
422  LOG_WARN("Instruction at address %1 not copied to true leg of preceding branch", address);
423  return true;
424  }
425 
426  // If delay_insn decoded to empty list ( NOP) or if it isn't a flag assign => Put delay inst
427  // first
428  if (delay_inst.rtl->empty() || !delay_inst.rtl->back()->isFlagAssign()) {
429  if (delay_inst.type != NOP) {
430  // Emit delay instr
431  // This is in case we have an in-edge to the branch. If the BB is split, we want the
432  // split to happen here, so this delay instruction is active on this path
433  delay_inst.rtl->setAddress(address);
434 
435  BB_rtls->push_back(std::move(delay_inst.rtl));
436  }
437 
438  // Now emit the branch BB
439  BB_rtls->push_back(std::move(inst.rtl));
440  BasicBlock *newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
441 
442  if (newBB == nullptr) {
443  return false;
444  }
445 
446  // "taken" branch
447  createJumpToAddress(jumpDest, newBB, cfg, tq, textLimit);
448 
449  // Skip the NCT/NOP instruction (we already emitted it)
450  cfg->addEdge(newBB, address + 8);
451  address += 8;
452  }
453  else if (canOptimizeDelayCopy(address, jumpDest, delta, textLimit)) {
454  // We can just branch to the instr before jumpDest. Adjust the destination of the branch
455  jumpStmt->adjustFixedDest(-4);
456  // Now emit the branch
457  BB_rtls->push_back(std::move(inst.rtl));
458  BasicBlock *newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
459  assert(newBB);
460 
461  createJumpToAddress(jumpDest - 4, newBB, cfg, tq, textLimit);
462  // Add the "false" leg: point to the delay inst
463  cfg->addEdge(newBB, address + 4);
464  address += 4; // Skip branch but not delay
465  }
466  else { // The CCs are affected, and we can't use the copy delay slot trick
467  // SCD, must copy delay instr to orphan
468  // Copy the delay instruction to the dest of the branch, as an orphan. First add the branch.
469  BB_rtls->push_back(std::move(inst.rtl));
470  // Make a BB for the current list of RTLs. We want to do this first, else ordering can go
471  // silly
472  BasicBlock *newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
473  assert(newBB);
474 
475  // Visit the target of the branch
476  tq.visit(cfg, jumpDest, newBB);
477  std::unique_ptr<RTLList> orphanBBRTLs(new RTLList);
478 
479  // Add a branch from the orphan instruction to the dest of the branch.
480  delay_inst.rtl->append(new GotoStatement(jumpDest));
481  orphanBBRTLs->push_back(std::move(delay_inst.rtl));
482 
483  BasicBlock *orphanBB = cfg->createBB(BBType::Oneway, std::move(orphanBBRTLs));
484 
485  // Add an out edge from the orphan as well
486  cfg->addEdge(orphanBB, jumpDest);
487 
488  // Add an out edge from the current RTL to the orphan. Put a label at the orphan
489  cfg->addEdge(newBB, orphanBB);
490 
491  // Add the "false" leg to the NCT
492  cfg->addEdge(newBB, address + 4);
493  // Don't skip the delay instruction, so it will be decoded next.
494  address += 4;
495  }
496 
497  assert(BB_rtls == nullptr);
498  return true;
499 }

502 bool SPARCFrontEnd::case_SCDAN(Address &address, ptrdiff_t delta, Interval<Address> textLimit,
503  DecodeResult &inst, DecodeResult &delayInst,
504  std::unique_ptr<RTLList> BB_rtls, ProcCFG *cfg, TargetQueue &tq)
505 {
506  // We may have to move the delay instruction to an orphan BB, which then branches to the target
507  // of the jump. Instead of moving the delay instruction to an orphan BB, we may have a duplicate
508  // of the delay instruction just before the target; if so, we can branch to that and not need
509  // the orphan. We do just a binary comparison; that may fail to make this optimisation if the
510  // instr has relative fields.
511  GotoStatement *jumpStmt = static_cast<GotoStatement *>(inst.rtl->back());
512  Address jumpDest = jumpStmt->getFixedDest();
513  BasicBlock *newBB = nullptr;
514 
515  if (canOptimizeDelayCopy(address, jumpDest, delta, textLimit)) {
516  // Adjust the destination of the branch
517  jumpStmt->adjustFixedDest(-4);
518  // Now emit the branch
519  BB_rtls->push_back(std::move(inst.rtl));
520  newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
521  assert(newBB);
522 
523  createJumpToAddress(jumpDest - 4, newBB, cfg, tq, textLimit);
524  }
525  else { // SCDAN; must move delay instr to orphan. Assume it's not a NOP (though if it is, no
526  // harm done)
527  // Move the delay instruction to the dest of the branch, as an orphan. First add the branch.
528  BB_rtls->push_back(std::move(inst.rtl));
529 
530  // Make a BB for the current list of RTLs. We want to do this first, else ordering can go
531  // silly
532  newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
533  assert(newBB);
534 
535  // Visit the target of the branch
536  tq.visit(cfg, jumpDest, newBB);
537 
538  std::unique_ptr<RTLList> orphanRTL(new RTLList);
539 
540  // Also add a branch from the orphan instruction to the dest of the branch
541  delayInst.rtl->append(new GotoStatement(jumpDest));
542  orphanRTL->push_back(std::move(delayInst.rtl));
543 
544  BasicBlock *orphanBB = cfg->createBB(BBType::Oneway, std::move(orphanRTL));
545 
546  // Add an out edge from the orphan as well. Set a label there.
547  cfg->addEdge(orphanBB, jumpDest);
548 
549  // Add an out edge from the current RTL to
550  // the orphan. Set a label there.
551  cfg->addEdge(newBB, orphanBB);
552  }
553 
554  // Both cases (orphan or not)
555  // Add the "false" leg: point past delay inst. Set a label there (see below)
556  cfg->addEdge(newBB, address + 8);
557 
558  // Could need a jump to the following BB, e.g. if jumpDest is the delay slot instruction itself!
559  // e.g. beq,a $+8
560 
561  address += 8; // Skip branch and delay
562  return true;
563 }

566 bool SPARCFrontEnd::processProc(UserProc *proc, Address pc)
567 {
568  // Declare an object to manage the queue of targets not yet processed yet.
569  // This has to be individual to the procedure! (so not a global)
570  TargetQueue _targetQueue(m_program->getProject()->getSettings()->traceDecoder);
571 
572  // Similarly, we have a set of CallStatement pointers. These may be
573  // disregarded if this is a speculative decode that fails (i.e. an illegal
574  // instruction is found). If not, this set will be used to add to the set
575  // of calls to be analysed in the cfg, and also to call prog.visitProc()
576  std::list<CallStatement *> callList;
577 
578  // Indicates whether or not the next instruction to be decoded is the
579  // lexical successor of the current one. Will be true for all NCTs and for
580  // CTIs with a fall through branch.
581  bool sequentialDecode = true;
582 
583  // The control flow graph of the current procedure
584  ProcCFG *cfg = proc->getCFG();
585  assert(cfg);
586 
587  // Initialise the queue of control flow targets that have yet to be decoded.
588  _targetQueue.initial(pc);
589 
590  // Get the next address from which to continue decoding and go from
591  // there. Exit the loop if there are no more addresses or they all
592  // correspond to locations that have been decoded.
593  while ((pc = _targetQueue.getNextAddress(*cfg)) != Address::INVALID) {
594  // The list of RTLs for the current basic block
595  std::unique_ptr<RTLList> BB_rtls(new RTLList);
596  DecodeResult inst;
597 
598  // Keep decoding sequentially until a CTI
599  // without a fall through branch is decoded
600  while (sequentialDecode) {
601  // Decode and classify the current source instruction
602  if (m_program->getProject()->getSettings()->traceDecoder) {
603  LOG_MSG("*%1", pc);
604  }
605 
606  // Check if this is an already decoded jump instruction (from a previous pass with
607  // propagation etc) If so, we don't need to decode this instruction
608  std::map<Address, RTL *>::iterator ff = m_previouslyDecoded.find(pc);
609 
610  if (ff != m_previouslyDecoded.end()) {
611  inst.rtl.reset(ff->second);
612  inst.valid = true;
613  inst.type = DD; // E.g. decode the delay slot instruction
614  }
615  else if (!decodeSingleInstruction(pc, inst)) {
616  warnInvalidInstruction(pc);
617  sequentialDecode = false;
618  continue;
619  }
620 
621  // Display RTL representation if asked
622  if (m_program->getProject()->getSettings()->printRTLs) {
623  QString tgt;
624  OStream st(&tgt);
625  inst.rtl->print(st);
626  LOG_MSG(tgt);
627  }
628 
629  assert(inst.numBytes == 4); // all instructions have the same length
630 
631  // Need to construct a new list of RTLs if a basic block has just been finished but
632  // decoding is continuing from its lexical successor
633  if (BB_rtls == nullptr) {
634  BB_rtls.reset(new RTLList);
635  }
636 
637  // Define aliases to the RTLs so that they can be treated as a high level types where
638  // appropriate.
639  RTL *rtl = inst.rtl.get();
640  GotoStatement *jumpStmt = nullptr;
641  Statement *last = nullptr;
642 
643  if (!rtl->empty()) {
644  last = rtl->back();
645  jumpStmt = static_cast<GotoStatement *>(last);
646  }
647 
648  switch (inst.type) {
649  case NCT: {
650  // Ret/restore epilogues are handled as ordinary RTLs now
651  if (last && last->getKind() == StmtType::Ret) {
652  sequentialDecode = false;
653  }
654  }
655  // fallthrough
656 
657  case NOP: {
658  // Always put the NOP into the BB. It may be needed if it is the
659  // the destinsation of a branch. Even if not the start of a BB,
660  // some other branch may be discovered to it later.
661  BB_rtls->push_back(std::move(inst.rtl));
662  pc += inst.numBytes;
663  break;
664  }
665 
666  case SKIP: {
667  // We can't simply ignore the skipped delay instruction as there
668  // will most likely be a branch to it so we simply set the jump
669  // to go to one past the skipped instruction.
670  if (jumpStmt) {
671  jumpStmt->setDest(pc + 2 * inst.numBytes);
672  }
673 
674  BB_rtls->push_back(std::move(inst.rtl));
675  BasicBlock *newBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
676  assert(newBB);
677 
678  createJumpToAddress(pc + 2 * inst.numBytes, newBB, cfg, _targetQueue,
679  m_program->getBinaryFile()->getImage()->getLimitText());
680 
681  // There is no fall through branch.
682  sequentialDecode = false;
683  break;
684  }
685 
686  case SU: {
687  // Ordinary, non-delay branch.
688  BB_rtls->push_back(std::move(inst.rtl));
689 
690  BasicBlock *newBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
691  assert(newBB);
692 
693  if (jumpStmt) {
694  createJumpToAddress(jumpStmt->getFixedDest(), newBB, cfg, _targetQueue,
695  m_program->getBinaryFile()->getImage()->getLimitText());
696  }
697 
698  // There is no fall through branch.
699  sequentialDecode = false;
700  break;
701  }
702 
703  case SD: {
704  // This includes "call" and "ba". If a "call", it might be a move_call_move idiom,
705  // or a call to .stret4
706  DecodeResult delayInst;
707  if (!decodeSingleInstruction(pc + inst.numBytes, delayInst)) {
708  warnInvalidInstruction(pc + inst.numBytes);
709  sequentialDecode = false;
710  continue;
711  }
712 
713  if (m_program->getProject()->getSettings()->traceDecoder) {
714  LOG_MSG("*%1", pc + 4);
715  }
716 
717  if (!last) {
718  LOG_ERROR("Cannot decode Static Delayed branch at address %1: "
719  "semantics are empty",
720  rtl->getAddress());
721  break;
722  }
723  else if (last->getKind() == StmtType::Call) {
724  // Check the delay slot of this call. First case of interest is when the
725  // instruction is a restore, e.g.
726  // 142c8: 40 00 5b 91 call exit
727  // 142cc: 91 e8 3f ff restore %g0, -1, %o0
728  const ptrdiff_t delta = m_program->getBinaryFile()->getImage()->getTextDelta();
729  if (m_decoder->isSPARCRestore(pc + inst.numBytes, delta)) {
730  // Give the address of the call; I think that this is actually important, if
731  // faintly annoying
732  delayInst.rtl->setAddress(pc);
733  BB_rtls->push_back(std::move(delayInst.rtl));
734 
735  // The restore means it is effectively followed by a return (since the
736  // resore semantics chop off one level of return address)
737  static_cast<CallStatement *>(last)->setReturnAfterCall(true);
738  sequentialDecode = false;
739  case_CALL(pc, inst, nop_inst, BB_rtls, proc, callList, true);
740  break;
741  }
742 
743  // Next class of interest is if it assigns to %o7 (could be a move, add, and
744  // possibly others). E.g.:
745  // 14e4c: 82 10 00 0f mov %o7, %g1
746  // 14e50: 7f ff ff 60 call blah
747  // 14e54: 9e 10 00 01 mov %g1, %o7
748  // Note there could be an unrelated instruction between the first move and the
749  // call (move/x/call/move in UQBT terms). In boomerang, we leave the semantics
750  // of the moves there (to be likely removed by dataflow analysis) and merely
751  // insert a return BB after the call Note that if an add, there may be an
752  // assignment to a temp register first. So look at last RT
753  // TODO: why would delay_inst.rtl->empty() be empty here ?
754  Statement *a = delayInst.rtl->empty() ? nullptr : delayInst.rtl->back();
755 
756  if (a && a->isAssign()) {
757  SharedExp lhs = static_cast<Assign *>(a)->getLeft();
758 
759  if (lhs->isRegN(REG_SPARC_O7)) {
760  // If it's an add, this is special. Example:
761  // call foo
762  // add %o7, K, %o7
763  // is equivalent to call foo / ba .+K
764  SharedExp rhs = static_cast<Assign *>(a)->getRight();
765  auto o7(Location::regOf(REG_SPARC_O7));
766 
767  if (rhs->getOper() == opPlus && rhs->access<Exp, 2>()->isIntConst() &&
768  *rhs->getSubExp1() == *o7) {
769  // Get the constant
770  const int K = rhs->access<Const, 2>()->getInt();
771  case_CALL(pc, inst, delayInst, BB_rtls, proc, callList, true);
772 
773  // We don't generate a goto; instead, we just decode from the new
774  // address Note: the call to case_CALL has already incremented
775  // address by 8, so don't do again
776  pc += K;
777  break;
778  }
779  else {
780  // We assume this is some sort of move/x/call/move pattern. The
781  // overall effect is to pop one return address, we we emit a return
782  // after this call
783  static_cast<CallStatement *>(last)->setReturnAfterCall(true);
784  sequentialDecode = false;
785  case_CALL(pc, inst, delayInst, BB_rtls, proc, callList, true);
786  break;
787  }
788  }
789  }
790  }
791 
792  const RTL *delayRTL = delayInst.rtl.get();
793 
794  switch (delayInst.type) {
795  case NOP:
796  case NCT:
797 
798  // Ordinary delayed instruction. Since NCT's can't affect unconditional jumps,
799  // we put the delay instruction before the jump or call
800  if (last->getKind() == StmtType::Call) {
801  // This is a call followed by an NCT/NOP
802  sequentialDecode = case_CALL(pc, inst, delayInst, BB_rtls, proc, callList);
803  }
804  else {
805  // This is a non-call followed by an NCT/NOP
806  case_SD(pc, m_program->getBinaryFile()->getImage()->getTextDelta(),
807  m_program->getBinaryFile()->getImage()->getLimitText(), inst,
808  delayInst, std::move(BB_rtls), cfg, _targetQueue);
809 
810  // There is no fall through branch.
811  sequentialDecode = false;
812  }
813 
814  break;
815 
816  case SKIP:
817  case_unhandled_stub(pc);
818  pc += 2 * inst.numBytes;
819  break;
820 
821  case SU: {
822  // SD/SU.
823  // This will be either BA or CALL followed by BA,A. Our interpretation is that
824  // it is as if the SD (i.e. the BA or CALL) now takes the destination of the SU
825  // (i.e. the BA,A). For example:
826  // call 1000, ba,a 2000
827  // is really like:
828  // call 2000.
829 
830  // Just so that we can check that our interpretation is correct the first time
831  // we hit this case...
832  case_unhandled_stub(pc);
833 
834  // Adjust the destination of the SD and emit it.
835  const GotoStatement *delayJump = static_cast<const GotoStatement *>(
836  delayRTL->back());
837  const Address dest = pc + inst.numBytes + delayJump->getFixedDest();
838  jumpStmt->setDest(dest);
839  BB_rtls->push_back(std::move(inst.rtl));
840 
841  // Create the appropriate BB
842  if (last->getKind() == StmtType::Call) {
843  BasicBlock *newBB = cfg->createBB(BBType::Call, std::move(BB_rtls));
844  assert(newBB);
845 
846  createCallToAddress(dest, pc, newBB, cfg, 2 * inst.numBytes);
847  pc += 2 * inst.numBytes;
848 
849  // Add this call site to the set of call sites which need to be analyzed
850  // later.
851  callList.push_back(static_cast<CallStatement *>(inst.rtl->back()));
852  }
853  else {
854  BasicBlock *newBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
855  assert(newBB);
856  createJumpToAddress(dest, newBB, cfg, _targetQueue,
857  m_program->getBinaryFile()->getImage()->getLimitText());
858 
859  // There is no fall through branch.
860  sequentialDecode = false;
861  }
862 
863  break;
864  }
865 
866  default:
867  case_unhandled_stub(pc);
868  pc += 2 * inst.numBytes; // Skip the pair
869  break;
870  }
871 
872  break;
873  }
874 
875  case DD: {
876  DecodeResult delayInst;
877  if (!decodeSingleInstruction(pc + inst.numBytes, delayInst)) {
878  warnInvalidInstruction(pc + inst.numBytes);
879  sequentialDecode = false;
880  continue;
881  }
882 
883  switch (delayInst.type) {
884  case NOP:
885  case NCT:
886  sequentialDecode = case_DD(
887  pc, m_program->getBinaryFile()->getImage()->getTextDelta(), inst, delayInst,
888  std::move(BB_rtls), _targetQueue, proc, callList);
889  break;
890 
891  default: case_unhandled_stub(pc); break;
892  }
893 
894  break;
895  }
896 
897  case SCD: {
898  // Always execute the delay instr, and branch if condition is met.
899  // Normally, the delayed instruction moves in front of the branch. But if it affects
900  // the condition codes, we may have to duplicate it as an orphan in the true leg of
901  // the branch, and fall through to the delay instruction in the "false" leg. Instead
902  // of moving the delay instruction to an orphan BB, we may have a duplicate of the
903  // delay instruction just before the target; if so, we can branch to that and not
904  // need the orphan. We do just a binary comparison; that may fail to make this
905  // optimisation if the instr has relative fields.
906 
907  DecodeResult delayInst;
908  if (!decodeSingleInstruction(pc + inst.numBytes, delayInst)) {
909  warnInvalidInstruction(pc + inst.numBytes);
910  sequentialDecode = false;
911  continue;
912  }
913 
914  switch (delayInst.type) {
915  case NOP:
916  case NCT:
917  sequentialDecode = case_SCD(
918  pc, m_program->getBinaryFile()->getImage()->getTextDelta(),
919  m_program->getBinaryFile()->getImage()->getLimitText(), inst, delayInst,
920  std::move(BB_rtls), cfg, _targetQueue);
921  break;
922 
923  default:
924 
925  if (delayInst.rtl->back()->getKind() == StmtType::Call) {
926  // Assume it's the move/call/move pattern
927  sequentialDecode = case_SCD(
928  pc, m_program->getBinaryFile()->getImage()->getTextDelta(),
929  m_program->getBinaryFile()->getImage()->getLimitText(), inst, delayInst,
930  std::move(BB_rtls), cfg, _targetQueue);
931  break;
932  }
933 
934  case_unhandled_stub(pc);
935  break;
936  }
937 
938  break;
939  }
940 
941  case SCDAN: {
942  // Execute the delay instruction if the branch is taken; skip (anull) the delay
943  // instruction if branch not taken.
944  DecodeResult delayInst;
945  if (!decodeSingleInstruction(pc + inst.numBytes, delayInst)) {
946  warnInvalidInstruction(pc + inst.numBytes);
947  sequentialDecode = false;
948  continue;
949  }
950 
951  switch (delayInst.type) {
952  case NOP: {
953  // This is an ordinary two-way branch. Add the branch to the list of RTLs for
954  // this BB
955  BB_rtls->push_back(std::move(inst.rtl));
956  // Create the BB and add it to the CFG
957  BasicBlock *newBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
958  assert(newBB);
959 
960  // Visit the destination of the branch; add "true" leg
961  const Address jumpDest = jumpStmt ? jumpStmt->getFixedDest() : Address::INVALID;
962  createJumpToAddress(jumpDest, newBB, cfg, _targetQueue,
963  m_program->getBinaryFile()->getImage()->getLimitText());
964 
965  // Add the "false" leg: point past the delay inst
966  cfg->addEdge(newBB, pc + 8);
967  pc += 2 * inst.numBytes; // Skip branch and delay
968  break;
969  }
970 
971  case NCT:
972  sequentialDecode = case_SCDAN(
973  pc, m_program->getBinaryFile()->getImage()->getTextDelta(),
974  m_program->getBinaryFile()->getImage()->getLimitText(), inst, delayInst,
975  std::move(BB_rtls), cfg, _targetQueue);
976  break;
977 
978  default:
979  case_unhandled_stub(pc);
980  pc += 2 * inst.numBytes;
981  break;
982  }
983 
984  break;
985  }
986 
987  default: // Others are non SPARC cases
988  LOG_WARN("Encountered instruction class '%1' which is invalid for SPARC",
989  (int)inst.type);
990  break;
991  }
992 
993  // If sequentially decoding, check if the next address happens to be the start of an
994  // existing BB. If so, finish off the current BB (if any RTLs) as a fallthrough, and no
995  // need to decode again (unless it's an incomplete BB, then we do decode it). In fact,
996  // mustn't decode twice, because it will muck up the coverage, but also will cause
997  // subtle problems like add a call to the list of calls to be processed, then delete the
998  // call RTL (e.g. Pentium 134.perl benchmark)
999  if (sequentialDecode && cfg->isStartOfBB(pc)) {
1000  // Create the fallthrough BB, if there are any RTLs at all
1001  if (BB_rtls) {
1002  BasicBlock *newBB = cfg->createBB(BBType::Fall, std::move(BB_rtls));
1003  assert(newBB);
1004  // Add an out edge to this address
1005  cfg->addEdge(newBB, pc);
1006  }
1007 
1008  // Pick a new address to decode from, if the BB is complete
1009  if (!cfg->isStartOfIncompleteBB(pc)) {
1010  sequentialDecode = false;
1011  }
1012  }
1013  } // while (sequentialDecode)
1014 
1015  sequentialDecode = true;
1016  } // End huge while loop
1017 
1018  // Add the callees to the set of CallStatements to proces for parameter recovery, and also to
1019  // the Prog object
1020  for (CallStatement *call : callList) {
1021  Address dest = call->getFixedDest();
1022 
1023  // Don't visit the destination of a register call
1024  if (dest != Address::INVALID) {
1025  Function *callee = proc->getProg()->getOrCreateFunction(dest);
1026  if (callee) {
1027  proc->addCallee(callee);
1028  }
1029  }
1030  }
1031 
1032  // MVE: Not 100% sure this is the right place for this
1033  proc->setEntryBB();
1034 
1035  return true;
1036 }

1076 bool SPARCFrontEnd::isHelperFunc(Address dest, Address addr, RTLList &lrtl)
1077 {
1078  const BinarySymbol *sym = m_program->getBinaryFile()->getSymbols()->findSymbolByAddress(dest);
1079 
1080  if (!(sym && sym->isImportedFunction())) {
1081  return false;
1082  }
1083 
1084  QString name = m_program->getSymbolNameByAddr(dest);
1085 
1086  if (name.isEmpty()) {
1087  LOG_ERROR("Can't find symbol for PLT address %1", dest);
1088  return false;
1089  }
1090 
1091  SharedExp rhs;
1092 
1093  if (name == ".umul") {
1094  // %o0 * %o1
1095  rhs = Binary::get(opMult, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1096  }
1097  else if (name == ".mul") {
1098  // %o0 *! %o1
1099  rhs = Binary::get(opMults, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1100  }
1101  else if (name == ".udiv") {
1102  // %o0 / %o1
1103  rhs = Binary::get(opDiv, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1104  }
1105  else if (name == ".div") {
1106  // %o0 /! %o1
1107  rhs = Binary::get(opDivs, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1108  }
1109  else if (name == ".urem") {
1110  // %o0 % %o1
1111  rhs = Binary::get(opMod, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1112  }
1113  else if (name == ".rem") {
1114  // %o0 %! %o1
1115  rhs = Binary::get(opMods, Location::regOf(REG_SPARC_O0), Location::regOf(REG_SPARC_O1));
1116  // } else if (name.substr(0, 6) == ".stret") {
1117  // // No operation. Just use %o0
1118  // rhs->push(idRegOf); rhs->push(idIntConst); rhs->push(8);
1119  }
1120  else if (name == "_Q_mul") {
1121  // Pointers to args are in %o0 and %o1; ptr to result at [%sp+64]
1122  // So semantics is m[m[r[14]] = m[r[8]] *f m[r[9]]
1123  quadOperation(addr, lrtl, opFMult);
1124  return true;
1125  }
1126  else if (name == "_Q_div") {
1127  quadOperation(addr, lrtl, opFDiv);
1128  return true;
1129  }
1130  else if (name == "_Q_add") {
1131  quadOperation(addr, lrtl, opFPlus);
1132  return true;
1133  }
1134  else if (name == "_Q_sub") {
1135  quadOperation(addr, lrtl, opFMinus);
1136  return true;
1137  }
1138  else {
1139  // Not a (known) helper function
1140  return false;
1141  }
1142 
1143  // Need to make an RTAssgn with %o0 = rhs
1144  SharedExp lhs = Location::regOf(REG_SPARC_O0);
1145 
1146  lrtl.push_back(std::unique_ptr<RTL>(new RTL(addr, { new Assign(lhs, rhs) })));
1147  return true;
1148 }

1151 void SPARCFrontEnd::gen32op32gives64(OPER op, RTLList &lrtl, Address addr)
1152 {
1153  std::list<Statement *> *ls = new std::list<Statement *>;
1154 #if V9_ONLY
1155  // tmp[tmpl] = sgnex(32, 64, r8) op sgnex(32, 64, r9)
1156  Statement *a = new Assign(
1157  64, Location::tempOf(Const::get("tmpl")),
1158  Binary::get(op, // opMult or opMults
1159  new Ternary(opSgnEx, Const(32), Const(64), Location::regOf(REG_SPARC_O0)),
1160  new Ternary(opSgnEx, Const(32), Const(64), Location::regOf(REG_SPARC_O1))));
1161  ls->push_back(a);
1162  // r8 = truncs(64, 32, tmp[tmpl]);
1163  a = new Assign(32, Location::regOf(REG_SPARC_O0),
1164  new Ternary(opTruncs, Const::get(64), Const::get(32),
1165  Location::tempOf(Const::get("tmpl"))));
1166  ls->push_back(a);
1167  // r9 = r[tmpl]@32:63;
1168  a = new Assign(
1169  32, Location::regOf(REG_SPARC_O1),
1170  new Ternary(opAt, Location::tempOf(Const::get("tmpl")), Const::get(32), Const::get(63)));
1171  ls->push_back(a);
1172 #else
1173  // BTL: The .umul and .mul support routines are used in V7 code. We implsment these using the V8
1174  // UMUL and SMUL instructions. BTL: In SPARC V8, UMUL and SMUL perform 32x32 -> 64 bit
1175  // multiplies.
1176  // The 32 high-order bits are written to %Y and the 32 low-order bits are written to
1177  // r[rd]. This is also true on V9 although the high-order bits are also written into the
1178  // 32 high-order bits of the 64 bit r[rd].
1179 
1180  // r[tmp] = r8 op r9
1181  Assign *a = new Assign(Location::tempOf(Const::get(const_cast<char *>("tmp"))),
1182  Binary::get(op, // opMult or opMults
1183  Location::regOf(REG_SPARC_O0),
1184  Location::regOf(REG_SPARC_O1)));
1185  ls->push_back(a);
1186  // r8 = r[tmp]; /* low-order bits */
1187  a = new Assign(Location::regOf(REG_SPARC_O0),
1188  Location::tempOf(Const::get(const_cast<char *>("tmp"))));
1189  ls->push_back(a);
1190  // r9 = %Y; /* high-order bits */
1191  a = new Assign(Location::regOf(REG_SPARC_O0),
1192  Unary::get(opMachFtr, Const::get(const_cast<char *>("%Y"))));
1193  ls->push_back(a);
1194 #endif /* V9_ONLY */
1195 
1196  lrtl.push_back(std::unique_ptr<RTL>(new RTL(addr, ls)));

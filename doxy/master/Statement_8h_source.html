39 /// Types of Statements, or high-level register transfer lists.
40 enum class StmtType : uint8_t
41 {
42  INVALID = 0,
43  Assign = 1, ///< Ordinary Assignment x := y
44  PhiAssign, ///< x := phi(a, b, c)
45  ImpAssign, ///< Implicit assignment (potential function parameter) x := -
46  BoolAssign, ///< For "setCC" instructions
47  Call, ///< Call statements rets := CALL foo(args)
48  Ret, ///< Return
49  Branch, ///< Conditional jump / if-then
50  Goto, ///< Unconditional jump / goto

28 bool UnusedStatementRemovalPass::execute(UserProc *proc)
29 {
30  // Only remove unused statements after decompiling as much as possible of the proc
31  // Remove unused statements
32  RefCounter refCounts; // The map
33  // Count the references first
34  updateRefCounts(proc, refCounts);
35 
36  // Now remove any that have no used
37  if (proc->getProg()->getProject()->getSettings()->removeNull) {
38  remUnusedStmtEtc(proc, refCounts);
39  removeNullStatements(proc);
40  proc->debugPrintAll("after removing unused and null statements pass 1");
41  }
42 
43  return true;
44 }

47 void UnusedStatementRemovalPass::updateRefCounts(UserProc *proc, RefCounter &refCounts)
48 {
49  StatementList stmts;
50  proc->getStatements(stmts);
51 
52  for (Statement *s : stmts) {
53  // Don't count uses in implicit statements. There is no RHS of course,
54  // but you can still have x from m[x] on the LHS and so on, but these are not real uses
55  if (s->isImplicit()) {
56  continue;
57  }
58 
59  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
60  LOG_MSG("Counting references in %1", s);
61  }
62 
63  LocationSet refs;
64  s->addUsedLocs(refs, false); // Ignore uses in collectors
65 
66  for (const SharedExp &rr : refs) {
67  if (rr->isSubscript()) {
68  Statement *def = rr->access<RefExp>()->getDef();
69 
70  // Used to not count implicit refs here (def->getNumber() == 0), meaning that
71  // implicit definitions get removed as dead code! But these are the ideal place to
72  // read off final parameters, and it is guaranteed now that implicit statements are
73  // sorted out for us by now (for dfa type analysis)
74  if (def /* && def->getNumber() */) {
75  refCounts[def]++;
76 
77  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
78  LOG_MSG("counted ref to %1", rr);
79  }
80  }
81  }
82  }
83  }
84 
85  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
86  LOG_MSG("### Reference counts for %1:", proc->getName());
87 
88  for (RefCounter::iterator rr = refCounts.begin(); rr != refCounts.end(); ++rr) {
89  LOG_MSG(" %1: %2", rr->first->getNumber(), rr->second);
90  }
91 
92  LOG_MSG("### End reference counts");
93  }
94 }

97 void UnusedStatementRemovalPass::remUnusedStmtEtc(UserProc *proc, RefCounter &refCounts)
98 {
99  StatementList stmts;
100  proc->getStatements(stmts);
101  bool change;
102 
103  do { // FIXME: check if this is ever needed
104  change = false;
105  StatementList::iterator ll = stmts.begin();
106 
107  while (ll != stmts.end()) {
108  Statement *s = *ll;
109 
110  if (!s->isAssignment()) {
111  // Never delete a statement other than an assignment (e.g. nothing "uses" a Jcond)
112  ++ll;
113  continue;
114  }
115 
116  const Assignment *as = static_cast<const Assignment *>(s);
117  SharedConstExp asLeft = as->getLeft();
118 
119  if (asLeft && (asLeft->getOper() == opGlobal)) {
120  // assignments to globals must always be kept
121  ++ll;
122  continue;
123  }
124 
125  // If it's a memof and renameable it can still be deleted
126  if (asLeft->isMemOf() && !proc->canRename(asLeft)) {
127  // Assignments to memof-anything-but-local must always be kept.
128  ++ll;
129  continue;
130  }
131 
132  if (asLeft->isMemberOf() || asLeft->isArrayIndex()) {
133  // can't say with these; conservatively never remove them
134  ++ll;
135  continue;
136  }
137 
138  // Care not to insert unnecessarily
139  if ((refCounts.find(s) == refCounts.end()) || (refCounts[s] == 0)) {
140  // First adjust the counts, due to statements only referenced by statements that are
141  // themselves unused. Need to be careful not to count two refs to the same def as
142  // two; refCounts is a count of the number of statements that use a definition, not
143  // the total number of refs
144  StatementSet stmtsRefdByUnused;
145  LocationSet components;
146  // Second parameter false to ignore uses in collectors
147  s->addUsedLocs(components, false);
148 
149  for (const SharedExp &component : components) {
150  if (component->isSubscript() && component->access<RefExp>()->getDef()) {
151  stmtsRefdByUnused.insert(component->access<RefExp>()->getDef());
152  }
153  }
154 
155  for (Statement *refd : stmtsRefdByUnused) {
156  if (refd == nullptr) {
157  continue;
158  }
159 
160  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
161  LOG_MSG("Decrementing ref count of %1 because %2 is unused",
162  refd->getNumber(), s->getNumber());
163  }
164 
165  refCounts[refd]--;
166  }
167 
168  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
169  LOG_MSG("Removing unused statement %1 %2", s->getNumber(), s);
170  }
171 
172  proc->removeStatement(s);
173  ll = stmts.erase(ll); // So we don't try to re-remove it
174  change = true;
175  continue; // Don't call getNext this time
176  }
177 
178  ++ll;
179  }
180  } while (change);
181 
182  // Recalulate at least the livenesses. Example: first call to printf in test/pentium/fromssa2,
183  // eax used only in a removed statement, so liveness in the call needs to be removed
184  PassManager::get()->executePass(PassID::CallLivenessRemoval, proc);
185  PassManager::get()->executePass(PassID::BlockVarRename, proc);
186 
187  // Now fully decompiled (apart from one final pass, and transforming out of SSA form)
188  proc->setStatus(ProcStatus::FinalDone);
189 }

192 bool UnusedStatementRemovalPass::removeNullStatements(UserProc *proc)
193 {
194  bool change = false;
195  StatementList stmts;
196  proc->getStatements(stmts);
197 
198  // remove null code
199  for (Statement *s : stmts) {
200  if (s->isNullStatement()) {
201  // A statement of the form x := x
202  LOG_VERBOSE("Removing null statement: %1 %2", s->getNumber(), s);
203 
204  proc->removeStatement(s);
205  change = true;
206  }
207  }
208 
209  return change;
210 }

32 static SharedExp defineAll = Terminal::get(opDefineAll); // An expression representing <all>
33 
34 // There is an entry in stacks[defineAll] that represents the latest definition
35 // from a define-all source. It is needed for variables that don't have a definition as yet
36 // (i.e. stacks[x].empty() is true). As soon as a real definition to x appears,
37 // stacks[defineAll] does not apply for variable x. This is needed to get correct
38 // operation of the use collectors in calls.
39 
40 // Care with the Stacks object (a map from expression to stack);
41 // using Stacks[q].empty() can needlessly insert an empty stack
42 #define STACKS_EMPTY(q) (stacks.find(q) == stacks.end() || stacks[q].empty())

45 // Subscript dataflow variables
46 bool BlockVarRenamePass::renameBlockVars(
47  UserProc *proc, int n, std::map<SharedExp, std::deque<Statement *>, lessExpStar> &stacks)
48 {
49  if (proc->getCFG()->getNumBBs() == 0) {
50  return false;
51  }
52 
53  bool changed = false;
54  const bool assumeABICompliance = proc->getProg()->getProject()->getSettings()->assumeABI;
55 
56  // For each statement S in block n
57  BasicBlock::RTLIterator rit;
58  StatementList::iterator sit;
59  BasicBlock *bb = proc->getDataFlow()->nodeToBB(n);
60 
61  for (Statement *S = bb->getFirstStmt(rit, sit); S; S = bb->getNextStmt(rit, sit)) {
62  {
63  // For each use of some variable x in S (not just assignments)
64  LocationSet locs;
65 
66  if (S->isPhi()) {
67  PhiAssign *pa = static_cast<PhiAssign *>(S);
68  SharedExp phiLeft = pa->getLeft();
69 
70  if (phiLeft->isMemOf() || phiLeft->isRegOf()) {
71  phiLeft->getSubExp1()->addUsedLocs(locs);
72  }
73 
74  // A phi statement may use a location defined in a childless call,
75  // in which case its use collector needs updating
76  for (auto &pp : *pa) {
77  Statement *def = pp->getDef();
78 
79  if (def && def->isCall()) {
80  static_cast<CallStatement *>(def)->useBeforeDefine(phiLeft->clone());
81  }
82  }
83  }
84  else { // Not a phi assignment
85  S->addUsedLocs(locs);
86  }
87 
88  for (SharedExp location : locs) {
89  // Don't rename memOfs that are not renamable according to the current policy
90  if (!proc->canRename(location)) {
91  continue;
92  }
93 
94  Statement *def = nullptr;
95 
96  if (location->isSubscript()) { // Already subscripted?
97  // No renaming required, but redo the usage analysis, in case this is a new
98  // return, and also because we may have just removed all call livenesses
99  // Update use information in calls, and in the proc (for parameters)
100  SharedExp base = location->getSubExp1();
101  def = location->access<RefExp>()->getDef();
102 
103  if (def && def->isCall()) {
104  // Calls have UseCollectors for locations that are used before definition at
105  // the call
106  static_cast<CallStatement *>(def)->useBeforeDefine(base->clone());
107  continue;
108  }
109 
110  // Update use collector in the proc (for parameters)
111  if (def == nullptr) {
112  proc->markAsInitialParam(base->clone());
113  }
114 
115  continue; // Don't re-rename the renamed variable
116  }
117 
118  if (!STACKS_EMPTY(location)) {
119  def = stacks[location].back();
120  }
121  else if (!STACKS_EMPTY(defineAll)) {
122  def = stacks[defineAll].back();
123  }
124  else {
125  // If the both stacks are empty, use a nullptr definition. This will be changed
126  // into a pointer to an implicit definition at the start of type analysis, but
127  // not until all the m[...] have stopped changing their expressions (complicates
128  // implicit assignments considerably).
129  def = nullptr;
130  // Update the collector at the start of the UserProc
131  proc->markAsInitialParam(location->clone());
132  }
133 
134 
135  if (def && def->isCall()) {
136  // Calls have UseCollectors for locations that are used before definition
137  // at the call
138  static_cast<CallStatement *>(def)->useBeforeDefine(location->clone());
139  }
140 
141  // Replace the use of x with x{def} in S
142  changed = true;
143 
144  if (S->isPhi()) {
145  SharedExp phiLeft = static_cast<PhiAssign *>(S)->getLeft();
146  phiLeft->setSubExp1(phiLeft->getSubExp1()->expSubscriptVar(location, def));
147  }
148  else {
149  subscriptVar(S, location, def);
150  }
151  }
152  }
153 
154  // MVE: Check for Call and Return Statements; these have DefCollector objects that need to
155  // be updated
156  // Do before the below, so CallStatements have not yet processed their defines
157  if (S->isCall() || S->isReturn()) {
158  DefCollector *col;
159 
160  if (S->isCall()) {
161  col = static_cast<CallStatement *>(S)->getDefCollector();
162  }
163  else {
164  col = static_cast<ReturnStatement *>(S)->getCollector();
165  }
166 
167  col->updateDefs(stacks, proc);
168  }
169 
170  // For each definition of some variable a in S
171  LocationSet defs;
172  S->getDefinitions(defs, assumeABICompliance);
173 
174  for (SharedExp a : defs) {
175  // Don't consider a if it cannot be renamed
176  const bool suitable = proc->canRename(a);
177 
178  if (suitable) {
179  // Push i onto Stacks[a]
180  // Note: we clone a because otherwise it could be an expression
181  // that gets deleted through various modifications.
182  // This is necessary because we do several passes of this algorithm
183  // to sort out the memory expressions.
184  if (stacks.find(a) != stacks.end()) { // expression exists, no need for clone ?
185  stacks[a].push_back(S);
186  }
187  else {
188  stacks[a->clone()].push_back(S);
189  }
190 
191  // Replace definition of 'a' with definition of a_i in S (we don't do this)
192  }
193 
194  // FIXME: MVE: do we need this awful hack?
195  if (a->isLocal()) {
196  SharedConstExp a1 = S->getProc()->expFromSymbol(a->access<Const, 1>()->getStr());
197  assert(a1);
198 
199  // Stacks already has a definition for a (as just the bare local)
200  if (suitable) {
201  stacks[a1->clone()].push_back(S);
202  }
203  }
204  }
205 
206  // Special processing for define-alls (presently, only childless calls).
207  // But note that only 'everythings' at the current memory level are defined!
208  if (S->isCall() && static_cast<const CallStatement *>(S)->isChildless() &&
209  !proc->getProg()->getProject()->getSettings()->assumeABI) {
210  // S is a childless call (and we're not assuming ABI compliance)
211  stacks[defineAll]; // Ensure that there is an entry for defineAll
212 
213  for (auto &elem : stacks) {
214  // if (dd->first->isMemDepth(memDepth))
215  elem.second.push_back(S); // Add a definition for all vars
216  }
217  }
218  }
219 
220  // For each successor Y of block n
221  for (BasicBlock *Ybb : bb->getSuccessors()) {
222  // For each phi-function in Y
223  for (Statement *St = Ybb->getFirstStmt(rit, sit); St; St = Ybb->getNextStmt(rit, sit)) {
224  if (!St->isPhi()) {
225  continue;
226  }
227 
228  PhiAssign *pa = static_cast<PhiAssign *>(St);
229 
230  // Suppose the jth operand of the phi is 'a'
231  // For now, just get the LHS
232  SharedExp a = pa->getLeft();
233 
234  // Only consider variables that can be renamed
235  if (!proc->canRename(a)) {
236  continue;
237  }
238 
239  Statement *def = nullptr; // assume No reaching definition
240 
241  if (!STACKS_EMPTY(a)) {
242  def = stacks[a].back();
243  }
244 
245  // "Replace jth operand with a_i"
246  pa->putAt(bb, def, a);
247  }
248  }
249 
250  // For each child X of n
251  // Note: linear search!
252  const int numBB = proc->getCFG()->getNumBBs();
253 
254  for (int X = 0; X < numBB; X++) {
255  const int idom = proc->getDataFlow()->getIdom(X);
256  if (idom == n && X != n) { // if 'n' is immediate dominator of X
257  renameBlockVars(proc, X, stacks);
258  }
259  }
260 
261  // For each statement S in block n
262  // NOTE: Because of the need to pop childless calls from the Stacks, it is important in my
263  // algorithm to process the statments in the BB *backwards*. (It is not important in Appel's
264  // algorithm, since he always pushes a definition for every variable defined on the Stacks).
265  BasicBlock::RTLRIterator rrit;
266  StatementList::reverse_iterator srit;
267 
268  for (Statement *S = bb->getLastStmt(rrit, srit); S; S = bb->getPrevStmt(rrit, srit)) {
269  // For each definition of some variable a in S
270  LocationSet defs;
271  S->getDefinitions(defs, assumeABICompliance);
272 
273  for (const auto &def : defs) {
274  if (!proc->canRename(def)) {
275  continue;
276  }
277 
278  auto stackIt = stacks.find(def);
279 
280  if (stackIt == stacks.end()) {
281  LOG_FATAL("Tried to pop '%1' from Stacks; does not exist", def);
282  }
283 
284  stackIt->second.pop_back();
285  }
286 
287  // Pop all defs due to childless calls
288  if (S->isCall() && static_cast<const CallStatement *>(S)->isChildless()) {
289  for (auto &stack : stacks) {
290  if (!stack.second.empty() && (stack.second.back() == S)) {
291  stack.second.pop_back();
292  }
293  }
294  }
295  }
296 
297  return changed;
298 }

301 bool BlockVarRenamePass::execute(UserProc *proc)
302 {
303  /// The stack which remembers the last definition of an expression.
304  std::map<SharedExp, std::deque<Statement *>, lessExpStar> stacks;
305  BasicBlock *entryBB = proc->getCFG()->getEntryBB();
306 
307  if (entryBB == nullptr) {
308  return false;
309  }
310 
311  return renameBlockVars(proc, proc->getDataFlow()->pbbToNode(entryBB), stacks);
312 }

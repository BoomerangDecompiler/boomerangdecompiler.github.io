23 /**
24  * This class implements an ordered or unordered set of statements.
25  * \tparam T Statement type (Statement, Assign, CallStatement etc.)
26  * \tparam Sorter Binary functor type that defines the sorting order.
27  * If Sorter == void, the set is unordered.
28  */
29 template<typename T, typename Sorter = void,
30  typename Enabler = std::enable_if<std::is_base_of<Statement, T>::value>>
31 class StmtSet
32 {
33  using Set = typename std::conditional<std::is_void<Sorter>::value, std::unordered_set<T *>,
34  std::set<T *, Sorter>>::type;
35

37  typedef typename Set::iterator iterator;
38  typedef typename Set::const_iterator const_iterator;
39 
40 public:

42  iterator end() { return m_set.end(); }
43 
44  const_iterator begin() const { return m_set.begin(); }
45  const_iterator end() const { return m_set.end(); }
46 
47 public:

49  void clear() { m_set.clear(); }
50  int size() const { return m_set.size(); }

52  void insert(T *stmt)
53  {
54  assert(stmt != nullptr);
55  m_set.insert(stmt);
56  }

58  /// Remove this Statement.
59  /// \returns true if removed, false if not found
60  bool remove(T *s)
61  {

63  m_set.erase(s);
64  return true;
65  }
66 
67  return false;
68  }

73  /// \returns true if any statement in this set defines \p loc
74  bool definesLoc(SharedExp loc) const
75  {
76  if (!loc) {
77  return false;
78  }
79

84  /// \returns true if this set is a subset of \p other
85  bool isSubSetOf(const StmtSet &other)
86  {
87  if (m_set.size() > other.m_set.size()) {
88  return false;
89  }
90 
91  for (T *stmt : *this) {
92  if (!other.contains(stmt)) {
93  return false;
94  }
95  }
96 
97  return true;
98  }

100  /// Set union: this = this union \p other
101  void makeUnion(const StmtSet &other)
102  {
103  for (T *stmt : other) {
104  m_set.insert(stmt);
105  }
106  }
107 
108  /// Set intersection: this = this intersect other
109  void makeIsect(const StmtSet &other)
110  {
111  for (auto it = m_set.begin(); it != m_set.end();) {
112  if (!other.contains(*it)) {
113  // Not in both sets
114  it = m_set.erase(it);
115  }
116  else {
117  ++it;
118  }
119  }
120  }
121 
122  /// Set difference: this = this - other
123  void makeDiff(const StmtSet &other)
124  {
125  if (&other == this) {
126  m_set.clear(); // A \ A == empty set
127  return;
128  }
129 
130  for (T *stmt : other) {
131  m_set.erase(stmt);
132  }
133  }
134 
135  /// Find a definition for \p loc on the LHS of each assignment in this set.
136  /// If found, return pointer to the Assign with that LHS (else return nullptr)
137  template<typename = std::enable_if<std::is_base_of<Assign, T>::value>>
138  Assign *lookupLoc(SharedExp loc)
139  {
140  if (!loc) {
141  return nullptr;
142  }
143 
144  Assign as(loc, Terminal::get(opWild));
145  iterator ff = m_set.find(&as);
146 
147  return (ff != end()) ? *ff : nullptr;
148  }
149

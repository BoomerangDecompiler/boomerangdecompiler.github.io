152 bool UserProc::removeStatement(Statement *stmt)
153 {
154  if (!stmt) {
155  return false;
156  }
157 
158  // remove anything proven about this statement
159  for (auto provenIt = m_provenTrue.begin(); provenIt != m_provenTrue.end();) {
160  LocationSet refs;
161  provenIt->second->addUsedLocs(refs);
162  provenIt->first->addUsedLocs(
163  refs); // Could be say m[esp{99} - 4] on LHS and we are deleting stmt 99
164 
165  auto it = std::find_if(refs.begin(), refs.end(), [stmt](const SharedExp &ref) {
166  return ref->isSubscript() && ref->access<RefExp>()->getDef() == stmt;
167  });
168 
169  if (it != refs.end()) {
170  LOG_VERBOSE("Removing proven true exp %1 = %2 that uses statement being removed.",
171  provenIt->first, provenIt->second);
172 
173  provenIt = m_provenTrue.erase(provenIt);
174  continue;
175  }
176 
177  ++provenIt;
178  }
179 
180  // remove from BB/RTL
181  BasicBlock *bb = stmt->getBB(); // Get our enclosing BB
182  if (!bb) {
183  return false;
184  }
185 
186  for (auto &rtl : *bb->getRTLs()) {
187  for (RTL::iterator it = rtl->begin(); it != rtl->end(); ++it) {
188  if (*it == stmt) {
189  rtl->erase(it);
190  return true;
191  }
192  }
193  }
194 
195  return false;
196 }

199 Assign *UserProc::insertAssignAfter(Statement *s, SharedExp left, SharedExp right)
200 {
201  BasicBlock *bb = nullptr;
202  Assign *as = new Assign(left, right);
203 
204  if (s == nullptr) {
205  // This means right is supposed to be a parameter.
206  // We can insert the assignment at the start of the entryBB
207  bb = m_cfg->getEntryBB();
208  }
209  else {
210  // An ordinary definition; put the assignment right after s
211  bb = s->getBB();
212  }
213 
214  as->setProc(this);
215  as->setBB(bb);
216 
217  if (s) {
218  // Insert the new assignment directly after s,
219  // or near the end of the existing BB if s has been removed already.
220  for (auto &rtl : *bb->getRTLs()) {
221  for (auto it = rtl->begin(); it != rtl->end(); ++it) {
222  if (*it == s) {
223  rtl->insert(++it, as);
224  return as;
225  }
226  }
227  }
228  }
229 
230  auto &lastRTL = bb->getRTLs()->back();
231  if (lastRTL->empty() || lastRTL->back()->isAssignment()) {
232  lastRTL->append(as);
233  }
234  else {
235  // do not insert after a Branch statement etc.
236  lastRTL->insert(std::prev(lastRTL->end()), as);
237  }
238  return as;
239 }

242 bool UserProc::insertStatementAfter(Statement *afterThis, Statement *stmt)
243 {
244  assert(!afterThis->isBranch());
245 
246  for (BasicBlock *bb : *m_cfg) {
247  RTLList *rtls = bb->getRTLs();
248 
249  if (rtls == nullptr) {
250  continue; // e.g. bb is (as yet) invalid
251  }
252 
253  for (const auto &rtl : *rtls) {
254  for (RTL::iterator ss = rtl->begin(); ss != rtl->end(); ++ss) {
255  if (*ss == afterThis) {
256  rtl->insert(std::next(ss), stmt);
257  stmt->setBB(bb);
258  return true;
259  }
260  }
261  }
262  }
263 
264  return false;
265 }

268 Assign *UserProc::replacePhiByAssign(const PhiAssign *orig, const SharedExp &rhs)
269 {
270  // I believe we always want to propagate to these ex-phi's; check!
271  SharedExp newRhs = rhs->propagateAll();
272 
273  for (BasicBlock *bb : *m_cfg) {
274  for (const auto &rtl : *bb->getRTLs()) {
275  for (RTL::iterator ss = rtl->begin(); ss != rtl->end(); ++ss) {
276  if (*ss == orig) {
277  // convert *ss to an Assign
278  Assign *asgn(new Assign(orig->getLeft()->clone(), newRhs));
279 
280  asgn->setType(orig->getType()->clone());
281  asgn->setNumber(orig->getNumber());
282  asgn->setProc(orig->getProc());
283  asgn->setBB(bb);
284 
285  Statement *toDelete = *ss;
286  *ss = asgn;
287 
288  StatementList stmts;
289  getStatements(stmts);
290 
291  // replace all refs orig -> asgn
292  for (Statement *stmt : stmts) {
293  StmtSubscriptReplacer stmtMod(orig, asgn);
294 
295  stmt->accept(&stmtMod);
296  }
297 
298  SymbolMap newSymbols;
299 
300  for (auto it = m_symbolMap.begin(); it != m_symbolMap.end();) {
301  SharedExp exp = (*it).first->clone();
302  ExpSubscriptReplacer esr(orig, asgn);
303  exp->acceptModifier(&esr);
304 
305  if (esr.isModified()) {
306  SharedExp local = it->second;
307  it = m_symbolMap.erase(it);
308  newSymbols.insert({ exp, local });
309  }
310  else {
311  ++it;
312  }
313  }
314 
315  for (auto elem : newSymbols) {
316  m_symbolMap.insert(elem);
317  }
318 
319  delete toDelete;
320  return asgn;
321  }
322  }
323  }
324  }
325 
326  return nullptr;
327 }

339 void UserProc::insertParameter(SharedExp e, SharedType ty)
340 {
341  if (filterParams(e)) {
342  return; // Filtered out
343  }
344 
345  // Used to filter out preserved locations here: no!
346  // Propagation and dead code elimination solve the problem.
347  // See test/pentium/restoredparam for an example
348  // where you must not remove restored locations
349 
350  // Wrap it in an implicit assignment; DFA based TA should update the type later
351  ImplicitAssign *as = new ImplicitAssign(ty->clone(), e->clone());
352 
353  auto it = std::lower_bound(m_parameters.begin(), m_parameters.end(), as,
354  [this](const Statement *stmt, const Assignment *a) {
355  return m_signature->argumentCompare(
356  static_cast<const Assignment &>(*stmt), *a);
357  });
358 
359  if (it == m_parameters.end() ||
360  *static_cast<ImplicitAssign *>(*it)->getLeft() != *as->getLeft()) {
361  // not a duplicate
362  m_parameters.insert(it, as);
363  }
364 
365  // update the signature
366  m_signature->setNumParams(0);
367  int i = 1;
368 
369  for (Statement *param : m_parameters) {
370  Assignment *a = static_cast<Assignment *>(param);
371  QString paramName = QString("param%1").arg(i++);
372 
373  m_signature->addParameter(paramName, a->getLeft(), a->getType());
374  }
375 }

441 bool UserProc::filterParams(SharedExp e)
442 {
443  switch (e->getOper()) {
444  case opPC: return true;
445  case opTemp: return true;
446  case opRegOf: {
447  if (!e->isRegOfConst()) {
448  return false;
449  }
450  const int sp = Util::getStackRegisterIndex(m_prog);
451  return e->access<Const, 1>()->getInt() == sp;
452  }
453 
454  case opMemOf: {
455  SharedExp addr = e->getSubExp1();
456 
457  if (addr->isIntConst()) {
458  return true; // Global memory location
459  }
460 
461  if (addr->isSubscript() && addr->access<RefExp>()->isImplicitDef()) {
462  auto reg = addr->getSubExp1();
463  int sp = 999;
464 
465  if (m_signature) {
466  sp = Util::getStackRegisterIndex(m_prog);
467  }
468 
469  if (reg->isRegN(sp)) {
470  return true; // Filter out m[sp{-}] assuming it is the return address
471  }
472  }
473 
474  return false; // Might be some weird memory expression that is not a local
475  }
476 
477  case opGlobal:
478  return true; // Never use globals as argument locations (could appear on RHS of args)
479 
480  default: return false;
481  }
482 }

499 bool UserProc::filterReturns(SharedExp e)
500 {
501  if (isPreserved(e)) {
502  // If it is preserved, then it can't be a return (since we don't change it)
503  return true;
504  }
505 
506  switch (e->getOper()) {
507  case opPC: return true; // Ignore %pc
508 
509  case opDefineAll: return true; // Ignore <all>
510 
511  case opTemp:
512  return true; // Ignore all temps (should be local to one instruction)
513 
514  // Would like to handle at least %ZF, %CF one day. For now, filter them out
515  case opZF:
516  case opCF:
517  case opNF:
518  case opOF:
519  case opFlags: return true;
520 
521  case opMemOf:
522  // Actually, surely all sensible architectures will only every return
523  // in registers. So for now, just filter out all mem-ofs
524  // return signature->isStackLocal(prog, e); // Filter out local variables
525  return true;
526 
527  case opGlobal: return true; // Never return in globals
528 
529  default: return false;
530  }
531 }

559 void UserProc::ensureExpIsMappedToLocal(const std::shared_ptr<RefExp> &ref)
560 {
561  if (!lookupSymFromRefAny(ref).isEmpty()) {
562  return; // Already have a symbol for ref
563  }
564 
565  Statement *def = ref->getDef();
566 
567  if (!def) {
568  return; // TODO: should this be logged ?
569  }
570 
571  SharedExp base = ref->getSubExp1();
572  SharedType ty = def->getTypeForExp(base);
573  // No, get its name from the front end
574  QString locName = nullptr;
575 
576  if (base->isRegOf()) {
577  locName = getRegName(base);
578 
579  // Create a new local, for the base name if it doesn't exist yet,
580  // so we don't need several names for the same combination of location
581  // and type. However if it does already exist, addLocal will allocate
582  // a new name.
583  // Example: r8{0}->argc type int, r8->o0 type int, now r8->o0_1 type char*.
584  if (existsLocal(locName)) {
585  locName = newLocalName(ref);

587  }
588  else {
589  locName = newLocalName(ref);
590  }
591 
592  addLocal(ty, locName, base);
593 }

596 SharedExp UserProc::getSymbolExp(SharedExp le, SharedType ty, bool lastPass)
597 {
598  assert(ty != nullptr);
599 
600  SharedExp e = nullptr;
601 
602  // check for references to the middle of a local
603  if (le->isMemOf() && le->getSubExp1()->getOper() == opMinus &&
604  le->getSubExp1()->getSubExp1()->isSubscript() &&
605  le->getSubExp1()->getSubExp1()->getSubExp1()->isRegN(m_signature->getStackRegister()) &&
606  le->getSubExp1()->getSubExp2()->isIntConst()) {
607  for (auto &elem : m_symbolMap) {
608  if (!(elem).second->isLocal()) {
609  continue;
610  }
611 
612  QString name = elem.second->access<Const, 1>()->getStr();
613 
614  if (m_locals.find(name) == m_locals.end()) {
615  continue;
616  }
617 
618  SharedType lty = m_locals[name];
619  SharedConstExp loc = elem.first;
620 
621  if (loc->isMemOf() && loc->getSubExp1()->getOper() == opMinus &&
622  loc->access<Exp, 1, 1>()->isSubscript() &&
623  loc->access<Exp, 1, 1, 1>()->isRegN(m_signature->getStackRegister()) &&
624  loc->access<Exp, 1, 2>()->isIntConst()) {
625  const int byteOffset = le->access<Const, 1, 2>()->getInt() -
626  loc->access<Const, 1, 2>()->getInt();
627 
628  if (Util::inRange(byteOffset, 1, static_cast<int>(lty->getSize() / 8))) {
629  e = Location::memOf(Binary::get(opPlus,
630  Unary::get(opAddrOf, elem.second->clone()),
631  Const::get(byteOffset)));
632  LOG_VERBOSE("Seems %1 is in the middle of %2 returning %3", le, loc, e);
633  return e;
634  }
635  }
636  }
637  }
638 
639  if (m_symbolMap.find(le) != m_symbolMap.end()) {
640  return getSymbolFor(le, ty);
641  }
642 
643  if (*ty == *VoidType::get() && lastPass) {
644  ty = IntegerType::get(STD_SIZE);
645  }
646 
647  // the default of just assigning an int type is bad..
648  // if the locals is not an int then assigning it this
649  // type early results in aliases to this local not being recognised
650 
651  // Exp* base = le;
652  // if (le->isSubscript())
653  // base = ((RefExp*)le)->getSubExp1();
654  // NOTE: using base below instead of le does not enhance anything, but causes us to lose def
655  // information
656  e = createLocal(ty->clone(), le);
657  mapSymbolTo(le->clone(), e);
658  return e->clone();
659 }

702 bool UserProc::isLocalOrParamPattern(SharedConstExp e) const
703 {
704  if (!e->isMemOf()) {
705  return false; // Don't want say a register
706  }
707  else if (!m_signature || !m_signature->isPromoted()) {
708  return false; // Prevent an assert failure if using -E
709  }
710 
711  const int sp = m_signature->getStackRegister();
712  const auto initSp(RefExp::get(Location::regOf(sp), nullptr)); // sp{-}
713 
714  SharedConstExp addrExp = e->getSubExp1();
715  if (*addrExp == *initSp) {
716  return true; // Accept m[sp{-}]
717  }
718  else if (addrExp->getOper() != opPlus && addrExp->getOper() != opMinus) {
719  return false;
720  }
721 
722  return (*addrExp->getSubExp1() == *initSp) && addrExp->getSubExp2()->isIntConst();
723 }

739 void UserProc::mapSymbolTo(const SharedConstExp &from, SharedExp to)
740 {
741  assert(from && to);
742 
743  SymbolMap::iterator it = m_symbolMap.find(from);
744 
745  while (it != m_symbolMap.end() && *it->first == *from) {
746  if (*it->second == *to) {

751  }

1161 
1162 bool UserProc::proveEqual(const SharedExp &queryLeft, const SharedExp &queryRight, bool conditional)
1163 {
1164  if ((m_provenTrue.find(queryLeft) != m_provenTrue.end()) &&
1165  (*m_provenTrue[queryLeft] == *queryRight)) {
1166  if (m_prog->getProject()->getSettings()->debugProof) {
1167  LOG_MSG("found true in provenTrue cache %1 in %2",
1168  Binary::get(opEquals, queryLeft, queryRight), getName());
1169  }
1170 
1171  return true;
1172  }
1173 
1174  const SharedExp origLeft = queryLeft;
1175  const SharedExp origRight = queryRight;
1176 
1177  SharedExp query = Binary::get(opEquals, queryLeft->clone(), queryRight->clone());
1178 
1179  // subscript locs on the right with {-} (nullptr reference)
1180  LocationSet locs;
1181  query->getSubExp2()->addUsedLocs(locs);
1182 
1183  for (const SharedExp &xx : locs) {
1184  query->setSubExp2(query->getSubExp2()->expSubscriptValNull(xx));
1185  }
1186 
1187  if (!query->getSubExp1()->isSubscript()) {
1188  bool gotDef = false;
1189 
1190  // replace expression from return set with expression in the collector of the return
1191  if (m_retStatement) {
1192  auto def = m_retStatement->findDefFor(query->getSubExp1());
1193 
1194  if (def) {
1195  query->setSubExp1(def);
1196  gotDef = true;
1197  }
1198  }
1199 
1200  if (!gotDef) {
1201  // OK, the thing I'm looking for isn't in the return collector, but perhaps there is an
1202  // entry for <all> If this is proved, then it is safe to say that x == x for any x with
1203  // no definition reaching the exit
1204  auto right = origRight->clone()->simplify(); // In case it's sp+0
1205 
1206  if ((*origLeft == *right) && // x == x
1207  (origLeft->getOper() != opDefineAll) && // Beware infinite recursion
1208  proveEqual(defAll, defAll)) { // Recurse in case <all> not proven yet
1209  if (m_prog->getProject()->getSettings()->debugProof) {
1210  LOG_MSG("Using all=all for %1", query->getSubExp1());
1211  LOG_MSG("Prove returns true");
1212  }
1213 
1214  m_provenTrue[origLeft->clone()] = right;
1215  return true;
1216  }
1217 
1218  if (m_prog->getProject()->getSettings()->debugProof) {
1219  LOG_MSG("Not in return collector: %1", query->getSubExp1());
1220  LOG_MSG("Prove returns false");
1221  }
1222 
1223  return false;
1224  }
1225  }
1226 
1227  if (m_recursionGroup) { // If in involved in a recursion cycle
1228  // then save the original query as a premise for bypassing calls
1229  m_recurPremises[origLeft->clone()] = origRight;
1230  }
1231 
1232  std::set<PhiAssign *> lastPhis;
1233  std::map<PhiAssign *, SharedExp> cache;
1234  bool result = prover(query, lastPhis, cache);
1235 
1236  if (m_recursionGroup) {
1237  killPremise(origLeft); // Remove the premise, regardless of result
1238  }
1239 
1240  if (m_prog->getProject()->getSettings()->debugProof) {
1241  LOG_MSG("Prove returns %1 for %2 in %3", (result ? "true" : "false"), query,
1242  this->getName());
1243  }
1244 
1245  if (result && !conditional) {
1246  m_provenTrue[origLeft] = origRight; // Save the now proven equation
1247  }
1248 
1249  return result;
1250 }

1252 
1253 bool UserProc::prover(SharedExp query, std::set<PhiAssign *> &lastPhis,
1254  std::map<PhiAssign *, SharedExp> &cache, PhiAssign *lastPhi /* = nullptr */)
1255 {
1256  // A map that seems to be used to detect loops in the call graph:
1257  std::map<CallStatement *, SharedExp> called;
1258  auto phiInd = query->getSubExp2()->clone();
1259 
1260  if (lastPhi && (cache.find(lastPhi) != cache.end()) && (*cache[lastPhi] == *phiInd)) {
1261  if (m_prog->getProject()->getSettings()->debugProof) {
1262  LOG_MSG("true - in the phi cache");
1263  }
1264 
1265  return true;
1266  }
1267 
1268  std::set<Statement *> refsTo;
1269 
1270  query = query->clone();
1271  bool change = true;
1272  bool swapped = false;
1273 
1274  while (change) {
1275  if (m_prog->getProject()->getSettings()->debugProof) {
1276  LOG_MSG("%1", query);
1277  }
1278 
1279  change = false;
1280 
1281  if (query->isEquality()) {
1282  // same left and right means true
1283  if (*query->getSubExp1() == *query->getSubExp2()) {
1284  query = Terminal::get(opTrue);
1285  change = true;
1286  }
1287 
1288  // move constants to the right
1289  if (!change) {
1290  auto plus = query->getSubExp1();
1291  auto s1s2 = plus ? plus->getSubExp2() : nullptr;
1292 
1293  if (plus && s1s2) {
1294  if ((plus->getOper() == opPlus) && s1s2->isIntConst()) {
1295  query->setSubExp2(Binary::get(opPlus, query->getSubExp2(),
1296  Unary::get(opNeg, s1s2->clone())));
1297  query->setSubExp1(plus->getSubExp1());
1298  change = true;
1299  }
1300 
1301  if ((plus->getOper() == opMinus) && s1s2->isIntConst()) {
1302  query->setSubExp2(Binary::get(opPlus, query->getSubExp2(), s1s2->clone()));
1303  query->setSubExp1(plus->getSubExp1());
1304  change = true;
1305  }
1306  }
1307  }
1308 
1309  // substitute using a statement that has the same left as the query
1310  if (!change && (query->getSubExp1()->isSubscript())) {
1311  auto r = query->access<RefExp, 1>();
1312  Statement *s = r->getDef();
1313  CallStatement *call = dynamic_cast<CallStatement *>(s);
1314 
1315  if (call) {
1316  // See if we can prove something about this register.
1317  UserProc *destProc = dynamic_cast<UserProc *>(call->getDestProc());
1318  SharedExp base = r->getSubExp1();
1319 
1320  if (destProc && !destProc->isLib() && (destProc->m_recursionGroup != nullptr) &&
1321  (destProc->m_recursionGroup->find(this) !=
1322  destProc->m_recursionGroup->end())) {
1323  // The destination procedure may not have preservation proved as yet,
1324  // because it is involved in our recursion group. Use the conditional
1325  // preservation logic to determine whether query is true for this procedure
1326  SharedExp provenTo = destProc->getProven(base);
1327 
1328  if (provenTo) {
1329  // There is a proven preservation. Use it to bypass the call
1330  auto queryLeft = call->localiseExp(provenTo->clone());
1331  query->setSubExp1(queryLeft);
1332 
1333  // Now try everything on the result
1334  return prover(query, lastPhis, cache, lastPhi);
1335  }
1336  else {
1337  // Check if the required preservation is one of the premises already
1338  // assumed
1339  SharedExp premisedTo = destProc->getPremised(base);
1340 
1341  if (premisedTo) {
1342  if (m_prog->getProject()->getSettings()->debugProof) {
1343  LOG_MSG("Conditional preservation for call from %1 to %2, "
1344  "allows bypassing",
1345  getName(), destProc->getName());
1346  }
1347 
1348  auto queryLeft = call->localiseExp(premisedTo->clone());
1349  query->setSubExp1(queryLeft);
1350  return prover(query, lastPhis, cache, lastPhi);
1351  }
1352  else {
1353  // There is no proof, and it's not one of the premises. It may yet
1354  // succeed, by making another premise! Example: try to prove esp,
1355  // depends on whether ebp is preserved, so recurse to check ebp's
1356  // preservation. Won't infinitely loop because of the premise map
1357  // FIXME: what if it needs a rx = rx + K preservation?
1358  destProc->setPremise(base);
1359 
1360  if (m_prog->getProject()->getSettings()->debugProof) {
1361  LOG_MSG("New required premise '%1' for %2",
1362  Binary::get(opEquals, base, base), destProc->getName());
1363  }
1364 
1365  // Pass conditional as true, since even if proven, this is
1366  // conditional on other things
1367  bool result = destProc->proveEqual(base->clone(), base->clone(),
1368  true);
1369  destProc->killPremise(base);
1370 
1371  if (result) {
1372  if (m_prog->getProject()->getSettings()->debugProof) {
1373  LOG_MSG("Conditional preservation with new premise '%1' "
1374  "succeeds for %2",
1375  Binary::get(opEquals, base, base),
1376  destProc->getName());
1377  }
1378 
1379  // Use the new conditionally proven result
1380  auto queryLeft = call->localiseExp(base->clone());
1381  query->setSubExp1(queryLeft);
1382  return destProc->prover(query, lastPhis, cache, lastPhi);
1383  }
1384  else {
1385  if (m_prog->getProject()->getSettings()->debugProof) {
1386  LOG_MSG(
1387  "Conditional preservation required premise '%1' fails!",
1388  Binary::get(opEquals, base, base));
1389  }
1390 
1391  // Do nothing else; the outer proof will likely fail
1392  }
1393  }
1394  }
1395  } // End call involved in this recursion group
1396 
1397  // Seems reasonable that recursive procs need protection from call loops too
1398  // getProven returns the right side of what is
1399  auto right = call->getProven(r->getSubExp1());
1400 
1401  if (right) { // proven about r (the LHS of query)
1402  right = right->clone();
1403 
1404  if ((called.find(call) != called.end()) && (*called[call] == *query)) {
1405  LOG_MSG("Found call loop to %1 %2", call->getDestProc()->getName(),
1406  query);
1407  query = Terminal::get(opFalse);
1408  change = true;
1409  }
1410  else {
1411  called[call] = query->clone();
1412 
1413  if (m_prog->getProject()->getSettings()->debugProof) {
1414  LOG_MSG("Using proven for %1 %2 = %3",
1415  call->getDestProc()->getName(), r->getSubExp1(), right);
1416  }
1417 
1418  right = call->localiseExp(right);
1419 
1420  if (m_prog->getProject()->getSettings()->debugProof) {
1421  LOG_MSG("Right with subs: %1", right);
1422  }
1423 
1424  query->setSubExp1(right); // Replace LHS of query with right
1425  change = true;
1426  }
1427  }
1428  }
1429  else if (s && s->isPhi()) {
1430  // for a phi, we have to prove the query for every statement
1431  PhiAssign *pa = static_cast<PhiAssign *>(s);
1432  bool ok = true;
1433 
1434  if ((lastPhis.find(pa) != lastPhis.end()) || (pa == lastPhi)) {
1435  ok = (*query->getSubExp2() == *phiInd);
1436 
1437  if (m_prog->getProject()->getSettings()->debugProof) {
1438  if (ok) {
1439  // FIXME: induction??!
1440  LOG_MSG("Phi loop detected (set true due to induction)");
1441  }
1442  else {
1443  LOG_MSG("Phi loop detected (set false: %1 != %2)",
1444  query->getSubExp2(), phiInd);
1445  }
1446  }
1447  }
1448  else {
1449  if (m_prog->getProject()->getSettings()->debugProof) {
1450  LOG_MSG("Found %1 prove for each, ", s);
1451  }
1452 
1453  for (const std::shared_ptr<RefExp> &pi : *pa) {
1454  SharedExp e = query->clone();
1455  std::shared_ptr<RefExp> r1 = e->access<RefExp, 1>();
1456  r1->setDef(pi->getDef());
1457 
1458  if (m_prog->getProject()->getSettings()->debugProof) {
1459  LOG_MSG("proving for %1", e);
1460  }
1461 
1462  lastPhis.insert(lastPhi);
1463

81

83  CallStatement *call = dynamic_cast<CallStatement *>(s);
84

86  ReturnStatement *rs = call->getCalleeReturn();

88 
89  if (!rs || !uc) {
90  continue;
91  }
92 
93  LOG_VERBOSE(" returns:");
94

96  // Intersect the callee's returns with the live locations at the call,
97  // i.e. make sure that they exist in *uc

99  if (!assgn) {
100  continue;
101  }
102 
103  if (!uc->exists(assgn->getLeft())) {
104  continue; // Intersection fails
105  }
106 
107  LOG_VERBOSE(" %1 %2", assgn->getType()->getCtype(), assgn->getLeft());

126 void DFATypeRecovery::replaceArrayIndices(Statement *s)
127 {
128  UserProc *proc = s->getProc();
129  Prog *prog = proc->getProg();
130 
131  std::list<SharedExp> result;
132  s->searchAll(scaledArrayPat, result);
133 
134  // We have m[idx*stride + base]
135  // Rewrite it as globalN[idx] with addr(globalN) == base
136  // with a global array globalN with base type size \e stride
137  for (SharedExp arrayExp : result) {
138  const Address base = arrayExp->access<Const, 1, 2>()->getAddr();
139  // const int stride = arrayExp->access<Const, 1, 1, 2>()->getInt();
140  SharedExp idx = arrayExp->access<Exp, 1, 1, 1>();
141 
142  // Replace with the array expression
143  QString name = prog->getGlobalNameByAddr(base);
144  if (name.isEmpty()) {
145  name = prog->newGlobalName(base);
146  }
147 
148  SharedExp array = Binary::get(opArrayIndex, Location::global(name, proc), idx);
149 
150  if (s->searchAndReplace(scaledArrayPat, array)) {
151  if (s->isImplicit()) {
152  // Register an array of appropriate type
153  prog->markGlobalUsed(
154  base, ArrayType::get(static_cast<const ImplicitAssign *>(s)->getType()));
155  }
156  else if (s->isCall()) {
157  // array of function pointers
158  prog->markGlobalUsed(base, ArrayType::get(PointerType::get(FuncType::get())));
159  }
160  }
161  }

165 void DFATypeRecovery::dfa_analyze_implict_assigns(Statement *s)
166 {
167  if (!s->isImplicit()) {
168  return;
169  }
170 
171  UserProc *proc = s->getProc();
172  assert(proc);
173  Prog *prog = proc->getProg();
174  assert(prog);
175 
176  SharedExp lhs = static_cast<const ImplicitAssign *>(s)->getLeft();
177  // Note: parameters are not explicit any more
178  // if (lhs->isParam()) { // }
179 
180  bool allZero = false;
181  SharedExp slhs = lhs->clone()->removeSubscripts(allZero);
182  SharedType implicitType = static_cast<const ImplicitAssign *>(s)->getType();
183  int i = proc->getSignature()->findParam(slhs);
184 
185  if (i != -1) {
186  proc->setParamType(i, implicitType);
187  }
188  else if (lhs->isMemOf()) {
189  SharedExp sub = lhs->getSubExp1();
190 
191  if (sub->isIntConst()) {
192  // We have a m[K] := -
193  Address K = sub->access<Const>()->getAddr();
194  prog->markGlobalUsed(K, implicitType);
195  }
196  }
197  else if (lhs->isGlobal()) {
198  assert(std::dynamic_pointer_cast<Location>(lhs) != nullptr);
199  QString gname = lhs->access<Const, 1>()->getStr();
200  prog->setGlobalType(gname, implicitType);
201  }

205 void DFATypeRecovery::recoverFunctionTypes(Function *function)
206 {
207  if (function->isLib()) {
208  LOG_VERBOSE("Not using DFA type analysis on library function '%1'", function->getName());
209  return;
210  }
211 
212  if (function->getProg()->getProject()->getSettings()->debugTA) {
213  LOG_VERBOSE("--- Start data flow based type analysis for %1 ---", getName());
214  }
215 
216  bool first = true;
217  UserProc *up = dynamic_cast<UserProc *>(function);
218  assert(up != nullptr);
219 
220  do {
221  if (first) {
222  // Subscript the discovered extra parameters
223  PassManager::get()->executePass(PassID::BlockVarRename, up);
224  PassManager::get()->executePass(PassID::StatementPropagation, up);
225  PassManager::get()->executePass(PassID::ImplicitPlacement, up);
226  }
227 
228  first = false;
229  dfaTypeAnalysis(up);
230 
231  // There used to be a pass here to insert casts. This is best left until global type
232  // analysis is complete, so do it just before translating from SSA form (which is the where
233  // type information becomes inaccessible)
234  } while (doEllipsisProcessing(up));
235 
236  PassManager::get()->executePass(PassID::BBSimplify, up); // In case there are new struct members
237 
238  if (function->getProg()->getProject()->getSettings()->debugTA) {
239  LOG_VERBOSE("=== End type analysis for %1 ===", getName());
240  }
241 }

244 void DFATypeRecovery::dfaTypeAnalysis(UserProc *proc)
245 {
246  ProcCFG *cfg = proc->getCFG();
247  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, "Before DFA type analysis");
248 
249  // First use the type information from the signature.
250  // Sometimes needed to split variables (e.g. argc as a
251  // int and char* in sparc/switch_gcc)
252  bool ch = dfaTypeAnalysis(proc->getSignature().get(), cfg);
253  StatementList stmts;
254  proc->getStatements(stmts);
255 
256  int iter = 0;
257 
258  for (iter = 1; iter <= DFA_ITER_LIMIT; ++iter) {
259  ch = false;
260 
261  for (Statement *stmt : stmts) {
262  Statement *before = nullptr;
263 
264  if (proc->getProg()->getProject()->getSettings()->debugTA) {
265  before = stmt->clone();
266  }
267 
268  DFATypeAnalyzer ana;
269  stmt->accept(&ana);
270  const bool thisCh = ana.hasChanged();
271 
272  if (thisCh) {
273  ch = true;
274 
275  if (proc->getProg()->getProject()->getSettings()->debugTA) {
276  LOG_VERBOSE(" Caused change:\n"
277  " FROM: %1\n"
278  " TO: %2",
279  before, stmt);
280  }
281  }
282 
283  if (proc->getProg()->getProject()->getSettings()->debugTA) {
284  delete before;
285  }
286  }
287 
288  if (!ch) {
289  // No more changes: round robin algorithm terminates
290  break;
291  }
292  }
293 
294  if (ch) {
295  LOG_VERBOSE("Iteration limit exceeded for dfaTypeAnalysis of procedure '%1'",
296  proc->getName());
297  }
298 
299  if (proc->getProg()->getProject()->getSettings()->debugTA) {
300  LOG_MSG("### Results for data flow based type analysis for %1 ###", proc->getName());
301  printResults(stmts, iter);
302  LOG_MSG("### End results for Data flow based type analysis for %1 ###", proc->getName());
303  }
304 
305  // Now use the type information gathered
306 
307  proc->getProg()->getProject()->alertDecompileDebugPoint(
308  proc, "Before other uses of DFA type analysis");
309  proc->debugPrintAll("Before other uses of DFA type analysis");
310 
311  Prog *_prog = proc->getProg();
312  DataIntervalMap localsMap(proc); // map of all local variables of proc
313 
314  for (Statement *s : stmts) {
315  // 1) constants
316  std::list<std::shared_ptr<Const>> constList;
317  findConstantsInStmt(s, constList);
318 
319  for (const std::shared_ptr<Const> &con : constList) {
320  if (!con || con->isStrConst() || con->isFuncPtrConst()) {
321  continue;
322  }
323 
324  SharedType t = con->getType();
325  int val = con->getInt();
326 
327  if (t && t->resolvesToPointer()) {
328  auto pt = t->as<PointerType>();
329  SharedType baseType = pt->getPointsTo();
330 
331  if (baseType->resolvesToChar()) {
332  // Convert to a string MVE: check for read-only?
333  // Also, distinguish between pointer to one char, and ptr to many?
334  const char *str = _prog->getStringConstant(Address(val), true);
335 
336  if (str) {
337  // Make a string
338  con->setOper(opStrConst);
339  con->setRawStr(str);
340  }
341  }
342  else if (baseType->resolvesToInteger() || baseType->resolvesToFloat() ||
343  baseType->resolvesToSize()) {
344  Address addr = Address(con->getInt()); // TODO: use getAddr
345  _prog->markGlobalUsed(addr, baseType);
346  QString gloName = _prog->getGlobalNameByAddr(addr);
347 
348  if (!gloName.isEmpty()) {
349  Address r = addr - _prog->getGlobalAddrByName(gloName);
350  SharedExp ne;
351 
352  if (!r.isZero()) { // TODO: what if r is NO_ADDR ?
353  SharedExp g = Location::global(gloName, proc);
354  ne = Location::memOf(
355  Binary::get(opPlus, Unary::get(opAddrOf, g), Const::get(r)), proc);
356  }
357  else {
358  SharedType ty = _prog->getGlobalType(gloName);
359  Assign *assgn = dynamic_cast<Assign *>(s);
360 
361  if (assgn && s->isAssign() && assgn->getType()) {
362  Type::Size bits = assgn->getType()->getSize();
363 
364  if ((ty == nullptr) || (ty->getSize() == 0)) {
365  _prog->setGlobalType(gloName, IntegerType::get(bits));
366  }
367  }
368 
369  SharedExp g = Location::global(gloName, proc);
370 
371  if (g && ty && ty->resolvesToArray()) {
372  ne = Binary::get(opArrayIndex, g, Const::get(0));
373  }
374  else {
375  ne = g;
376  }
377  }
378 
379  SharedExp memof = Location::memOf(con);
380 
381  if (!s->searchAndReplace(*memof, ne)) {
382  ne = nullptr;
383  }
384  }
385  }
386  else if (baseType->resolvesToArray()) {
387  // We have found a constant in s which has type pointer to array of alpha. We
388  // can't get the parent of con, but we can find it with the pattern
389  // unscaledArrayPat.
390  std::list<SharedExp> result;
391  s->searchAll(unscaledArrayPat, result);
392 
393  for (auto &elem : result) {
394  // idx + K
395  auto bin_rr = std::dynamic_pointer_cast<Binary>(elem);
396 
397  if (!bin_rr) {
398  assert(false);
399  return;
400  }
401 
402  auto constK = bin_rr->access<Const, 2>();
403 
404  // Note: keep searching till we find the pattern with this constant, since
405  // other constants may not be used as pointer to array type.
406  if (constK != con) {
407  continue;
408  }
409 
410  Address K = Address(constK->getInt());
411  SharedExp idx = bin_rr->getSubExp1();
412  SharedExp arr = Unary::get(
413  opAddrOf,
414  Binary::get(opArrayIndex,
415  Location::global(_prog->getGlobalNameByAddr(K), proc),
416  idx));
417  // Beware of changing expressions in implicit assignments...
418  // map can become invalid
419  const bool isImplicit = s->isImplicit();
420 
421  if (isImplicit) {
422  cfg->removeImplicitAssign(static_cast<ImplicitAssign *>(s)->getLeft());
423  }
424 
425  if (!s->searchAndReplace(unscaledArrayPat, arr)) {
426  arr = nullptr; // remove if not emplaced in s
427  }
428 
429  // s will likely have an m[a[array]], so simplify
430  s->simplifyAddr();
431 
432  if (isImplicit) {
433  // Replace the implicit assignment entry. Note that s' lhs has changed
434  cfg->findOrCreateImplicitAssign(
435  static_cast<ImplicitAssign *>(s)->getLeft());
436  }
437 
438  // Ensure that the global is declared
439  // Ugh... I think that arrays and pointers to arrays are muddled!
440  _prog->markGlobalUsed(K, baseType);
441  }
442  }
443  }
444  else if (t->resolvesToFloat()) {
445  if (con->isIntConst()) {
446  // Reinterpret as a float (and convert to double)
447  // con->setFlt(reinterpret_cast<float>(con->getInt()));
448  int tmp = con->getInt();
449  con->setFlt(*reinterpret_cast<float *>(&tmp));
450  con->setOper(opFltConst);
451  con->setType(FloatType::get(64)); // TODO: why double ? float might do
452  }
453 
454  // MVE: more work if double?
455  }
456  else { /* if (t->resolvesToArray()) */
457  _prog->markGlobalUsed(Address(val), t);
458  }
459  }
460 
461  // 2) Search for the scaled array pattern and replace it with an array use m[idx*K1 + K2]
462  replaceArrayIndices(s);
463 
464  // 3) Check implicit assigns for parameter and global types.
465  dfa_analyze_implict_assigns(s);
466 
467  // 4) Add the locals (soon globals as well) to the localTable, to sort out the overlaps
468  if (s->isTyping()) {
469  SharedExp addrExp = nullptr;
470  SharedType typeExp = nullptr;
471 
472  assert(s->isAssignment());
473  SharedExp lhs = static_cast<Assignment *>(s)->getLeft();
474 
475  if (lhs->isMemOf()) {
476  addrExp = lhs->getSubExp1();
477  typeExp = static_cast<Assignment *>(s)->getType();
478  }
479 
480  const int spIndex = Util::getStackRegisterIndex(_prog);
481  if (addrExp && proc->getSignature()->isAddrOfStackLocal(spIndex, addrExp)) {
482  int localAddressOffset = 0;
483 
484  if ((addrExp->getArity() == 2) &&
485  proc->getSignature()->isOpCompatStackLocal(addrExp->getOper())) {
486  auto K = addrExp->access<Const, 2>();
487 
488  if (K->isConst()) {
489  localAddressOffset = K->getInt();
490 
491  if (addrExp->getOper() == opMinus) {
492  localAddressOffset = -localAddressOffset;
493  }
494  }
495  }
496 
497  const SharedType &ty = static_cast<TypingStatement *>(s)->getType();
498  LOG_VERBOSE2(
499  "Type analysis for '%1': Adding addrExp '%2' with type %3 to local table",
500  proc->getName(), addrExp, ty);
501  SharedExp loc_mem = Location::memOf(addrExp);
502  localsMap.insertItem(Address(localAddressOffset), proc->lookupSym(loc_mem, ty),
503  typeExp);
504  }
505  }
506  }
507 
508  proc->debugPrintAll("After application of DFA type analysis");
509  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, "After DFA type analysis");
510 }

514 {

547 bool DFATypeRecovery::doEllipsisProcessing(UserProc *proc)
548 {
549  bool ch = false;
550 
551  for (BasicBlock *bb : *proc->getCFG()) {
552  BasicBlock::RTLRIterator rrit;
553  StatementList::reverse_iterator srit;
554  CallStatement *c = dynamic_cast<CallStatement *>(bb->getLastStmt(rrit, srit));
555 
556  // Note: we may have removed some statements, so there may no longer be a last statement!
557  if (c == nullptr) {
558  continue;
559  }
560 
561  ch |= c->ellipsisProcessing(proc->getProg());
562  }
563 
564  if (ch) {
565  PassManager::get()->executePass(PassID::CallAndPhiFix, proc);
566  }
567

56 bool ST20BinaryLoader::loadFromMemory(QByteArray &arr)
57 {
58  const DWord fileSize = arr.size();
59 
60  if (fileSize < 8 + 2) { // sizeof(magic) + 2 bytes for the negative jump
61  return false;
62  }
63 
64  m_image = new char[fileSize];
65  memcpy(m_image, arr.constData(), fileSize);
66 
67  Address codeStart = ROM_HIGH - fileSize;
68  BinarySection *section = m_binaryImage->createSection("$CODE", codeStart, ROM_HIGH);
69  section->setCode(true);
70  section->setHostAddr(HostAddress(m_image));
71 
72  return true;
73 }

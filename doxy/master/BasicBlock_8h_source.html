33 /// Kinds of basic block nodes
34 /// reordering these will break the save files - trent
35 enum class BBType
36 {
37  Invalid = -1, ///< invalid instruction
38  Fall = 0, ///< fall-through node
39  Oneway = 1, ///< unconditional branch (jmp)
40  Twoway = 2, ///< conditional branch (jXX)
41  Nway = 3, ///< case branch (jmp [off + 4*eax])

43  Ret = 5, ///< return (ret)
44  CompJump = 6, ///< computed jump

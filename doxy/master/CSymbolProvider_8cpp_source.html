34 bool CSymbolProvider::readLibraryCatalog(const Prog *prog, const QString &filePath)
35 {
36  // TODO: this is a work for generic semantics provider plugin : HeaderReader
37  QFile file(filePath);
38 
39  if (!file.open(QFile::ReadOnly | QFile::Text)) {
40  LOG_ERROR("Cannot open library signature catalog `%1'", filePath);
41  return false;
42  }
43 
44  QTextStream is(&file);
45 
46  while (!is.atEnd()) {
47  QString sigFilePath;
48  is >> sigFilePath;
49  sigFilePath = sigFilePath.mid(0, sigFilePath.indexOf('#')); // cut the line to first '#'
50 
51  if ((sigFilePath.size() > 0) && sigFilePath.endsWith('\n')) {
52  sigFilePath = sigFilePath.mid(0, sigFilePath.size() - 1);
53  }
54 
55  if (sigFilePath.isEmpty()) {
56  continue;
57  }
58 
59  CallConv cc = CallConv::C; // Most APIs are C calling convention
60 
61  if (sigFilePath == "windows.h") {
62  cc = CallConv::Pascal; // One exception
63  }
64 
65  if (sigFilePath == "mfc.h") {
66  cc = CallConv::ThisCall; // Another exception
67  }
68 
69  const QString sig_path = QFileInfo(filePath).absoluteDir().absoluteFilePath(sigFilePath);
70  if (!readLibrarySignatures(qPrintable(sig_path), prog, cc)) {
71  return false;
72  }
73  }
74 
75  return true;
76 }

97 bool CSymbolProvider::addSymbolsFromSymbolFile(Prog *prog, const QString &fname)
98 {
99  AnsiCParserDriver driver;
100  const CallConv cc = prog->isWin32() ? CallConv::Pascal : CallConv::C;
101 
102  if (driver.parse(fname, prog->getMachine(), cc) != 0) {
103  LOG_ERROR("Cannot read symbol file '%1': %2", fname);
104  return false;
105  }
106 
107  for (std::shared_ptr<Symbol> &sym : driver.symbols) {
108  if (sym->sig) {
109  QString name = sym->sig->getName();
110  Module *targetModule = prog->getOrInsertModuleForSymbol(name);
111 
112  auto bin_sym = prog->getBinaryFile()->getSymbols()->findSymbolByAddress(sym->addr);
113  const bool isLib = (bin_sym && bin_sym->isImportedFunction()) ||
114  // NODECODE isn't really the right modifier; perhaps we should have a
115  // LIB modifier, to specifically specify that this function obeys
116  // library calling conventions
117  sym->mods.noDecode;
118  Function *p = targetModule->createFunction(name, sym->addr, isLib);
119 
120  if (!sym->mods.incomplete) {
121  p->setSignature(sym->sig->clone());
122  p->getSignature()->setForced(true);
123  }
124  }
125  else {
126  QString name = sym->name;
127  SharedType ty = sym->ty;
128 
129  prog->createGlobal(sym->addr, sym->ty, sym->name);
130  }
131  }
132 
133  for (std::shared_ptr<SymbolRef> &ref : driver.refs) {
134  prog->getFrontEnd()->addRefHint(ref->addr, ref->name);
135  }
136 
137  return true;
138 }

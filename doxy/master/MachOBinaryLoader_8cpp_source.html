97 bool MachOBinaryLoader::loadFromMemory(QByteArray &img)
98 {
99  QBuffer fp(&img);
100 
101  if (!fp.open(QFile::ReadOnly)) {
102  return false;
103  }
104 
105  unsigned int imgoffs = 0;
106 
107  unsigned char *magic = reinterpret_cast<uint8_t *>(img.data());
108  struct mach_header *header; // The Mach-O header
109 
110  if (Util::testMagic(magic, { 0xca, 0xfe, 0xba, 0xbe })) {
111  const int nimages = Util::readDWord(magic + 4, Endian::Big);
112  DEBUG_PRINT("Binary is universal with %1 images", nimages);
113 
114  for (int i = 0; i < nimages; i++) {
115  int fbh = 8 + i * 5 * 4;
116  unsigned int cputype = Util::readDWord(magic + fbh + 0, Endian::Big);
117  unsigned int offset = Util::readDWord(magic + fbh + 8, Endian::Big);
118  DEBUG_PRINT("cputype: %1", cputype);
119  DEBUG_PRINT("cpusubtype: %1", BE4(fbh + 4));
120  DEBUG_PRINT("offset: %1", BE4(fbh + 8));
121  DEBUG_PRINT("size: %1", BE4(fbh + 12));
122  DEBUG_PRINT("pad: %1", BE4(fbh + 16));
123 
124  if (cputype == 0x7) { // i386
125  imgoffs = offset;
126  }
127  }
128  }
129 
130  header = reinterpret_cast<mach_header *>(img.data() + imgoffs); // new mach_header;
131  // fp.read((char *)header, sizeof(mach_header));
132 
133  if ((header->magic != MH_MAGIC) && (READ4_BE(header->magic) != MH_MAGIC)) {
134  LOG_ERROR("Error loading file, bad Mach-O magic");
135  return false;
136  }
137 
138  // check for swapped bytes
139  swap_bytes = (READ4_BE(header->magic) == MH_MAGIC);
140 
141  // Determine CPU type
142  if (BMMH(header->cputype) == 0x07) {
143  machine = Machine::PENTIUM;
144  }
145  else {
146  machine = Machine::PPC;
147  }
148 
149  sections.clear();
150  std::vector<segment_command> segments;
151  std::vector<nlist> symbols;
152  // uint32_t startundef, nundef;
153  // uint32_t startlocal, nlocal,ndef, startdef;
154  std::vector<section> stubs_sects;
155 
156  std::vector<char> strtbl;
157  std::vector<DWord> indirectsymtbl;
158 
159  Address objc_symbols = Address::INVALID;
160  Address objc_modules = Address::INVALID;
161  Address objc_strings = Address::INVALID;
162  Address objc_refs = Address::INVALID;
163  unsigned objc_modules_size = 0;
164 
165  fp.seek(imgoffs + sizeof(*header));
166 
167  for (unsigned i = 0; i < BMMH(header->ncmds); i++) {
168  load_command cmd;
169  long pos = fp.pos();
170  fp.read(reinterpret_cast<char *>(&cmd), sizeof(load_command));
171  fp.seek(pos);
172 
173  switch (BMMH(cmd.cmd)) {
174  case LC_SEGMENT: {
175  segment_command seg;
176  fp.read(reinterpret_cast<char *>(&seg), sizeof(seg));
177  segments.push_back(seg);
178  DEBUG_PRINT("seg addr %1 size %2 fileoff %3 filesize %4 flags %5", BMMH(seg.vmaddr),
179  BMMH(seg.vmsize), BMMH(seg.fileoff), BMMH(seg.filesize), BMMH(seg.flags));
180 
181  for (DWord n = 0; n < BMMH(seg.nsects); n++) {
182  section sect;
183  fp.read(reinterpret_cast<char *>(&sect), sizeof(sect));
184  sections.push_back(sect);
185  DEBUG_PRINT(" sectname %1 segname %2 addr %3 size %4 flags %5", sect.sectname,
186  sect.segname, BMMH(sect.addr), BMMH(sect.size), BMMH(sect.flags));
187 
188  if ((BMMH(sect.flags) & SECTION_TYPE) == S_SYMBOL_STUBS) {
189  stubs_sects.push_back(sect);
190  DEBUG_PRINT(" symbol stubs section, start index %1, stub size %2",
191  BMMH(sect.reserved1), BMMH(sect.reserved2));
192  }
193 
194  if (!strcmp(sect.sectname, SECT_OBJC_SYMBOLS)) {
195  assert(objc_symbols == Address::INVALID);
196  objc_symbols = Address(BMMH(sect.addr));
197  }
198 
199  if (!strcmp(sect.sectname, SECT_OBJC_MODULES)) {
200  assert(objc_modules == Address::INVALID);
201  objc_modules = Address(BMMH(sect.addr));
202  objc_modules_size = BMMH(sect.size);
203  }
204 
205  if (!strcmp(sect.sectname, SECT_OBJC_STRINGS)) {
206  assert(objc_strings == Address::INVALID);
207  objc_strings = Address(BMMH(sect.addr));
208  }
209 
210  if (!strcmp(sect.sectname, SECT_OBJC_REFS)) {
211  assert(objc_refs == Address::INVALID);
212  objc_refs = Address(BMMH(sect.addr));
213  }
214  }
215  } break;
216 
217  case LC_SYMTAB: {
218  symtab_command syms;
219  fp.read(reinterpret_cast<char *>(&syms), sizeof(syms));
220  fp.seek(imgoffs + BMMH(syms.stroff));
221  strtbl.resize(BMMH(syms.strsize));
222  fp.read(&strtbl[0], BMMH(syms.strsize));
223  fp.seek(imgoffs + BMMH(syms.symoff));
224 
225  for (unsigned n = 0; n < BMMH(syms.nsyms); n++) {
226  nlist sym;
227  fp.read(reinterpret_cast<char *>(&sym), sizeof(sym));
228  symbols.push_back(sym);
229  // DEBUG_PRINT(stdout, "got sym %s flags %x value %x\n", strtbl +
230  // BMMH(sym.n_un.n_strx), sym.n_type, BMMH(sym.n_value));
231  }
232 
233  DEBUG_PRINT("symtab contains %1 symbols", BMMH(syms.nsyms));
234  } break;
235 
236  case LC_DYSYMTAB: {
237  struct dysymtab_command syms;
238  fp.read(reinterpret_cast<char *>(&syms), sizeof(syms));
239  DEBUG_PRINT("dysymtab local %1 %2 defext %3 %4 undef %5 %6", BMMH(syms.ilocalsym),
240  BMMH(syms.nlocalsym), BMMH(syms.iextdefsym), BMMH(syms.nextdefsym),
241  BMMH(syms.iundefsym), BMMH(syms.nundefsym));
242 
243  // TODO: find uses for values below
244  // startlocal = BMMH(syms.ilocalsym);
245  // nlocal = BMMH(syms.nlocalsym);
246  // startdef = BMMH(syms.iextdefsym);
247  // ndef = BMMH(syms.nextdefsym);
248  // startundef = BMMH(syms.iundefsym);
249  // nundef = BMMH(syms.nundefsym);
250 
251  DEBUG_PRINT("dysymtab has %1 indirect symbols: ", BMMH(syms.nindirectsyms));
252  indirectsymtbl.resize(BMMH(syms.nindirectsyms));
253  fp.seek(imgoffs + BMMH(syms.indirectsymoff));
254  fp.read(reinterpret_cast<char *>(&indirectsymtbl[0]),
255  indirectsymtbl.size() * sizeof(DWord));
256 
257  for (unsigned j = 0; j < BMMH(syms.nindirectsyms); j++) {
258  DEBUG_PRINT(" %1 ", BMMH(indirectsymtbl[j]));
259  }
260  } break;
261 
262  default:
263  DEBUG_PRINT("Not handled load command %1", BMMH(cmd.cmd));
264  // yep, there's lots of em
265  break;
266  }
267 
268  fp.seek(pos + BMMH(cmd.cmdsize));
269  }
270 
271  segment_command *lowest = &segments[0], *highest = &segments[0];
272 
273  for (unsigned i = 1; i < segments.size(); i++) {
274  if (BMMH(segments[i].vmaddr) < BMMH(lowest->vmaddr)) {
275  lowest = &segments[i];
276  }
277 
278  if (BMMH(segments[i].vmaddr) > BMMH(highest->vmaddr)) {
279  highest = &segments[i];
280  }
281  }
282 
283  loaded_addr = Address(BMMH(lowest->vmaddr));
284  loaded_size = BMMH(highest->vmaddr) - BMMH(lowest->vmaddr) + BMMH(highest->vmsize);
285 
286  base = reinterpret_cast<char *>(malloc(loaded_size));
287 
288  if (!base) {
289  LOG_ERROR("Cannot allocate memory for copy of image");
290  return false;
291  }
292 
293  for (unsigned i = 0; i < segments.size(); i++) {
294  fp.seek(imgoffs + BMMH(segments[i].fileoff));
295  Address a = Address(BMMH(segments[i].vmaddr));
296  unsigned sz = BMMH(segments[i].vmsize);
297  unsigned fsz = BMMH(segments[i].filesize);
298  memset(base + a.value() - loaded_addr.value(), 0, sz);
299  fp.read(base + a.value() - loaded_addr.value(), fsz);
300  DEBUG_PRINT("loaded segment %1 %2 in mem %3 in file", a.value(), sz, fsz);
301 
302  QString name = QByteArray(segments[i].segname, 17);
303  BinarySection *sect = Image->createSection(
304  name, Address(BMMH(segments[i].vmaddr)),
305  Address(BMMH(segments[i].vmaddr) + BMMH(segments[i].vmsize)));
306  assert(sect);
307  sect->setHostAddr(HostAddress(base) + BMMH(segments[i].vmaddr) - loaded_addr);
308  assert(sect->getHostAddr() + sect->getSize() <= HostAddress(base) + loaded_size);
309 
310  unsigned long l = BMMH(segments[i].initprot);
311  sect->setBss(false); // TODO
312  sect->setEndian((machine == Machine::PPC) ? Endian::Big : Endian::Little);
313  sect->setCode((l & VM_PROT_EXECUTE) != 0);
314  sect->setData((l & VM_PROT_READ) != 0);
315  sect->setReadOnly((l & VM_PROT_WRITE) == 0);
316 
317  for (size_t s_idx = 0; s_idx < sections.size(); s_idx++) {
318  if (strcmp(sections[s_idx].segname, segments[i].segname) != 0) {
319  continue;
320  }
321 
322  if ((0 == strcmp(sections[s_idx].sectname, "__cfstring")) ||
323  (0 == strcmp(sections[s_idx].sectname, "__cstring"))) {
324  sect->setAttributeForRange(
325  "StringsSection", true, Address(BMMH(sections[s_idx].addr)),
326  Address(BMMH(sections[s_idx].addr) + BMMH(sections[s_idx].size)));
327  }
328 
329  sect->setAttributeForRange(
330  "ReadOnly", (BMMH(sections[i].flags) & VM_PROT_WRITE) ? true : false,
331  Address(BMMH(sections[s_idx].addr)),
332  Address(BMMH(sections[s_idx].addr) + BMMH(sections[s_idx].size)));
333  }
334 
335  DEBUG_PRINT("loaded segment %1 %2 in mem %3 in file code=%4 data=%5 readonly=%6", a.value(),
336  sz, fsz, sect->isCode(), sect->isData(), sect->isReadOnly());
337  }
338 
339  // process stubs_sects
340  for (unsigned j = 0; j < stubs_sects.size(); j++) {
341  for (unsigned i = 0; i < BMMH(stubs_sects[j].size) / BMMH(stubs_sects[j].reserved2); i++) {
342  unsigned startidx = BMMH(stubs_sects[j].reserved1);
343  unsigned symbol = BMMH(indirectsymtbl[startidx + i]);
344  Address addr = Address(BMMH(stubs_sects[j].addr) + i * BMMH(stubs_sects[j].reserved2));
345 
346  DEBUG_PRINT("stub for %1 at %2", strtbl + BMMH(symbols[symbol].n_un.n_strx),
347  addr.value());
348  char *name = &strtbl.at(BMMH(symbols[symbol].n_un.n_strx));
349 
350  if (*name == '_') { // we want printf not _printf
351  name++;
352  }
353 
354  BinarySymbol *sym = Symbols->createSymbol(addr, name);
355  sym->setAttribute("Function", true);
356  sym->setAttribute("Imported", true);
357  }
358  }
359 
360  // process the remaining symbols
361  for (unsigned i = 0; i < symbols.size(); i++) {
362  char *name = &strtbl.at(BMMH(symbols[i].n_un.n_strx));
363 
364  if ((BMMH(symbols[i].n_un.n_strx) != 0) && (BMMH(symbols[i].n_value) != 0) &&
365  (*name != 0)) {
366  uint8_t sym_type = symbols[i].n_type;
367  bool is_stab = (sym_type & N_STAB) != 0;
368 
369  if (is_stab) {
370  continue; // TODO: handle stab symbols !
371  }
372 
373  DEBUG_PRINT("symbol %s at %x type %x\n", name, BMMH(symbols[i].n_value),
374  sym_type & N_TYPE);
375 
376  if (*name == '_') { // we want main not _main
377  name++;
378  }
379 
380  Symbols->createSymbol(Address(BMMH(symbols[i].n_value)), name);
381  }
382  }
383 
384  // process objective-c section
385  if (objc_modules != Address::INVALID) {
386  DEBUG_PRINT("Processing objective-c section");
387 
388  for (unsigned i = 0; i < objc_modules_size;) {
389  struct objc_module *module = reinterpret_cast<struct objc_module *>(
390  (HostAddress(base) + objc_modules - loaded_addr + i).value());
391  char *name = reinterpret_cast<char *>(intptr_t(base) + BMMH(module->name) -
392  loaded_addr.value());
393  Symtab symtab = reinterpret_cast<Symtab>(
394  (HostAddress(base) + BMMH(module->symtab) - loaded_addr).value());
395 
396 #ifdef DEBUG_MACHO_LOADER_OBJC
397  LOG_MSG("module %1 (%2 classes)", name, BMMHW(symtab->cls_def_cnt));
398 #endif
399  ObjcModule *m = &modules[name];
400  m->name = name;
401 
402  for (unsigned j = 0; j < BMMHW(symtab->cls_def_cnt); j++) {
403  struct objc_class *def = reinterpret_cast<struct objc_class *>(
404  base + BMMH(symtab->defs[j]) - loaded_addr.value());
405  char *_name = reinterpret_cast<char *>(Address::value_type(base) + BMMH(def->name) -
406  loaded_addr.value());
407 
408 #ifdef DEBUG_MACHO_LOADER_OBJC
409  LOG_MSG(" class %1", name);
410 #endif
411 
412  ObjcClass *cl = &m->classes[_name];
413  cl->name = _name;
414  struct objc_ivar_list *ivars = reinterpret_cast<struct objc_ivar_list *>(
415  base + BMMH(def->ivars) - loaded_addr.value());
416 
417  for (unsigned k = 0; k < static_cast<unsigned int>(BMMH(ivars->ivar_count)); k++) {
418  struct objc_ivar *ivar = &ivars->ivar_list[k];
419  char *name2 = reinterpret_cast<char *>(
420  Address::value_type(base) + BMMH(ivar->ivar_name) - loaded_addr.value());
421  char *types = reinterpret_cast<char *>(
422  Address::value_type(base) + BMMH(ivar->ivar_type) - loaded_addr.value());
423 
424 #ifdef DEBUG_MACHO_LOADER_OBJC
425  LOG_MSG(" ivar %1 %2 %3", name, types, BMMH(ivar->ivar_offset));
426 #endif
427 
428  ObjcIvar *iv = &cl->ivars[name2];
429  iv->name = name2;
430  iv->type = types;
431  iv->offset = BMMH(ivar->ivar_offset);
432  }
433 
434  // this is weird, why is it defined as a ** in the struct but used as a * in otool?
435  struct objc_method_list *methods = reinterpret_cast<struct objc_method_list *>(
436  intptr_t(base) + BMMH(def->methodLists) - loaded_addr.value());
437 
438  for (unsigned k = 0; k < static_cast<unsigned int>(BMMH(methods->method_count));
439  k++) {
440  struct objc_method *method = &methods->method_list[k];
441  char *name3 = reinterpret_cast<char *>(
442  intptr_t(base) + BMMH(method->method_name) - loaded_addr.value());
443  char *types = reinterpret_cast<char *>(
444  intptr_t(base) + BMMH(method->method_types) - loaded_addr.value());
445 
446 #ifdef DEBUG_MACHO_LOADER_OBJC
447  LOG_MSG(" method %1 %2 %3", name, types, BMMH((void *)method->method_imp));
448 #endif
449  ObjcMethod *me = &cl->methods[name3];
450  me->name = name3;
451  me->types = types;
452  me->addr = Address(BMMH(method->method_imp));
453  }
454  }
455 
456  i += BMMH(module->size);
457  }
458  }
459 
460  entrypoint = getMainEntryPoint();
461  return true;
462 }

465 int MachOBinaryLoader::canLoad(QIODevice &dev) const
466 {
467  Byte buf[8];
468 
469  dev.read(reinterpret_cast<char *>(buf), sizeof(buf));
470 
471  if (((buf[0] == 0xfe) && (buf[1] == 0xed) && (buf[2] == 0xfa) && (buf[3] == 0xce)) ||
472  ((buf[0] == 0xce) && (buf[1] == 0xfa) && (buf[2] == 0xed) && (buf[3] == 0xfe)) ||
473  ((buf[0] == 0xca) && (buf[1] == 0xfe) && (buf[2] == 0xba) && (buf[3] == 0xbe))) {
474  /* Mach-O Mac OS-X binary */
475  return 4;
476  }
477 
478  return 0;

106 bool Project::loadBinaryFile(const QString &filePath)
107 {
108  LOG_MSG("Loading binary file '%1'", filePath);
109 
110  // Find loader plugin to load file
111  IFileLoader *loader = getBestLoader(filePath);
112 
113  if (loader == nullptr) {
114  LOG_WARN("Cannot load '%1': Unrecognized binary file format.", filePath);
115  return false;
116  }
117 
118  if (isBinaryLoaded()) {
119  loader->unload();
120  unloadBinaryFile();
121  }
122 
123  QFile srcFile(filePath);
124  if (!srcFile.open(QFile::ReadOnly)) {
125  LOG_WARN("Opening '%1' failed");
126  return false;
127  }
128 
129  m_loadedBinary.reset(new BinaryFile(srcFile.readAll(), loader));
130 
131  if (loader->loadFromFile(m_loadedBinary.get()) == false) {
132  return false;
133  }
134 
135  m_loadedBinary->getImage()->updateTextLimits();
136 
137  return createProg(m_loadedBinary.get(), QFileInfo(filePath).baseName()) != nullptr;
138 }

168 bool Project::decodeBinaryFile()
169 {
170  if (!getProg()) {
171  LOG_ERROR("Cannot decode binary file: No binary file is loaded.");
172  return false;
173  }
174  else if (!m_fe) {
175  LOG_ERROR("Cannot decode binary file: No suitable frontend found.");
176  return false;
177  }
178 
179  loadSymbols();
180 
181  if (!getSettings()->m_entryPoints.empty()) { // decode only specified procs
182  // decode entry points from -e (and -E) switch(es)
183  for (auto &elem : getSettings()->m_entryPoints) {
184  LOG_MSG("Decoding specified entrypoint at address %1", elem);
185  m_prog->decodeEntryPoint(elem);
186  }
187  }
188  else if (!decodeAll()) { // decode everything
189  return false;
190  }
191 
192  this->alertEndDecode();
193 
194  LOG_MSG("Found %1 procs", m_prog->getNumFunctions());
195 
196  if (getSettings()->generateSymbols) {
197  ProgSymbolWriter().writeSymbolsToFile(getProg(), "symbols.h");
198  }
199 
200  if (getSettings()->generateCallGraph) {
201  CallGraphDotWriter().writeCallGraph(getProg(), "callgraph.dot");
202  }
203 
204  return true;
205 }

266 IFrontEnd *Project::createFrontEnd()
267 {
268  try {
269  Plugin *plugin = nullptr;
270 
271  switch (getLoadedBinaryFile()->getMachine()) {
272  case Machine::PENTIUM:
273  plugin = m_pluginManager->getPluginByName("X86 FrontEnd plugin");
274  break;
275  case Machine::SPARC:
276  plugin = m_pluginManager->getPluginByName("SPARC FrontEnd plugin");
277  break;
278  case Machine::PPC: plugin = m_pluginManager->getPluginByName("PPC FrontEnd plugin"); break;
279  case Machine::ST20:
280  plugin = m_pluginManager->getPluginByName("ST20 FrontEnd plugin");
281  break;
282  default: LOG_ERROR("Machine architecture not supported!"); break;
283  }
284 
285  if (!plugin) {
286  throw std::runtime_error("Plugin not found.");
287  }
288 
289  IFrontEnd *fe = plugin->getIfc<IFrontEnd>();
290  if (!fe->initialize(this)) {
291  throw std::runtime_error("FrontEnd initialization failed.");
292  }
293  return fe;
294  }
295  catch (const std::runtime_error &err) {
296  LOG_ERROR("Cannot create FrontEnd: %1", err.what());
297  }
298 
299  return nullptr;
300 }

319 bool Project::decodeAll()
320 {
321  if (getSettings()->decodeMain) {
322  LOG_MSG("Decoding entry point...");
323  }
324 
325  if (!m_fe || !m_fe->decodeEntryPointsRecursive(getSettings()->decodeMain)) {
326  LOG_ERROR("Aborting load due to decode failure");
327  return false;
328  }
329 
330  bool gotMain = false;
331  Address mainAddr = m_fe->findMainEntryPoint(gotMain);
332  if (gotMain) {
333  m_prog->addEntryPoint(mainAddr);
334  }
335 
336  if (getSettings()->decodeChildren) {
337  // this causes any undecoded userprocs to be decoded
338  LOG_MSG("Decoding anything undecoded...");
339  if (!m_fe->decodeUndecoded()) {
340  LOG_ERROR("Aborting load due to decode failure");
341  return false;
342  }
343  }
344 
345  return true;
346 }

520 IFileLoader *Project::getBestLoader(const QString &filePath) const
521 {
522  QFile inputBinary(filePath);
523 
524  if (!inputBinary.open(QFile::ReadOnly)) {
525  LOG_ERROR("Unable to open binary file: %1", filePath);
526  return nullptr;
527  }
528 
529  IFileLoader *bestLoader = nullptr;
530  int bestScore = 0;
531 
532  // get the best plugin for loading this file
533  for (Plugin *p : m_pluginManager->getPluginsByType(PluginType::FileLoader)) {
534  inputBinary.seek(0); // reset the file offset for the next plugin
535  IFileLoader *loader = p->getIfc<IFileLoader>();
536 
537  int score = loader->canLoad(inputBinary);
538 
539  if (score > bestScore) {
540  bestScore = score;
541  bestLoader = loader;
542  }
543  }
544 
545  return bestLoader;
546 }

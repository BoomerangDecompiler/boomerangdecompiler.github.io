97 /*
98  * getting all the classes in the application...
99  *
100  * int objc_getClassList(buffer, bufferLen)
101  * classes is an array of Class values (which are pointers)
102  * which will be filled by the function; if this
103  * argument is nullptr, no copying is done, only the
104  * return value is returned
105  * bufferLen is the number of Class values the given buffer
106  * can hold; if the buffer is not large enough to
107  * hold all the classes, the buffer is filled to
108  * the indicated capacity with some arbitrary subset
109  * of the known classes, which could be different
110  * from call to call
111  * returns the number of classes, which is the number put
112  * in the buffer if the buffer was large enough,
113  * or the length the buffer should have been
114  *
115  * int numClasses = 0, newNumClasses = objc_getClassList(nullptr, 0);
116  * Class *classes = nullptr;
117  * while (numClasses < newNumClasses) {
118  * numClasses = newNumClasses;
119  * classes = realloc(classes, sizeof(Class) * numClasses);
120  * newNumClasses = objc_getClassList(classes, numClasses);
121  * }
122  * // now, can use the classes list; if nullptr, there are no classes
123  * free(classes);
124  *
125  */
126 OBJC_EXPORT int objc_getClassList(Class *buffer, int bufferLen);

121 bool PassManager::executePass(IPass *pass, UserProc *proc)
122 {
123  assert(pass != nullptr);
124  LOG_VERBOSE("Executing pass '%1' for '%2'", pass->getName(), proc->getName());
125 
126  const bool changed = pass->execute(proc);
127 
128  QString msg = QString("after executing pass '%1'").arg(pass->getName());
129  proc->debugPrintAll(qPrintable(msg));
130  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, qPrintable(msg));
131 
132  return changed;
133 }

136 bool PassManager::executePassGroup(const QString &name, UserProc *proc)
137 {
138  auto it = m_passGroups.find(name);
139  if (it == m_passGroups.end()) {
140  throw std::invalid_argument(
141  QString("Pass group '%1' does not exist").arg(name).toStdString());
142  }
143 
144  const PassGroup &group = it.value();
145  bool changed = false;
146 
147  LOG_VERBOSE("Executing pass group '%1' for '%2'", name, proc->getName());
148  for (IPass *pass : group) {
149  changed |= executePass(pass, proc);
150  }
151 
152  return changed;
153 }

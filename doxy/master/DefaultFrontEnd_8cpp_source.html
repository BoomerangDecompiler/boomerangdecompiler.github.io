63 bool DefaultFrontEnd::decodeEntryPointsRecursive(bool decodeMain)
64 {
65  if (!decodeMain) {
66  return true;
67  }
68 
69  BinaryImage *image = m_program->getBinaryFile()->getImage();
70 
71  Interval<Address> extent(image->getLimitTextLow(), image->getLimitTextHigh());
72 
73  m_program->getProject()->alertStartDecode(extent.lower(),
74  (extent.upper() - extent.lower()).value());
75 
76  bool gotMain;
77  Address a = findMainEntryPoint(gotMain);
78  LOG_VERBOSE("start: %1, gotMain: %2", a, (gotMain ? "true" : "false"));
79 
80  if (a == Address::INVALID) {
81  std::vector<Address> entrypoints = findEntryPoints();
82 
83  for (auto &entrypoint : entrypoints) {
84  if (!decodeRecursive(entrypoint)) {
85  return false;
86  }
87  }
88 
89  return true;
90  }
91 
92  decodeRecursive(a);
93  m_program->addEntryPoint(a);
94 
95  if (!gotMain) {
96  return true; // Decoded successfully, but patterns don't match a known main() pattern
97  }
98 
99  static const char *mainName[] = { "main", "WinMain", "DriverEntry" };
100  QString name = m_program->getSymbolNameByAddr(a);
101 
102  if (name == nullptr) {
103  name = mainName[0];
104  }
105 
106  for (auto &elem : mainName) {
107  if (name != elem) {
108  continue;
109  }
110 
111  Function *proc = m_program->getFunctionByAddr(a);
112 
113  if (proc == nullptr) {
114  LOG_WARN("No proc found for address %1", a);
115  return false;
116  }
117 
118  auto fty = std::dynamic_pointer_cast<FuncType>(Type::getNamedType(name));
119 
120  if (!fty) {
121  LOG_WARN("Unable to find signature for known entrypoint %1", name);
122  }
123  else {
124  proc->setSignature(fty->getSignature()->clone());
125  proc->getSignature()->setName(name);
126  // proc->getSignature()->setFullSig(true); // Don't add or remove parameters
127  proc->getSignature()->setForced(true); // Don't add or remove parameters
128  }
129 
130  break;
131  }
132 
133  return true;
134 }

137 bool DefaultFrontEnd::decodeRecursive(Address addr)
138 {
139  assert(addr != Address::INVALID);
140 
141  Function *newProc = m_program->getOrCreateFunction(addr);
142 
143  // Sometimes, we have to adjust the entry address since
144  // the instruction at addr is just a jump to another address.
145  addr = newProc->getEntryAddress();
146  LOG_MSG("Starting decode at address %1", addr);
147  UserProc *proc = static_cast<UserProc *>(m_program->getFunctionByAddr(addr));
148 
149  if (proc == nullptr) {
150  LOG_MSG("No proc found at address %1", addr);
151  return false;
152  }
153  else if (proc->isLib()) {
154  LOG_MSG("NOT decoding library proc at address %1", addr);
155  return false;
156  }
157 
158  if (processProc(proc, addr)) {
159  proc->setDecoded();
160  }
161 
162  return m_program->isWellFormed();
163 }

166 bool DefaultFrontEnd::decodeUndecoded()
167 {
168  bool change = true;
169  LOG_MSG("Looking for undecoded procedures to decode...");
170 
171  while (change) {
172  change = false;
173 
174  for (const auto &m : m_program->getModuleList()) {
175  for (Function *function : *m) {
176  if (function->isLib()) {
177  continue;
178  }
179 
180  UserProc *userProc = static_cast<UserProc *>(function);
181 
182  if (userProc->isDecoded()) {
183  continue;
184  }
185 
186  // undecoded userproc.. decode it
187  change = true;
188 
189  if (!processProc(userProc, userProc->getEntryAddress())) {
190  return false;
191  }
192 
193  userProc->setDecoded();
194 
195  // Break out of the loops if not decoding children
196  if (!m_program->getProject()->getSettings()->decodeChildren) {
197  break;
198  }
199  }
200  }
201 
202  if (!m_program->getProject()->getSettings()->decodeChildren) {
203  break;
204  }
205  }
206 
207  return m_program->isWellFormed();
208 }

221 bool DefaultFrontEnd::processProc(UserProc *proc, Address addr)
222 {
223  BasicBlock *currentBB;
224 
225  LOG_VERBOSE("### Decoding proc '%1' at address %2 ###", proc->getName(), addr);
226 
227  // We have a set of CallStatement pointers. These may be disregarded if this is a speculative
228  // decode that fails (i.e. an illegal instruction is found). If not, this set will be used to
229  // add to the set of calls to be analysed in the ProcCFG, and also to call newProc()
230  std::list<CallStatement *> callList;
231 
232  // Indicates whether or not the next instruction to be decoded is the lexical successor of the
233  // current one. Will be true for all NCTs and for CTIs with a fall through branch.
234  bool sequentialDecode = true;
235 
236  ProcCFG *cfg = proc->getCFG();
237  assert(cfg);
238 
239  // Initialise the queue of control flow targets that have yet to be decoded.
240  m_targetQueue.initial(addr);
241 
242  // Clear the pointer used by the caller prologue code to access the last call rtl of this
243  // procedure decoder.resetLastCall();
244 
245  int numBytesDecoded = 0;
246  Address startAddr = addr;
247  Address lastAddr = addr;
248 
249  while ((addr = m_targetQueue.getNextAddress(*cfg)) != Address::INVALID) {
250  // The list of RTLs for the current basic block
251  std::unique_ptr<RTLList> BB_rtls(new RTLList);
252 
253  // Keep decoding sequentially until a CTI without a fall through branch is decoded
254  DecodeResult inst;
255 
256  while (sequentialDecode) {
257  // Decode and classify the current source instruction
258  if (m_program->getProject()->getSettings()->traceDecoder) {
259  LOG_MSG("*%1", addr);
260  }
261 
262  if (!decodeSingleInstruction(addr, inst)) {
263  // Do not throw away previously decoded instrucions before the invalid one
264  if (BB_rtls && !BB_rtls->empty()) {
265  cfg->createBB(BBType::Fall, std::move(BB_rtls));
266  }
267 
268  QString message;
269  BinaryImage *image = m_program->getBinaryFile()->getImage();
270 
271  Byte insnData[4] = { 0 };
272  bool print = true;
273  for (int i = 0; i < 4; i++) {
274  if (!image->readNative1(addr + i, insnData[i])) {
275  print = false;
276  break;
277  }
278  }
279 
280  if (print) {
281  // clang-format off
282  message.sprintf("Encountered invalid instruction at address %s: "
283  "0x%02X 0x%02X 0x%02X 0x%02X",
284  qPrintable(addr.toString()),
285  insnData[0],
286  insnData[1],
287  insnData[2],
288  insnData[3]);
289  // clang-format on
290  LOG_WARN(message);
291  }
292 
293  assert(!inst.valid);
294  break; // try next instruction in queue
295  }
296  else if (!inst.rtl || inst.rtl->empty()) {
297  LOG_VERBOSE("Instruction at address %1 is a no-op!", addr);
298  if (!inst.rtl) {
299  inst.rtl.reset(new RTL(addr));
300  }
301  }
302 
303  // Need to construct a new list of RTLs if a basic block has just been finished but
304  // decoding is continuing from its lexical successor
305  if (BB_rtls == nullptr) {
306  BB_rtls.reset(new std::list<std::unique_ptr<RTL>>());
307  }
308 
309  if (!inst.valid) {
310  // Alert the watchers to the problem
311  m_program->getProject()->alertBadDecode(addr);
312 
313  // An invalid instruction. Most likely because a call did not return (e.g. call
314  // _exit()), etc. Best thing is to emit an INVALID BB, and continue with valid
315  // instructions Emit the RTL anyway, so we have the address and maybe some other
316  // clues
317  BB_rtls->push_back(std::make_unique<RTL>(addr));
318  cfg->createBB(BBType::Invalid, std::move(BB_rtls));
319  break; // try the next instruction in the queue
320  }
321 
322  // alert the watchers that we have decoded an instruction
323  m_program->getProject()->alertInstructionDecoded(addr, inst.numBytes);
324  numBytesDecoded += inst.numBytes;
325 
326  // Check if this is an already decoded jump instruction (from a previous pass with
327  // propagation etc) If so, we throw away the just decoded RTL (but we still may have
328  // needed to calculate the number of bytes.. ick.)
329  std::map<Address, RTL *>::iterator ff = m_previouslyDecoded.find(addr);
330 
331  if (ff != m_previouslyDecoded.end()) {
332  inst.rtl.reset(ff->second);
333  }
334 
335  if (!inst.rtl) {
336  // This can happen if an instruction is "cancelled", e.g. call to __main in a hppa
337  // program Just ignore the whole instruction
338  if (inst.numBytes > 0) {
339  addr += inst.numBytes;
340  }
341 
342  continue;
343  }
344 
345  // Display RTL representation if asked
346  if (m_program->getProject()->getSettings()->printRTLs) {
347  QString tgt;
348  OStream st(&tgt);
349  inst.rtl->print(st);
350  LOG_MSG(tgt);
351  }
352 
353  // Make a copy (!) of the list. This is needed temporarily to work around the following
354  // problem. We are currently iterating an RTL, which could be a return instruction. The
355  // RTL is passed to createReturnBlock; if this is not the first return statement, it
356  // will get cleared, and this will cause problems with the current iteration. The
357  // effects seem to be worse for MSVC/Windows. This problem will likely be easier to cope
358  // with when the RTLs are removed, and there are special Statements to mark the start of
359  // instructions (and their native address).
360  // FIXME: However, this workaround breaks logic below where a GOTO is changed to a CALL
361  // followed by a return if it points to the start of a known procedure
362  std::list<Statement *> sl(inst.rtl->getStatements());
363 
364  for (auto ss = sl.begin(); ss != sl.end(); ++ss) {
365  Statement *s = *ss;
366  s->setProc(proc); // let's do this really early!
367 
368  if (m_refHints.find(inst.rtl->getAddress()) != m_refHints.end()) {
369  const QString &name(m_refHints[inst.rtl->getAddress()]);
370  Address globAddr = m_program->getGlobalAddrByName(name);
371 
372  if (globAddr != Address::INVALID) {
373  s->searchAndReplace(Const(globAddr),
374  Unary::get(opAddrOf, Location::global(name, proc)));
375  }
376  }
377 
378  s->simplify();
379  GotoStatement *jumpStmt = dynamic_cast<GotoStatement *>(s);
380 
381  // Check for a call to an already existing procedure (including self recursive
382  // jumps), or to the PLT (note that a LibProc entry for the PLT function may not yet
383  // exist)
384  if (s->getKind() == StmtType::Goto) {
385  preprocessProcGoto(ss, jumpStmt->getFixedDest(), sl, inst.rtl.get());
386  s = *ss; // *ss can be changed within processProc
387  }
388 
389  switch (s->getKind()) {
390  case StmtType::Goto: {
391  Address jumpDest = jumpStmt->getFixedDest();
392 
393  // Handle one way jumps and computed jumps separately
394  if (jumpDest != Address::INVALID) {
395  BB_rtls->push_back(std::move(inst.rtl));
396  sequentialDecode = false;
397 
398  currentBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
399 
400  // Exit the switch now if the basic block already existed
401  if (currentBB == nullptr) {
402  break;
403  }
404 
405  // Add the out edge if it is to a destination within the
406  // procedure
407  if (jumpDest < m_program->getBinaryFile()->getImage()->getLimitTextHigh()) {
408  m_targetQueue.visit(cfg, jumpDest, currentBB);
409  cfg->addEdge(currentBB, jumpDest);
410  }
411  else {
412  LOG_WARN("Goto instruction at address %1 branches beyond end of "
413  "section, to %2",
414  addr, jumpDest);
415  }
416  }
417  } break;
418 
419  case StmtType::Case: {
420  SharedExp jumpDest = jumpStmt->getDest();
421 
422  if (jumpDest == nullptr) { // Happens if already analysed (now redecoding)
423  BB_rtls->push_back(std::move(inst.rtl));
424 
425  // processSwitch will update num outedges
426  currentBB = cfg->createBB(BBType::Nway, std::move(BB_rtls));
427 
428  // decode arms, set out edges, etc
429  IndirectJumpAnalyzer().processSwitch(currentBB, proc);
430  sequentialDecode = false; // Don't decode after the jump
431  break; // Just leave it alone
432  }
433 
434  // Check for indirect calls to library functions, especially in Win32 programs
435  if (refersToImportedFunction(jumpDest)) {
436  LOG_VERBOSE("Jump to a library function: %1, replacing with a call/ret.",
437  jumpStmt);
438 
439  // jump to a library function
440  // replace with a call ret
441  const BinarySymbol
442  *sym = m_program->getBinaryFile()->getSymbols()->findSymbolByAddress(
443  jumpDest->access<Const, 1>()->getAddr());
444  assert(sym != nullptr);
445  QString func = sym->getName();
446  CallStatement *call = new CallStatement;
447  call->setDest(jumpDest->clone());
448  LibProc *lp = proc->getProg()->getOrCreateLibraryProc(func);
449 
450  if (lp == nullptr) {
451  LOG_FATAL("getLibraryProc() returned nullptr");
452  }
453 
454  call->setDestProc(lp);
455  BB_rtls->push_back(
456  std::unique_ptr<RTL>(new RTL(inst.rtl->getAddress(), { call })));
457 
458  currentBB = cfg->createBB(BBType::Call, std::move(BB_rtls));
459  appendSyntheticReturn(currentBB, proc, inst.rtl.get());
460  sequentialDecode = false;
461 
462  if (inst.rtl->getAddress() == proc->getEntryAddress()) {
463  // it's a thunk
464  // Proc *lp = prog->findProc(func.c_str());
465  func = "__imp_" + func;
466  proc->setName(func);
467  // lp->setName(func.c_str());
468  m_program->getProject()->alertSignatureUpdated(proc);
469  }
470 
471  callList.push_back(call);
472  ss = sl.end();
473  ss--; // get out of the loop
474  break;
475  }
476 
477  BB_rtls->push_back(std::move(inst.rtl));
478 
479  // We create the BB as a COMPJUMP type, then change to an NWAY if it turns out
480  // to be a switch stmt
481  cfg->createBB(BBType::CompJump, std::move(BB_rtls));
482 
483  LOG_VERBOSE2("COMPUTED JUMP at address %1, jumpDest = %2", addr, jumpDest);
484 
485  sequentialDecode = false;
486  break;
487  }
488 
489  case StmtType::Branch: {
490  Address jumpDest = jumpStmt->getFixedDest();
491  BB_rtls->push_back(std::move(inst.rtl));
492 
493  currentBB = cfg->createBB(BBType::Twoway, std::move(BB_rtls));
494 
495  // Stop decoding sequentially if the basic block already existed otherwise
496  // complete the basic block
497  if (currentBB == nullptr) {
498  sequentialDecode = false;
499  }
500  else {
501  // Add the out edge if it is to a destination within the section
502  if (jumpDest < m_program->getBinaryFile()->getImage()->getLimitTextHigh()) {
503  m_targetQueue.visit(cfg, jumpDest, currentBB);
504  cfg->addEdge(currentBB, jumpDest);
505  }
506  else {
507  LOG_WARN("Branch instruction at address %1 branches beyond end of "
508  "section, to %2",
509  addr, jumpDest);
510  currentBB->setType(BBType::Oneway);
511  }
512 
513  // Add the fall-through outedge
514  cfg->addEdge(currentBB, addr + inst.numBytes);
515  }
516  } break;
517 
518  case StmtType::Call: {
519  CallStatement *call = static_cast<CallStatement *>(s);
520 
521  // Check for a dynamic linked library function
522  if (refersToImportedFunction(call->getDest())) {
523  // Dynamic linked proc pointers are treated as static.
524  Address linkedAddr = call->getDest()->access<Const, 1>()->getAddr();
525  QString name = m_program->getBinaryFile()
526  ->getSymbols()
527  ->findSymbolByAddress(linkedAddr)
528  ->getName();
529 
530  Function *function = proc->getProg()->getOrCreateLibraryProc(name);
531  call->setDestProc(function);
532  call->setIsComputed(false);
533 
534  if (function->isNoReturn() || isNoReturnCallDest(function->getName())) {
535  sequentialDecode = false;
536  }
537  }
538 
539  Address functionAddr = getAddrOfLibraryThunk(call, proc);
540  if (functionAddr != Address::INVALID) {
541  // Yes, it's a library function. Look up its name.
542  QString name = m_program->getBinaryFile()
543  ->getSymbols()
544  ->findSymbolByAddress(functionAddr)
545  ->getName();
546 
547  // Assign the proc to the call
548  Function *p = proc->getProg()->getOrCreateLibraryProc(name);
549 
550  if (call->getDestProc()) {
551  // prevent unnecessary __imp procs
552  m_program->removeFunction(call->getDestProc()->getName());
553  }
554 
555  call->setDestProc(p);
556  call->setIsComputed(false);
557  call->setDest(Location::memOf(Const::get(functionAddr)));
558 
559  if (p->isNoReturn() || isNoReturnCallDest(p->getName())) {
560  sequentialDecode = false;
561  }
562  }
563 
564  // Treat computed and static calls separately
565  if (call->isComputed()) {
566  BB_rtls->push_back(std::move(inst.rtl));
567  currentBB = cfg->createBB(BBType::CompCall, std::move(BB_rtls));
568 
569  // Stop decoding sequentially if the basic block already
570  // existed otherwise complete the basic block
571  if (currentBB == nullptr) {
572  sequentialDecode = false;
573  }
574  else {
575  cfg->addEdge(currentBB, addr + inst.numBytes);
576  }
577 
578  // Add this call to the list of calls to analyse. We won't
579  // be able to analyse it's callee(s), of course.
580  callList.push_back(call);
581  }
582  else {
583  // Static call
584  Address callAddr = call->getFixedDest();
585 
586  // Calls with 0 offset (i.e. call the next instruction) are simply
587  // pushing the PC to the stack. Treat these as non-control flow
588  // instructions and continue.
589  if (callAddr == addr + inst.numBytes) {
590  break;
591  }
592 
593  // Call the virtual helper function. If implemented, will check for
594  // machine specific funcion calls
595  if (isHelperFunc(callAddr, addr, *BB_rtls)) {
596  // We have already added to BB_rtls
597  inst.rtl.reset(); // Discard the call semantics
598  break;
599  }
600 
601  RTL *rtl = inst.rtl.get();
602  BB_rtls->push_back(std::move(inst.rtl));
603 
604  // Add this non computed call site to the set of call sites which need
605  // to be analysed later.
606  callList.push_back(call);
607 
608  // Record the called address as the start of a new procedure if it
609  // didn't already exist.
610  if (!callAddr.isZero() && (callAddr != Address::INVALID) &&
611  (proc->getProg()->getFunctionByAddr(callAddr) == nullptr)) {
612  callList.push_back(call);
613 
614  if (m_program->getProject()->getSettings()->traceDecoder) {
615  LOG_MSG("p%1", callAddr);
616  }
617  }
618 
619  // Check if this is the _exit or exit function. May prevent us from
620  // attempting to decode invalid instructions, and getting invalid stack
621  // height errors
622  QString procName = m_program->getSymbolNameByAddr(callAddr);
623 
624  if (procName.isEmpty() && refersToImportedFunction(call->getDest())) {
625  Address a = call->getDest()->access<Const, 1>()->getAddr();
626  procName = m_program->getBinaryFile()
627  ->getSymbols()
628  ->findSymbolByAddress(a)
629  ->getName();
630  }
631 
632  if (!procName.isEmpty() && isNoReturnCallDest(procName)) {
633  // Make sure it has a return appended (so there is only one exit
634  // from the function)
635  currentBB = cfg->createBB(BBType::Call, std::move(BB_rtls));
636  appendSyntheticReturn(currentBB, proc, rtl);
637 
638  // Stop decoding sequentially
639  sequentialDecode = false;
640  }
641  else {
642  // Create the new basic block
643  currentBB = cfg->createBB(BBType::Call, std::move(BB_rtls));
644  BB_rtls = nullptr;
645 
646  if (call->isReturnAfterCall()) {
647  // Constuct the RTLs for the new basic block
648  std::unique_ptr<RTLList> rtls(new RTLList);
649  rtls->push_back(std::unique_ptr<RTL>(
650  new RTL(rtl->getAddress() + 1, { new ReturnStatement() })));
651  BasicBlock *returnBB = cfg->createBB(BBType::Ret, std::move(rtls));
652 
653  // Add out edge from call to return
654  cfg->addEdge(currentBB, returnBB);
655 
656  // Mike: do we need to set return locations?
657  // This ends the function
658  sequentialDecode = false;
659  }
660  else {
661  // Add the fall through edge if the block didn't
662  // already exist
663  if (currentBB != nullptr) {
664  cfg->addEdge(currentBB, addr + inst.numBytes);
665  }
666  }
667  }
668  }
669 
670  if (currentBB && currentBB->getRTLs()) {
671  extraProcessCall(call, *currentBB->getRTLs());
672  }
673 
674  // make sure we already moved the created RTL into a BB
675  assert(BB_rtls == nullptr);
676  break;
677  }
678 
679  case StmtType::Ret:
680  // Stop decoding sequentially
681  sequentialDecode = false;
682 
683  // Create the list of RTLs for the next basic block and
684  // continue with the next instruction.
685  createReturnBlock(proc, std::move(BB_rtls), std::move(inst.rtl));
686  break;
687 
688  case StmtType::BoolAssign:
689  // This is just an ordinary instruction; no control transfer
690  // Fall through
691  // FIXME: Do we need to do anything here?
692  case StmtType::Assign:
693  case StmtType::PhiAssign:
694  case StmtType::ImpAssign:
695  // Do nothing
696  break;
697  case StmtType::INVALID: assert(false); break;
698  }
699 
700  // This can happen if a high-level statement (e.g. a return)
701  // is not the last statement in a rtl. Ignore the following statements.
702  if (!sequentialDecode) {
703  break;
704  }
705  }
706 
707  if (BB_rtls != nullptr && inst.rtl != nullptr) {
708  // If non null, we haven't put this RTL into a the current BB as yet
709  BB_rtls->push_back(std::move(inst.rtl));
710  }
711 
712  if (inst.reDecode) {
713  // Special case: redecode the last instruction, without advancing addr by
714  // numBytes
715  continue;
716  }
717 
718  addr += inst.numBytes;
719 
720  if (addr > lastAddr) {
721  lastAddr = addr;
722  }
723 
724  // If sequentially decoding, check if the next address happens to be the start of an
725  // existing BB. If so, finish off the current BB (if any RTLs) as a fallthrough, and
726  // no need to decode again (unless it's an incomplete BB, then we do decode it). In
727  // fact, mustn't decode twice, because it will muck up the coverage, but also will
728  // cause subtle problems like add a call to the list of calls to be processed, then
729  // delete the call RTL (e.g. Pentium 134.perl benchmark)
730  if (sequentialDecode && cfg->isStartOfBB(addr)) {
731  // Create the fallthrough BB, if there are any RTLs at all
732  if (BB_rtls) {
733  BasicBlock *bb = cfg->createBB(BBType::Fall, std::move(BB_rtls));
734 
735  // Add an out edge to this address
736  if (bb) {
737  cfg->addEdge(bb, addr);
738  }
739  }
740 
741  // Pick a new address to decode from, if the BB is complete
742  if (!cfg->isStartOfIncompleteBB(addr)) {
743  sequentialDecode = false;
744  }
745  }
746  } // while sequentialDecode
747 
748  sequentialDecode = true;
749  } // while getNextAddress() != Address::INVALID
750 
751 
752  for (CallStatement *callStmt : callList) {
753  Address dest = callStmt->getFixedDest();
754 
755  // Don't visit the destination of a register call
756  Function *np = callStmt->getDestProc();
757 
758  if ((np == nullptr) && (dest != Address::INVALID)) {
759  // np = newProc(proc->getProg(), dest);
760  np = proc->getProg()->getOrCreateFunction(dest);
761  }
762 
763  if (np != nullptr) {
764  proc->addCallee(np);
765  }
766  }
767 
768  m_program->getProject()->alertFunctionDecoded(proc, startAddr, lastAddr, numBytesDecoded);
769 
770  LOG_VERBOSE("### Finished decoding proc '%1' ###", proc->getName());
771 
772  return true;
773 }

913 BasicBlock *DefaultFrontEnd::createReturnBlock(UserProc *proc, std::unique_ptr<RTLList> BB_rtls,
914  std::unique_ptr<RTL> returnRTL)
915 {
916  ProcCFG *cfg = proc->getCFG();
917 
918  // Add the RTL to the list; this has the semantics for the return instruction as well as the
919  // ReturnStatement The last Statement may get replaced with a GotoStatement
920  if (BB_rtls == nullptr) {
921  BB_rtls.reset(new RTLList); // In case no other semantics
922  }
923 
924  RTL *retRTL = returnRTL.get();
925  BB_rtls->push_back(std::move(returnRTL));
926  Address retAddr = proc->getRetAddr();
927  BasicBlock *newBB = nullptr;
928 
929  if (retAddr == Address::INVALID) {
930  // Create the basic block
931  newBB = cfg->createBB(BBType::Ret, std::move(BB_rtls));
932  if (newBB) {
933  Statement *s = retRTL->back(); // The last statement should be the ReturnStatement
934  proc->setRetStmt(static_cast<ReturnStatement *>(s), retRTL->getAddress());
935  }
936  }
937  else {
938  // We want to replace the *whole* RTL with a branch to THE first return's RTL. There can
939  // sometimes be extra semantics associated with a return (e.g. Pentium return adds to
940  // the stack pointer before setting %pc and branching). Other semantics (e.g. SPARC
941  // returning a value as part of the restore instruction) are assumed to appear in a
942  // previous RTL. It is assumed that THE return statement will have the same semantics
943  // (NOTE: may not always be valid). To avoid this assumption, we need branches to
944  // statements, not just to native addresses (RTLs).
945  BasicBlock *retBB = proc->getCFG()->findRetNode();
946  assert(retBB);
947 
948  if (retBB->getFirstStmt()->isReturn()) {
949  // ret node has no semantics, clearly we need to keep ours
950  assert(!retRTL->empty());
951  retRTL->pop_back();
952  }
953  else {
954  retRTL->clear();
955  }
956 
957  retRTL->append(new GotoStatement(retAddr));
958  newBB = cfg->createBB(BBType::Oneway, std::move(BB_rtls));
959 
960  if (newBB) {
961  cfg->ensureBBExists(retAddr, retBB);
962  cfg->addEdge(newBB, retBB);
963 
964  // Visit the return instruction. This will be needed in most cases to split the
965  // return BB (if it has other instructions before the return instruction).
966  m_targetQueue.visit(cfg, retAddr, newBB);
967  }
968  }
969 
970  // make sure the RTLs have been moved into the BB
971  assert(BB_rtls == nullptr);
972  return newBB;
973 }

71 Address DOS4GWBinaryLoader::getMainEntryPoint()
72 {
73  const BinarySymbol *sym = m_symbols->findSymbolByName("main");
74 
75  if (sym) {
76  return sym->getLocation();
77  }
78 
79  sym = m_symbols->findSymbolByName("__CMain");
80 
81  if (sym) {
82  return sym->getLocation();
83  }
84 
85  // Search with this crude pattern: call, sub ebp, ebp, call __Cmain in the first 0x300 bytes
86  // Start at program entry point
87  unsigned p = READ4_LE(m_LXHeader.eip);
88  unsigned lim = p + 0x300;
89  Address addr;
90 
91  // unsigned lastOrdCall = 0; //TODO: identify the point of setting this variable
92  bool gotSubEbp = false; // True if see sub ebp, ebp
93  bool lastWasCall = false; // True if the last instruction was a call
94 
95  BinarySection *textSection = m_image->getSectionByName(
96  "seg0"); // Assume the first section is text
97 
98  if (textSection == nullptr) {
99  textSection = m_image->getSectionByName(".text");
100  }
101 
102  if (textSection == nullptr) {
103  textSection = m_image->getSectionByName("CODE");
104  }
105 
106  if (textSection == nullptr) {
107  LOG_ERROR("Could not find text (code) section!");
108  return Address::INVALID;
109  }
110 
111  const Address nativeOrigin = textSection->getSourceAddr();
112  const unsigned textSize = textSection->getSize();
113 
114  if (textSize < 0x300) {
115  lim = p + textSize;
116  }
117 
118  while (p < lim) {
119  const Byte op1 = Util::readByte(&m_imageBase[p + 0]);
120  const Byte op2 = Util::readByte(&m_imageBase[p + 1]);
121 
122  switch (op1) {
123  case 0xE8:
124 
125  // An ordinary call
126  if (gotSubEbp) {
127  // This is the call we want. Get the offset from the call instruction
128  addr = nativeOrigin + p + 5 + READ4_LE_P(&m_imageBase[p + 1]);
129  LOG_VERBOSE("Found __CMain at address %1", addr);
130  return addr;
131  }
132 
133  // lastOrdCall = p;
134  lastWasCall = true;
135  break;
136 
137  case 0x2B: // 0x2B 0xED is sub ebp,ebp
138 
139  if ((op2 == 0xED) && lastWasCall) {
140  gotSubEbp = true;
141  }
142 
143  lastWasCall = false;
144  break;
145 
146  default:
147  gotSubEbp = false;
148  lastWasCall = false;
149  break;
150 
151  case 0xEB: // Short relative jump
152 
153  if (op2 >= 0x80) { // Branch backwards?
154  break; // Yes, just ignore it
155  }
156 
157  // Otherwise, actually follow the branch. May have to modify this some time...
158  p += op2 + 2; // +2 for the instruction itself, and op2 for the displacement
159  continue; // Don't break, we have the new "pc" set already
160  }
161 
162  int size = microX86Dis(&m_imageBase[p + 0]);
163 
164  if (size == 0x40) {
165  LOG_WARN("Microdisassembler out of step at offset %1", p);
166  size = 1;
167  }
168 
169  p += size;
170  }
171 
172  return Address::INVALID;
173 }

176 bool DOS4GWBinaryLoader::loadFromMemory(QByteArray &data)
177 {
178  QBuffer buf(&data);
179 
180  if (!buf.open(QBuffer::ReadOnly)) {
181  return false;
182  }
183 
184  DWord lxoffLE, lxoff;
185 
186  if (!buf.seek(0x3c)) {
187  return false;
188  }
189 
190  buf.read(reinterpret_cast<char *>(&lxoffLE), 4); // Note: peoffLE will be in Little Endian
191  lxoff = READ4_LE(lxoffLE);
192 
193  if (!buf.seek(lxoff)) {
194  return false;
195  }
196 
197  if (!buf.read(reinterpret_cast<char *>(&m_LXHeader), sizeof(LXHeader))) {
198  return false;
199  }
200 
201  if (!Util::testMagic(&m_LXHeader.sigLo, { 'L', 'X' }) &&
202  !Util::testMagic(&m_LXHeader.sigLo, { 'L', 'E' })) {
203  LOG_ERROR("Error loading file: bad LE/LX magic");
204  return false;
205  }
206 
207  const DWord objTableOffset = READ4_LE(m_LXHeader.objtbloffset);
208  if (!Util::inRange(lxoff, sizeof(LXHeader), buf.size()) ||
209  !Util::inRange(objTableOffset, 0U, (DWord)buf.size()) ||
210  !Util::inRange(lxoff + objTableOffset, sizeof(LXHeader), buf.size())) {
211  LOG_ERROR("Cannot read LX file: Object table extends past file boundary");
212  return false;
213  }
214  else if (!buf.seek(lxoff + READ4_LE(m_LXHeader.objtbloffset))) {
215  LOG_ERROR("Cannot read LX object table");
216  return false;
217  }
218 
219  const DWord numObjsInModule = READ4_LE(m_LXHeader.numobjsinmodule);
220  if (!Util::inRange(numObjsInModule, 0UL,
221  (buf.size() - lxoff - objTableOffset) / sizeof(LXObject))) {
222  LOG_ERROR("Cannot read LX file: Object table extends past file boundary");
223  return false;
224  }
225 
226  m_LXObjects.resize(numObjsInModule);
227 
228  buf.read(reinterpret_cast<char *>(&m_LXObjects[0]), numObjsInModule * sizeof(LXObject));
229 
230  unsigned npages = 0;
231  int cbImage = 0;
232 
233  for (unsigned n = 0; n < numObjsInModule; n++) {
234  if (READ4_LE(m_LXObjects[n].ObjectFlags) & 0x40) {
235  if (READ4_LE(m_LXObjects[n].PageTblIdx) + READ4_LE(m_LXObjects[n].NumPageTblEntries) -
236  1 >
237  npages) {
238  npages = READ4_LE(m_LXObjects[n].PageTblIdx) +
239  READ4_LE(m_LXObjects[n].NumPageTblEntries) - 1;
240  }
241 
242  cbImage = READ4_LE(m_LXObjects[n].RelocBaseAddr) + READ4_LE(m_LXObjects[n].VirtualSize);
243  }
244  }
245 
246  if (numObjsInModule > 0) {
247  cbImage -= READ4_LE(m_LXObjects[0].RelocBaseAddr);
248  }
249 
250  if (!Util::inRange(cbImage, 0, buf.size())) {
251  LOG_ERROR("Cannot allocate %1 bytes for copy of LX image", cbImage);
252  return false;
253  }
254 
255  m_imageBase.resize(cbImage);
256 
257  uint32_t numSections = READ4_LE(m_LXHeader.numobjsinmodule);
258  std::vector<SectionParam> params;
259 
260  for (unsigned n = 0; n < numSections; n++) {
261  if (READ4_LE(m_LXObjects[n].ObjectFlags) & 0x40) {
262  printf("vsize %x reloc %x flags %x page %u npage %u\n",
263  READ4_LE(m_LXObjects[n].VirtualSize), READ4_LE(m_LXObjects[n].RelocBaseAddr),
264  READ4_LE(m_LXObjects[n].ObjectFlags), READ4_LE(m_LXObjects[n].PageTblIdx),
265  READ4_LE(m_LXObjects[n].NumPageTblEntries));
266 
267  SectionParam sect;
268  DWord Flags = READ4_LE(m_LXObjects[n].ObjectFlags);
269 
270  sect.Name = QString("seg%1").arg(n); // no section names in LX
271  sect.from = Address(READ4_LE(m_LXObjects[n].RelocBaseAddr));
272  sect.ImageAddress = HostAddress(&m_imageBase[0]) +
273  (sect.from - params.front().from).value();
274  sect.Size = READ4_LE(m_LXObjects[n].VirtualSize);
275  sect.Bss = 0; // TODO
276  sect.Code = (Flags & 0x4) ? true : false;
277  sect.Data = (Flags & 0x4) ? false : true;
278  sect.ReadOnly = (Flags & 0x1) ? false : true;
279 
280  buf.seek(m_LXHeader.datapagesoffset +
281  (READ4_LE(m_LXObjects[n].PageTblIdx) - 1) * READ4_LE(m_LXHeader.pagesize));
282  char *p = &m_imageBase[0] + READ4_LE(m_LXObjects[n].RelocBaseAddr) -
283  READ4_LE(m_LXObjects[0].RelocBaseAddr);
284  buf.read(p, READ4_LE(m_LXObjects[n].NumPageTblEntries) * READ4_LE(m_LXHeader.pagesize));
285  }
286  }
287 
288  for (SectionParam par : params) {
289  BinarySection *sect = m_image->createSection(par.Name, par.from, par.from + par.Size);
290 
291  if (sect) {
292  sect->setBss(par.Bss);
293  sect->setCode(par.Code);
294  sect->setData(par.Data);
295  sect->setReadOnly(par.ReadOnly);
296  sect->setHostAddr(par.ImageAddress);
297  }
298  }
299 
300  // TODO: decode entry tables
301 
302  // fixups
303  if (!buf.seek(READ4_LE(m_LXHeader.fixuppagetbloffset) + lxoff)) {
304  return false;
305  }
306 
307  unsigned int *fixuppagetbl = new unsigned int[npages + 1];
308  buf.read(reinterpret_cast<char *>(fixuppagetbl), sizeof(unsigned int) * (npages + 1));
309 
310  // for (unsigned n = 0; n < npages; n++)
311  // printf("offset for page %i: %x\n", n + 1, fixuppagetbl[n]);
312  // printf("offset to end of fixup rec: %x\n", fixuppagetbl[npages]);
313 
314  buf.seek(READ4_LE(m_LXHeader.fixuprecordtbloffset) + lxoff);
315  LXFixup fixup;
316  unsigned srcpage = 0;
317 
318  do {
319  buf.read(reinterpret_cast<char *>(&fixup), sizeof(fixup));
320 
321  if ((fixup.src != 7) || (fixup.flags & ~0x50)) {
322  LOG_WARN("Unknown fixup type %1 %2", QString("%1").arg(fixup.src, 2, 16, QChar('0')),
323  QString("%1").arg(fixup.flags, 2, 16, QChar('0')));
324 
325  return false;
326  }
327 
328  // printf("srcpage = %i srcoff = %x object = %02x trgoff = %x\n", srcpage + 1, fixup.srcoff,
329  // fixup.object, fixup.trgoff);
330  unsigned long src = srcpage * READ4_LE(m_LXHeader.pagesize) + READ2_LE(fixup.srcoff);
331  unsigned short object = 0;
332 
333  if (fixup.flags & 0x40) {
334  buf.read(reinterpret_cast<char *>(&object), 2);
335  }
336  else {
337  buf.read(reinterpret_cast<char *>(&object), 1);
338  }
339 
340  unsigned int trgoff = 0;
341 
342  if (fixup.flags & 0x10) {
343  buf.read(reinterpret_cast<char *>(&trgoff), 4);
344  }
345  else {
346  buf.read(reinterpret_cast<char *>(&trgoff), 2);
347  }
348 
349  unsigned long target = READ4_LE(m_LXObjects[object - 1].RelocBaseAddr) + READ2_LE(trgoff);
350  // printf("relocate dword at %x to point to %x\n", src, target);
351  Util::writeDWord(&m_imageBase[src], target, Endian::Little);
352 
353  while (buf.pos() - (READ4_LE(m_LXHeader.fixuprecordtbloffset) + lxoff) >=
354  READ4_LE(fixuppagetbl[srcpage + 1])) {
355  srcpage++;
356  }
357  } while (srcpage < npages);
358 
359  return true;
360 }

363 int DOS4GWBinaryLoader::canLoad(QIODevice &fl) const
364 {
365  LXHeader testHeader;
366 
367  if (fl.read(reinterpret_cast<char *>(&testHeader), sizeof(testHeader)) != sizeof(LXHeader)) {
368  return 0; // file too small
369  }
370  else if (!Util::testMagic(&testHeader.sigLo, { 'M', 'Z' })) {
371  return 0; // not a DOS-based file
372  }
373 
374  // read offset 0x3C from beginning of file (pointer to PE header)
375  const DWord peOffset = READ4_LE(testHeader.loadersectionchksum);
376  if (!fl.seek(peOffset)) {
377  return 0; // file corrupted / too small
378  }
379 
380  Byte buf[2];
381  if (fl.read(reinterpret_cast<char *>(buf), 2) != 2) {
382  return 0;
383  }
384 
385  if (Util::testMagic(buf, { 'L', 'E' })) {
386  return 2 + 4 + 2; // Win32 VxD (Linear Executable) or DOS4GW app
387  }
388 
389  return 0; // unrecognized file format
390 }

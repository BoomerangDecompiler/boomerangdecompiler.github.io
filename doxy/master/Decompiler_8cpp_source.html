66 void Decompiler::loadInputFile(const QString &inputFile, const QString &outputPath)
67 {
68  m_project.getSettings()->setOutputDirectory(outputPath);
69  emit loadingStarted();
70 
71  bool ok = m_project.loadBinaryFile(inputFile);
72  if (!ok) {
73  emit machineTypeChanged(QString("Unavailable: Load Failed!"));
74  return;
75  }
76 
77  switch (m_project.getLoadedBinaryFile()->getMachine()) {
78  case Machine::PENTIUM: emit machineTypeChanged("pentium"); break;
79  case Machine::SPARC: emit machineTypeChanged("sparc"); break;
80  case Machine::PPC: emit machineTypeChanged("ppc"); break;
81  case Machine::ST20: emit machineTypeChanged("st20"); break;
82  case Machine::UNKNOWN:
83  case Machine::INVALID: emit machineTypeChanged("UNKNOWN"); break;
84  }
85 
86  IFrontEnd *fe = m_project.getProg()->getFrontEnd();
87  assert(fe != nullptr);
88  std::vector<Address> entrypoints = fe->findEntryPoints();
89 
90  for (Address entryPoint : entrypoints) {
91  m_userEntrypoints.push_back(entryPoint);
92  emit entryPointAdded(entryPoint, m_project.getProg()->getSymbolNameByAddr(entryPoint));
93  }
94 
95  for (const BinarySection *section : *m_project.getLoadedBinaryFile()->getImage()) {
96  emit sectionAdded(section->getName(), section->getSourceAddr(),
97  section->getSourceAddr() + section->getSize());
98  }
99 
100  emit loadCompleted();
101 }

183 void Decompiler::onFunctionCreated(Function *function)
184 {
185  if (function->isLib()) {
186  QString params;
187 
188  if ((function->getSignature() == nullptr) || function->getSignature()->isUnknown()) {
189  params = "<unknown>";
190  }
191  else {
192  for (int i = 0; i < function->getSignature()->getNumParams(); i++) {
193  auto ty = function->getSignature()->getParamType(i);
194  params.append(ty->getCtype());
195  params.append(" ");
196  params.append(function->getSignature()->getParamName(i));
197 
198  if (i != function->getSignature()->getNumParams() - 1) {
199  params.append(", ");
200  }
201  }
202  }
203 
204  emit libProcCreated(function->getName(), params);
205  }
206  else {
207  emit userProcCreated(function->getName(), function->getEntryAddress());
208  }
209 }

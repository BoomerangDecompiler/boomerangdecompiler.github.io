112 /**
113  * Convert a list of actual parameters in the form of a STL list of Exps
114  * into one expression (using opList)
115  * \note The expressions in the list are not cloned;
116  * they are simply copied to the new opList
117  *
118  * \param le the list of expressions
119  * \returns The opList Expression
120  */
121 SharedExp listExpToExp(std::list<SharedExp> *le)
122 {
123  SharedExp e;
124  SharedExp *cur = &e;
125  SharedExp end = Terminal::get(opNil); // Terminate the chain
126 
127  for (auto &elem : *le) {
128  *cur = Binary::get(opList, elem, end);
129  // cur becomes the address of the address of the second subexpression
130  // In other words, cur becomes a reference to the second subexp ptr
131  // Note that declaring cur as a reference doesn't work (remains a reference to e)
132  cur = &(*cur)->refSubExp2();
133  }
134 
135  return e;
136 }

139 /**
140  * Expand tables in an RTL and save to dictionary
141  * \note This may generate many entries
142  *
143  * \param iname Parser object representing the instruction name
144  * \param params Parser object representing the instruction params
145  * \param o_rtlist Original rtlist object (before expanding)
146  * \param Dict Ref to the dictionary that will contain the results of the parse
147  */
148 bool SSL2ParserDriver::expandTables(const std::shared_ptr<InsNameElem> &iname,
149  const std::shared_ptr<std::list<QString>> &params,
150  SharedRTL o_rtlist, RTLInstDict *dict)
151 {
152  const int m = iname->getNumInstructions();
153  iname->reset();
154 
155  // Expand the tables (if any) in this instruction
156  for (int i = 0; i < m; i++, iname->increment()) {
157  const QString instructionName = iname->getInstruction();
158 
159  // Need to make substitutions to a copy of the RTL
160  RTL rtl(*o_rtlist); // deep copy of contents
161 
162  if (dict->insert(instructionName, *params, rtl) != 0) {
163  LOG_ERROR("Pattern %1 conflicts with an earlier declaration of %2.",
164  iname->getInsPattern(), instructionName);
165  return false;
166  }
167  }
168 
169  indexrefmap.erase(indexrefmap.begin(), indexrefmap.end());
170  return true;
171 }

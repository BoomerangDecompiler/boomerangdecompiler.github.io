25 bool UsedLocsFinder::preVisit(const std::shared_ptr<Location> &exp, bool &visitChildren)
26 {
27  if (!m_memOnly) {
28  m_used->insert(exp->shared_from_this()); // All locations visited are used
29  }
30 
31  if (exp->isMemOf()) {
32  // Example: m[r28{10} - 4] we use r28{10}
33  SharedExp child = exp->access<Exp, 1>();
34  // Care! Need to turn off the memOnly flag for work inside the m[...], otherwise everything
35  // will get ignored
36  bool wasMemOnly = m_memOnly;
37  m_memOnly = false;
38  child->acceptVisitor(this);
39  m_memOnly = wasMemOnly;
40  visitChildren = false; // Already looked inside child
41  }
42  else {
43  visitChildren = true;
44  }
45 
46  return true; // Continue looking for other locations
47 }

50 bool UsedLocsFinder::visit(const std::shared_ptr<Terminal> &exp)
51 {
52  if (m_memOnly) {
53  return true; // Only interested in m[...]
54  }
55 
56  switch (exp->getOper()) {
57  case opPC:
58  case opFlags:
59  case opFflags:
60  case opDefineAll:
61  // Fall through
62  // The carry flag can be used in some SPARC idioms, etc
63  case opDF:
64  case opCF:
65  case opZF:
66  case opNF:
67  case opOF: // also these

70

72  }
73 
74  return true; // Always continue recursion

78 bool UsedLocsFinder::preVisit(const std::shared_ptr<RefExp> &exp, bool &visitChildren)
79 {
80  if (m_memOnly) {
81  visitChildren = true; // Look inside the ref for m[...]
82  return true; // Don't count this reference
83  }
84 
85  std::shared_ptr<RefExp> e = exp;
86  m_used->insert(e); // This location is used
87 
88  // However, e's subexpression is NOT used ...
89  visitChildren = false;
90  // ... unless that is a m[x], array[x] or .x, in which case x (not m[x]/array[x]/refd.x) is used
91  SharedExp refd = e->getSubExp1();
92 
93  if (refd->isMemOf()) {
94  refd->getSubExp1()->acceptVisitor(this);
95  }
96  else if (refd->isArrayIndex()) {
97  refd->getSubExp1()->acceptVisitor(this);
98  refd->getSubExp2()->acceptVisitor(this);
99  }
100  else if (refd->isMemberOf()) {
101  refd->getSubExp1()->acceptVisitor(this);
102  }
103 
104  return true;

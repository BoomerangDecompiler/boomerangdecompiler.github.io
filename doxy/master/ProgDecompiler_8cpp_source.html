32 void ProgDecompiler::decompile()
33 {
34  assert(!m_prog->getModuleList().empty());
35  LOG_VERBOSE("%1 procedures", m_prog->getNumFunctions(false));
36 
37  // Start decompiling each entry point
38  for (UserProc *up : m_prog->getEntryProcs()) {
39  LOG_MSG("Decompiling entry point '%1'", up->getName());
40  up->decompileRecursive();
41  }
42 
43  // Just in case there are any Procs not in the call graph.
44 
45  if (m_prog->getProject()->getSettings()->decodeMain &&
46  m_prog->getProject()->getSettings()->decodeChildren) {
47  bool foundone = true;
48 
49  while (foundone) {
50  foundone = false;
51 
52  for (const auto &module : m_prog->getModuleList()) {
53  for (Function *pp : *module) {
54  if (pp->isLib()) {
55  continue;
56  }
57 
58  UserProc *proc = static_cast<UserProc *>(pp);
59 
60  if (proc->isDecompiled()) {
61  continue;
62  }
63  proc->decompileRecursive();
64  foundone = true;
65  }
66  }
67  }
68  }
69 
70  globalTypeAnalysis();
71 
72  if (m_prog->getProject()->getSettings()->removeReturns) {
73  // Repeat until no change. Not 100% sure if needed.
74  while (removeUnusedParamsAndReturns()) {
75  for (auto &module : m_prog->getModuleList()) {
76  for (Function *proc : *module) {
77  if (proc->isLib()) {
78  continue;
79  }
80 
81  PassManager::get()->executePass(PassID::BranchAnalysis,
82  static_cast<UserProc *>(proc));
83  }
84  }
85  }
86  }
87 
88  globalTypeAnalysis();
89 
90  // Now it is OK to transform out of SSA form
91  fromSSAForm();
92  removeUnusedGlobals();
93 
94  LOG_MSG("Compressing CFG...");
95 
96  for (const auto &module : m_prog->getModuleList()) {
97  for (Function *func : *module) {
98  if (!func->isLib()) {
99  CFGCompressor().compressCFG(static_cast<UserProc *>(func)->getCFG());
100  }
101  }
102  }
103 
104  LOG_MSG("Decompilation finished.");
105 }

108 void ProgDecompiler::globalTypeAnalysis()
109 {
110  LOG_MSG("Performing global type analysis...");
111 
112  if (m_prog->getProject()->getSettings()->debugTA) {
113  LOG_VERBOSE("### Start global data-flow-based type analysis ###");
114  }
115 
116  for (const auto &module : m_prog->getModuleList()) {
117  for (Function *pp : *module) {
118  UserProc *proc = dynamic_cast<UserProc *>(pp);
119 
120  if (!proc || !proc->isDecoded()) {
121  continue;
122  }
123 
124  // FIXME: this just does local TA again. Need to meet types for all parameter/arguments,
125  // and return/results! This will require a repeat until no change loop
126  LOG_VERBOSE("Global type analysis for '%1'", proc->getName());
127  PassManager::get()->executePass(PassID::LocalTypeAnalysis, proc);
128  }
129  }
130 
131  if (m_prog->getProject()->getSettings()->debugTA) {
132  LOG_VERBOSE("### End type analysis ###");
133  }
134 }

137 void ProgDecompiler::removeUnusedGlobals()
138 {
139  LOG_MSG("Removing unused global variables...");
140 
141  // seach for used globals
142  std::list<SharedExp> usedGlobals;
143 
144  for (const auto &module : m_prog->getModuleList()) {
145  for (Function *func : *module) {
146  if (func->isLib()) {
147  continue;
148  }
149 
150  UserProc *proc = static_cast<UserProc *>(func);
151  Location search(opGlobal, Terminal::get(opWild), proc);
152  // Search each statement in u, excepting implicit assignments (their uses don't count,
153  // since they don't really exist in the program representation)
154  StatementList stmts;
155  StatementList::iterator ss;
156  proc->getStatements(stmts);
157 
158  for (Statement *s : stmts) {
159  if (s->isImplicit()) {
160  continue; // Ignore the uses in ImplicitAssigns
161  }
162 
163  bool found = s->searchAll(search, usedGlobals);
164 
165  if (found && m_prog->getProject()->getSettings()->debugUnused) {
166  LOG_VERBOSE("A global is used by stmt %1", s->getNumber());
167  }
168  }
169  }
170  }
171 
172  // make a map to find a global by its name (could be a global var too)
173  QMap<QString, std::shared_ptr<Global>> namedGlobals;
174 
175  for (auto &g : m_prog->getGlobals()) {
176  namedGlobals[g->getName()] = g;
177  }
178 
179  // Rebuild the globals vector. Delete the unused globals only after re-inserting them
180  Prog::GlobalSet oldGlobals = m_prog->getGlobals();
181  m_prog->getGlobals().clear();
182 
183  for (const SharedExp &e : usedGlobals) {
184  if (m_prog->getProject()->getSettings()->debugUnused) {
185  LOG_MSG(" %1 is used", e);
186  }
187 
188  QString name(e->access<Const, 1>()->getStr());
189  auto &usedGlobal = namedGlobals[name];
190 
191  if (usedGlobal) {
192  m_prog->getGlobals().insert(usedGlobal);
193  }
194  else {
195  LOG_WARN("An expression refers to a nonexistent global");
196  }
197  }
198 }

209 {

211 
212  for (const auto &module : m_prog->getModuleList()) {
213  for (Function *pp : *module) {
214  if (pp->isLib()) {
215  continue;
216  }
217 
218  UserProc *proc = static_cast<UserProc *>(pp);
219  proc->numberStatements();
220  PassManager::get()->executePass(PassID::FromSSAForm, proc);
221  }
222  }

178 {
179  std::list<SharedExp *> li;
180  result = nullptr; // In case it fails; don't leave it unassigned
181  // The search requires a reference to a pointer to this object.
182  // This isn't needed for searches, only for replacements, but we want to re-use the same search
183  // routine
184  SharedExp top = shared_from_this();
185  doSearch(pattern, top, li, false);
186 
187  if (!li.empty()) {
188  result = *li.front();
189  return true;
190  }
191 
192  return false;

220 SharedExp Exp::searchReplaceAll(const Exp &pattern, const SharedExp &replace, bool &change,
221  bool once /* = false */)
222 {
223  // TODO: consider working on base object, and only in case when we find the search, use clone
224  // call to return the new object ?
225  if (*this == pattern) {
226  change = true;
227  return replace->clone();
228  }
229 
230  std::list<SharedExp *> matches;
231  SharedExp top = shared_from_this(); // top may change; that's why we have to return it
232  doSearch(pattern, top, matches, false);
233 
234  for (SharedExp *pexp : matches) {
235  *pexp = replace->clone(); // Do the replacement
236 
237  if (once) {
238  change = true;
239  return top;
240  }
241  }
242 
243  change = !matches.empty();
244  return top;

248 void Exp::doSearch(const Exp &pattern, SharedExp &toSearch, std::list<SharedExp *> &matches,
249  bool once)
250 {
251  const bool compare = (pattern == *toSearch);
252 
253  if (compare) {
254  matches.push_back(&toSearch); // Success
255 
256  if (once) {
257  return; // No more to do
258  }
259  }
260 
261  // Either want to find all occurrences, or did not match at this level
262  // Recurse into children, unless a matching opSubscript
263  if (!compare || (toSearch->m_oper != opSubscript)) {
264  toSearch->doSearchChildren(pattern, matches, once);
265  }
266 }

278 void Exp::partitionTerms(std::list<SharedExp> &positives, std::list<SharedExp> &negatives,
279  std::vector<int> &integers, bool negate)
280 {
281  SharedExp p1, p2;
282 
283  switch (m_oper) {
284  case opPlus:
285  p1 = getSubExp1();
286  p2 = getSubExp2();
287  p1->partitionTerms(positives, negatives, integers, negate);
288  p2->partitionTerms(positives, negatives, integers, negate);
289  break;
290 
291  case opMinus:
292  p1 = getSubExp1();
293  p2 = getSubExp2();
294  p1->partitionTerms(positives, negatives, integers, negate);
295  p2->partitionTerms(positives, negatives, integers, !negate);
296  break;
297 
298  case opTypedExp:
299  p1 = getSubExp1();
300  p1->partitionTerms(positives, negatives, integers, negate);
301  break;
302 
303  case opIntConst: {
304  const int k = access<Const>()->getInt();
305 
306  if (negate) {
307  integers.push_back(-k);
308  }
309  else {
310  integers.push_back(k);
311  }
312 
313  break;
314  }
315 
316  default:
317  // These can be any other expression tree
318  if (negate) {
319  negatives.push_back(shared_from_this());
320  }
321  else {
322  positives.push_back(shared_from_this());
323  }
324  }
325 }

328 SharedExp Exp::accumulate(std::list<SharedExp> &exprs)
329 {
330  if (exprs.empty()) {
331  return Const::get(0);
332  }
333 
334  if (exprs.size() == 1) {
335  return exprs.front()->clone();
336  }
337 
338  std::list<SharedExp> cloned_list;
339  Util::clone(exprs, cloned_list);
340 
341  SharedExp last_val = cloned_list.back();
342  cloned_list.pop_back();
343  auto res = Binary::get(opPlus, cloned_list.back(), last_val);
344  cloned_list.pop_back();
345 
346  while (!cloned_list.empty()) {
347  res = Binary::get(opPlus, cloned_list.back(), res);
348  cloned_list.pop_back();
349  }
350 
351  return res;
352 }

413 SharedExp Exp::fixSuccessor()
414 {
415  SharedExp result;
416  UniqExp search_expression(new Unary(opSuccessor, Terminal::get(opWildRegOf)));
417 
418  // Assume only one successor function in any 1 expression
419  if (search(*search_expression, result)) {
420  // Result has the matching expression, i.e. succ(r[K])
421  SharedExp sub1 = result->getSubExp1();
422  assert(sub1->isRegOf());
423  SharedExp sub2 = sub1->getSubExp1();
424  assert(sub2->isIntConst());
425  // result sub1 sub2
426  // succ( r[ Const K ])
427  // Note: we need to clone the r[K] part, since it will be deleted as
428  // part of the searchReplace below
429  auto replace = sub1->clone();
430  auto c = replace->access<Const, 1>();
431  c->setInt(c->getInt() + 1); // Do the increment
432  bool change = false;
433  SharedExp res = searchReplace(*result, replace, change);
434  return res;
435  }
436 
437  return shared_from_this();
438 }

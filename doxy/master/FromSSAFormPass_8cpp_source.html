38 bool FromSSAFormPass::execute(UserProc *proc)
39 {
40  proc->getProg()->getProject()->alertDecompiling(proc);
41 
42  StatementList stmts;
43  proc->getStatements(stmts);
44 
45  for (Statement *s : stmts) {
46  // Map registers to initial local variables
47  mapRegistersToLocals(s);
48 
49  // Insert casts where needed, as types are about to become inaccessible
50  insertCastsForStmt(s);
51  }
52 
53  // First split the live ranges where needed by reason of type incompatibility, i.e. when the
54  // type of a subscripted variable is different to its previous type. Start at the top, because
55  // we don't want to rename parameters (e.g. argc)
56  typedef std::pair<SharedType, SharedExp> FirstTypeEnt;
57  typedef std::map<SharedExp, FirstTypeEnt, lessExpStar> FirstTypesMap;
58 
59  FirstTypesMap firstTypes;
60 
61  ConnectionGraph ig; // The interference graph; these can't have the same local variable
62  ConnectionGraph pu; // The Phi Unites: these need the same local variable or copies
63  const bool assumeABICompliance = proc->getProg()->getProject()->getSettings()->assumeABI;
64 
65  for (Statement *s : stmts) {
66  LocationSet defs;
67  s->getDefinitions(defs, assumeABICompliance);
68 
69  for (SharedExp defdByS : defs) {
70  SharedType ty = s->getTypeForExp(defdByS);
71 
72  if (ty == nullptr) { // Can happen e.g. when getting the type for %flags
73  ty = VoidType::get();
74  }
75 
76  if (proc->getProg()->getProject()->getSettings()->debugLiveness) {
77  LOG_MSG("Got type %1 for %2 from %3", ty->getCtype(), defdByS, s);
78  }
79 
80  FirstTypesMap::iterator ff = firstTypes.find(defdByS);
81  SharedExp ref = RefExp::get(defdByS, s);
82 
83  if (ff == firstTypes.end()) {
84  // There is no first type yet. Record it.
85  firstTypes[defdByS] = { ty, ref };
86  }
87  else if (ff->second.first && !ty->isCompatibleWith(*ff->second.first)) {
88  if (proc->getProg()->getProject()->getSettings()->debugLiveness) {
89  LOG_WARN("Def of %1 at '%2' is not compatible with first type %3.", defdByS,
90  s->toString(), ff->second.first);
91  }
92 
93  // There already is a type for defdByS, and it is different to the type for this
94  // definition. Record an "interference" so it will get a new variable
95  if (!ty->isVoid()) { // just ignore void interferences ??!!
96  ig.add(ref, ff->second.second);
97  }
98  }
99  }
100  }
101  assert(ig.allRefsHaveDefs());
102 
103  // Find the interferences generated by more than one version of a variable being live at the
104  // same program point
105  InterferenceFinder(proc->getCFG()).findInterferences(ig);
106  assert(ig.allRefsHaveDefs());
107 
108  // Find the set of locations that are "united" by phi-functions
109  // FIXME: are these going to be trivially predictable?
110  findPhiUnites(proc, pu);
111 
112  if (proc->getProg()->getProject()->getSettings()->debugLiveness) {
113  LOG_MSG("## ig interference graph:");
114 
115  for (const auto &[from, to] : ig) {
116  LOG_MSG(" ig %1 -> %2", from, to);
117  }
118 
119  LOG_MSG("## pu phi unites graph:");
120 
121  for (const auto &[from, to] : pu) {
122  LOG_MSG(" pu %1 -> %2", from, to);
123  }
124  }
125 
126  // Choose one of each interfering location to give a new name to
127  assert(ig.allRefsHaveDefs());
128 
129  for (const auto &[first, second] : ig) {
130  auto ref1 = first->access<RefExp>();
131  auto ref2 = second->access<RefExp>(); // r1 -> r2 and vice versa
132  QString name1 = proc->lookupSymFromRefAny(ref1);
133  QString name2 = proc->lookupSymFromRefAny(ref2);
134 
135  if (!name1.isEmpty() && !name2.isEmpty() && (name1 != name2)) {
136  continue; // Already different names, probably because of the redundant mapping
137  }
138 
139  std::shared_ptr<RefExp> rename;
140 
141  if (ref1->isImplicitDef()) {
142  // If r1 is an implicit definition, don't rename it (it is probably a parameter, and
143  // should retain its current name)
144  rename = ref2;
145  }
146  else if (ref2->isImplicitDef()) {
147  rename = ref1; // Ditto for r2
148  }
149 
150  if (rename == nullptr) {
151  if (ref2->getDef()->isPhi()) { // Prefer the destinations of phis
152  rename = ref2;
153  }
154  else {
155  rename = ref1;
156  }
157  }
158 
159  SharedType ty = rename->getDef()->getTypeForExp(rename->getSubExp1());
160  SharedExp local = proc->createLocal(ty, rename);
161 
162  if (proc->getProg()->getProject()->getSettings()->debugLiveness) {
163  LOG_MSG("Renaming %1 to %2", rename, local);
164  }
165 
166  proc->mapSymbolTo(rename, local);
167  }
168 
169  // Implement part of the Phi Unites list, where renamings or parameters have broken them, by
170  // renaming. The rest of them will be done as phis are removed The idea is that where l1 and l2
171  // have to unite, and exactly one of them already has a local/name, you can implement the
172  // unification by giving the unnamed one the same name as the named one, as long as they don't
173  // interfere
174  for (const auto &[first, second] : pu) {
175  auto ref1 = first->access<RefExp>();
176  auto ref2 = second->access<RefExp>();
177  QString name1 = proc->lookupSymFromRef(ref1);
178  QString name2 = proc->lookupSymFromRef(ref2);
179 
180  if (!name1.isEmpty() && !name2.isEmpty() && !ig.isConnected(ref1, *ref2)) {
181  // There is a case where this is unhelpful, and it happen in test/pentium/fromssa2. We
182  // have renamed the destination of the phi to ebx_1, and that leaves the two phi
183  // operands as ebx. However, we attempt to unite them here, which will cause one of the
184  // operands to become ebx_1, so the neat oprimisation of replacing the phi with one copy
185  // doesn't work. The result is an extra copy. So check of r1 is a phi and r2 one of its
186  // operands, and all other operands for the phi have the same name. If so, don't rename.
187  Statement *def1 = ref1->getDef();
188 
189  if (def1->isPhi()) {
190  bool allSame = true;
191  bool r2IsOperand = false;
192  QString firstName;
193  PhiAssign *pa = static_cast<PhiAssign *>(def1);
194 
195  for (const std::shared_ptr<RefExp> &refExp : *pa) {
196  auto re(RefExp::get(refExp->getSubExp1(), refExp->getDef()));
197 
198  if (*re == *ref2) {
199  r2IsOperand = true;
200  }
201 
202  if (firstName.isEmpty()) {
203  firstName = proc->lookupSymFromRefAny(re);
204  }
205  else {
206  QString tmp = proc->lookupSymFromRefAny(re);
207 
208  if (tmp.isEmpty() || (firstName != tmp)) {
209  allSame = false;
210  break;
211  }
212  }
213  }
214 
215  if (allSame && r2IsOperand) {
216  continue; // This situation has happened, don't map now
217  }
218  }
219 
220  proc->mapSymbolTo(ref2, Location::local(name1, proc));
221  continue;
222  }
223  }
224 
225  /* * * * * * * * * * * * * * *\
226  * *
227  * IR gets changed with hard locals and params here *
228  * *
229  \* * * * * * * * * * * * * * */
230 
231  // First rename the variables (including phi's, but don't remove).
232  // NOTE: it is not possible to postpone renaming these locals till the back end, since the same
233  // base location may require different names at different locations, e.g. r28{0} is local0,
234  // r28{16} is local1
235  // Update symbols and parameters, particularly for the stack pointer inside
236  // memofs.
237  // NOTE: the ordering of the below operations is critical! Re-ordering may well prevent
238  // e.g. parameters from renaming successfully.
239  assert(proc->allPhisHaveDefs());
240  nameParameterPhis(proc);
241  PassManager::get()->executePass(PassID::LocalAndParamMap, proc);
242  mapParameters(proc);
243  removeSubscriptsFromSymbols(proc);
244  removeSubscriptsFromParameters(proc);
245 
246  for (Statement *s : stmts) {
247  // The last part of the fromSSA logic:
248  // replace subscripted locations with suitable local variables
249  ExpSSAXformer esx(proc);
250  StmtSSAXformer ssx(&esx, proc);
251 
252  s->accept(&ssx);
253  }
254 
255  // Now remove the phis
256  for (Statement *s : stmts) {
257  if (!s->isPhi()) {
258  continue;
259  }
260 
261  // Check if the base variables are all the same
262  PhiAssign *phi = static_cast<PhiAssign *>(s);
263 
264  if (phi->begin() == phi->end()) {
265  // no params to this phi, just remove it
266  LOG_VERBOSE("Phi with no params, removing: %1", s);
267 
268  proc->removeStatement(s);
269  continue;
270  }
271 
272  LocationSet refs;
273  phi->addUsedLocs(refs);
274  bool phiParamsSame = true;
275  SharedExp first = nullptr;
276 
277  if (phi->getNumDefs() > 1) {
278  for (const std::shared_ptr<RefExp> &pi : *phi) {
279  if (pi->getSubExp1() == nullptr) {
280  continue;
281  }
282 
283  if (first == nullptr) {
284  first = pi->getSubExp1();
285  continue;
286  }
287 
288  if (!(*(pi->getSubExp1()) == *first)) {
289  phiParamsSame = false;
290  break;
291  }
292  }
293  }
294 
295  if (phiParamsSame && first) {
296  // Is the left of the phi assignment the same base variable as all the operands?
297  if (*phi->getLeft() == *first) {
298  if (proc->getProg()->getProject()->getSettings()->debugLiveness ||
299  proc->getProg()->getProject()->getSettings()->debugUnused) {
300  LOG_MSG("Removing phi: left and all refs same or 0: %1", s);
301  }
302 
303  // Just removing the refs will work, or removing the whole phi
304  // NOTE: Removing the phi here may cause other statments to be not used.
305  proc->removeStatement(s);
306  }
307  else {
308  // Need to replace the phi by an expression,
309  // e.g. local0 = phi(r24{3}, r24{5}) becomes
310  // local0 = r24
311  proc->replacePhiByAssign(phi, first->clone());
312  }
313  }
314  else {
315  // Need new local(s) for phi operands that have different names from the lhs
316 
317  // This way is costly in copies, but has no problems with extending live ranges
318  // Exp* tempLoc = newLocal(pa->getType());
319  SharedExp tempLoc = proc->getSymbolExp(RefExp::get(phi->getLeft(), phi),
320  phi->getType());
321 
322  if (proc->getProg()->getProject()->getSettings()->debugLiveness) {
323  LOG_MSG("Phi statement %1 requires local, using %2", s, tempLoc);
324  }
325 
326  // For each definition ref'd in the phi
327  for (const std::shared_ptr<RefExp> &pi : *phi) {
328  if (pi->getSubExp1() == nullptr) {
329  continue;
330  }
331 
332  proc->insertAssignAfter(pi->getDef(), tempLoc, pi->getSubExp1());
333  }
334 
335  // Replace the RHS of the phi with tempLoc
336  proc->replacePhiByAssign(phi, tempLoc);
337  }
338  }
339 
340  return true;
341 }

344 void FromSSAFormPass::nameParameterPhis(UserProc *proc)
345 {
346  StatementList stmts;
347  proc->getStatements(stmts);
348 
349  for (Statement *insn : stmts) {
350  if (!insn->isPhi()) {
351  continue; // Might be able to optimise this a bit
352  }
353 
354  PhiAssign *pi = static_cast<PhiAssign *>(insn);
355  // See if the destination has a symbol already
356  SharedExp lhs = pi->getLeft();
357  auto lhsRef = RefExp::get(lhs, pi);
358 
359  if (proc->findFirstSymbol(lhsRef) != nullptr) {
360  continue; // Already mapped to something
361  }
362 
363  bool multiple = false; // True if find more than one unique parameter
364  QString firstName; // The name for the first parameter found
365  SharedType ty = pi->getType();
366 
367  for (const std::shared_ptr<RefExp> &v : *pi) {
368  if (v->getDef()->isImplicit()) {
369  QString name = proc->lookupSym(RefExp::get(v->getSubExp1(), v->getDef()), ty);
370 
371  if (!name.isEmpty()) {
372  if (!firstName.isEmpty() && (firstName != name)) {
373  multiple = true;
374  break;
375  }
376 
377  firstName = name; // Remember this candidate
378  }
379  }
380  }
381 
382  if (multiple || firstName.isEmpty()) {
383  continue;
384  }
385 
386  proc->mapSymbolTo(lhsRef, Location::param(firstName, proc));
387  }
388 }

391 void FromSSAFormPass::mapParameters(UserProc *proc)
392 {
393  // Replace the parameters with their mappings
394  StatementList::iterator pp;
395 
396  for (pp = proc->getParameters().begin(); pp != proc->getParameters().end(); ++pp) {
397  SharedExp lhs = static_cast<Assignment *>(*pp)->getLeft();
398  QString mappedName = proc->lookupParam(lhs);
399 
400  if (mappedName.isEmpty()) {
401  LOG_WARN("No symbol mapping for parameter %1", lhs);
402  bool allZero;
403  SharedExp clean = lhs->clone()->removeSubscripts(allZero);
404 
405  if (allZero) {
406  static_cast<Assignment *>(*pp)->setLeft(clean);
407  }
408 
409  // Else leave them alone
410  }
411  else {
412  static_cast<Assignment *>(*pp)->setLeft(Location::param(mappedName, proc));
413  }
414  }
415 }

418 void FromSSAFormPass::removeSubscriptsFromSymbols(UserProc *proc)
419 {
420  // Basically, use the symbol map to map the symbols in the symbol map!
421  // However, do not remove subscripts from the outer level; they are still needed for comments in
422  // the output and also for when removing subscripts from parameters (still need the {0}) Since
423  // this will potentially change the ordering of entries, need to copy the map
424  UserProc::SymbolMap sm2 = proc->getSymbolMap(); // Object copy
425 
426  proc->getSymbolMap().clear();
427  ExpSSAXformer esx(proc);
428 
429  for (const auto &[first, second] : sm2) {
430  SharedExp from = std::const_pointer_cast<Exp>(first);
431 
432  if (from->isSubscript()) {
433  // As noted above, don't touch the outer level of subscripts
434  SharedExp &sub = from->refSubExp1();
435  sub = sub->acceptModifier(&esx);
436  }
437  else {
438  from = from->acceptModifier(&esx);
439  }
440 
441  proc->mapSymbolTo(from, second);
442  }
443 }

458 void FromSSAFormPass::findPhiUnites(UserProc *proc, ConnectionGraph &pu)
459 {
460  StatementList stmts;
461  proc->getStatements(stmts);
462 
463  for (Statement *stmt : stmts) {
464  if (!stmt->isPhi()) {
465  continue;
466  }
467 
468  PhiAssign *pa = static_cast<PhiAssign *>(stmt);
469  SharedExp lhs = pa->getLeft();
470  auto reLhs = RefExp::get(lhs, pa);
471 
472  for (const std::shared_ptr<RefExp> &v : *pa) {
473  assert(v->getSubExp1());
474  auto re = RefExp::get(v->getSubExp1(), v->getDef());
475  pu.connect(reLhs, re);
476  }
477  }

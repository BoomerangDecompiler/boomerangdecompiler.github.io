71 bool ST20Decoder::decodeInstruction(Address pc, ptrdiff_t delta, DecodeResult &result)
72 {
73  int total = 0; // Total value from all prefixes
74 
75  result.reset();
76  result.rtl = std::make_unique<RTL>(pc);
77 
78  while (true) {
79  const Byte instructionData = Util::readByte(
80  (const void *)(pc.value() + delta + result.numBytes));
81  const Byte functionCode = (instructionData >> 4) & 0xF;
82  const Byte oper = instructionData & 0xF;
83 
84  result.numBytes++;
85 
86  switch (functionCode) {
87  case 0: { // unconditional jump
88  total += oper;
89  const Address jumpDest = pc + result.numBytes + total;
90 
91  GotoStatement *jump = new GotoStatement();
92  jump->setDest(jumpDest);
93  result.rtl->append(jump);
94  } break;
95 
96  case 1:
97  case 3:
98  case 4:
99  case 5:
100  case 7:
101  case 8:
102  case 11:
103  case 12:
104  case 13:
105  case 14: {
106  total += oper;
107  result.rtl = instantiate(pc, functionNames[functionCode], { Const::get(total) });
108  } break;
109 
110  case 2: { // prefix
111  total = (total + oper) << 4;
112  continue;
113  }
114  case 6: { // negative prefix
115  total = (total + ~oper) << 4;
116  continue;
117  }
118 
119  case 9: { // call
120  total += oper;
121  const Address callDest = Address(pc + result.numBytes + total);
122  result.rtl = instantiate(pc, "call", { Const::get(callDest) });
123 
124  CallStatement *newCall = new CallStatement;
125  newCall->setIsComputed(false);
126  newCall->setDest(callDest);
127 
128  if (m_prog) {
129  Function *callee = m_prog->getOrCreateFunction(callDest);
130  if (callee && callee != reinterpret_cast<Function *>(-1)) {
131  newCall->setDestProc(callee);
132  }
133  }
134 
135  result.rtl->append(newCall);
136  } break;
137 
138  case 10: { // cond jump
139  total += oper;
140  BranchStatement *br = new BranchStatement();
141  br->setDest(pc + result.numBytes + total);
142  br->setCondExpr(Binary::get(opEquals, Location::regOf(REG_ST20_A), Const::get(0)));
143 
144  result.rtl->append(br);
145  } break;
146 
147  case 15: { // operate
148  total += oper;
149  const char *insnName = getInstructionName(total);
150  if (!insnName) {
151  // invalid or unknown instruction
152  result.valid = false;
153  return false;
154  }
155 
156  result.rtl = instantiate(pc, insnName);
157 
158  const bool isRet = strcmp(insnName, "ret") == 0 || strcmp(insnName, "iret") == 0 ||
159  strcmp(insnName, "tret") == 0;
160  if (isRet) {
161  result.rtl->append(new ReturnStatement);
162  }
163  else if (strcmp(insnName, "gcall") == 0) {
164  CallStatement *call = new CallStatement;
165  call->setDest(Location::tempOf(Const::get(QString("tmp"))));
166  call->setIsComputed(true);
167  result.rtl->append(call);
168  }
169  } break;
170 
171  default: assert(false);
172  }
173 
174  break;
175  }
176 
177  return result.valid;
178 }

365 std::unique_ptr<RTL> ST20Decoder::instantiate(Address pc, const char *name,
366  const std::initializer_list<SharedExp> &args)
367 {
368  // Take the argument, convert it to upper case and remove any .'s
369  const QString sanitizedName = QString(name).remove(".").toUpper();
370 
371  // Put the operands into a vector
372  std::vector<SharedExp> actuals(args);
373 
374  if (m_prog && m_prog->getProject()->getSettings()->debugDecoder) {
375  OStream q_cout(stdout);
376  // Display a disassembly of this instruction if requested
377  q_cout << pc << ": " << name << " ";
378 
379  for (const SharedExp &itd : actuals) {
380  if (itd->isIntConst()) {
381  int val = itd->access<Const>()->getInt();
382 
383  if ((val > 100) || (val < -100)) {
384  q_cout << "0x" << QString::number(val, 16);
385  }
386  else {
387  q_cout << val;
388  }
389  }
390  else {
391  itd->print(q_cout);
392  }
393 
394  q_cout << " ";
395  }
396 
397  q_cout << '\n';
398  }
399 
400  std::unique_ptr<RTL> rtl = m_rtlDict.instantiateRTL(sanitizedName, pc, actuals);
401  if (!rtl) {
402  LOG_ERROR("Cannot find semantics for instruction '%1' at address %2, "
403  "treating instruction as NOP",
404  name, pc);
405  return m_rtlDict.instantiateRTL("NOP", pc, {});
406  }
407 
408  return rtl;
409 }

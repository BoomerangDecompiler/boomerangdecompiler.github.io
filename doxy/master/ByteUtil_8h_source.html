90 /**
91  * Sign-extend \p src into \a TgtType.
92  * Example:
93  * signExtend<int>((unsigned char)0xFF) == -1
94  *
95  * \param src number to sign-extend
96  * \param numSrcBits Number of Bits in the source type
97  * (Mainly to counter int-promotion in (blabla & 0xFF))
98  */
99 template<typename TgtType = int, typename SrcType>
100 TgtType signExtend(const SrcType &src, std::size_t numSrcBits = 8 * sizeof(SrcType))
101 {
102  static_assert(std::is_integral<SrcType>::value, "Source type must be an integer!");
103  static_assert(std::is_integral<TgtType>::value && std::is_signed<TgtType>::value,
104  "Target type must be a signed integer!");
105 
106  // size difference, in bits
107  const int sizeDifference = 8 * sizeof(TgtType) - numSrcBits;
108  return (static_cast<TgtType>(static_cast<TgtType>(src) << sizeDifference)) >> sizeDifference;
109 }

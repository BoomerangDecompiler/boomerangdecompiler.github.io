28 bool StringInstructionProcessor::processStringInstructions()
29 {
30  std::list<std::pair<RTL *, BasicBlock *>> stringInstructions;
31 
32  for (BasicBlock *bb : *m_proc->getCFG()) {
33  RTLList *bbRTLs = bb->getRTLs();
34 
35  if (bbRTLs == nullptr) {
36  continue;
37  }
38 
39  Address prev, addr = Address::ZERO;
40 
41  for (auto &rtl : *bbRTLs) {
42  prev = addr;
43  addr = rtl->getAddress();
44 
45  if (!rtl->empty()) {
46  Statement *firstStmt = rtl->front();
47  if (firstStmt->isAssign()) {
48  SharedExp lhs = static_cast<Assign *>(firstStmt)->getLeft();
49 
50  if (lhs->isMachFtr()) {
51  QString str = lhs->access<Const, 1>()->getStr();
52 
53  if (str.startsWith("%SKIP")) {
54  stringInstructions.push_back({ rtl.get(), bb });
55 
56  // Assume there is only 1 string instruction per BB
57  // This might not be true, but can be migitated
58  // by calling processStringInstructions multiple times
59  // to catch all string instructions.
60  break;
61  }
62  }
63  }
64  }
65  }
66  }
67 
68  for (auto p : stringInstructions) {
69  RTL *skipRTL = p.first;
70  BasicBlock *bb = p.second;
71 
72  BranchStatement *skipBranch = new BranchStatement;
73 
74  assert(skipRTL->size() >= 4); // They vary; at least 5 or 6
75 
76  Statement *s1 = *skipRTL->begin();
77  Statement *s6 = *(--skipRTL->end());
78  if (s1->isAssign()) {
79  skipBranch->setCondExpr(static_cast<Assign *>(s1)->getRight());
80  }
81  else {
82  skipBranch->setCondExpr(nullptr);
83  }
84  skipBranch->setDest(skipRTL->getAddress() + 2);
85 
86  BranchStatement *rptBranch = new BranchStatement;
87  if (s6->isAssign()) {
88  rptBranch->setCondExpr(static_cast<Assign *>(s6)->getRight());
89  }
90  else {
91  rptBranch->setCondExpr(nullptr);
92  }
93  rptBranch->setDest(skipRTL->getAddress());
94 
95  splitForBranch(bb, skipRTL, skipBranch, rptBranch);
96  }
97 
98  return !stringInstructions.empty();
99 }

102 BasicBlock *StringInstructionProcessor::splitForBranch(BasicBlock *bb, RTL *stringRTL,
103  BranchStatement *skipBranch,
104  BranchStatement *rptBranch)
105 {
106  Address stringAddr = stringRTL->getAddress();
107  RTLList::iterator stringIt = std::find_if(
108  bb->getRTLs()->begin(), bb->getRTLs()->end(),
109  [stringRTL](const std::unique_ptr<RTL> &ptr) { return stringRTL == ptr.get(); });
110 
111  assert(stringIt != bb->getRTLs()->end());
112 
113  const bool haveA = (stringIt != bb->getRTLs()->begin());
114  const bool haveB = (std::next(stringIt) != bb->getRTLs()->end());
115  BasicBlock *aBB = nullptr;
116  BasicBlock *bBB = nullptr;
117 
118  const std::vector<BasicBlock *> oldPredecessors = bb->getPredecessors();
119  const std::vector<BasicBlock *> oldSuccessors = bb->getSuccessors();
120 
121  if (haveA) {
122  aBB = bb;
123  bb = m_proc->getCFG()->splitBB(aBB, stringAddr);
124  assert(aBB->getLowAddr() < bb->getLowAddr());
125  }
126  stringIt = bb->getRTLs()->begin();
127  if (haveB) {
128  Address splitAddr = (*std::next(stringIt))->getAddress();
129  bBB = m_proc->getCFG()->splitBB(bb, splitAddr);
130  assert(bb->getLowAddr() < bBB->getLowAddr());
131  }
132  else {
133  // this means the original BB has a fallthrough branch to its successor.
134  // Just pretend the successor is the split off B bb.
135  bBB = bb->getSuccessor(0);
136  }
137 
138  assert(bb->getRTLs()->size() == 1); // only the string instruction
139  assert(bb->getRTLs()->front()->getAddress() == stringAddr);
140 
141  // Make an RTL for the skip and the rpt branch instructions.
142  std::unique_ptr<RTLList> skipBBRTLs(new RTLList);
143  std::unique_ptr<RTLList> rptBBRTLs(new RTLList);
144  skipBBRTLs->push_back(std::unique_ptr<RTL>(new RTL(stringAddr, { skipBranch })));
145  rptBBRTLs->push_back(std::unique_ptr<RTL>(new RTL(**stringIt)));
146 
147  rptBBRTLs->front()->setAddress(stringAddr + 1);
148  rptBBRTLs->front()->pop_front();
149  rptBBRTLs->front()->back() = rptBranch;
150 
151  // remove the original string instruction from the CFG.
152  bb->removeAllPredecessors();
153 
154  // remove connection between the string instruction and the B part
155  for (BasicBlock *succ : oldSuccessors) {
156  bb->removeSuccessor(succ);
157  succ->removePredecessor(bb);
158  }
159 
160  const bool entryBBNeedsUpdate = !haveA && bb == m_proc->getCFG()->getEntryBB();
161  m_proc->getCFG()->removeBB(bb);
162 
163  BasicBlock *skipBB = m_proc->getCFG()->createBB(BBType::Twoway, std::move(skipBBRTLs));
164  BasicBlock *rptBB = m_proc->getCFG()->createBB(BBType::Twoway, std::move(rptBBRTLs));
165 
166  assert(skipBB && rptBB);
167 
168  if (haveA) {
169  aBB->removeAllSuccessors();
170  aBB->setType(BBType::Fall);
171  m_proc->getCFG()->addEdge(aBB, skipBB);
172  }
173  else {
174  for (BasicBlock *pred : oldPredecessors) {
175  for (int i = 0; i < pred->getNumSuccessors(); i++) {
176  if (pred->getSuccessor(i) == bb) {
177  pred->setSuccessor(i, skipBB);
178  skipBB->addPredecessor(pred);
179  }
180  }
181  }
182  }
183 
184  bBB->removePredecessor(bb);
185  m_proc->getCFG()->addEdge(skipBB, bBB);
186  m_proc->getCFG()->addEdge(skipBB, rptBB);
187  m_proc->getCFG()->addEdge(rptBB, bBB);
188  m_proc->getCFG()->addEdge(rptBB, rptBB);
189 
190  if (entryBBNeedsUpdate) {
191  m_proc->getCFG()->setEntryAndExitBB(skipBB);
192  }
193 
194  return haveB ? bBB : rptBB;
195 }

30 bool DataIntervalMap::isClear(Address addr, unsigned size) const
31 {
32  VariableMap::const_iterator it1, it2;
33  std::tie(it1, it2) = m_varMap.equalRange(addr, addr + size);
34 
35  if ((it1 == m_varMap.end()) && (it2 == m_varMap.end())) {
36  return true; // no variable blocking
37  }
38  else if (it2 == m_varMap.end()) {
39  return false;
40  }
41 
42  for (VariableMap::const_iterator it = it1; it != it2; ++it) {
43  // since we don't know the length of unbounded arrays (yet),
44  // it is possible that the array ends before \p addr.
45  // Therefore, allow these intervals to be marked as "clear"
46  if (!it->second.type->isArray() || !it->second.type->as<ArrayType>()->isUnbounded()) {
47  return false;
48  }
49  }
50 
51  return true; // only unbounded arrays "blocking"
52 }

82 DataIntervalMap::iterator DataIntervalMap::insertItem(Address baseAddr, QString name,
83  SharedType type, bool forced)
84 {
85  if (name.isEmpty()) {
86  name = "<noname>";
87  }
88 
89  const Interval<Address> newTypeRange(baseAddr, baseAddr + type->getSize() / 8);
90 
91  iterator it1, it2;
92  std::tie(it1, it2) = m_varMap.equalRange(newTypeRange);
93 
94  if ((it1 == end()) && (it2 == end())) {
95  // not overlapped by any variable -> just insert directly
96  return m_varMap.insert(newTypeRange, TypedVariable(baseAddr, name, type));
97  }
98 
99  // try to adjust types such that the type can be inserted
100  while (it1 != it2) {
101  TypedVariable &var = it1->second;
102 
103  if (it1->first.containsInterval(newTypeRange)) {
104  if (newTypeRange.containsInterval(it1->first)) {
105  // both types are of equal size
106  bool changed;
107  it1->second.type = it1->second.type->meetWith(type, changed);
108  return it1;
109  }
110  else {
111  // new type may be part of an existing type
112  insertComponentType(&var, baseAddr, name, type, forced);
113  return it1;
114  }
115  }
116  else if (newTypeRange.containsInterval(it1->first)) {
117  // the new type is a larger/derived type which contains the old type
118  return replaceComponents(baseAddr, name, type, forced);
119  }
120  else {
121  // the new and the old type do not completely overlap
122  // -> fail or insert anyway if forced
123  if (forced) {
124  clearRange(newTypeRange);
125  assert(isClear(newTypeRange));
126  return m_varMap.insert(newTypeRange, TypedVariable(baseAddr, name, type));
127  }
128  else {
129  LOG_ERROR("TYPE ERROR: Cannot insert variable of type %1 at address %2",
130  type->getCtype(), baseAddr);
131  LOG_ERROR("TYPE ERROR: because it conflicts with variable of type %1 at address %2",
132  var.type->getCtype(), var.baseAddr);
133  return end();
134  }
135  }
136  }
137 
138  return end();
139 }

142 void DataIntervalMap::insertComponentType(TypedVariable *existingVar, Address addr,
143  const QString & /*name*/, SharedType type,
144  bool /*forced*/)
145 {
146  assert(existingVar);
147  assert(existingVar->baseAddr <= addr);
148 
149  if (existingVar->type->resolvesToCompound()) {
150  const uint64 bitOffset = (addr - existingVar->baseAddr).value() * 8;
151  SharedType memberType = existingVar->type->as<CompoundType>()->getMemberTypeByOffset(
152  bitOffset);
153 
154  if (!memberType || !memberType->isCompatibleWith(*type)) {
155  LOG_ERROR("TYPE ERROR: At address %1 type %2 is not compatible with existing structure "
156  "type %3",
157  addr, type->getCtype(), existingVar->type->getCtype());
158  }
159 
160  // types are compatible
161  bool ch;
162  memberType = memberType->meetWith(type, ch);
163  existingVar->type->as<CompoundType>()->setMemberTypeByOffset(bitOffset, memberType);
164  }
165  else if (existingVar->type->resolvesToArray()) {
166  SharedType baseType = existingVar->type->as<ArrayType>()->getBaseType();
167  assert(baseType);
168 
169  if (!baseType->isCompatibleWith(*type)) {
170  LOG_ERROR("TYPE ERROR: At address %1 type %2 is not compatible with existing array "
171  "member type %3",
172  addr, type->getCtype(), baseType->getCtype());
173  return;
174  }
175 
176  // types are compatible -> change it
177  bool ch;
178  existingVar->type->as<ArrayType>()->setBaseType(baseType->meetWith(type, ch));
179  }
180  else {
181  LOG_ERROR("TYPE ERROR: Existing type at address %1 is not structure or array type",
182  existingVar->baseAddr);
183  }
184 }

187 DataIntervalMap::iterator DataIntervalMap::replaceComponents(Address addr, const QString &name,
188  SharedType ty, bool /*forced*/)
189 {
190  // This is the byte address just past the type to be inserted
191  const Address endAddr = addr + ty->getSize() / 8;
192 
193  VariableMap::const_iterator it1, it2;
194 
195  // First check that the new entry will be compatible with everything it will overlap
196  if (ty->resolvesToCompound()) {
197  std::tie(it1, it2) = m_varMap.equalRange(addr, endAddr);
198 
199  for (VariableMap::const_iterator it = it1; it != it2; ++it) {
200  unsigned bitOffset = (it->second.baseAddr - addr).value() * 8;
201 
202  SharedType memberType = ty->as<CompoundType>()->getMemberTypeByOffset(bitOffset);
203 
204  if (!memberType->isCompatibleWith(*it->second.type, true)) {
205  LOG_ERROR("TYPE ERROR: At address %1 struct type %2 is not compatible with "
206  "existing type %3",
207  addr, ty->getCtype(), it->second.type->getCtype());
208  return end();
209  }
210 
211  LOG_VERBOSE("%1", this->toString());
212  LOG_VERBOSE("%1 %2", memberType->getCtype(), it->second.type->getCtype());
213 
214  bool ch;
215  memberType = it->second.type->meetWith(memberType, ch);
216  ty->as<CompoundType>()->setMemberTypeByOffset(bitOffset, memberType);
217  }
218  }
219  else if (ty->resolvesToArray()) {
220  SharedType memberType = ty->as<ArrayType>()->getBaseType();
221  std::tie(it1, it2) = m_varMap.equalRange(addr, endAddr);
222 
223  for (VariableMap::const_iterator it = it1; it != it2; ++it) {
224  if (memberType->isCompatibleWith(*it->second.type, true)) {
225  bool ch;
226  memberType = memberType->meetWith(it->second.type, ch);
227  ty->as<ArrayType>()->setBaseType(memberType);
228  }
229  else {
230  LOG_ERROR("TYPE ERROR: At address %1 array type %2 is not compatible with existing "
231  "type %3",
232  addr, ty->getCtype(), it->second.type->getCtype());
233  return end();
234  }
235  }
236  }
237  else {
238  // Just make sure it doesn't overlap anything
239  if (!isClear(addr, (ty->getSize() + 7) / 8)) {
240  LOG_ERROR("TYPE ERROR: at address %1, overlapping type %2 "
241  "does not resolve to compound or array",
242  addr, ty->getCtype());
243  return end();
244  }
245  }
246 
247  // The compound or array type is compatible. Remove the items that it will overlap with
248  std::tie(it1, it2) = m_varMap.equalRange(addr, endAddr);
249 
250  // Check for existing locals that need to be updated
251  if (ty->resolvesToCompound() || ty->resolvesToArray()) {
252  SharedExp rsp = Location::regOf(m_proc->getSignature()->getStackRegister());
253  auto rsp0 = RefExp::get(rsp, m_proc->getCFG()->findTheImplicitAssign(rsp)); // sp{0}
254 
255  for (VariableMap::const_iterator it = it1; it != it2; ++it) {
256  // Check if there is an existing local here
257  SharedExp locl = Location::memOf(
258  Binary::get(opPlus, rsp0->clone(), Const::get(it->second.baseAddr.native())));
259  locl->simplifyArith(); // Convert m[sp{0} + -4] to m[sp{0} - 4]
260  SharedType elemTy;
261  int bitOffset = (it->second.baseAddr - addr).value() / 8;
262 
263  if (ty->resolvesToCompound()) {
264  elemTy = ty->as<CompoundType>()->getMemberTypeByOffset(bitOffset);
265  }
266  else {
267  elemTy = ty->as<ArrayType>()->getBaseType();
268  }
269 
270  QString locName = m_proc->findLocal(locl, elemTy);
271 
272  if (!locName.isEmpty() && ty->resolvesToCompound()) {
273  auto c = ty->as<CompoundType>();
274  // want s.m where s is the new compound object and m is the member at offset
275  // bitOffset
276  QString memName = c->getMemberNameByOffset(bitOffset);
277  SharedExp s = Location::memOf(Binary::get(opPlus, rsp0->clone(), Const::get(addr)));
278  s->simplifyArith();
279  SharedExp memberExp = Binary::get(opMemberAccess, s, Const::get(memName));
280  m_proc->mapSymbolTo(locl, memberExp);
281  }
282  else {
283  // FIXME: to be completed
284  }
285  }
286  }
287 
288  m_varMap.eraseAll(Interval<Address>(addr, addr + ty->getSizeInBytes()));
289 
290  return m_varMap.insert(addr, addr + ty->getSizeInBytes(), TypedVariable(addr, name, ty));
291 }

331 void DataIntervalMap::clearRange(const Interval<Address> &interval)
332 {
333  iterator it, it2;
334 
335  std::tie(it, it2) = m_varMap.equalRange(interval);
336 
337  while (it != it2) {
338  TypedVariable &var = it->second;
339  const Interval<Address> typeRange(var.baseAddr, var.baseAddr + 8 * var.size);
340 
341  if (var.type->resolvesToArray() && (var.baseAddr < interval.lower()) &&
342  var.type->as<ArrayType>()->isUnbounded()) {
343  // unbounded array -> just adjust the range of the array
344  // to not overlap with the interval
345  uint64 newSize = interval.lower().value() - var.baseAddr.value();
346 
347  // we have a whole number as number of elements;
348  newSize %= var.type->as<ArrayType>()->getBaseType()->getSize();
349 
350  LOG_VERBOSE("Adjusting size of unbounded array at address %1 to %2 bytes", var.baseAddr,
351  newSize / 8);
352  var.size = newSize / 8;
353  var.type->as<ArrayType>()->setLength(
354  8 * var.size / var.type->as<ArrayType>()->getBaseType()->getSize());
355 
356  it = std::next(it);
357  }
358  else {
359  // type is in the way -> just delete it
360  it = m_varMap.erase(it);
361  }
362  }

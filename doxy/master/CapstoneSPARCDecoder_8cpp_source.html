42 /**
43  * Translates Capstone register IDs to Boomerang internal register IDs.
44  * \returns RegNumSpecial if register not found.
45  */
46 RegNum CapstoneSPARCDecoder::fixRegNum(const cs::cs_insn *insn, int opIdx) const
47 {
48  assert(insn != nullptr);
49  assert(opIdx < insn->detail->sparc.op_count);
50  assert(insn->detail->sparc.operands[opIdx].type == cs::SPARC_OP_REG);
51 
52  const int csRegID = insn->detail->sparc.operands[opIdx].reg;
53 
54  if (csRegID >= cs::SPARC_REG_F0 && csRegID <= cs::SPARC_REG_F31) {
55  if (getRegOperandSize(insn, opIdx) == 64) {
56  return REG_SPARC_F0TO1 + (csRegID - cs::SPARC_REG_F0) / 2;
57  }
58  else if (getRegOperandSize(insn, opIdx) == 128) {
59  return REG_SPARC_F0TO3 + (csRegID - cs::SPARC_REG_F0) / 4;
60  }
61  else { // single float
62  return REG_SPARC_F0 + (csRegID - cs::SPARC_REG_F0);
63  }
64  }
65 
66  return fixRegNum(csRegID);
67 }

70 RegNum CapstoneSPARCDecoder::fixRegNum(int csRegID) const
71 {
72  if (csRegID >= cs::SPARC_REG_G0 && csRegID <= cs::SPARC_REG_G7) {
73  return REG_SPARC_G0 + (csRegID - cs::SPARC_REG_G0);
74  }
75  // Workaround for bug in Capstone (o0..o7 are not numbered sequentially).
76  // o6 and o7 are handled by oldRegMap
77  else if (csRegID >= cs::SPARC_REG_O0 && csRegID <= cs::SPARC_REG_O5) {
78  return REG_SPARC_O0 + (csRegID - cs::SPARC_REG_O0);
79  }
80  else if (csRegID >= cs::SPARC_REG_I0 && csRegID <= cs::SPARC_REG_I7) {
81  return REG_SPARC_I0 + (csRegID - cs::SPARC_REG_I0);
82  }
83  else if (csRegID >= cs::SPARC_REG_L0 && csRegID <= cs::SPARC_REG_L7) {
84  return REG_SPARC_L0 + (csRegID - cs::SPARC_REG_L0);
85  }
86  else if (csRegID >= cs::SPARC_REG_F0 && csRegID <= cs::SPARC_REG_F31) {
87  return REG_SPARC_F0 + (csRegID - cs::SPARC_REG_F0);
88  }
89  else if (csRegID >= cs::SPARC_REG_F32 && csRegID <= cs::SPARC_REG_F62) {
90  return REG_SPARC_F0TO1 + (csRegID - cs::SPARC_REG_F32);
91  }
92 
93  auto it = oldRegMap.find((cs::sparc_reg)csRegID);
94  return (it != oldRegMap.end()) ? it->second : RegNumSpecial;
95 }

99 {
100  assert(insn != nullptr);
101  assert(opIdx < insn->detail->sparc.op_count);
102  assert(insn->detail->sparc.operands[opIdx].type == cs::SPARC_OP_REG);

131 bool CapstoneSPARCDecoder::decodeInstruction(Address pc, ptrdiff_t delta, DecodeResult &result)
132 {
133  const Byte *instructionData = reinterpret_cast<const Byte *>((HostAddress(delta) + pc).value());
134  const Byte *oldInstructionData = instructionData;
135 
136  cs::cs_detail insnDetail;
137  cs::cs_insn decodedInstruction;
138  decodedInstruction.detail = &insnDetail;
139 
140  size_t bufsize = SPARC_INSTRUCTION_LENGTH;
141  uint64_t addr = pc.value();
142  result.valid = cs::cs_disasm_iter(m_handle, &instructionData, &bufsize, &addr,
143  &decodedInstruction);
144 
145  if (!result.valid) {
146  // HACK: Capstone does not support ldd and std for gpr destinations,
147  // so we have to test for it manually.
148  const uint32_t insn = Util::readDWord(oldInstructionData, Endian::Big);
149 
150  result.valid = decodeLDD(&decodedInstruction, insn);
151  if (!result.valid) {
152  result.valid = decodeSTD(&decodedInstruction, insn);
153  if (!result.valid) {
154  return false;
155  }
156  }
157 
158  decodedInstruction.address = pc.value();
159  }
160 
161  result.type = getInstructionType(&decodedInstruction);
162  result.numBytes = SPARC_INSTRUCTION_LENGTH;
163  result.reDecode = false;
164  result.rtl = createRTLForInstruction(pc, &decodedInstruction);
165  result.forceOutEdge = Address::ZERO;
166  result.valid = (result.rtl != nullptr);
167 
168  if (result.rtl->empty()) {
169  // Force empty unrecognized instructions to have NOP type instead of NCT
170  result.type = NOP;
171  }
172 
173  return true;
174 }

189 bool CapstoneSPARCDecoder::isSPARCRestore(Address pc, ptrdiff_t delta) const
190 {
191  const Byte *instructionData = reinterpret_cast<const Byte *>((HostAddress(delta) + pc).value());
192 
193  cs::cs_insn *decodedInstruction;
194  size_t numInstructions = cs_disasm(m_handle, instructionData, SPARC_INSTRUCTION_LENGTH,
195  pc.value(), 1, &decodedInstruction);
196 
197  if (numInstructions < 1) {
198  return false;
199  }
200 
201  return decodedInstruction->id == cs::SPARC_INS_RESTORE;

205 SharedExp CapstoneSPARCDecoder::operandToExp(const cs::cs_insn *instruction, int opIdx) const
206 {
207  const cs::cs_sparc_op &operand = instruction->detail->sparc.operands[opIdx];
208  switch (operand.type) {
209  case cs::SPARC_OP_IMM: {
210  return Const::get(Address(operand.imm));
211  }
212  case cs::SPARC_OP_REG: {
213  return getRegExp(instruction, opIdx);
214  }
215  case cs::SPARC_OP_MEM: {
216  SharedExp memExp = getRegExp(operand.mem.base);
217 
218  if (operand.mem.index != cs::SPARC_REG_INVALID) {
219  memExp = Binary::get(opPlus, memExp, getRegExp(operand.mem.index));
220  }
221 
222  memExp = Binary::get(opPlus, memExp, Const::get(operand.mem.disp));
223  return Location::memOf(memExp)->simplifyArith();
224  }
225  default: LOG_ERROR("Unknown sparc instruction operand type %1", (int)operand.type); break;
226  }
227 
228  return nullptr;
229 }

232 std::unique_ptr<RTL> CapstoneSPARCDecoder::createRTLForInstruction(Address pc,
233  cs::cs_insn *instruction)
234 {
235  const int numOperands = instruction->detail->sparc.op_count;
236  cs::cs_sparc_op *operands = instruction->detail->sparc.operands;
237 
238  QString insnID = instruction->mnemonic;
239 
240  // chop off branch prediction hints
241  if (insnID.endsWith(",pn") || insnID.endsWith(",pt")) {
242  insnID.chop(3);
243  }
244 
245  insnID = insnID.remove(',').toUpper();
246 
247  if (instruction->id == cs::SPARC_INS_LDD) {
248  const bool isFloatReg = instruction->detail->sparc.operands[1].reg >= cs::SPARC_REG_F0 &&
249  instruction->detail->sparc.operands[1].reg <= cs::SPARC_REG_F62;
250 
251  if (isFloatReg) {
252  insnID = "LDDF";
253  }
254  }
255  else if (instruction->id == cs::SPARC_INS_STD) {
256  const bool isFloatReg = instruction->detail->sparc.operands[0].reg >= cs::SPARC_REG_F0 &&
257  instruction->detail->sparc.operands[0].reg <= cs::SPARC_REG_F62;
258 
259  if (isFloatReg) {
260  insnID = "STDF";
261  }
262  }
263 
264  std::unique_ptr<RTL> rtl = instantiateRTL(pc, qPrintable(insnID), instruction);
265 
266  if (rtl == nullptr) {
267  LOG_ERROR("Cannot find semantics for instruction '%1' at address %2, "
268  "treating instruction as NOP",
269  insnID, pc);
270  return std::make_unique<RTL>(pc);
271  }
272 
273  if (insnID == "BA" || insnID == "BAA" || insnID == "BN" || insnID == "BNA") {
274  rtl->clear();
275  rtl->append(new GotoStatement(Address(operands[0].imm)));
276  }
277  else if (insnID == "FBA" || insnID == "FBAA" || insnID == "FBN" || insnID == "FBNA") {
278  rtl->clear();
279  rtl->append(new GotoStatement(Address(operands[0].imm)));
280  }
281  else if (instruction->id == cs::SPARC_INS_B) {
282  rtl->clear();
283  BranchStatement *branch = new BranchStatement;
284  branch->setDest(Address(operands[numOperands - 1].imm));
285  branch->setIsComputed(false);
286 
287  BranchType bt = BranchType::INVALID;
288 
289  switch (instruction->detail->sparc.cc) {
290  case cs::SPARC_CC_ICC_NE: bt = BranchType::JNE; break;
291  case cs::SPARC_CC_ICC_E: bt = BranchType::JE; break;
292  case cs::SPARC_CC_ICC_G: bt = BranchType::JSG; break;
293  case cs::SPARC_CC_ICC_LE: bt = BranchType::JSLE; break;
294  case cs::SPARC_CC_ICC_GE: bt = BranchType::JSGE; break;
295  case cs::SPARC_CC_ICC_L: bt = BranchType::JSL; break;
296  case cs::SPARC_CC_ICC_GU: bt = BranchType::JUG; break;
297  case cs::SPARC_CC_ICC_LEU: bt = BranchType::JULE; break;
298  case cs::SPARC_CC_ICC_CC: bt = BranchType::JUGE; break;
299  case cs::SPARC_CC_ICC_CS: bt = BranchType::JUL; break;
300  case cs::SPARC_CC_ICC_POS: bt = BranchType::JPOS; break;
301  case cs::SPARC_CC_ICC_NEG: bt = BranchType::JMI; break;
302  default: break;
303  }
304 
305  branch->setCondType(bt);
306  rtl->append(branch);
307  }
308  else if (instruction->id == cs::SPARC_INS_FB) {
309  rtl->clear();
310  BranchStatement *branch = new BranchStatement;
311  branch->setDest(Address(operands[0].imm));
312  branch->setIsComputed(false);
313 
314  BranchType bt = BranchType::INVALID;
315 
316  switch (instruction->detail->sparc.cc) {
317  case cs::SPARC_CC_FCC_NE: bt = BranchType::JNE; break;
318  case cs::SPARC_CC_FCC_E: bt = BranchType::JE; break;
319  case cs::SPARC_CC_FCC_G: bt = BranchType::JSG; break;
320  case cs::SPARC_CC_FCC_LE: bt = BranchType::JSLE; break;
321  case cs::SPARC_CC_FCC_GE: bt = BranchType::JSGE; break;
322  case cs::SPARC_CC_FCC_L: bt = BranchType::JSL; break;
323  case cs::SPARC_CC_FCC_UG: bt = BranchType::JSG; break;
324  case cs::SPARC_CC_FCC_UL: bt = BranchType::JSL; break;
325  case cs::SPARC_CC_FCC_LG: bt = BranchType::JNE; break;
326  case cs::SPARC_CC_FCC_UE: bt = BranchType::JE; break;
327  case cs::SPARC_CC_FCC_UGE: bt = BranchType::JSGE; break;
328  case cs::SPARC_CC_FCC_ULE: bt = BranchType::JSLE; break;
329  default: break;
330  }
331 
332  branch->setCondType(bt, true);
333  rtl->append(branch);
334  }
335  else if (instruction->id == cs::SPARC_INS_CALL) {
336  rtl->clear();
337  CallStatement *call = new CallStatement;
338  if (operands[0].type == cs::SPARC_OP_IMM) {
339  const Address callDest = Address(operands[0].imm);
340 
341  call->setIsComputed(false);
342  call->setDest(callDest);
343 
344  if (m_prog) {
345  Function *destProc = m_prog->getOrCreateFunction(callDest);
346 
347  if (destProc == reinterpret_cast<Function *>(-1)) {
348  destProc = nullptr;
349  }
350 
351  call->setDestProc(destProc);
352  }
353  }
354  else { // mem / reg
355  SharedExp callDest = Unary::get(opAddrOf, operandToExp(instruction, 0))->simplify();
356  if (callDest->isConst()) {
357  call->setIsComputed(false);
358  call->setDest(callDest->access<Const>()->getAddr());
359 
360  if (m_prog) {
361  Function *destProc = m_prog->getOrCreateFunction(
362  callDest->access<Const>()->getAddr());
363 
364  if (destProc == reinterpret_cast<Function *>(-1)) {
365  destProc = nullptr;
366  }
367 
368  call->setDestProc(destProc);
369  }
370  }
371  else {
372  call->setIsComputed(true);
373  call->setDest(callDest);
374  }
375  }
376 
377  rtl->append(call);
378  }
379  else if (instruction->id == cs::SPARC_INS_JMPL || instruction->id == cs::SPARC_INS_JMP) {
380  rtl->clear();
381  CaseStatement *caseStmt = new CaseStatement;
382  caseStmt->setIsComputed(true);
383 
384  // Capstone returns the operand as SPARC_OP_MEM, so we have to "undo" the outermost memof
385  // returned by operandToExp by an addrof
386  caseStmt->setDest(Unary::get(opAddrOf, operandToExp(instruction, 0))->simplify());
387  rtl->append(caseStmt);
388  }
389  else if (instruction->id == cs::SPARC_INS_RET || instruction->id == cs::SPARC_INS_RETL ||
390  instruction->id == cs::SPARC_INS_RETT) {
391  rtl->clear();
392  ReturnStatement *retStmt = new ReturnStatement;
393  rtl->append(retStmt);
394  }
395 
396  return rtl;
397 }

400 std::unique_ptr<RTL> CapstoneSPARCDecoder::instantiateRTL(Address pc, const char *instructionID,
401  const cs::cs_insn *instruction)
402 {
403  const int numOperands = instruction->detail->sparc.op_count;
404  std::vector<SharedExp> args(numOperands);
405 
406  for (int i = 0; i < numOperands; i++) {
407  args[i] = operandToExp(instruction, i);
408  }
409 
410  if (m_debugMode) {
411  QString argNames;
412  for (int i = 0; i < numOperands; i++) {
413  if (i != 0) {
414  argNames += " ";
415  }
416  argNames += args[i]->toString();
417  }
418 
419  LOG_MSG("Instantiating RTL at %1: %2 %3", pc, instructionID, argNames);
420  }
421 
422  // Take the argument, convert it to upper case and remove any .'s
423  const QString sanitizedName = QString(instructionID).remove(".").toUpper();
424  return m_dict.instantiateRTL(sanitizedName, pc, args);
425 }

499 ICLASS CapstoneSPARCDecoder::getInstructionType(const cs::cs_insn *instruction)
500 {
501  if (instruction->id == cs::SPARC_INS_NOP) {
502  return ICLASS::NOP;
503  }
504  else if (instruction->id == cs::SPARC_INS_UNIMP) {
505  return ICLASS::NOP;
506  }
507  else if (instruction->id == cs::SPARC_INS_CALL &&
508  instruction->detail->sparc.operands[0].type == cs::SPARC_OP_MEM) {
509  if (instruction->detail->sparc.operands[0].mem.base == cs::SPARC_REG_G0) {
510  return ICLASS::SD; // call %g0+foo, %o3. This is a static call
511  }
512  else {
513  return ICLASS::DD; // computed call
514  }
515  }
516  else if ((instruction->id == cs::SPARC_INS_JMP || instruction->id == cs::SPARC_INS_JMPL) &&
517  instruction->detail->sparc.operands[0].type == cs::SPARC_OP_MEM) {
518  if (instruction->detail->sparc.operands[0].mem.base == cs::SPARC_REG_G0) {
519  return ICLASS::SD;
520  }
521  else {
522  return ICLASS::DD;
523  }
524  }
525 
526  // FIXME: This code should check instruction->detail.sparc instead, however Casptone
527  // still has some bugs wrt. condition codes of branches, e.g. ba has cc invalid instead of 'a'
528  QString insMnemonic = QString(instruction->mnemonic);
529  if (insMnemonic.endsWith(",pn") || insMnemonic.endsWith(",pt")) {
530  insMnemonic.chop(3);
531  }
532 
533  const auto it = g_instructionTypes.find(insMnemonic);
534  return it != g_instructionTypes.end() ? it->second : ICLASS::NCT;
535 }

538 int CapstoneSPARCDecoder::getRegOperandSize(const cs::cs_insn *instruction, int opIdx) const
539 {
540  switch (instruction->id) {
541  // these always have 32 bit operands
542  case cs::SPARC_INS_FSTOI:
543  case cs::SPARC_INS_FITOS:
544  case cs::SPARC_INS_FSQRTS:
545  return 32;
546 
547  // these always have 64 bit operands
548  case cs::SPARC_INS_FCMPD:
549  case cs::SPARC_INS_FCMPED:
550  case cs::SPARC_INS_FDIVD:
551  case cs::SPARC_INS_FMULD:
552  case cs::SPARC_INS_FSQRTD:
553  case cs::SPARC_INS_FSUBD:
554  case cs::SPARC_INS_LDD: // LDDF
555  case cs::SPARC_INS_STD: // STDF
556  return 64;
557 
558  // these always have 128 bit operands
559  case cs::SPARC_INS_FCMPQ:
560  case cs::SPARC_INS_FDIVQ:
561  case cs::SPARC_INS_FMULQ:
562  case cs::SPARC_INS_FSQRTQ:
563  case cs::SPARC_INS_FSUBQ: return 128;
564 
565  case cs::SPARC_INS_FDTOI:
566  case cs::SPARC_INS_FDTOS: return (opIdx == 0) ? 64 : 32;
567 
568  case cs::SPARC_INS_FQTOS:
569  case cs::SPARC_INS_FQTOI: return (opIdx == 0) ? 128 : 32;
570 
571  case cs::SPARC_INS_FQTOD: return (opIdx == 0) ? 128 : 64;
572 
573  case cs::SPARC_INS_FDTOQ: return (opIdx == 0) ? 64 : 128;
574 
575  case cs::SPARC_INS_FITOD:
576  case cs::SPARC_INS_FSTOD: return (opIdx == 0) ? 32 : 64;
577 
578  case cs::SPARC_INS_FITOQ:

587 cs::sparc_reg fixSparcReg(uint8 code)
588 {
589  if (code == 30) {
590  return cs::SPARC_REG_FP;
591  }

603  }

610 bool CapstoneSPARCDecoder::decodeLDD(cs::cs_insn *decodedInstruction, uint32_t insn) const
611 {
612  if (((insn >> 19) & 0b1100000111111) != 0b1100000000011) {
613  return false; // not ldd
614  }
615 
616  const cs::sparc_reg rd = fixSparcReg((insn >> 25) & 0x1F);
617  const cs::sparc_reg rs1 = fixSparcReg((insn >> 14) & 0x1F);
618  const bool hasImm = ((insn >> 13) & 1) != 0;
619 
620  decodedInstruction->id = cs::SPARC_INS_LDD;
621  decodedInstruction->size = SPARC_INSTRUCTION_LENGTH;
622 
623  decodedInstruction->detail->sparc.cc = cs::SPARC_CC_INVALID;
624  decodedInstruction->detail->sparc.hint = cs::SPARC_HINT_INVALID;
625  decodedInstruction->detail->sparc.op_count = 2;
626 
627  decodedInstruction->detail->sparc.operands[0].type = cs::SPARC_OP_MEM;
628  decodedInstruction->detail->sparc.operands[0].mem.base = rs1;
629 
630  if (hasImm) {
631  const int simm = Util::signExtend(insn & 0x1FFF, 13);
632  decodedInstruction->detail->sparc.operands[0].mem.index = cs::SPARC_REG_INVALID;
633  decodedInstruction->detail->sparc.operands[0].mem.disp = simm;
634  std::sprintf(decodedInstruction->op_str, "[%s + %d], %s", cs::cs_reg_name(m_handle, rs1),
635  simm, cs::cs_reg_name(m_handle, rd));
636  }
637  else { // reg offset
638  const cs::sparc_reg rs2 = fixSparcReg(insn & 0x1F);
639  decodedInstruction->detail->sparc.operands[0].mem.index = rs2;
640  decodedInstruction->detail->sparc.operands[0].mem.disp = 0;
641  std::sprintf(decodedInstruction->op_str, "[%s + %s], %s", cs::cs_reg_name(m_handle, rs1),
642  cs::cs_reg_name(m_handle, rs2), cs::cs_reg_name(m_handle, rd));
643  }
644 
645  decodedInstruction->detail->sparc.operands[1].type = cs::SPARC_OP_REG;
646  decodedInstruction->detail->sparc.operands[1].reg = rd;
647 
648  Util::writeDWord(&decodedInstruction->bytes, insn, Endian::Little);
649  decodedInstruction->bytes[4] = 0;
650  std::strcpy(decodedInstruction->mnemonic, "ldd");
651  return true;

120 Address Win32BinaryLoader::getMainEntryPoint()
121 {
122  const BinarySymbol *mainSymbol = m_symbols->findSymbolByName("main");
123 
124  if (mainSymbol) {
125  return mainSymbol->getLocation();
126  }
127 
128  mainSymbol = m_symbols->findSymbolByName("_main"); // Example: MinGW
129 
130  if (mainSymbol) {
131  return mainSymbol->getLocation();
132  }
133 
134  mainSymbol = m_symbols->findSymbolByName("WinMain"); // Example: MinGW
135 
136  if (mainSymbol) {
137  return mainSymbol->getLocation();
138  }
139 
140  // This is a bit of a hack, but no more than the rest of Windows :-O The pattern is to look for
141  // an indirect call (opcode FF 15) to exit; within 10 instructions before that should be the
142  // call to WinMain (with no other calls inbetween). This pattern should work for "old style" and
143  // "new style" PE executables, as well as console mode PE files.
144 
145  // Start at program entry point
146  const Address imageBase = Address(READ4_LE(m_peHeader->Imagebase));
147  unsigned rva = READ4_LE(m_peHeader->EntrypointRVA);
148  Address addr;
149  unsigned lastOrdCall = 0;
150  int gap; // Number of instructions from the last ordinary call
151  int borlandState = 0; // State machine for Borland
152 
153  BinarySection *section = m_binaryImage->getSectionByName(".text");
154 
155  if (section == nullptr) {
156  section = m_binaryImage->getSectionByName("CODE");
157 
158  if (section == nullptr) {
159  LOG_ERROR("Cannot find a section containing code!");
160  return Address::INVALID;
161  }
162  }
163 
164 #define MAIN_RANGE (0x200U) // number of bytes to look for main/WinMain from start of entry point
165 
166  const unsigned int textSize = section->getSize();
167  const unsigned int searchLimit = rva + std::min(MAIN_RANGE, textSize);
168 
169  if (m_peHeader->Subsystem == 1) {
170  // native -> _start == main
171  return imageBase + READ4_LE(m_peHeader->EntrypointRVA);
172  }
173 
174  gap = 0xF0000000; // Large positive number (in case no ordinary calls)
175 
176  while (rva + 1 < searchLimit) { // make sure not to read past the end of the section
177  const Byte op1 = Util::readByte(m_image + rva + 0);
178  const Byte op2 = Util::readByte(m_image + rva + 1);
179 
180  LOG_VERBOSE("At %1, ops 0x%2, 0x%3", QString::number(rva, 16), QString::number(op1, 16),
181  QString::number(op2, 16));
182 
183  switch (op1) {
184  case 0xE8:
185  // An ordinary call; this could be to winmain/main
186  lastOrdCall = rva;
187  gap = 0;
188 
189  if (borlandState == 1) {
190  borlandState++;
191  }
192  else {
193  borlandState = 0;
194  }
195 
196  break;
197 
198  case 0xFF:
199  if (op2 == 0x15) { // Opcode FF 15 is indirect call
200  // Get the 4 byte address from the instruction
201  addr = Address(READ4_LE_P(m_image + rva + 2));
202  // const char *c = dlprocptrs[addr].c_str();
203  // printf("Checking %x finding %s\n", addr, c);
204  const BinarySymbol *calleeSym = m_symbols->findSymbolByAddress(addr);
205 
206  if (calleeSym && (calleeSym->getName() == "exit")) {
207  if (gap <= 10) {
208  // This is it. The instruction at lastOrdCall is (win)main
209  addr = Address(READ4_LE_P(m_image + lastOrdCall + 1));
210  addr += lastOrdCall + 5; // Addr is dest of call
211  return imageBase + addr;
212  }
213  }
214  }
215  else {
216  borlandState = 0;
217  }
218 
219  break;
220 
221  // Short relative jump, e.g. Borland
222  case 0xEB:
223  if (op2 >= 0x80) { // Branch backwards?
224  break; // Yes, just ignore it
225  }
226 
227  // Otherwise, actually follow the branch. May have to modify this some time...
228  rva += op2 + 2; // +2 for the instruction itself, and op2 for the displacement
229  gap++;
230  continue;
231 
232  case 0x6A:
233  if (op2 == 0) { // Push 00
234  // Borland pattern: Borland state before:
235  // push 0 0
236  // call __ExceptInit 1
237  // pop ecx 2
238  // push offset mainInfo 3
239  // push 0 4
240  if (borlandState == 0) {
241  borlandState = 1;
242  }
243  else if (borlandState == 4) {
244  // Borland pattern succeeds. p-4 has the offset of mainInfo
245  Address mainInfo = Address(READ4_LE(*(m_image + rva - 4)));
246 
247  // Address of main is at mainInfo+0x18
248  Address main = Address::INVALID;
249  if (m_binaryImage->readNativeAddr4(mainInfo + 0x18, main)) {
250  return main;
251  }
252  else {
253  return Address::INVALID;
254  }
255  }
256  }
257  else {
258  borlandState = 0;
259  }
260 
261  break;
262 
263  case 0x59: // Pop ecx
264  if (borlandState == 2) {
265  borlandState = 3;
266  }
267  else {
268  borlandState = 0;
269  }
270 
271  break;
272 
273  case 0x68: // Push 4 byte immediate
274  if (borlandState == 3) {
275  borlandState++;
276  }
277  else {
278  borlandState = 0;
279  }
280 
281  break;
282 
283  default: borlandState = 0; break;
284  }
285 
286  const int size = microX86Dis(rva + m_image);
287 
288  if (size == 0x40) {
289  LOG_WARN("Microdisassembler out of step at offset %1", rva);
290  break;
291  }
292 
293  rva += size;
294  gap++;
295  }
296 
297  // VS.NET release console mode pattern
298  rva = READ4_LE(m_peHeader->EntrypointRVA);
299 
300  if ((Util::readByte(m_image + rva + 0x20) == 0xff) &&
301  (Util::readByte(m_image + rva + 0x21) == 0x15)) {
302  Address desti = Address(READ4_LE_P(m_image + rva + 0x22));
303  const BinarySymbol *dest_sym = m_symbols->findSymbolByAddress(desti);
304 
305  if (dest_sym && (dest_sym->getName() == "GetVersionExA")) {
306  if ((Util::readByte(m_image + rva + 0x6d) == 0xff) &&
307  (Util::readByte(m_image + rva + 0x6e) == 0x15)) {
308  desti = Address(READ4_LE_P(m_image + rva + 0x6f));
309  dest_sym = m_symbols->findSymbolByAddress(desti);
310 
311  if (dest_sym && (dest_sym->getName() == "GetModuleHandleA")) {
312  if (Util::readByte(m_image + rva + 0x16e) == 0xe8) {
313  Address dest = Address(rva + 0x16e + 5 + READ4_LE_P(rva + m_image + 0x16f));
314  return imageBase + dest;
315  }
316  }
317  }
318  }
319  }
320 
321  // For VS.NET, need an old favourite: find a call with three pushes in the first 100 instuctions
322  int count = 100;
323  int numPushes = 0;
324  rva = READ4_LE(m_peHeader->EntrypointRVA);
325 
326  while (count > 0 && rva + 1 < m_imageSize) {
327  count--;
328  const Byte op1 = Util::readByte(m_image + rva + 0);
329  const Byte op2 = Util::readByte(m_image + rva + 1);
330 
331  if (op1 == 0xE8) { // CALL opcode
332  if (numPushes == 3) {
333  // Get the offset
334  int off = READ4_LE(*(m_image + rva + 1));
335  Address dest = Address(rva + 5 + off);
336 
337  // Check for a jump there
338  const Byte destOp = *reinterpret_cast<Byte *>(m_image + dest.value());
339 
340  if (destOp == 0xE9) {
341  // Follow that jump
342  off = READ4_LE(*(m_image + dest.value() + 1));
343  dest += off + 5;
344  }
345 
346  return dest + READ4_LE(m_peHeader->Imagebase);
347  }
348  else {
349  numPushes = 0; // Assume pushes don't accumulate over calls
350  }
351  }
352  else if ((op1 >= 0x50) && (op1 <= 0x57)) { // PUSH opcode
353  numPushes++;
354  }
355  else if (op1 == 0xFF) {
356  // FF 35 is push m[K]
357 
358  if (op2 == 0x35) {
359  numPushes++;
360  }
361  }
362  else if (op1 == 0xE9) {
363  // Follow the jump
364  int off = READ4_LE(*(m_image + rva + 1));
365  rva += off + 5;
366  continue;
367  }
368 
369  int size = microX86Dis(rva + m_image);
370 
371  if (size == 0x40) {
372  LOG_WARN("Microdisassembler out of step at offset %1", rva);
373  break;
374  }
375 
376  rva += size;
377 
378  if (rva >= textSize) {
379  break;
380  }
381  }
382 
383  // mingw pattern
384  rva = READ4_LE(m_peHeader->EntrypointRVA);
385  bool in_mingw_CRTStartup = false;
386  Address lastcall = Address::ZERO;
387  Address lastlastcall = Address::ZERO;
388 
389  while (true) {
390  const Byte op1 = Util::readByte(m_image + rva);
391 
392  if (in_mingw_CRTStartup && op1 == 0xC3) {
393  break;
394  }
395 
396  if (op1 == 0xE8) { // CALL opcode
397  unsigned int dest = rva + 5 + READ4_LE_P(m_image + rva + 1);
398  if (Util::inRange(dest, 0U, m_imageSize)) {
399  const Byte op2 = Util::readByte(m_image + dest);
400 
401  if (in_mingw_CRTStartup) {
402  const Byte op2a = Util::readByte(m_image + dest + 1);
403  Address desti = Address(READ4_LE_P(m_image + dest + 2));
404 
405  // skip all the call statements until we hit a call to an indirect call to
406  // ExitProcess; main is the 2nd call before this one
407  if (op2 == 0xff && op2a == 0x25) {
408  const BinarySymbol *dest_sym = m_symbols->findSymbolByAddress(desti);
409 
410  if (dest_sym && (dest_sym->getName() == "ExitProcess")) {
411  m_mingwMain = true;
412  return Address(READ4_LE(m_peHeader->Imagebase)) + lastlastcall + 5 +
413  READ4_LE_P(m_image + lastlastcall.value() + 1);
414  }
415  }
416 
417  lastlastcall = lastcall;
418  lastcall = Address(rva);
419  }
420  else {
421  rva = dest;
422  in_mingw_CRTStartup = true;
423  continue;
424  }
425  }
426  }
427 
428  int size = microX86Dis(m_image + rva);
429 
430  if (size == 0x40) {
431  LOG_WARN("Microdisassembler out of step at offset %1", rva);
432  break;
433  }
434 
435  rva += size;
436 
437  if (rva >= textSize) {
438  break;
439  }
440  }
441 
442  // Microsoft VisualC 2-6/net runtime
443  rva = READ4_LE(m_peHeader->EntrypointRVA);
444  bool gotGMHA = false; // has GetModuleHandleA been found?
445 
446  while (rva < textSize) {
447  const Byte op1 = Util::readByte(m_image + rva + 0);
448  const Byte op2 = Util::readByte(m_image + rva + 1);
449 
450  if (op1 == 0xFF && op2 == 0x15) { // indirect CALL opcode
451  const Address destAddr = Address(READ4_LE_P(m_image + rva + 2));
452  const BinarySymbol *destSym = m_symbols->findSymbolByAddress(destAddr);
453 
454  if (destSym && (destSym->getName() == "GetModuleHandleA")) {
455  gotGMHA = true;
456  }
457  }
458 
459  if (op1 == 0xE8 && gotGMHA) { // CALL opcode
460  Address dest = Address(rva + 5 + READ4_LE(*(m_image + rva + 1)));
461  m_symbols->createSymbol(dest + READ4_LE(m_peHeader->Imagebase), "WinMain");
462  return dest + READ4_LE(m_peHeader->Imagebase);
463  }
464 
465  if (op1 == 0xc3) { // ret ends search
466  break;
467  }
468 
469  int size = microX86Dis(rva + m_image);
470 
471  if (size == 0x40) {
472  LOG_WARN("Microdisassembler out of step at offset %1", rva);
473  break;
474  }
475 
476  rva += size;
477  }
478 
479  return Address::INVALID;
480 }

491 void Win32BinaryLoader::processIAT()
492 {
493  PEImportDtor *id = reinterpret_cast<PEImportDtor *>(
494  (m_image + READ4_LE(m_peHeader->ImportTableRVA)));
495 
496  // If any import table entry exists
497  if (m_peHeader->ImportTableRVA == 0 || id->name == 0) {
498  return;
499  }
500 
501  do {
502  const DWord nameOffset = READ4_LE(id->name);
503  const char *dllName = m_image + nameOffset;
504  if (!Util::inRange(nameOffset, 0U, m_imageSize)) {
505  LOG_WARN("Cannot read IAT entry: name offset out or range");
506  continue;
507  }
508 
509  const DWord originalFirstThunk = READ4_LE(id->originalFirstThunk);
510  const DWord firstThunk = READ4_LE(id->firstThunk);
511 
512  if (!Util::inRange(originalFirstThunk, 0U, m_imageSize) ||
513  !Util::inRange(firstThunk, 0U, m_imageSize)) {
514  LOG_WARN("Cannot read IAT entry: thunk offset out of range");
515  continue;
516  }
517 
518  const DWord thunk = (originalFirstThunk != 0) ? originalFirstThunk : firstThunk;
519 
520  const DWord *iat = reinterpret_cast<const DWord *>(m_image + thunk);
521  DWord iatEntry = Util::readDWord(iat, Endian::Little);
522  Address paddr = Address(READ4_LE(m_peHeader->Imagebase) + firstThunk);
523 
524  while (iatEntry != 0) {
525  if ((char *)iat > m_image + m_imageSize) {
526  LOG_WARN("Cannot read IAT entry: entry extends past file size");
527  break;
528  }
529 
530  if ((iatEntry >> 31) != 0) {
531  // This is an ordinal number (stupid idea)
532  // Dots can't be in identifiers
533  QString nodots = QString(dllName).replace(".", "_");
534  nodots = QString("%1_%2").arg(nodots).arg(iatEntry & ~(1 << 31));
535  BinarySymbol *sym = m_symbols->createSymbol(paddr, nodots);
536  sym->setAttribute("Imported", true);
537  sym->setAttribute("Function", true);
538  }
539  else {
540  // Normal case (IMAGE_IMPORT_BY_NAME). Skip the useless hint (2 bytes)
541  if (!Util::inRange(iatEntry + 2, 0U, m_imageSize)) {
542  LOG_WARN("Cannot read IAT entry: entry name offset out of range");
543  break;
544  }
545 
546  QString name = m_image + iatEntry + 2;
547 
548  BinarySymbol *sym = m_symbols->createSymbol(paddr, name);
549  sym->setAttribute("Imported", true);
550  sym->setAttribute("Function", true);
551  Address old_loc = Address(HostAddress(iat).value() - HostAddress(m_image).value() +
552  READ4_LE(m_peHeader->Imagebase));
553 
554  if (paddr != old_loc) { // add both possibilities
555  BinarySymbol *symbol = m_symbols->createSymbol(old_loc, QString("old_") + name);
556  symbol->setAttribute("Imported", true);
557  symbol->setAttribute("Function", true);
558  }
559  }
560 
561  iat++;
562  iatEntry = READ4_LE_P(iat);
563  paddr += 4;
564  }
565  } while ((++id)->name != 0);
566 }

569 void Win32BinaryLoader::readDebugData(QString exename)
570 {
571 #if defined(_WIN32) && !defined(__MINGW32__)
572  // attempt to load symbols for the exe or dll
573 
574  DWORD error;
575  HANDLE currProcess = GetCurrentProcess();
576 
577  dbghelp::SymSetOptions(SYMOPT_LOAD_LINES);
578 
579  if (!dbghelp::SymInitialize(currProcess, nullptr, FALSE)) {
580  error = GetLastError();
581  printf("SymInitialize returned error : %d\n", error);
582  return;
583  }
584 
585  DWORD64 dwBaseAddr = 0;
586  dwBaseAddr = dbghelp::SymLoadModule64(currProcess, nullptr, qPrintable(exename), nullptr,
587  dwBaseAddr, 0);
588 
589  if (dwBaseAddr != 0) {
590  assert(dwBaseAddr == m_peHeader->Imagebase);
591  bool found = false;
592  dbghelp::SymEnumSourceFiles(currProcess, dwBaseAddr, 0, lookforsource, &found);
593  m_hasDebugInfo = found;
594  }
595  else {
596  // SymLoadModule64 failed
597  error = GetLastError();
598  printf("SymLoadModule64 returned error : %d\n", error);
599  return;
600  }
601 #else
602  Q_UNUSED(exename);
603  LOG_WARN("Loading PE debug information is only available on Windows!");
604 #endif
605 }

610 bool Win32BinaryLoader::loadFromMemory(QByteArray &arr)
611 {
612  const char *fileData = arr.constData();
613  const DWord fileSize = arr.size();
614 
615  if (DOS_HEADER_SIZE + 4 /* ptr to PE header */ + sizeof(PEHeader) > fileSize) {
616  LOG_ERROR("Invalid PE: File size too small");
617  return false;
618  }
619 
620  DWord peHeaderOffset = Util::readDWord(fileData + DOS_HEADER_SIZE, Endian::Little);
621  if (peHeaderOffset + sizeof(PEHeader) > fileSize) {
622  LOG_ERROR("Invalid PE: PE header extends past file boundary");
623  return false;
624  }
625 
626  const PEHeader *peHdr = reinterpret_cast<const PEHeader *>(fileData + peHeaderOffset);
627 
628  try {
629  const DWord imageSize = READ4_LE(peHdr->ImageSize);
630  m_image = new char[imageSize];
631  m_imageSize = imageSize;
632  }
633  catch (const std::bad_alloc &) {
634  LOG_ERROR("Cannot allocate memory for copy of image");
635  return false;
636  }
637 
638  const DWord dosHeaderSize = READ4_LE(peHdr->HeaderSize);
639  if (dosHeaderSize >= fileSize) {
640  LOG_ERROR("Invalid PE: DOS header extends past file boundary");
641  return false;
642  }
643 
644  memcpy(m_image, fileData, dosHeaderSize);
645  m_header = reinterpret_cast<Header *>(m_image);
646 
647  if (!Util::testMagic((Byte *)m_header, { 'M', 'Z' })) {
648  LOG_ERROR("Invalid PE: Bad magic");
649  return false;
650  }
651 
652  m_peHeader = reinterpret_cast<PEHeader *>(m_image + peHeaderOffset);
653  if (!Util::testMagic((Byte *)m_peHeader, { 'P', 'E' })) {
654  LOG_ERROR("Invalid PE: Bad PE magic");
655  return false;
656  }
657 
658  const SWord ntHeaderSize = Util::readWord(&m_peHeader->NtHdrSize, Endian::Little);
659  const PEObject *o = reinterpret_cast<const PEObject *>((Byte *)(m_peHeader) + ntHeaderSize +
660  24);
661 
662  std::vector<SectionParam> params;
663 
664  const DWord numSections = Util::readWord(&m_peHeader->numObjects, Endian::Little);
665 
666  for (DWord i = 0; i < numSections; i++, o++) {
667  const DWord rva = READ4_LE(o->RVA);
668  const DWord size = READ4_LE(o->VirtualSize);
669  const DWord physOff = READ4_LE(o->PhysicalOffset);
670  const DWord physSize = READ4_LE(o->PhysicalSize);
671 
672  SectionParam sect;
673  // TODO: Check for unreadable sections (!IMAGE_SCN_MEM_READ)?
674  // FIXME Using std::min fixes the crash but does not solve the root issue.
675  // This needs further consideration.
676  memset(m_image + rva, 0, size);
677  memcpy(m_image + rva, fileData + physOff, std::min(physSize, size));
678 
679  sect.Name = QByteArray(o->ObjectName, 8);
680  sect.From = Address(READ4_LE(m_peHeader->Imagebase) + rva);
681  sect.ImageAddress = HostAddress(m_image) + rva;
682  sect.Size = size;
683  sect.PhysSize = physSize;
684 
685  // clang-format off
686  const DWord peFlags = READ4_LE(o->Flags);
687  sect.Bss = (peFlags & IMAGE_SCN_CNT_UNINITIALIZED_DATA) ? true : false;
688  sect.Code = (peFlags & IMAGE_SCN_CNT_CODE) ? true : false;

745 {

767 /**
768  * \internal Used above for a hack to find jump instructions pointing to IATs.
769  * Heuristic: start just before the "start" entry point looking for FF 25 opcodes followed by a
770  * pointer to an import entry. E.g. FF 25 58 44 40 00 where 00404458 is the IAT for _ftol. Note:
771  * some are on 0x10 byte boundaries, some on 2 byte boundaries (6 byte jumps packed), and there are
772  * often up to 0x30 bytes of statically linked library code (e.g. _atexit, __onexit) with sometimes
773  * two static libs in a row. So keep going until there is about 0x60 bytes with no match. Note:
774  * slight chance of coming across a misaligned match; probability is about 1/65536 times dozens in
775  * 2^32 ~= 10^-13
776  */
777 void Win32BinaryLoader::findJumps(Address curr)
778 {
779  BinarySection *section = m_binaryImage->getSectionByName(".text");
780  if (section == nullptr) {
781  section = m_binaryImage->getSectionByName("CODE");
782  if (section == nullptr) {
783  return;
784  }
785  }
786 
787  // Add to native addr to get host:
788  ptrdiff_t delta = (section->getHostAddr() - section->getSourceAddr()).value();
789 
790  int cnt = 0; // Count of bytes with no match
791  while (cnt < 0x60) { // Max of 0x60 bytes without a match
792  curr -= 2; // Has to be on 2-byte boundary
793  cnt += 2;
794 
795  if (curr < section->getSourceAddr()) {
796  break; // stepped out of section
797  }
798 
799  const HostAddress instrPtr(curr, delta);
800  const Byte *opcode = reinterpret_cast<const Byte *>(instrPtr.value());
801  if (opcode[0] != 0xFF || opcode[1] != 0x25) { // only check FF 25 jumps
802  continue;
803  }
804 
805  Address operand = Address(READ4_LE_P(HostAddress(curr, delta + 2)));
806  const BinarySymbol *symbol = m_symbols->findSymbolByAddress(operand);
807 
808  if (symbol == nullptr) {
809  continue;
810  }
811 
812  // try to rename symbol
813  const QString oldName = symbol->getName();
814  if (m_symbols->renameSymbol(oldName, "__imp_" + oldName) == false) {
815  continue;
816  }
817 
818  BinarySymbol *sym = m_symbols->createSymbol(curr, oldName);
819  sym->setAttribute("Function", true);
820  sym->setAttribute("Imported", true);
821  curr -= 4; // Next match is at least 4+2 bytes away
822  cnt = 0;
823  }
824 }

61 bool PentiumSignature::qualified(UserProc *p, Signature & /*candidate*/)
62 {
63  if (p->getProg()->getMachine() != Machine::PENTIUM) {
64  return false;
65  }
66 
67  LOG_VERBOSE2("Consider promotion to stdc pentium signature for %1", p->getName());
68 
69 #if 1
70  LOG_VERBOSE2("Promotion qualified: always true");
71  return true; // For now, always pass
72 #else
73  bool gotcorrectret1 = false;
74  bool gotcorrectret2 = false;
75  StatementList internal;
76  // p->getInternalStatements(internal);
77  internal.append(*p->getCFG()->getReachExit());
78  StatementList::iterator it;
79 
80  for (Statement *s : internal) {
81  Assign *e = dynamic_cast<Assign *>(s);
82 
83  if (e == nullptr) {
84  continue;
85  }
86 
87  if (e->getLeft()->isPC()) {
88  if (e->getRight()->isMemOf() && e->getRight()->getSubExp1()->isRegN(REG_PENT_ESP)) {
89  LOG_VERBOSE("Got pc = m[r[28]]");
90  gotcorrectret1 = true;
91  }
92  }
93  else if (e->getLeft()->isRegOfConst() &&
94  (e->getLeft()->getSubExp1()->access<Const>()->getInt() == REG_PENT_ESP)) {
95  if ((e->getRight()->getOper() == opPlus) &&
96  e->getRight()->getSubExp1()->isRegN(REG_PENT_ESP) &&
97  e->getRight()->getSubExp2()->isIntConst() &&
98  (e->getRight()->getSubExp2()->access<Const>()->getInt() == 4)) {
99  LOG_VERBOSE("Got r[28] = r[28] + 4");
100  gotcorrectret2 = true;
101  }
102  }
103  }
104 
105  LOG_VERBOSE("Promotion: %1", gotcorrectret1 && gotcorrectret2);
106  return gotcorrectret1 && gotcorrectret2;
107 #endif
108 }

166 SharedExp PentiumSignature::getProven(SharedExp left) const
167 {
168  if (left->isRegOfConst()) {
169  const int r = left->access<Const, 1>()->getInt();
170 
171  switch (r) {
172  case REG_PENT_ESP: // esp
173  return Binary::get(opPlus, Location::regOf(REG_PENT_ESP), Const::get(4)); // esp+4
174 
175  case REG_PENT_BX:
176  case REG_PENT_BP:
177  case REG_PENT_SI:
178  case REG_PENT_DI:
179  case REG_PENT_BL:
180  case REG_PENT_BH:
181  case REG_PENT_EBX:
182  case REG_PENT_EBP:
183  case REG_PENT_ESI:
184  case REG_PENT_EDI: return Location::regOf(r);
185  }
186  }
187 
188  return nullptr;

237 bool PentiumSignature::returnCompare(const Assignment &a, const Assignment &b) const
238 {
239  SharedConstExp la = a.getLeft();
240  SharedConstExp lb = b.getLeft();
241 
242  // Eax is the preferred return location
243  if (la->isRegN(REG_PENT_EAX)) {
244  return true; // r24 is less than anything
245  }
246  else if (lb->isRegN(REG_PENT_EAX)) {
247  return false; // Nothing is less than r24
248  }
249 
250  // Next best is floating point %st
251  if (la->isRegN(REG_PENT_ST0)) {
252  return true; // r30 is less than anything that's left
253  }
254  else if (lb->isRegN(REG_PENT_ST0)) {
255  return false; // Nothing left is less than r30
256  }
257 
258  // Else don't care about the order
259  return *la < *lb;
260 }

263 bool CallingConvention::StdC::PentiumSignature::argumentCompare(const Assignment &a,
264  const Assignment &b) const
265 {

267  SharedConstExp lb = b.getLeft();
268  int ma = Util::getStackOffset(la, REG_PENT_ESP);
269  int mb = Util::getStackOffset(lb, REG_PENT_ESP);
270 
271  if (ma && mb) {
272  return ma < mb;
273  }
274 
275  if (ma && !mb) {
276  return true; // m[sp-K] is less than anything else
277  }
278 
279  if (mb && !ma) {
280  return false; // Nothing else is less than m[sp-K]
281  }
282 
283  // Else don't care about the order
284  return *la < *lb;

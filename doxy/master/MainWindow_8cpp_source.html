178 void MainWindow::on_btnInputFileBrowse_clicked()
179 {
180  QString openFileDir = ".";
181 
182  if (ui->cbInputFile->currentIndex() != -1) {
183  // try to use the directory of the last opened file as starting directory.
184  QString lastUsedFile = ui->cbInputFile->itemText(ui->cbInputFile->currentIndex());
185 
186  if (!lastUsedFile.isEmpty()) {
187  QFileInfo fi(lastUsedFile);
188  openFileDir = fi.absolutePath();
189  }
190  }
191 
192  QString fileName = QFileDialog::getOpenFileName(
193  this, tr("Select a file to decompile..."), openFileDir,
194  "Windows Binaries (*.exe *.dll *.scr *.sys);;Other Binaries (*)");
195 
196  if (fileName.isEmpty()) {
197  // user cancelled
198  return;
199  }
200 
201  int existingIdx = ui->cbInputFile->findText(fileName);
202 
203  if (existingIdx == -1) {
204  // file not in the cache
205  ui->cbInputFile->insertItem(0, fileName);
206  existingIdx = 0;
207  saveSettings();
208  }
209 
210  ui->cbInputFile->setCurrentIndex(existingIdx);
211 
212  // we now have at least one input file
213  ui->btnToLoad->setEnabled(ui->cbOutputPath->count() > 0);
214 }

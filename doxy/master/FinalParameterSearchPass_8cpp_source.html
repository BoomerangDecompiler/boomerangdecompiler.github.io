30 {
31  qDeleteAll(proc->getParameters());
32  proc->getParameters().clear();
33 
34  if (proc->getSignature()->isForced()) {
35  // Copy from signature
36  int n = proc->getSignature()->getNumParams();
37  ImplicitConverter ic(proc->getCFG());
38 
39  for (int i = 0; i < n; ++i) {
40  SharedExp paramLoc = proc->getSignature()->getParamExp(i)->clone(); // E.g. m[r28 + 4]
41  LocationSet components;
42  paramLoc->addUsedLocs(components);
43 
44  for (const SharedExp &component : components) {
45  if (component != paramLoc) { // Don't subscript outer level
46  paramLoc->expSubscriptVar(component, nullptr); // E.g. r28 -> r28{-}
47  paramLoc->acceptModifier(&ic); // E.g. r28{-} -> r28{0}
48  }
49  }
50 
51  proc->getParameters().append(
52  new ImplicitAssign(proc->getSignature()->getParamType(i), paramLoc));
53  QString name = proc->getSignature()->getParamName(i);
54  SharedExp param = Location::param(name, proc);
55  SharedExp reParamLoc = RefExp::get(
56  paramLoc, proc->getCFG()->findOrCreateImplicitAssign(paramLoc));
57  proc->mapSymbolTo(reParamLoc, param); // Update name map
58  }
59 
60  return true;
61  }
62 
63  if (DEBUG_PARAMS) {
64  LOG_VERBOSE("Finding final parameters for %1", getName());
65  }
66 
67  // int sp = signature->getStackRegister();
68  proc->getSignature()->setNumParams(0); // Clear any old ideas
69  StatementList stmts;
70  proc->getStatements(stmts);
71 
72  for (Statement *s : stmts) {
73  // Assume that all parameters will be m[]{0} or r[]{0}, and in the implicit definitions at
74  // the start of the program
75  if (!s->isImplicit()) {
76  // Note: phis can get converted to assignments, but I hope that this is only later on:
77  // check this!
78  break; // Stop after reading all implicit assignments
79  }
80 
81  SharedExp e = static_cast<ImplicitAssign *>(s)->getLeft();
82 
83  if (proc->getSignature()->findParam(e) == -1) {
84  if (DEBUG_PARAMS) {
85  LOG_VERBOSE("Potential param %1", e);
86  }
87 
88  // I believe that the only true parameters will be registers or memofs that look like
89  // locals (stack pararameters)
90  if (!(e->isRegOf() || proc->isLocalOrParamPattern(e))) {
91  continue;
92  }
93 
94  if (DEBUG_PARAMS) {
95  LOG_VERBOSE("Found new parameter %1", e);
96  }
97

99  // Add this parameter to the signature (for now; creates parameter names)
100  proc->addParameterToSignature(e, ty);
101  // Insert it into the parameters StatementList, in sensible order
102  proc->insertParameter(e, ty);
103  }
104  }
105

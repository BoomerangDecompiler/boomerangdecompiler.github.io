44 /**
45  * \class Exp
46  * An expression class, though it will probably be used to hold many other things (e.g. perhaps
47  * transformations). It is a standard tree representation. Exp itself is abstract. A special class
48  * Const is used for constants. Unary, Binary, and Ternary hold 1, 2, and 3 subexpressions
49  * respectively. For efficiency of representation, these have to be separate classes, derived from
50  * Exp.
51  *
52  * Main class hierarchy:
53  * Exp (abstract)
54  * ____/ | \
55  * / | \
56  * Unary Const Terminal
57  * TypedExp____/ | \
58  * RefExp____/ Binary Location
59  * |
60  * Ternary
61  */
62 class BOOMERANG_API Exp : public std::enable_shared_from_this<Exp>

220  /// Access sub-expressions recursively
221  template<class T, int SUB_IDX, int... Path>
222  std::shared_ptr<T> access()
223  {
224  // clang-format off
225  switch (SUB_IDX) {
226  case 1: return getSubExp1()->access<T, Path...>();
227  case 2: return getSubExp2()->access<T, Path...>();
228  case 3: return getSubExp3()->access<T, Path...>();
229  default: assert(false);
230  }
231  // clang-format on
232 
233  return nullptr;
234  }

356  /**
357  * Takes an expression consisting on only + and - operators and partitions
358  * its terms into positive non-integer fixed terms, negative non-integer
359  * fixed terms and integer terms. For example, given:
360  * %sp + 108 + n - %sp - 92
361  * the resulting partition will be:
362  * positives = { %sp, n }
363  * negatives = { %sp }
364  * integers = { 108, -92 }
365  *
366  * \note integers is a vector so we can use the accumulate func
367  * \note Expressions are NOT cloned. Therefore, do not delete the expressions in
368  * positives or negatives
369  *
370  * \param positives the list of positive terms
371  * \param negatives the list of negative terms
372  * \param integers the vector of integer terms
373  * \param negate determines whether or not to negate the whole expression, i.e. we are on the
374  * RHS of an opMinus
375  */
376  void partitionTerms(std::list<SharedExp> &positives, std::list<SharedExp> &negatives,
377  std::vector<int> &integers, bool negate);

82 SharedExp SPARCSignature::getArgumentExp(int n) const
83 {
84  if (n < static_cast<int>(m_params.size())) {
85  return Signature::getArgumentExp(n);
86  }
87 
88  SharedExp e;
89 
90  if (n >= 6) {
91  // SPARCs pass the seventh and subsequent parameters at m[%sp+92],
92  // m[%esp+96], etc.
93  e = Location::memOf(Binary::get(opPlus,
94  Location::regOf(REG_SPARC_SP), // %o6 == %sp
95  Const::get(92 + (n - 6) * 4)));
96  }
97  else {
98  e = Location::regOf(REG_SPARC_O0 + n);
99  }
100 
101  return e;
102 }

112 SharedExp SPARCSignature::getProven(SharedExp left) const
113 {
114  if (left->isRegOfConst()) {
115  int r = left->access<Const, 1>()->getInt();
116 
117  switch (r) {
118  // These registers are preserved in SPARC: i0-i7 (24-31), sp (14)
119  case REG_SPARC_SP: // sp
120  case REG_SPARC_I0:
121  case REG_SPARC_I1:
122  case REG_SPARC_I2:
123  case REG_SPARC_I3: // i0-i3
124  case REG_SPARC_I4:
125  case REG_SPARC_I5:
126  case REG_SPARC_I6:
127  case REG_SPARC_I7: // i4-i7
128  // NOTE: Registers %g2 to %g4 are NOT preserved in ordinary application (non library)
129  // code
130  return left;
131  }
132  }
133 
134  return nullptr;
135 }

228 bool SPARCSignature::isAddrOfStackLocal(RegNum spIndex, const SharedConstExp &e) const
229 {
230  if (!Signature::isAddrOfStackLocal(spIndex, e)) {
231  return false;
232  }
233 
234  // SPARC specific test: K must be < 92; else it is a parameter
235  SharedConstExp simplified = e->clone()->simplify();
236  if (!simplified->getSubExp2()) {
237  // bare sp
238  return true;
239  }
240  else if (!simplified->getSubExp2()->isIntConst()) {
241  return false;
242  }
243 
244  const int offsetFromSP = simplified->getSubExp2()->access<Const>()->getInt();
245  return simplified->getOper() == opMinus && offsetFromSP < 92;
246 }

250 bool SPARCSignature::returnCompare(const Assignment &a, const Assignment &b) const
251 {
252  SharedConstExp la = a.getLeft();
253  SharedConstExp lb = b.getLeft();
254 
255  // %o0 (r8) is the preferred return location
256  if (la->isRegN(REG_SPARC_O0)) {
257  return true; // r24 is less than anything
258  }
259 
260  if (lb->isRegN(REG_SPARC_O0)) {
261  return false; // Nothing is less than r24
262  }
263 
264  // Next best is %f0 (r32)
265  if (la->isRegN(REG_SPARC_F0)) {
266  return true; // r32 is less than anything that's left
267  }
268 
269  if (lb->isRegN(REG_SPARC_F0)) {
270  return false; // Nothing left is less than r32
271  }
272 
273  // Next best is %f0-1 (r64)
274  if (la->isRegN(REG_SPARC_F0TO1)) {
275  return true; // r64 is less than anything that's left
276  }
277 
278  if (lb->isRegN(REG_SPARC_F0TO1)) {
279  return false; // Nothing left is less than r64
280  }
281 
282  // Next best is m[esp{-}+64]
283  if (*la == spPlus64) {
284  return true; // m[esp{-}+64] is less than anything that's left
285  }
286 
287  if (*lb == spPlus64) {
288  return false; // Nothing left is less than m[esp{-}+64]
289  }
290 
291  // Else don't care about the order
292  return *la < *lb;
293 }

296 bool SPARCSignature::argumentCompare(const Assignment &a, const Assignment &b) const
297 {
298  SharedConstExp la = a.getLeft();
299  SharedConstExp lb = b.getLeft();
300  // %o0-$o5 (r8-r13) are the preferred argument locations
301  int ra = 0, rb = 0;
302 
303  if (la->isRegOf()) {
304  int r = la->access<Const, 1>()->getInt();
305 
306  if ((r >= REG_SPARC_O0) && (r <= REG_SPARC_O5)) {
307  ra = r;
308  }
309  }
310 
311  if (lb->isRegOf()) {
312  int r = lb->access<Const, 1>()->getInt();
313 
314  if ((r >= REG_SPARC_O0) && (r <= REG_SPARC_O5)) {
315  rb = r;
316  }
317  }
318 
319  if (ra && rb) {
320  return ra < rb; // Both r8-r13: compare within this set
321  }
322 
323  if (ra && (rb == 0)) {
324  return true; // r8-r13 less than anything else
325  }
326 
327  if (rb && (ra == 0)) {
328  return false; // Nothing else is less than r8-r13
329  }
330 
331  const int ma = Util::getStackOffset(la, REG_SPARC_FP);
332  const int mb = Util::getStackOffset(lb, REG_SPARC_FP);
333 
334  if (ma && mb) {
335  return ma < mb; // Both m[sp + K]: order by memory offset
336  }
337 
338  if (ma && !mb) {
339  return true; // m[sp+K] less than anything left
340  }
341 
342  if (mb && !ma) {
343  return false; // nothing left is less than m[sp+K]
344  }
345 
346  return *la < *lb; // Else order arbitrarily
347 }

167 Function *Prog::getOrCreateFunction(Address startAddress)
168 {
169  if (startAddress == Address::INVALID) {
170  return nullptr;
171  }
172 
173  // Check if we already have this proc
174  Function *existingFunction = getFunctionByAddr(startAddress);
175 
176  if (existingFunction) { // Exists already ?
177  return existingFunction; // Yes, we are done
178  }
179 
180  Address tgt = m_binaryFile ? m_binaryFile->getJumpTarget(startAddress) : Address::INVALID;
181 
182 
183  if (tgt != Address::INVALID) {
184  startAddress = tgt;
185  }
186 
187  existingFunction = getFunctionByAddr(startAddress);
188 
189  if (existingFunction) { // Exists already ?
190  return existingFunction; // Yes, we are done
191  }
192 
193  QString procName;
194  bool isLibFunction = false;
195  const BinarySymbol *sym = m_binaryFile
196  ? m_binaryFile->getSymbols()->findSymbolByAddress(startAddress)
197  : nullptr;
198 
199  if (sym) {
200  isLibFunction = sym->isImportedFunction() || sym->isStaticFunction();
201  procName = sym->getName();
202  }
203 
204  if (procName.isEmpty()) {
205  // No name. Give it the name of the start address.
206  procName = QString("proc_%1").arg(startAddress.toString());
207  LOG_VERBOSE("Assigning name %1 to address %2", procName, startAddress);
208  }
209 
210  return m_rootModule->createFunction(procName, startAddress, isLibFunction);
211 }

352  Plugin *plugin = m_project->getPluginManager()->getPluginByName("C Symbol Provider plugin");

359  prov->readLibraryCatalog(this, dataDir.absoluteFilePath("signatures/common.hs"));
360 
361  QString libCatalogName;

363  case Machine::PENTIUM: libCatalogName = "signatures/pentium.hs"; break;

365  case Machine::PPC: libCatalogName = "signatures/ppc.hs"; break;
366  case Machine::ST20: libCatalogName = "signatures/st20.hs"; break;
367  default: libCatalogName = ""; break;

371  prov->readLibraryCatalog(this, dataDir.absoluteFilePath(libCatalogName));
372  }

374  if (isWin32()) {
375  prov->readLibraryCatalog(this, dataDir.absoluteFilePath("signatures/win32.hs"));

377 
378  // TODO: change this to BinaryLayer query ("FILE_FORMAT","MACHO")

427 const char *Prog::getStringConstant(Address addr, bool knownString /* = false */) const
428 {
429  if (!m_binaryFile || addr == Address::INVALID) {
430  return nullptr;
431  }
432 
433  const BinarySection *sect = m_binaryFile->getImage()->getSectionByAddr(addr);
434 
435  // Too many compilers put constants, including string constants,
436  // into read/write sections, so we cannot check if the address is in a readonly section
437  if (!sect || sect->isAddressBss(addr)) {
438  return nullptr;
439  }
440 
441  // At this stage, only support ascii, null terminated, non unicode strings.
442  // At least 4 of the first 6 chars should be printable ascii
443  const char *p = reinterpret_cast<const char *>(
444  (sect->getHostAddr() - sect->getSourceAddr() + addr).value());
445 
446  if (knownString) {
447  // No need to guess... this is hopefully a known string
448  return p;
449  }
450 
451  // this address is not known to be a string -> use heuristic
452  int numPrintables = 0;
453  int numControl = 0; // Control characters like \n, \r, \t
454  int numTotal = 0;
455 
456  for (int i = 0; i < 6; i++, numTotal++) {
457  if (p[i] == 0) {
458  break;
459  }
460  else if (std::isprint(static_cast<Byte>(p[i]))) {
461  numPrintables++;
462  }
463  else if (*p == '\n' || *p == '\t' || *p == '\r') {
464  numControl++;
465  }
466  }
467 
468  if (numTotal == 0) {
469  return "";
470  }
471  else if (numTotal - numPrintables - numControl < 2) {
472  return p;
473  }
474 
475  return nullptr;
476 }

592 bool Prog::decodeEntryPoint(Address entryAddr)
593 {
594  Function *func = getFunctionByAddr(entryAddr);
595 
596  if (!func || (!func->isLib() && !static_cast<UserProc *>(func)->isDecoded())) {
597  if (!Util::inRange(entryAddr, m_binaryFile->getImage()->getLimitTextLow(),
598  m_binaryFile->getImage()->getLimitTextHigh())) {
599  LOG_WARN("Attempt to decode entrypoint at address %1 outside text area", entryAddr);
600  return false;
601  }
602 
603  m_fe->decodeRecursive(entryAddr);
604  }
605 
606  if (!func) {
607  func = getFunctionByAddr(entryAddr);
608 
609  // Chek if there is a library thunk at entryAddr
610  if (!func) {
611  Address jumpTarget = m_binaryFile->getJumpTarget(entryAddr);
612 
613  if (jumpTarget != Address::INVALID) {
614  func = getFunctionByAddr(jumpTarget);
615  }
616  }
617  }
618 
619  if (!func) {
620  return false;
621  }
622 
623  if (!func->isLib()) { // -sf procs marked as __nodecode are treated as library procs (?)
624  m_entryProcs.push_back(static_cast<UserProc *>(func));
625  }
626 
627  return true;
628 }

712 bool Prog::markGlobalUsed(Address uaddr, SharedType knownType)
713 {
714  for (auto &glob : m_globals) {
715  if (glob->containsAddress(uaddr)) {
716  if (knownType) {
717  glob->meetType(knownType);
718  }
719 
720  return true;
721  }
722  }
723 
724  if (!m_binaryFile || m_binaryFile->getImage()->getSectionByAddr(uaddr) == nullptr) {
725  LOG_VERBOSE("Refusing to create a global at address %1 "
726  "that is in no known section of the binary",
727  uaddr);
728  return false;
729  }
730 
731  QString name = newGlobalName(uaddr);
732  SharedType ty;
733 
734  if (knownType) {
735  ty = knownType;
736 
737  if (ty->resolvesToArray() && ty->as<ArrayType>()->isUnbounded()) {
738  SharedType baseType = ty->as<ArrayType>()->getBaseType();
739  int baseSize = 0;
740 
741  if (baseType) {
742  baseSize = baseType->getSizeInBytes();
743  }
744 
745  auto symbol = m_binaryFile->getSymbols()->findSymbolByName(name);
746  int sz = symbol ? symbol->getSize() : 0;
747 
748  if (sz && baseSize) {
749  // Note: since ty is a pointer and has not been cloned, this will also set the type
750  // for knownType
751  ty->as<ArrayType>()->setLength(sz / baseSize);
752  }
753  }
754  }
755  else {
756  ty = guessGlobalType(name, uaddr);
757  }
758 
759  m_globals.insert(std::make_shared<Global>(ty, uaddr, name, this));
760 
761  LOG_VERBOSE("globalUsed: name %1, address %2, %3 type %4", name, uaddr,
762  knownType ? "known" : "guessed", ty->getCtype());
763 
764  return true;
765 }

768 std::shared_ptr<ArrayType> Prog::makeArrayType(Address startAddr, SharedType baseType)
769 {
770  QString name = newGlobalName(startAddr);
771 
772  // TODO: fix the case of missing symbol table interface
773  const BinarySymbol *symbol = m_binaryFile ? m_binaryFile->getSymbols()->findSymbolByName(name)
774  : nullptr;
775 
776  if (!symbol || (symbol->getSize() == 0)) {
777  return ArrayType::get(baseType); // An "unbounded" array
778  }
779 
780  size_t size = symbol->getSize();
781  int elemSize = baseType->getSizeInBytes();
782  if (elemSize < 1) {
783  elemSize = 1;
784  }

96 bool ElfBinaryLoader::loadFromMemory(QByteArray &img)
97 {
98  m_loadedImageSize = img.size();
99 
100  // Allocate memory to hold the file
101  m_loadedImage = reinterpret_cast<Byte *>(img.data());
102  m_elfHeader = reinterpret_cast<Elf32_Ehdr *>(img.data()); // Save a lot of casts
103 
104  if (m_loadedImageSize < sizeof(Elf32_Ehdr)) {
105  LOG_ERROR("Cannot load ELF file: File size too small");
106  return false;
107  }
108 
109  // Basic checks
110  if ((m_elfHeader->e_ident[EI_MAGO] != ELFMAG0) || (m_elfHeader->e_ident[EI_MAG1] != ELFMAG1) ||
111  (m_elfHeader->e_ident[EI_MAG2] != ELFMAG2) || (m_elfHeader->e_ident[EI_MAG3] != ELFMAG3)) {
112  LOG_ERROR("Cannot load ELF file: Bad magic %1 %2 %3 %4", m_elfHeader->e_ident[EI_MAGO],
113  m_elfHeader->e_ident[EI_MAG1], m_elfHeader->e_ident[EI_MAG2],
114  m_elfHeader->e_ident[EI_MAG3]);
115  return false;
116  }
117 
118  switch (m_elfHeader->e_ident[EI_CLASS]) {
119  case ELFCLASS32: m_binaryFile->setBitness(32); break;
120  case ELFCLASS64: m_binaryFile->setBitness(64); break;
121  default:
122  LOG_ERROR("Cannot load ELF file: Unknown ELF class %1", m_elfHeader->e_ident[EI_CLASS]);
123  return false;
124  }
125 
126  // endianness
127  switch (m_elfHeader->e_ident[EI_DATA]) {
128  case ELFDATA2LSB: m_endian = Endian::Little; break;
129  case ELFDATA2MSB: m_endian = Endian::Big; break;
130 
131  default:
132  LOG_ERROR("Cannot load ELF file: Unknown ELF endianness %1", m_elfHeader->e_ident[EI_DATA]);
133  return false;
134  }
135 
136  // Set up program and section header pointer (in case needed)
137  const Elf32_Off phOffset = elfRead4(&m_elfHeader->e_phoff);
138  const Elf32_Off shOffset = elfRead4(&m_elfHeader->e_shoff);
139 
140  if (!Util::inRange(phOffset, 1UL, m_loadedImageSize)) {
141  LOG_ERROR("Cannot load ELF file: Invalid program header offset %1", phOffset);
142  return false;
143  }
144  else if (!Util::inRange(shOffset, 1UL, m_loadedImageSize)) {
145  LOG_ERROR("Cannot load ELF file: Invalid section header offset %1", shOffset);
146  return false;
147  }
148 
149  m_programHdrs = reinterpret_cast<Elf32_Phdr *>(m_loadedImage + phOffset);
150  m_sectionHdrs = reinterpret_cast<Elf32_Shdr *>(m_loadedImage + shOffset);
151 
152  // Number of sections
153  const Elf32_Half numSections = elfRead2(&m_elfHeader->e_shnum);
154  if (numSections == 0) {
155  LOG_ERROR("Cannot load ELF file: No sections found");
156  return false;
157  }
158  else if ((const Byte *)(m_sectionHdrs + numSections) > m_loadedImage + m_loadedImageSize) {
159  LOG_ERROR("Cannot load ELF file: Section header information extends past end of file");
160  return false;
161  }
162 
163  // Set up the m_sh_link and m_sh_info arrays
164 
165  try {
166  m_shLink.reset(new Elf32_Word[numSections]);
167  m_shInfo.reset(new Elf32_Word[numSections]);
168  }
169  catch (const std::bad_alloc &) {
170  m_shLink.reset();
171  m_shInfo.reset();
172  LOG_ERROR("Cannot load ELF file: Not enough memory");
173  return false;
174  }
175 
176  // Set up section header string table pointer
177  const Elf32_Half stringSectionIndex = elfRead2(&m_elfHeader->e_shstrndx);
178 
179  if (!Util::inRange(stringSectionIndex, static_cast<Elf32_Half>(1), numSections)) {
180  LOG_ERROR("Cannot load ELF file: Invalid string section index %1", stringSectionIndex);
181  return false;
182  }
183 
184  const Elf32_Off stringSectionOffset = elfRead4(&m_sectionHdrs[stringSectionIndex].sh_offset);
185  if (!Util::inRange(stringSectionOffset, 1UL, m_loadedImageSize)) {
186  LOG_ERROR("Cannot load ELF file: Invalid string section offset %1", stringSectionOffset);
187  }
188 
189  m_strings = reinterpret_cast<const char *>(m_loadedImage + stringSectionOffset);
190 
191  bool seenCode = false; // True when have seen a code sect
192  Address arbitaryLoadAddr = Address(0x80000000);
193 
194  for (Elf32_Half i = 0; i < numSections; i++) {
195  // Get section information.
196  const Elf32_Shdr *sectionHeader = m_sectionHdrs + i;
197 
198  // Check if this section header entry is fully contained in this file
199  if (reinterpret_cast<const Byte *>(sectionHeader + 1) > m_loadedImage + m_loadedImageSize) {
200  LOG_ERROR(
201  "Cannot load ELF file: Section header for section %1 extends past image boundary",
202  i);
203  return false;
204  }
205 
206  const char *sectionName = m_strings + elfRead4(&sectionHeader->sh_name);
207 
208  if (reinterpret_cast<const Byte *>(sectionName) > m_loadedImage + m_loadedImageSize) {
209  LOG_ERROR("Cannot load ELF file: Section name for section %1 is outside of image size",
210  i);
211  return false;
212  }
213 
214  SectionParam newSection;
215  newSection.Name = sectionName;
216 
217  // Since the bss section is a special section, just assume it is always present
218  newSection.Bss = (strcmp(sectionName, ".bss") == 0);
219  newSection.Code = false;
220  newSection.Data = false;
221  newSection.ReadOnly = false;
222 
223  Elf32_Off _off = elfRead4(&sectionHeader->sh_offset);
224  if (!Util::inRange(_off, 0UL, m_loadedImageSize)) {
225  LOG_ERROR("Cannot load ELF file: Section data for section %1 is outside of image size",
226  i);
227  return false;
228  }
229  else if (_off != 0) {
230  newSection.imagePtr = HostAddress(m_loadedImage) + _off;
231  }
232 
233  newSection.SourceAddr = Address(elfRead4(&sectionHeader->sh_addr));
234  newSection.Size = elfRead4(&sectionHeader->sh_size);
235 
236  if (_off + newSection.Size > m_loadedImageSize && !newSection.Bss) {
237  LOG_ERROR("Cannot load ELF file: Section %1 extends past image boundary", i);
238  return false;
239  }
240 
241  if (newSection.SourceAddr.isZero() && !QString(sectionName).startsWith(".rel")) {
242  const Elf32_Word align = elfRead4(&sectionHeader->sh_addralign);
243 
244  if (align > 1) {
245  if (arbitaryLoadAddr.value() % align != 0) {
246  arbitaryLoadAddr += align - (arbitaryLoadAddr.value() % align);
247  }
248  }
249 
250  newSection.SourceAddr = arbitaryLoadAddr;
251  arbitaryLoadAddr += newSection.Size ? newSection.Size : 1;
252  }
253 
254  newSection.sectionType = elfRead4(&sectionHeader->sh_type);
255  newSection.entry_size = elfRead4(&sectionHeader->sh_entsize);
256  m_shLink[i] = elfRead4(&sectionHeader->sh_link);
257  m_shInfo[i] = elfRead4(&sectionHeader->sh_info);
258 
259  if (newSection.SourceAddr + newSection.Size > m_nextExtern) {
260  m_firstExtern = m_nextExtern = newSection.SourceAddr + newSection.Size;
261  }
262 
263  if ((elfRead4(&sectionHeader->sh_flags) & SHF_WRITE) == 0) {
264  newSection.ReadOnly = true;
265  }
266 
267  if (elfRead4(&sectionHeader->sh_flags) & SHF_EXECINSTR) {
268  newSection.Code = true;
269  seenCode = true; // We've got to a code section
270  }
271 
272  // Deciding what is data and what is not is actually quite tricky but important.
273  // For example, it's crucial to flag the .exception_ranges section as data,
274  // otherwise there is a "hole" in the allocation map, that means
275  // that there is more than one "delta" from a read-only section to a page,
276  // and in the end using -C results in a file that looks OK
277  // but when run just says "Killed". So we use the Elf designations;
278  // it seems that ALLOC.!EXEC -> data
279  // But we don't want sections before the .text section, like .interp, .hash, etc etc.
280  // Hence seenCode.
281  //
282  // NOTE: this ASSUMES that sections appear in a sensible order in the input binary file:
283  // junk, code, rodata, data, bss
284  if (seenCode &&
285  ((elfRead4(&sectionHeader->sh_flags) & (SHF_EXECINSTR | SHF_ALLOC)) == SHF_ALLOC) &&
286  (elfRead4(&sectionHeader->sh_type) != SHT_NOBITS)) {
287  newSection.Data = true;
288  }
289 
290  m_elfSections.push_back(newSection);
291  }
292 
293  // assign arbitary addresses to .rel.* sections too
294  for (SectionParam &sect : m_elfSections) {
295  if (sect.SourceAddr.isZero() && sect.Name.startsWith(".rel")) {
296  sect.SourceAddr = arbitaryLoadAddr;
297  arbitaryLoadAddr += (sect.Size > 0) ? sect.Size : 1;
298  }
299  }
300 
301  // Inform Boomerang about new sections
302  for (SectionParam par : m_elfSections) {
303  if (par.Size == 0) {
304  // this is most probably the NULL section; if it is not, warn the user
305  if (par.Name != "") {
306  LOG_WARN("Not adding 0 sized section '%1'", par.Name);
307  }
308 
309  continue;
310  }
311 
312  const Address lowAddr = par.SourceAddr;
313  const Address highAddr = par.SourceAddr + par.Size;
314 
315  BinarySection *sect = m_binaryFile->getImage()->createSection(par.Name, lowAddr, highAddr);
316 
317  if (sect) {
318  sect->setBss(par.Bss);
319  sect->setCode(par.Code);
320  sect->setData(par.Data);
321  sect->setEndian(m_endian);
322  sect->setHostAddr(par.imagePtr);
323  sect->setEntrySize(par.entry_size);
324  sect->setReadOnly(par.ReadOnly);
325 
326  if (!(par.Bss || par.SourceAddr.isZero())) {
327  sect->addDefinedArea(par.SourceAddr, par.SourceAddr + par.Size);
328  }
329 
330  if (par.sectionType == SHT_STRTAB) {
331  sect->setAttributeForRange("StringsSection", true, sect->getSourceAddr(),
332  sect->getSourceAddr() + sect->getSize());
333  }
334  }
335  }
336 
337  // Add symbol info. Note that some symbols will be in the main table only, and others in the
338  // dynamic table only. So the best idea is to add symbols for all sections of the appropriate
339  // type
340  for (size_t i = 1; i < m_elfSections.size(); ++i) {
341  const unsigned int sectionType = m_elfSections[i].sectionType;
342 
343  if ((sectionType == SHT_SYMTAB) || (sectionType == SHT_DYNSYM)) {
344  addSymbolsForSection(i);
345  }
346  }
347 
348  // Save the relocation to symbol table info
349  const BinarySection *section = m_binaryFile->getImage()->getSectionByName(".rela.text");
350 
351  if (section) {
352  m_relocHasAddend = true; // Remember its a relA table
353  m_relocSection = reinterpret_cast<Elf32_Rel *>(
354  section->getHostAddr().value()); // Save pointer to reloc table
355  }
356  else {
357  m_relocHasAddend = false;
358  section = m_binaryFile->getImage()->getSectionByName(".rel.text");
359 
360  if (section) {
361  m_relocSection = reinterpret_cast<Elf32_Rel *>(
362  section->getHostAddr().value()); // Save pointer to reloc table
363  }
364  }
365 
366  // Find the PLT limits. Required for IsDynamicLinkedProc(), e.g.
367  const BinarySection *pltSection = m_binaryFile->getImage()->getSectionByName(".plt");
368 
369  if (pltSection) {
370  m_pltMin = pltSection->getSourceAddr();
371  m_pltMax = pltSection->getSourceAddr() + pltSection->getSize();
372  }
373 
374  // Apply relocations; important when the input program is not compiled with -fPIC
375  applyRelocations();
376  markImports();
377 
378  return true;
379 }

410 Address ElfBinaryLoader::findRelPltOffset(int i)
411 {
412  const BinarySection *siPlt = m_binaryFile->getImage()->getSectionByName(".plt");
413  Address addrPlt = siPlt ? siPlt->getSourceAddr() : Address::ZERO;
414  const BinarySection *siRelPlt = m_binaryFile->getImage()->getSectionByName(".rel.plt");
415  int sizeRelPlt = sizeof(Elf32_Rel); // Size of each entry in the .rel.plt table
416 
417  if (siRelPlt == nullptr) {
418  siRelPlt = m_binaryFile->getImage()->getSectionByName(".rela.plt");
419  sizeRelPlt = sizeof(Elf32_Rela); // Size of each entry in the .rela.plt table is 12 bytes
420  }
421 
422  HostAddress addrRelPlt = HostAddress::ZERO;
423  int numRelPlt = 0;
424 
425  if (siPlt == nullptr || siRelPlt == nullptr) {
426  return Address::INVALID; // neither .plt nor .rel.plt nor .rela.plt are available
427  }
428 
429  addrRelPlt = siRelPlt->getHostAddr();
430  numRelPlt = siRelPlt->getSize() / sizeRelPlt;
431 
432  int first = i;
433  if (first >= numRelPlt) {
434  first = numRelPlt - 1;
435  }
436 
437  int curr = first;
438  int pltEntrySize = siPlt->getEntrySize();
439 
440  if (pltEntrySize == 0) {
441  return Address::INVALID;
442  }
443 
444  do {
445  // Each entry is sizeRelPlt bytes, and will contain the offset, then the info
446  // (addend optionally follows)
447  DWord *pltEntry = reinterpret_cast<DWord *>((addrRelPlt + (curr * sizeRelPlt)).value());
448  const int entry = elfRead4(pltEntry + 1);
449  const int sym = entry >> 8; // The symbol index is in the top 24 bits (Elf32 only)
450  const int entryType = entry & 0xFF;
451 
452  if (sym == i) {
453  const BinarySection *targetSect = m_binaryFile->getImage()->getSectionByAddr(
454  Address(elfRead4(pltEntry)));
455 
456  if (targetSect) {
457  if (targetSect->getName().contains("got")) {
458  int c = elfRead4(pltEntry) - targetSect->getSourceAddr().value();
459  int plt_offset2 = elfRead4(
460  reinterpret_cast<DWord *>((targetSect->getHostAddr() + c).value()));
461  int plt_idx = (plt_offset2 % pltEntrySize);
462 
463  if (entryType == R_386_JMP_SLOT) {
464  return Address(plt_offset2 - 6);
465  }
466 
467  return addrPlt + plt_idx * pltEntrySize;
468  }
469 
470  const int plt_offset = elfRead4(pltEntry) - siPlt->getSourceAddr().value();
471  // Found! Now we want the native address of the associated PLT entry.
472  // For now, assume a size of 0x10 for each PLT entry, and assume that each entry in
473  // the .rel.plt section corresponds exactly to an entry in the .plt (except there is
474  // one dummy .plt entry)
475  return addrPlt + plt_offset;
476  }
477  }
478 
479  if (--curr < 0) {
480  curr = numRelPlt - 1;
481  }
482  } while (curr != first); // Will eventually wrap around to first if not present
483 
484  return Address::ZERO; // Exit if this happens
485 }

488 void ElfBinaryLoader::processSymbol(Translated_ElfSym &sym, int e_type, int i,
489  const QString &currentFile)
490 {
491  bool imported = sym.SectionIdx == SHT_NULL;
492  bool local = sym.Binding == STB_LOCAL || sym.Binding == STB_WEAK;
493  const BinarySection *siPlt = m_binaryFile->getImage()->getSectionByName(".plt");
494 
495  if (sym.Value.isZero() && siPlt) { // && i < max_i_for_hack) {
496  // Special hack for gcc circa 3.3.3: (e.g. test/pentium/settest). The value in the dynamic
497  // symbol table is zero! I was assuming that index i in the dynamic symbol table would
498  // always correspond to index i in the .plt section, but for fedora2_true, this doesn't
499  // work. So we have to look in the .rel[a].plt section. Thanks, gcc! Note that this hack
500  // can cause strange symbol names to appear
501  sym.Value = findRelPltOffset(i);
502  }
503  else if (e_type == ET_REL) {
504  if (sym.SectionIdx < m_elfSections.size()) {
505  sym.Value += m_elfSections[sym.SectionIdx].SourceAddr;
506  }
507  }
508 
509  // try to find given symbol, if it has Value of 0, try to use the name.
510  const BinarySymbol *symbol = sym.Value.isZero() ? m_symbols->findSymbolByName(sym.Name)
511  : m_symbols->findSymbolByAddress(sym.Value);
512 
513  // Ensure no overwriting (except functions)
514  if (symbol != nullptr) { // TODO: if symbol already exists
515  return;
516  }
517  else if (sym.Binding == STB_WEAK && sym.Type == STT_NOTYPE) {
518  return;
519  }
520  else if (sym.Type == STT_FILE) {
521  return;
522  }
523  else if (sym.Name.isEmpty()) {
524  return;
525  }
526 
527  if (sym.Value.isZero()) {
528  LOG_WARN("Skipping symbol %1 with unknown location", sym.Name);
529  return;
530  }
531 
532  // TODO: add more symbol information here (function/export etc. ) ?
533  BinarySymbol *new_symbol(m_symbols->createSymbol(sym.Value, sym.Name, local));
534  new_symbol->setSize(elfRead4(&m_symbolSection[i].st_size));
535 
536  if (imported) {
537  new_symbol->setAttribute("Imported", true);
538  }
539 
540  if (sym.Type == STT_FUNC) {
541  new_symbol->setAttribute("Function", true);
542  }
543 
544  if (!currentFile.isEmpty()) {
545  new_symbol->setAttribute("SourceFile", currentFile);
546  }
547 }

550 void ElfBinaryLoader::addSymbolsForSection(int secIndex)
551 {
552  if (!Util::inRange(secIndex, 0, (int)m_elfSections.size())) {
553  LOG_WARN("Cannot add symbols for section with invalid index %1", secIndex);
554  return;
555  }
556 
557  const SectionParam &section = m_elfSections[secIndex];
558 
559  if (!Util::inRange(section.entry_size, sizeof(Elf32_Sym), m_loadedImageSize)) {
560  LOG_WARN("Cannot add symbols for section %1: Invalid section entry size %1",
561  section.entry_size);
562  return;
563  }
564  else if (section.Size == 0 || (section.Size % section.entry_size) != 0) {
565  LOG_WARN("Cannot add symbols for section %1: Invalid section size %2 "
566  "(zero or not divisible by %3)",
567  secIndex, section.Size, section.entry_size);
568  return;
569  }
570 
571  m_symbolSection = reinterpret_cast<const Elf32_Sym *>(
572  section.imagePtr.value()); // Pointer to symbols
573  if (!m_symbolSection) {
574  return;
575  }
576 
577  const SWord symbolType = elfRead2(&m_elfHeader->e_type);
578  const uint32 strSectionIdx = m_shLink[secIndex]; // sh_link points to the string table
579  if (!Util::inRange(strSectionIdx, 0UL, m_elfSections.size())) {
580  return; // cannot read symbol name from invalid string section
581  }
582 
583  const int numSymbols = section.Size / section.entry_size;
584  QString fileName;
585 
586  // Index 0 is a dummy entry
587  for (int i = 1; i < numSymbols; i++) {
588  Translated_ElfSym translatedSym;
589  Address val = Address(elfRead4(&m_symbolSection[i].st_value));
590  int nameIdx = elfRead4(&m_symbolSection[i].st_name);
591 
592  if (nameIdx == 0) { /* Silly symbols with no names */
593  continue;
594  }
595 
596  const QString symbolName = getStrPtr(strSectionIdx, nameIdx);
597  // Hack off the "@@GLIBC_2.0" of Linux, if present
598  translatedSym.Name = symbolName.left(symbolName.indexOf("@@"));
599  translatedSym.Type = ELF32_ST_TYPE(m_symbolSection[i].st_info);
600  translatedSym.Binding = ELF32_ST_BIND(m_symbolSection[i].st_info);
601  translatedSym.Visibility = ELF32_ST_VISIBILITY(m_symbolSection[i].st_other);
602  translatedSym.SymbolSize = m_symbolSection[i].st_size;
603  translatedSym.SectionIdx = elfRead2(&m_symbolSection[i].st_shndx);
604  translatedSym.Value = val;
605 
606  if (translatedSym.Type == STT_FILE) {
607  fileName = translatedSym.Name;
608  }
609 
610  if (translatedSym.Binding != STB_LOCAL && !fileName.isEmpty()) {
611  // first non-local symbol, clear the current_file
612  fileName.clear();
613  }
614 
615  processSymbol(translatedSym, symbolType, i, fileName);
616  }
617 
618  const Address addressOfMain = getMainEntryPoint();
619 
620  if ((addressOfMain != Address::INVALID) &&
621  (nullptr == m_symbols->findSymbolByAddress(addressOfMain))) {
622  // Ugh - main mustn't have the STT_FUNC attribute. Add it

628 void ElfBinaryLoader::addRelocsAsSyms(uint32_t relSecIdx)
629 {
630  if (relSecIdx >= m_elfSections.size()) {
631  // invalid section index
632  return;
633  }
634 
635  const SectionParam &section = m_elfSections[relSecIdx];
636 
637  m_relocSection = reinterpret_cast<const Elf32_Rel *>(
638  section.imagePtr.value()); // Pointer to symbols
639  const int symSecIdx = m_shLink[relSecIdx];
640  const int strSecIdx = m_shLink[symSecIdx];
641  const int numRelocs = section.Size / section.entry_size;
642 
643  // Index 0 is a dummy entry
644  for (int i = 1; i < numRelocs; i++) {
645  const Address val = Address(elfRead4(&m_relocSection[i].r_offset));
646  const int symIndex = elfRead4(&m_relocSection[i].r_info) >> 8;
647  const int flags = elfRead4(&m_relocSection[i].r_info) & 0xFF;
648 
649  if (flags == R_386_32) {
650  // Lookup the value of the symbol table entry
651  Address symbolAddr = Address(elfRead4(&m_symbolSection[symIndex].st_value));
652 
653  if (m_symbolSection[symIndex].st_info & STT_SECTION) {
654  const Elf32_Half shndx = elfRead2(&m_symbolSection[symIndex].st_shndx);
655  if (Util::inRange(shndx, 0, m_elfSections.size())) {
656  symbolAddr = m_elfSections[shndx].SourceAddr;
657  }
658  }
659 
660  // Overwrite the relocation value... ?
661  m_binaryFile->getImage()->writeNative4(val, symbolAddr.value());
662  continue;
663  }
664 
665  if ((flags & R_386_PC32) == 0) {
666  continue;
667  }
668 
669  if (symIndex == 0) { /* Silly symbols with no names */
670  continue;
671  }
672 
673  QString symbolName = getStrPtr(strSecIdx, elfRead4(&m_symbolSection[symIndex].st_name));
674 
675  // Hack off the "@@GLIBC_2.0" of Linux, if present
676  symbolName = symbolName.left(symbolName.indexOf("@@"));
677 
678  const BinarySymbol *symbol = m_symbols->findSymbolByName(symbolName);
679  // Add new extern
680  Address location = symbol ? symbol->getLocation() : m_nextExtern;
681 
682  if (nullptr == symbol) {
683  m_symbols->createSymbol(m_nextExtern, symbolName);
684  m_nextExtern += 4;
685  }
686 
687  m_binaryFile->getImage()->writeNative4(val, (location - val - 4).value());
688  }

848 void ElfBinaryLoader::applyRelocations()
849 {
850  int nextFakeLibAddr = -2; // See R_386_PC32 below; -1 sometimes used for main
851 
852  if (m_loadedImage == nullptr) {
853  return; // No file loaded
854  }
855 
856  const Elf32_Half machine = elfRead2(&m_elfHeader->e_machine);
857  const Elf32_Half e_type = elfRead2(&m_elfHeader->e_type);
858 
859  for (size_t i = 1; i < m_elfSections.size(); ++i) {
860  const SectionParam &ps(m_elfSections[i]);
861  if (ps.sectionType == SHT_RELA) {
862  const Elf32_Rela *relaEntries = reinterpret_cast<const Elf32_Rela *>(
863  ps.imagePtr.value());
864  const DWord numEntries = ps.Size / sizeof(Elf32_Rela);
865 
866  if (relaEntries == nullptr) {
867  LOG_WARN("Cannot read relocation entries from invalid section %1", i);
868  continue;
869  }
870  else if (ps.Size % sizeof(Elf32_Rela) != 0) {
871  LOG_WARN("Cannot read relocation entries from section %1 with invalid size %2 "
872  "(must be divisible by %3)",
873  i, ps.Size, sizeof(Elf32_Rela));
874  continue;
875  }
876 
877  switch (machine) {
878  case EM_SPARC:
879  // NOTE: the r_offset is different for .o files (E_REL in the e_type header field)
880  // than for exe's and shared objects!
881  for (DWord u = 0; u < numEntries; u++) {
882  Elf32_Byte relType = ELF32_R_TYPE(elfRead4(&relaEntries[u].r_info));
883  // Elf32_Word symTabIndex = ELF32_R_SYM(elfRead4(&relaEntries[u].r_info));
884 
885  switch (relType) {
886  case R_SPARC_NONE: // just ignore (common)
887  break;
888 
889  // TODO These relocation types need to be implemented.
890  case R_SPARC_HI22:
891  case R_SPARC_LO10:
892  case R_SPARC_COPY:
893  case R_SPARC_GLOB_DAT:
894  case R_SPARC_JMP_SLOT:
895  default: LOG_WARN("Unhandled SPARC relocation type %1", relType); break;
896  }
897  }
898  break;
899 
900  default: LOG_WARN("Unhandled relocation!"); break;
901  }
902  }
903  else if (ps.sectionType == SHT_REL) {
904  // A section such as .rel.dyn or .rel.plt (without an addend field).
905  // Each entry has 2 words: r_offet and r_info. The r_offset is just the offset from
906  // the beginning of the section (section given by the section header's sh_info) to
907  // the word to be modified. r_info has the type in the bottom byte, and a symbol
908  // table index in the top 3 bytes. A symbol table offset of 0 (STN_UNDEF) means use
909  // value 0. The symbol table involved comes from the section header's sh_link field.
910 
911  // NOTE: the r_offset is different for .o files (E_REL in the e_type header field)
912  // than for exe's and shared objects!
913  Address destNatOrigin = Address::ZERO;
914  HostAddress destHostOrigin = HostAddress::ZERO;
915 
916  if (e_type == ET_REL) {
917  Elf32_Word destSection = m_shInfo[i];
918  if (!Util::inRange(destSection, 0UL, m_elfSections.size())) {
919  continue;
920  }
921  destNatOrigin = m_elfSections[destSection].SourceAddr;
922  destHostOrigin = m_elfSections[destSection].imagePtr;
923  }
924 
925  // Section index for the associated symbol table
926  const uint32 symSectionIdx = m_shLink[i];
927  if (!Util::inRange(symSectionIdx, 0UL, m_elfSections.size())) {
928  continue;
929  }
930 
931  // Section index for the string section assoc with this
932  const uint32 strSectionIdx = m_shLink[symSectionIdx];
933  if (!Util::inRange(strSectionIdx, 0UL, m_elfSections.size())) {
934  continue;
935  }
936 
937  const char *strSection = reinterpret_cast<const char *>(
938  m_elfSections[strSectionIdx].imagePtr.value());
939  const Elf32_Sym *assocSymbols = symSectionIdx != 0
940  ? reinterpret_cast<const Elf32_Sym *>(
941  m_elfSections[symSectionIdx].imagePtr.value())
942  : nullptr;
943 
944  const Elf32_Rel *relEntries = reinterpret_cast<const Elf32_Rel *>(ps.imagePtr.value());
945  const DWord numEntries = ps.Size / sizeof(Elf32_Rel);
946  if (ps.Size % sizeof(Elf32_Rel) != 0) {
947  LOG_WARN("Invalid size %1 of relocation section %2 (must be divisible by %3)",
948  ps.Size, i, sizeof(Elf32_Rel));
949  continue;
950  }
951 
952  for (unsigned u = 0; u < numEntries; u++) {
953  const Elf32_Addr r_offset = elfRead4(&relEntries[u].r_offset);
954  const Elf32_Byte relType = ELF32_R_TYPE(elfRead4(&relEntries[u].r_info));
955  const Elf32_Word symbolIdx = ELF32_R_SYM(elfRead4(&relEntries[u].r_info));
956 
957  DWord *relocDestination; // Pointer to the word to be relocated
958 
959  if (e_type == ET_REL) {
960  if (!Util::inRange(r_offset, 0UL, m_loadedImageSize)) {
961  LOG_WARN("Not loading symbol number %1 due to invalid offset %2", u,
962  r_offset);
963  continue;
964  }
965  relocDestination = reinterpret_cast<DWord *>(
966  (destHostOrigin + r_offset).value());
967  }
968  else {
969  const BinarySection *destSec = m_binaryFile->getImage()->getSectionByAddr(
970  Address(r_offset));
971  if (!destSec) {
972  LOG_WARN("Not loading symbol number %1 due to invalid offset %2", u,
973  r_offset);
974  continue;
975  }
976  relocDestination = reinterpret_cast<DWord *>(
977  (destSec->getHostAddr() - destSec->getSourceAddr() + r_offset).value());
978  destNatOrigin = Address::ZERO;
979  }
980 
981  Address A = Address(elfRead4(relocDestination));
982  Address P = destNatOrigin + r_offset;
983  Address S = assocSymbols != nullptr
984  ? Address(elfRead4(&assocSymbols[symbolIdx].st_value))
985  : Address::ZERO;
986 
987  if (e_type == ET_REL && assocSymbols != nullptr) {
988  const Elf32_Half sectionIdx = elfRead2(&assocSymbols[symbolIdx].st_shndx);
989 
990  if (sectionIdx < m_elfSections.size()) {
991  S += m_elfSections[sectionIdx].SourceAddr;
992  }
993  }
994 
995  switch (machine) {
996  case EM_386:
997  switch (relType) {
998  case R_386_NONE: break;
999 
1000  case R_386_32: // S + A
1001  elfWrite4(relocDestination, (S + A).value());
1002  break;
1003 
1004  case R_386_PC32: // S + A - P
1005  if (assocSymbols == nullptr) {
1006  break; // cannot do relocation for this entry
1007  }
1008 
1009  if (ELF32_ST_TYPE(assocSymbols[symbolIdx].st_info) == STT_SECTION) {
1010  S = Address::ZERO;
1011  const Elf32_Half sectionIdx = elfRead2(
1012  &assocSymbols[symbolIdx].st_shndx);
1013 
1014  if (sectionIdx < m_elfSections.size()) {
1015  S += m_elfSections[sectionIdx].SourceAddr;
1016  }
1017  }
1018  else {
1019  S = Address(elfRead4(&assocSymbols[symbolIdx].st_value));
1020 
1021  if (S.isZero()) {
1022  // This means that the symbol doesn't exist in this module, and is
1023  // not accessed through the PLT, i.e. it will be statically linked,
1024  // e.g. strcmp. We have the name of the symbol right here in the
1025  // symbol table entry, but the only way to communicate with the
1026  // loader is through the target address of the call. So we use some
1027  // very improbable addresses (e.g. -1, -2, etc) and give them
1028  // entries in the symbol table
1029  DWord nameOffset = elfRead4(reinterpret_cast<const DWord *>(
1030  &assocSymbols[symbolIdx].st_name));
1031  const char *symbolName = strSection + nameOffset;
1032 
1033  // Allocate a new fake address
1034  S = Address((nextFakeLibAddr--) & Address::getSourceMask());
1035  BinarySymbol *newFunction = m_symbols->createSymbol(S, symbolName);
1036  newFunction->setAttribute("Function", true);
1037  newFunction->setAttribute("Imported", true);
1038  }
1039  else if (e_type == ET_REL) {
1040  const SWord sectionIdx = elfRead2(
1041  &assocSymbols[symbolIdx].st_shndx);
1042 
1043  if (sectionIdx < m_elfSections.size()) {
1044  S += m_elfSections[sectionIdx].SourceAddr;
1045  }
1046  }
1047  }
1048 
1049  elfWrite4(relocDestination, (S + A - P).value());
1050  break;
1051 
1052  case R_386_GLOB_DAT:
1053  case R_386_JMP_SLOT:
1054  case R_386_RELATIVE:
1055  // No need to do anything with these, if a shared object
1056  break;
1057 
1058  default:
1059  LOG_WARN("Unhandled x86 relocation type %1", static_cast<int>(relType));
1060  }
1061  break;
1062 
1063  default: LOG_WARN("Unhandled relocation!"); break;
1064  }
1065  }
1066  }
1067  }
1068 }

1080  switch (machine) {

1104

1109

1115  Address(r_offset));

1117  relocDestination = destSec->getSourceAddr() + r_offset;
1118  destNatOrigin = Address::ZERO;
1119  }
1120  }
1121 
1122  if (addr == relocDestination) {
1123  return true;
1124  }
1125  }
1126  }
1127  }
1128 
1129  break;
1130 
1131  default: LOG_WARN("Unhandled relocation!"); break; // Not implemented yet
1132  }
1133 
1134  return false;

36 int RTLInstDict::insert(const QString &name, std::list<QString> &params, const RTL &rtl)
37 {
38  QString opcode = name.toUpper();
39 
40  opcode.remove(".");
41 
42  if (m_instructions.find({ opcode, params.size() }) == m_instructions.end()) {
43  std::pair<QString, int> key{ opcode, params.size() };
44  m_instructions.emplace(key, TableEntry(params, rtl));
45  }
46  else {
47  return m_instructions[{ opcode, params.size() }].appendRTL(params, rtl);
48  }
49 
50  return 0;

122 std::unique_ptr<RTL> RTLInstDict::instantiateRTL(const RTL &existingRTL, Address natPC,
123  const std::list<QString> &params,
124  const std::vector<SharedExp> &args)
125 {
126  assert(params.size() == args.size());
127 
128  // Get a deep copy of the template RTL
129  std::unique_ptr<RTL> newList(new RTL(existingRTL));
130  newList->setAddress(natPC);
131 
132  // Iterate through each Statement of the new list of stmts
133  for (Statement *ss : *newList) {
134  // Search for the formals and replace them with the actual arguments
135  auto arg = args.begin();
136 
137  for (QString paramName : params) {
138  /* Simple parameter - just construct the formal to search for */
139  Location param(opParam, Const::get(paramName), nullptr);
140  ss->searchAndReplace(param, *arg);
141  ++arg;
142  }
143  assert(arg == args.end());
144  fixSuccessorForStmt(ss);
145 
146  if (m_verboseOutput) {
147  LOG_MSG(" %1", ss);
148  }
149  }
150 
151  // Perform simplifications, e.g. *1 in Pentium addressing modes
152  for (Statement *s : *newList) {
153  s->simplify();
154  }
155 
156  return newList;
157 }

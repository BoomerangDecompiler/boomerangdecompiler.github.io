90 /**
91  * Converts an operand of an instruction (like %eax + 4*%esi + 0x10)
92  * into a SSL expression that can be recognized by the decompiler.
93  */
94 SharedExp operandToExp(const cs::cs_x86_op &operand)
95 {
96  switch (operand.type) {
97  case cs::X86_OP_MEM: {
98  SharedExp base = (operand.mem.base != cs::X86_REG_INVALID)
99  ? Location::regOf(fixRegNum(operand.mem.base))
100  : Const::get(0);
101  SharedExp index = (operand.mem.index != cs::X86_REG_INVALID)
102  ? Location::regOf(fixRegNum(operand.mem.index))
103  : Const::get(0);
104  SharedExp scale = Const::get(operand.mem.scale);
105 
106  // addrExp = base + index * scale
107  SharedExp addrExp = Binary::get(opPlus, base, Binary::get(opMult, scale, index));
108 
109  // for some reason the address expression is not simplified
110  // if disp == 0
111  if (operand.mem.disp != 0) {
112  const int64_t disp = operand.mem.disp;
113 
114  SharedExp dispExp = Const::get(Address(disp));
115  addrExp = Binary::get(opPlus, addrExp, dispExp);
116  }
117 
118  return Location::memOf(addrExp)->simplify();
119  }
120  case cs::X86_OP_REG: {
121  return Location::regOf(fixRegNum(operand.reg));
122  }
123  case cs::X86_OP_IMM: {
124  return Const::get(Address(operand.imm));
125  }
126  default: LOG_ERROR("Unknown x86 instruction operand type %1", (int)operand.type); break;
127  }
128  return nullptr;
129 }

163 bool CapstoneX86Decoder::decodeInstruction(Address pc, ptrdiff_t delta, DecodeResult &result)
164 {
165  const Byte *instructionData = reinterpret_cast<const Byte *>((HostAddress(delta) + pc).value());
166  size_t size = X86_MAX_INSTRUCTION_LENGTH;
167  uint64 addr = pc.value();
168 
169  result.valid = cs_disasm_iter(m_handle, &instructionData, &size, &addr, m_insn);
170 
171  if (!result.valid) {
172  return false;
173  }
174  else if (m_insn->id == cs::X86_INS_BSF || m_insn->id == cs::X86_INS_BSR) {
175  // special hack to give BSF/BSR the correct semantics since SSL does not support loops yet
176  const bool ok = genBSFR(pc, m_insn, result);
177  return ok;
178  }
179 
180  result.type = ICLASS::NOP; // ICLASS is irrelevant for x86
181  result.numBytes = m_insn->size;
182  result.reDecode = false;
183  result.rtl = createRTLForInstruction(pc, m_insn);
184  result.forceOutEdge = Address::ZERO;
185  result.valid = (result.rtl != nullptr);
186  return true;
187 }

209 std::unique_ptr<RTL> CapstoneX86Decoder::createRTLForInstruction(Address pc,
210  const cs::cs_insn *instruction)
211 {
212  const int numOperands = instruction->detail->x86.op_count;
213  const cs::cs_x86_op *operands = instruction->detail->x86.operands;
214 
215  QString insnID = cs::cs_insn_name(m_handle, instruction->id);
216 
217  switch (instruction->detail->x86.prefix[0]) {
218  case cs::X86_PREFIX_REP: insnID = "REP" + insnID; break;
219  case cs::X86_PREFIX_REPNE: insnID = "REPNE" + insnID; break;
220  }
221 
222  insnID = insnID.toUpper();
223 
224  for (int i = 0; i < numOperands; i++) {
225  // example: ".imm8"
226  QString operandName = "." + operandNames[operands[i].type] +
227  QString::number(operands[i].size * 8);
228 
229  insnID += operandName;
230  }
231 
232  std::unique_ptr<RTL> rtl;
233 
234  // special hack to ignore 'and esp, 0xfffffff0 in startup code
235  if (instruction->id == cs::X86_INS_AND && operands[0].type == cs::X86_OP_REG &&
236  operands[0].reg == cs::X86_REG_ESP && operands[1].type == cs::X86_OP_IMM &&
237  operands[1].imm == 0xFFFFFFF0) {
238  return instantiateRTL(pc, "NOP", 0, nullptr);
239  }
240  else {
241  rtl = instantiateRTL(pc, qPrintable(insnID), numOperands, operands);
242  if (!rtl) {
243  LOG_ERROR("Cannot find semantics for instruction '%1' at address %2, "
244  "treating instruction as NOP",
245  insnID, pc);
246  return instantiateRTL(pc, "NOP", 0, nullptr);
247  }
248  }
249 
250  if (isInstructionInGroup(instruction, cs::CS_GRP_RET) ||
251  isInstructionInGroup(instruction, cs::CS_GRP_IRET)) {
252  rtl->append(new ReturnStatement);
253  }
254  else if (isInstructionInGroup(instruction, cs::CS_GRP_CALL)) {
255  Assign *last = static_cast<Assign *>(rtl->back());
256  SharedExp callDest = last->getRight();
257 
258  if (callDest->isConst() && callDest->access<const Const>()->getAddr() == pc + 5) {
259  // This is a call to the next instruction
260  // Use the standard semantics, except for the last statement
261  // (just updates %pc)
262  rtl->pop_back();
263  // And don't make it a call statement
264  }
265  else {
266  // correct the assignment to %pc to be relative
267  if (!rtl->getStatements().empty() && rtl->getStatements().back()->isAssign()) {
268  Assign *asgn = static_cast<Assign *>(rtl->getStatements().back());
269  if (asgn->getLeft()->isPC() && asgn->getRight()->isConst() &&
270  !asgn->getRight()->isStrConst()) {
271  const Address absoluteAddr = asgn->getRight()->access<Const>()->getAddr();
272  const int delta = (absoluteAddr -
273  Address(instruction->address - instruction->size))
274  .value();
275 
276  asgn->setRight(Binary::get(opPlus, Terminal::get(opPC), Const::get(delta)));
277  }
278  }
279 
280  CallStatement *call = new CallStatement;
281  // Set the destination
282  call->setDest(callDest);
283  rtl->append(call);
284 
285  if (callDest->isConst()) {
286  Function *destProc = m_prog->getOrCreateFunction(
287  callDest->access<Const>()->getAddr());
288 
289  if (destProc == reinterpret_cast<Function *>(-1)) {
290  destProc = nullptr; // In case a deleted Proc
291  }
292 
293  call->setDestProc(destProc);
294  }
295  else {
296  call->setIsComputed(true);
297  }
298  }
299  }
300  else if (isInstructionInGroup(instruction, cs::X86_GRP_JUMP)) {
301  Assign *last = static_cast<Assign *>(rtl->back());
302  assert(last->getLeft()->isPC());
303  SharedExp guard = last->getGuard();
304 
305  const bool isComputedJump = !last->getRight()->isConst();
306 
307  if (guard == nullptr) {
308  if (isComputedJump) {
309  // unconditional computed jump (switch statement)
310  CaseStatement *cs = new CaseStatement();
311  cs->setDest(last->getRight());
312  cs->setIsComputed(true);
313  rtl->pop_back();
314  rtl->append(cs);
315  }
316  else {
317  // unconditional jump
318  GotoStatement *gs = new GotoStatement;
319  gs->setDest(last->getRight());
320  rtl->pop_back();
321  rtl->append(gs);
322  }
323  }
324  else {
325  // conditional jump
326  BranchStatement *branch = new BranchStatement();
327  branch->setDest(last->getRight());
328  branch->setCondExpr(guard);
329  branch->setIsComputed(true);
330 
331  BranchType bt = BranchType::INVALID;
332  switch (instruction->id) {
333  case cs::X86_INS_JE: bt = BranchType::JE; break;
334  case cs::X86_INS_JNE: bt = BranchType::JNE; break;
335  case cs::X86_INS_JL: bt = BranchType::JSL; break; // signed less
336  case cs::X86_INS_JLE: bt = BranchType::JSLE; break;
337  case cs::X86_INS_JGE: bt = BranchType::JSGE; break;
338  case cs::X86_INS_JG: bt = BranchType::JSG; break;
339  case cs::X86_INS_JB: bt = BranchType::JUL; break; // unsigned less
340  case cs::X86_INS_JBE: bt = BranchType::JULE; break;
341  case cs::X86_INS_JAE: bt = BranchType::JUGE; break;
342  case cs::X86_INS_JA: bt = BranchType::JUG; break;
343  case cs::X86_INS_JS: bt = BranchType::JMI; break;
344  case cs::X86_INS_JNS: bt = BranchType::JPOS; break;
345  case cs::X86_INS_JO: bt = BranchType::JOF; break;
346  case cs::X86_INS_JNO: bt = BranchType::JNOF; break;
347  case cs::X86_INS_JP: bt = BranchType::JPAR; break;
348  case cs::X86_INS_JNP: bt = BranchType::JNPAR; break;
349  case cs::X86_INS_JCXZ: bt = BranchType::JE; break;
350  case cs::X86_INS_JECXZ: bt = BranchType::JE; break; // because jmp condition is %ecx = 0
351  case cs::X86_INS_LOOP: bt = BranchType::JNE; break;
352  case cs::X86_INS_LOOPE: bt = BranchType::JE; break;
353  case cs::X86_INS_LOOPNE: bt = BranchType::JNE; break;
354  default: assert(false); break;
355  }
356 
357  branch->setCondType(bt, false);
358  branch->setIsComputed(isComputedJump);
359 
360  // Need to fix up the conditional expression here...
361  // setCondType() assigns the wrong expression for jumps that do not depend on flags
362  switch (instruction->id) {
363  case cs::X86_INS_JCXZ: {
364  branch->setCondExpr(
365  Binary::get(opEquals, Location::regOf(REG_PENT_CX), Const::get(0)));
366  break;
367  }
368  case cs::X86_INS_JECXZ: {
369  branch->setCondExpr(
370  Binary::get(opEquals, Location::regOf(REG_PENT_ECX), Const::get(0)));
371  break;
372  }
373  case cs::X86_INS_LOOP: {
374  // FIXME wrong for 16 bit programs
375  branch->setCondExpr(
376  Binary::get(opNotEqual, Location::regOf(REG_PENT_ECX), Const::get(0)));
377  break;
378  }
379  case cs::X86_INS_LOOPE: {
380  // FIXME wrong for 16 bit programs
381  // clang-format off
382  branch->setCondExpr(Binary::get(opAnd,
383  Binary::get(opNotEqual,
384  Location::regOf(REG_PENT_ECX),
385  Const::get(0)),
386  Binary::get(opEquals,
387  Terminal::get(opZF),
388  Const::get(1))));
389  // clang-format on
390  break;
391  }
392  case cs::X86_INS_LOOPNE: {
393  // FIXME wrong for 16 bit programs
394  // clang-format off
395  branch->setCondExpr(Binary::get(opAnd,
396  Binary::get(opNotEqual,
397  Location::regOf(REG_PENT_ECX),
398  Const::get(0)),
399  Binary::get(opEquals,
400  Terminal::get(opZF),
401  Const::get(0))));
402  // clang-format on
403  break;
404  }
405  default: break;
406  }
407 
408  rtl->pop_back();
409  rtl->append(branch);
410  }
411  }
412  else if (insnID.startsWith("SET")) {
413  BoolAssign *bas = new BoolAssign(8);
414  bas->setCondExpr(static_cast<Assign *>(rtl->front())->getRight()->clone());
415  bas->setLeft(static_cast<Assign *>(rtl->front())->getLeft()->clone());
416 
417  BranchType bt = BranchType::INVALID;
418  switch (instruction->id) {
419  case cs::X86_INS_SETE: bt = BranchType::JE; break;
420  case cs::X86_INS_SETNE: bt = BranchType::JNE; break;
421  case cs::X86_INS_SETL: bt = BranchType::JSL; break; // signed less
422  case cs::X86_INS_SETLE: bt = BranchType::JSLE; break;
423  case cs::X86_INS_SETGE: bt = BranchType::JSGE; break;
424  case cs::X86_INS_SETG: bt = BranchType::JSG; break;
425  case cs::X86_INS_SETB: bt = BranchType::JUL; break; // unsigned less
426  case cs::X86_INS_SETBE: bt = BranchType::JULE; break;
427  case cs::X86_INS_SETAE: bt = BranchType::JUGE; break;
428  case cs::X86_INS_SETA: bt = BranchType::JUG; break;
429  case cs::X86_INS_SETS: bt = BranchType::JMI; break;
430  case cs::X86_INS_SETNS: bt = BranchType::JPOS; break;
431  case cs::X86_INS_SETO: bt = BranchType::JOF; break;
432  case cs::X86_INS_SETNO: bt = BranchType::JNOF; break;
433  case cs::X86_INS_SETP: bt = BranchType::JPAR; break;
434  case cs::X86_INS_SETNP: bt = BranchType::JNPAR; break;
435  default: assert(false); break;
436  }
437 
438  bas->setCondType(bt);
439  if (rtl->size() > 1) {
440  LOG_WARN(
441  "%1 additional statements in RTL for instruction '%2'; results may be inaccurate",
442  rtl->size() - 1, insnID);
443  }
444 
445  rtl->clear();
446  rtl->append(bas);
447  }
448 
449  return rtl;
450 }

453 std::unique_ptr<RTL> CapstoneX86Decoder::instantiateRTL(Address pc, const char *instructionID,
454  int numOperands,
455  const cs::cs_x86_op *operands)
456 {
457  // Take the argument, convert it to upper case and remove any .'s
458  const QString sanitizedName = QString(instructionID).remove(".").toUpper();
459 
460  std::vector<SharedExp> args(numOperands);
461  for (int i = 0; i < numOperands; i++) {
462  args[i] = operandToExp(operands[i]);
463  }
464 
465  if (m_debugMode) {
466  QString argNames;
467  for (int i = 0; i < numOperands; i++) {
468  if (i != 0) {
469  argNames += " ";
470  }
471  argNames += args[i]->toString();
472  }
473 
474  LOG_MSG("Instantiating RTL at %1: %2 %3", pc, instructionID, argNames);
475  }
476 
477  return m_dict.instantiateRTL(sanitizedName, pc, args);
478 }

481 bool CapstoneX86Decoder::genBSFR(Address pc, const cs::cs_insn *instruction, DecodeResult &result)
482 {
483  // Note the horrible hack needed here. We need initialisation code, and an extra branch, so the
484  // %SKIP/%RPT won't work. We need to emit 6 statements, but these need to be in 3 RTLs, since
485  // the destination of a branch has to be to the start of an RTL. So we use a state machine, and
486  // set numBytes to 0 for the first two times. That way, this instruction ends up emitting three
487  // RTLs, each with the semantics we need. Note: we don't use pentium.ssl for these.
488  //
489  // BSFR1:
490  // pc+0: *1* zf := 1
491  // pc+0: goto exit if src = 0
492  // BSFR2:
493  // pc+1: *1* zf := 0
494  // pc+1: dest := 0 if BSF else dest := opsize - 1
495  // BSFR3:
496  // pc+2: dest := dest op 1
497  // pc+2: goto pc+2 if src@[dest:dest]=0
498  // exit:
499  //
500 
501  BranchStatement *b = nullptr;
502  result.rtl = std::unique_ptr<RTL>(new RTL(pc + m_bsfrState));
503 
504  const cs::cs_x86_op &dstOp = instruction->detail->x86.operands[0];
505  const cs::cs_x86_op &srcOp = instruction->detail->x86.operands[1];
506 
507  assert(dstOp.size == srcOp.size);
508  const SharedExp dest = operandToExp(dstOp);
509  const SharedExp src = operandToExp(srcOp);
510  const int size = dstOp.size * 8;
511  const int init = instruction->id == cs::X86_INS_BSF ? 0 : size - 1;
512  const OPER incdec = instruction->id == cs::X86_INS_BSF ? opPlus : opMinus;
513 
514  switch (m_bsfrState) {
515  case 0:
516  result.rtl->append(new Assign(IntegerType::get(1), Terminal::get(opZF), Const::get(1)));
517  b = new BranchStatement;
518  b->setDest(pc + instruction->size);
519  b->setCondType(BranchType::JE);
520  b->setCondExpr(Binary::get(opEquals, src->clone(), Const::get(0)));
521  result.rtl->append(b);
522  break;
523 
524  case 1:
525  result.rtl->append(new Assign(IntegerType::get(1), Terminal::get(opZF), Const::get(0)));
526  result.rtl->append(new Assign(IntegerType::get(size), dest->clone(), Const::get(init)));
527  break;
528 
529  case 2:
530  result.rtl->append(new Assign(IntegerType::get(size), dest->clone(),
531  Binary::get(incdec, dest->clone(), Const::get(1))));
532  b = new BranchStatement;
533  b->setDest(pc + 2);
534  b->setCondType(BranchType::JE);
535  b->setCondExpr(Binary::get(opEquals,
536  Ternary::get(opAt, src->clone(), dest->clone(), dest->clone()),
537  Const::get(0)));
538  result.rtl->append(b);
539  break;
540 
541  default:
542  // Should never happen
543  LOG_FATAL("Unknown BSFR state %1", m_bsfrState);
544  }
545 
546  // Keep numBytes == 0 until the last state, so we re-decode this instruction 3 times
547  if (m_bsfrState != 3 - 1) {
548  // Let the number of bytes be 1. This is important at least for setting the fallthrough
549  // address for the branch (in the first RTL), which should point to the next RTL
550  result.numBytes = 1;
551  result.reDecode = true; // Decode this instuction again
552  }
553  else {
554  result.numBytes = instruction->size;
555  result.reDecode = false;
556  }
557 
558  if (m_debugMode) {
559  LOG_MSG("%1: BS%2%3%4", pc + m_bsfrState, (init == -1 ? "F" : "R"),
560  (size == 32 ? ".od" : ".ow"), m_bsfrState + 1);
561  }
562 
563  if (++m_bsfrState == 3) {
564  m_bsfrState = 0; // Ready for next time
565  }
566 
567  return true;
568 }

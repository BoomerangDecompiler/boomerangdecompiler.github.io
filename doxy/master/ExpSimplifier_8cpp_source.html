25 SharedExp ExpSimplifier::postModify(const std::shared_ptr<Unary> &exp)
26 {
27  bool &changed = m_modified;
28 
29  if (exp->getOper() == opBitNot || exp->getOper() == opLNot) {
30  OPER oper = exp->getSubExp1()->getOper();
31 
32  switch (oper) {
33  case opEquals: oper = opNotEqual; break;
34  case opNotEqual: oper = opEquals; break;
35  case opLess: oper = opGtrEq; break;
36  case opGtr: oper = opLessEq; break;
37  case opLessEq: oper = opGtr; break;
38  case opGtrEq: oper = opLess; break;
39  case opLessUns: oper = opGtrEqUns; break;
40  case opGtrUns: oper = opLessEqUns; break;
41  case opLessEqUns: oper = opGtrUns; break;
42  case opGtrEqUns: oper = opLessUns; break;
43  default: break;
44  }
45 
46  if (oper != exp->getSubExp1()->getOper()) {
47  changed = true;
48  exp->getSubExp1()->setOper(oper);
49  return exp->getSubExp1();
50  }
51  }
52 
53  if (exp->getOper() == opNeg || exp->getOper() == opBitNot || exp->getOper() == opLNot) {
54  if (exp->getSubExp1()->isIntConst()) {
55  // -k, ~k, or !k
56  int k = exp->access<Const, 1>()->getInt();
57 
58  switch (exp->getOper()) {
59  case opNeg: k = -k; break;
60  case opBitNot: k = ~k; break;
61  case opLNot: k = !k; break;
62  default: break;
63  }
64 
65  changed = true;
66  exp->access<Const, 1>()->setInt(k);
67  return exp->getSubExp1();
68  }
69  else if (exp->getOper() == exp->getSubExp1()->getOper()) {
70  return exp->access<Exp, 1, 1>();
71  }
72  }
73 
74  if ((exp->isMemOf() && exp->getSubExp1()->isAddrOf()) ||
75  (exp->isAddrOf() && exp->getSubExp1()->isMemOf())) {
76  changed = true;
77  return exp->getSubExp1()->getSubExp1();
78  }
79 
80  // Simplify e.g. ~(x comp y) -> !(x comp y)
81  const OPER myOper = exp->getOper();
82  if (myOper == opBitNot && exp->getSubExp1()->isLogExp()) {
83  changed = true;
84  exp->setOper(opLNot);
85  return exp;
86  }
87 
88  // if still not simplified, try De Morgan's laws
89  const OPER subOper = exp->getSubExp1()->getOper();
90 
91  if (myOper == opBitNot && (subOper == opBitAnd || subOper == opBitOr)) {
92  changed = true;
93  if (subOper == opBitAnd) {
94  return Binary::get(opBitOr, Unary::get(opBitNot, exp->access<Exp, 1, 1>()),
95  Unary::get(opBitNot, exp->access<Exp, 1, 2>()));
96  }
97  else {
98  return Binary::get(opBitAnd, Unary::get(opBitNot, exp->access<Exp, 1, 1>()),
99  Unary::get(opBitNot, exp->access<Exp, 1, 2>()));
100  }
101  }
102  else if (myOper == opLNot && (subOper == opAnd || subOper == opOr)) {
103  changed = true;
104  if (subOper == opAnd) {
105  return Binary::get(opOr, Unary::get(opLNot, exp->access<Exp, 1, 1>()),
106  Unary::get(opLNot, exp->access<Exp, 1, 2>()));
107  }
108  else {
109  return Binary::get(opAnd, Unary::get(opLNot, exp->access<Exp, 1, 1>()),
110  Unary::get(opLNot, exp->access<Exp, 1, 2>()));
111  }
112  }
113 
114  return exp;
115 }

118 SharedExp ExpSimplifier::postModify(const std::shared_ptr<Binary> &exp)
119 {
120  bool &changed = m_modified;
121 
122  OPER opSub1 = exp->getSubExp1()->getOper();
123  OPER opSub2 = exp->getSubExp2()->getOper();
124 
125  if (opSub1 == opIntConst && opSub2 == opIntConst) {
126  const int lhs = exp->access<Const, 1>()->getInt();
127  const int rhs = exp->access<Const, 2>()->getInt();
128 
129  switch (exp->getOper()) {
130  case opPlus: changed = true; return Const::get(lhs + rhs);
131  case opMinus: changed = true; return Const::get(lhs - rhs);
132  case opMults: changed = true; return Const::get(lhs * rhs);
133  case opShL: changed = true; return Const::get((rhs < 32) ? lhs << rhs : 0);
134  case opShR: changed = true; return Const::get((rhs < 32) ? lhs >> rhs : 0);
135 
136  case opBitAnd: changed = true; return Const::get(lhs & rhs);
137  case opBitOr: changed = true; return Const::get(lhs | rhs);
138  case opBitXor: changed = true; return Const::get(lhs ^ rhs);
139  case opEquals: changed = true; return Const::get(lhs == rhs);
140  case opNotEqual: changed = true; return Const::get(lhs != rhs);
141  case opLess: changed = true; return Const::get(lhs < rhs);
142  case opGtr: changed = true; return Const::get(lhs > rhs);
143  case opLessEq: changed = true; return Const::get(lhs <= rhs);
144  case opGtrEq: changed = true; return Const::get(lhs >= rhs);
145  case opMult: changed = true; return Const::get((int)((uint32)lhs * (uint32)rhs));
146  case opLessUns: changed = true; return Const::get((uint32)lhs < (uint32)rhs);
147  case opGtrUns: changed = true; return Const::get((uint32)lhs > (uint32)rhs);
148  case opLessEqUns: changed = true; return Const::get((uint32)lhs <= (uint32)rhs);
149  case opGtrEqUns: changed = true; return Const::get((uint32)lhs >= (uint32)rhs);
150 
151  case opShRA: {
152  if (rhs == 0) {
153  changed = true;
154  return Const::get(lhs);
155  }
156  else if (rhs >= 32) {
157  changed = true;
158  return Const::get((lhs < 0) ? -1 : 0);
159  }
160  else {
161  changed = true;
162  return Const::get(
163  (int)(lhs >> rhs | (lhs < 0 ? ~Util::getLowerBitMask(32 - rhs) : 0)));
164  }
165  }
166 
167  case opDivs:
168  if (rhs != 0) {
169  changed = true;
170  return Const::get(lhs / rhs);
171  }
172  break;
173 
174  case opMods:
175  if (rhs != 0) {
176  changed = true;
177  return Const::get(lhs % rhs);
178  }
179  break;
180 
181  case opDiv:
182  if (rhs != 0) {
183  changed = true;
184  return Const::get((int)((uint32)lhs / (uint32)rhs));
185  }
186  break;
187 
188  case opMod:
189  if (rhs != 0) {
190  changed = true;
191  return Const::get((int)((uint32)lhs % (uint32)rhs));
192  }
193  break;
194 
195  default: break;
196  }
197  }
198 
199  if ((exp->getOper() == opBitXor || exp->getOper() == opMinus) &&
200  *exp->getSubExp1() == *exp->getSubExp2()) {
201  // x ^ x or x - x: result is zero
202  changed = true;
203  return Const::get(0);
204  }
205 
206  if ((exp->getOper() == opBitOr || exp->getOper() == opBitAnd) &&
207  *exp->getSubExp1() == *exp->getSubExp2()) {
208  // x | x or x & x: result is x
209  changed = true;
210  return exp->getSubExp1();
211  }
212 
213  if (exp->getOper() == opEquals && *exp->getSubExp1() == *exp->getSubExp2()) {
214  // x == x: result is true
215  changed = true;
216  return std::make_shared<Terminal>(opTrue);
217  }
218  else if (exp->getOper() == opNotEqual && *exp->getSubExp1() == *exp->getSubExp2()) {
219  // x != x: result is false
220  changed = true;
221  return std::make_shared<Terminal>(opFalse);
222  }
223 
224  // Might want to commute to put an integer constant on the RHS
225  // Later simplifications can rely on this (add other ops as necessary)
226  if (opSub1 == opIntConst &&
227  (exp->getOper() == opPlus || exp->getOper() == opMult || exp->getOper() == opMults ||
228  exp->getOper() == opBitOr || exp->getOper() == opBitAnd || exp->getOper() == opEquals ||
229  exp->getOper() == opNotEqual)) {
230  exp->commute();
231  // Swap opSub1 and opSub2 as well
232  std::swap(opSub1, opSub2);
233  // This is not counted as a modification
234  }
235 
236  // Similarly for boolean constants
237  if (exp->getSubExp1()->isBoolConst() && !exp->getSubExp2()->isBoolConst() &&
238  (exp->isOr() || exp->isAnd())) {
239  exp->commute();
240  // Swap opSub1 and opSub2 as well
241  std::swap(opSub1, opSub2);
242  // This is not counted as a modification
243  }
244 
245  // Similarly for adding stuff to the addresses of globals
246  if (exp->getOper() == opPlus && exp->access<Exp, 2>()->isAddrOf() &&
247  exp->access<Exp, 2, 1>()->isSubscript() && exp->access<Exp, 2, 1, 1>()->isGlobal()) {
248  exp->commute();
249  // Swap opSub1 and opSub2 as well
250  std::swap(opSub1, opSub2);
251  // This is not counted as a modification
252  }
253 
254  // check for (x + a) + b where a and b are constants, becomes x + a+b
255  if (exp->getOper() == opPlus && opSub1 == opPlus && opSub2 == opIntConst &&
256  exp->getSubExp1()->getSubExp2()->isIntConst()) {
257  const int n = exp->access<Const, 2>()->getInt();
258  exp->getSubExp1()->setOper(opPlus);
259  exp->access<Const, 1, 2>()->setInt(exp->access<Const, 1, 2>()->getInt() + n);
260  changed = true;
261  return exp->getSubExp1();
262  }
263 
264  // check for (x - a) + b where a and b are constants, becomes x + -a+b
265  if (exp->getOper() == opPlus && opSub1 == opMinus && opSub2 == opIntConst &&
266  exp->access<Exp, 1, 2>()->isIntConst()) {
267  const int n = exp->access<Const, 2>()->getInt();
268  exp->getSubExp1()->setOper(opPlus);
269  exp->access<Const, 1, 2>()->setInt(-exp->access<Const, 1, 2>()->getInt() + n);
270  changed = true;
271  return exp->getSubExp1();
272  }
273 
274  SharedExp res = exp->shared_from_this();
275 
276  // check for (x * k) - x, becomes x * (k-1)
277  // same with +
278  if ((exp->getOper() == opMinus || exp->getOper() == opPlus) &&
279  (opSub1 == opMults || opSub1 == opMult) &&
280  *exp->getSubExp2() == *exp->getSubExp1()->getSubExp1()) {
281  res = res->getSubExp1();
282  res->setSubExp2(Binary::get(exp->getOper(), res->getSubExp2(), Const::get(1)));
283  changed = true;
284  return res;
285  }
286 
287  // check for x + (x * k), becomes x * (k+1)
288  if (exp->getOper() == opPlus && (opSub2 == opMults || opSub2 == opMult) &&
289  *exp->getSubExp1() == *exp->getSubExp2()->getSubExp1()) {
290  res = res->getSubExp2();
291  res->setSubExp2(Binary::get(opPlus, res->getSubExp2(), Const::get(1)));
292  changed = true;
293  return res;
294  }
295 
296  // Turn a + -K into a - K (K is int const > 0)
297  // Also a - -K into a + K (K is int const > 0)
298  // Does not count as a change
299  if ((exp->getOper() == opPlus || exp->getOper() == opMinus) && opSub2 == opIntConst &&
300  exp->access<Const, 2>()->getInt() < 0) {
301  exp->access<Const, 2>()->setInt(-exp->access<Const, 2>()->getInt());
302  exp->setOper(exp->getOper() == opPlus ? opMinus : opPlus);
303  }
304 
305  // Check for exp + 0 or exp - 0 or exp | 0
306  if ((exp->getOper() == opPlus || exp->getOper() == opMinus || exp->getOper() == opBitOr) &&
307  opSub2 == opIntConst && exp->access<Const, 2>()->getInt() == 0) {
308  changed = true;
309  return exp->getSubExp1();
310  }
311 
312  // Check for exp or false
313  if (exp->getOper() == opOr && exp->getSubExp2()->isFalse()) {
314  changed = true;
315  return exp->getSubExp1();
316  }
317 
318  // Check for SharedExp * 0 or exp & 0
319  if ((exp->getOper() == opMult || exp->getOper() == opMults || exp->getOper() == opBitAnd) &&
320  opSub2 == opIntConst && exp->access<Const, 2>()->getInt() == 0) {
321  changed = true;
322  return Const::get(0);
323  }
324 
325  // Check for exp and false
326  if (exp->getOper() == opAnd && exp->getSubExp2()->isFalse()) {
327  changed = true;
328  return Terminal::get(opFalse);
329  }
330 
331  // Check for SharedExp * 1
332  if ((exp->getOper() == opMult || exp->getOper() == opMults) && opSub2 == opIntConst &&
333  exp->access<Const, 2>()->getInt() == 1) {
334  changed = true;
335  return res->getSubExp1();
336  }
337 
338  // x xor 0 = x
339  if (exp->getOper() == opBitXor && opSub2 == opIntConst &&
340  exp->access<Const, 2>()->getInt() == 0) {
341  changed = true;
342  return res->getSubExp1();
343  }
344 
345  // Check for SharedExp (a * x) / x -> a
346  if ((exp->getOper() == opDiv || exp->getOper() == opDivs) &&
347  (opSub1 == opMult || opSub1 == opMults) &&
348  *exp->getSubExp2() == *exp->getSubExp1()->getSubExp2()) {
349  changed = true;
350  return res->getSubExp1()->getSubExp1();
351  }
352 
353  // Check for exp / 1, becomes exp
354  if ((exp->getOper() == opDiv || exp->getOper() == opDivs) && opSub2 == opIntConst &&
355  exp->access<Const, 2>()->getInt() == 1) {
356  changed = true;
357  return res->getSubExp1();
358  }
359 
360  // Check for exp % 1, becomes 0
361  if ((exp->getOper() == opMod || exp->getOper() == opMods) && opSub2 == opIntConst &&
362  exp->access<Const, 2>()->getInt() == 1) {
363  changed = true;
364  return Const::get(0);
365  }
366 
367  // Check for SharedExp (a * x) % x, becomes 0
368  if ((exp->getOper() == opMod || exp->getOper() == opMods) &&
369  (opSub1 == opMult || opSub1 == opMults) &&
370  (*exp->getSubExp2() == *exp->getSubExp1()->getSubExp2() ||
371  *exp->getSubExp2() == *exp->getSubExp1()->getSubExp1())) {
372  changed = true;
373  return Const::get(0);
374  }
375 
376  // Check for SharedExp x % x, becomes 0
377  if ((exp->getOper() == opMod || exp->getOper() == opMods) &&
378  *exp->getSubExp2() == *exp->getSubExp1()) {
379  changed = true;
380  return Const::get(0);
381  }
382 
383  // Check for exp AND -1 (bitwise AND)
384  if (exp->getOper() == opBitAnd && opSub2 == opIntConst &&
385  exp->access<Const, 2>()->getInt() == -1) {
386  changed = true;
387  return exp->getSubExp1();
388  }
389 
390  // Check for exp OR -1 (bitwise OR)
391  if (exp->getOper() == opBitOr && opSub2 == opIntConst &&
392  exp->access<Const, 2>()->getInt() == -1) {
393  changed = true;
394  return exp->getSubExp2();
395  }
396 
397  // Check for exp AND TRUE (logical AND)
398  if (exp->getOper() == opAnd &&
399  // Is the below really needed?
400  ((opSub2 == opIntConst && exp->access<Const, 2>()->getInt() != 0) ||
401  exp->getSubExp2()->isTrue())) {
402  changed = true;
403  return res->getSubExp1();
404  }
405 
406  // Check for exp OR TRUE (logical OR)
407  if (exp->getOper() == opOr &&
408  ((opSub2 == opIntConst && exp->access<Const, 2>()->getInt() != 0) ||
409  exp->getSubExp2()->isTrue())) {
410  changed = true;
411  return Terminal::get(opTrue);
412  }
413 
414  // Check for [exp] << k where k is a positive integer const
415  if (exp->getOper() == opShL && opSub2 == opIntConst) {
416  const int k = exp->access<Const, 2>()->getInt();
417 
418  if (Util::inRange(k, 0, 4)) { // do not express e.g. a << 4 as multiplication
419  exp->setOper(opMult);
420  exp->access<Const, 2>()->setInt(1 << k);
421  changed = true;
422  return exp;
423  }
424  }
425 
426  // Check for -x compare y, becomes x compare -y
427  if (exp->isComparison() && opSub1 == opNeg) {
428  exp->setSubExp1(exp->access<Exp, 1, 1>());
429  exp->setSubExp2(Unary::get(opNeg, exp->getSubExp2()));
430  changed = true;
431  return exp;
432  }
433 
434  // Check for (x + y) compare 0, becomes x compare -y
435  if (exp->isComparison() && opSub2 == opIntConst && exp->access<Const, 2>()->getInt() == 0) {
436  if (opSub1 == opPlus) {
437  exp->setSubExp2(Unary::get(opNeg, exp->access<Exp, 1, 2>()));
438  exp->setSubExp1(exp->access<Exp, 1, 1>());
439  changed = true;
440  return exp;
441  }
442  else if (opSub1 == opMinus) {
443  exp->setSubExp2(exp->access<Exp, 1, 2>());
444  exp->setSubExp1(exp->access<Exp, 1, 1>());
445  changed = true;
446  return exp;
447  }
448  }
449 
450  // Check for 0 <=u x, or for x >=u 0, becomes true
451  if ((exp->getOper() == opLessEqUns && exp->getSubExp1()->isIntConst() &&
452  exp->access<Const, 1>()->getInt() == 0) ||
453  (exp->getOper() == opGtrEqUns && exp->getSubExp2()->isIntConst() &&
454  exp->access<Const, 2>()->getInt() == 0)) {
455  changed = true;
456  return Const::get(1);
457  }
458 
459  // Check for 0 <u x, or for x >u 0, becomes x != 0
460  if ((exp->getOper() == opLessUns && exp->getSubExp1()->isIntConst() &&
461  exp->access<Const, 1>()->getInt() == 0) ||
462  (exp->getOper() == opGtrUns && exp->getSubExp2()->isIntConst() &&
463  exp->access<Const, 2>()->getInt() == 0)) {
464  changed = true;
465  exp->setOper(opNotEqual);
466  return exp;
467  }
468 
469  // Check for (x == y) == 1, becomes x == y
470  // Check for (x == y) == 0, becomes x != y
471  if (exp->getOper() == opEquals && opSub1 == opEquals && opSub2 == opIntConst) {
472  const int rightConst = exp->access<Const, 2>()->getInt();
473  changed = true;
474 
475  switch (rightConst) {
476  case 0: exp->getSubExp1()->setOper(opNotEqual); return exp->getSubExp1();
477 
478  case 1: return exp->getSubExp1();
479 
480  default: return Terminal::get(opFalse);
481  }
482  }
483 
484  // Check for (x == y) != 0, becomes x == y
485  // Check for (x == y) != 1, becomes x != y
486  if (exp->getOper() == opNotEqual && opSub1 == opEquals && opSub2 == opIntConst) {
487  const int rightConst = exp->access<Const, 2>()->getInt();
488  changed = true;
489 
490  switch (rightConst) {
491  case 0: return exp->getSubExp1();
492 
493  case 1: exp->getSubExp1()->setOper(opNotEqual); return exp->getSubExp1();
494 
495  default: return Terminal::get(opTrue);
496  }
497  }
498 
499  // Check for (x > y) == 0, becomes x <= y
500  if (exp->getOper() == opEquals && exp->getSubExp1()->isComparison() && opSub2 == opIntConst &&
501  exp->access<Const, 2>()->getInt() == 0) {
502  changed = true;
503  return Unary::get(opLNot, exp->getSubExp1());
504  }
505 
506  auto b1 = std::dynamic_pointer_cast<Binary>(exp->getSubExp1());
507  auto b2 = std::dynamic_pointer_cast<Binary>(exp->getSubExp2());
508 
509  if ((exp->getOper() == opOr) && (opSub2 == opEquals) &&
510  ((opSub1 == opGtrEq) || (opSub1 == opLessEq) || (opSub1 == opGtrEqUns) ||
511  (opSub1 == opLessEqUns)) &&
512  (((*b1->getSubExp1() == *b2->getSubExp1()) && (*b1->getSubExp2() == *b2->getSubExp2())) ||
513  ((*b1->getSubExp1() == *b2->getSubExp2()) && (*b1->getSubExp2() == *b2->getSubExp1())))) {
514  res = res->getSubExp1();
515  changed = true;
516  return res;
517  }
518 
519  // Check for (x < y) || (x == y), becomes x <= y
520  // Check for (x <= y) || (x == y), becomes x <= y
521  if (exp->isOr() && exp->getSubExp1()->isComparison() && exp->getSubExp2()->isComparison() &&
522  (exp->getSubExp1()->isEquality() || exp->getSubExp2()->isEquality()) &&
523  *exp->access<Exp, 1, 1>() == *exp->access<Exp, 2, 1>() && // x on left == x on right
524  *exp->access<Exp, 1, 2>() == *exp->access<Exp, 2, 2>()) { // y on left == y on right
525  OPER otherOper = exp->getSubExp1()->isEquality() ? exp->getSubExp2()->getOper()
526  : exp->getSubExp1()->getOper();
527 
528  switch (otherOper) {
529  case opGtr:
530  case opGtrEq: otherOper = opGtrEq; break;
531  case opGtrUns:
532  case opGtrEqUns: otherOper = opGtrEqUns; break;
533  case opLess:
534  case opLessEq: otherOper = opLessEq; break;
535  case opLessUns:
536  case opLessEqUns: otherOper = opLessEqUns; break;
537  case opEquals:
538  // x == y || x == y -> x == y
539  changed = true;
540  return exp->getSubExp1();
541  case opNotEqual:
542  // x == y || x != y -> true
543  changed = true;
544  return Terminal::get(opTrue);
545  default: break;
546  }
547 
548  changed = true;
549  exp->getSubExp1()->setOper(otherOper);
550  return exp->getSubExp1();
551  }
552 
553  // Check for (x compare y) || (x != y), becomes x compare y
554  // Note: Case (x == y) || (x != y) handled above
555  if ((exp->isOr() || exp->getOper() == opBitOr) && exp->getSubExp1()->isComparison() &&
556  exp->getSubExp2()->isComparison() &&
557  (exp->getSubExp1()->isNotEquality() || exp->getSubExp2()->isNotEquality()) &&
558  *exp->access<Exp, 1, 1>() == *exp->access<Exp, 2, 1>() && // x on left == x on right
559  *exp->access<Exp, 1, 2>() == *exp->access<Exp, 2, 2>()) { // y on left == y on right
560  if (exp->getSubExp1()->isNotEquality()) {
561  changed = true;
562  return exp->getSubExp2();
563  }
564  else {
565  changed = true;
566  return exp->getSubExp1();
567  }
568  }

570  // For (a || b) or (a && b) recurse on a and b
571  if ((exp->getOper() == opOr) || (exp->getOper() == opAnd)) {
572  exp->refSubExp1() = exp->getSubExp1()->acceptModifier(this);
573  exp->refSubExp2() = exp->getSubExp2()->acceptModifier(this);

580  }

584  exp->access<Exp, 1, 2>()->isIntConst()) {

586  const int m = exp->access<const Const, 2>()->getInt();

597  changed = true;
598  return res;
599  }

602  // by c becomes: (x * a/c) + (y * b/c)
603  if (exp->getOper() == opDiv && exp->getSubExp1()->getOper() == opPlus &&
604  exp->getSubExp2()->isIntConst()) {

607

611  const int b = rightOfPlus->access<Const, 2>()->getInt();
612  const int c = exp->access<Const, 2>()->getInt();
613

615  changed = true;
616  leftOfPlus->access<Const, 2>()->setInt(a / c);
617  rightOfPlus->access<Const, 2>()->setInt(b / c);
618 
619  return exp->getSubExp1();

622  }

625  // becomes: (y * b) % c if a divides evenly by c
626  // becomes: (x * a) % c if b divides evenly by c
627  // becomes: 0 if both a and b divide evenly by c
628  if (exp->getOper() == opMod && exp->getSubExp1()->getOper() == opPlus &&
629  exp->getSubExp2()->isIntConst()) {
630  SharedExp leftOfPlus = exp->getSubExp1()->getSubExp1();
631  SharedExp rightOfPlus = exp->getSubExp1()->getSubExp2();
632 
633  if (leftOfPlus->getOper() == opMult && rightOfPlus->getOper() == opMult &&
634  leftOfPlus->getSubExp2()->isIntConst() && rightOfPlus->getSubExp2()->isIntConst()) {
635  const int a = leftOfPlus->access<Const, 2>()->getInt();
636  const int b = rightOfPlus->access<Const, 2>()->getInt();
637  const int c = exp->access<Const, 2>()->getInt();
638 
639  if (c != 0) {
640  if ((a % c == 0) && (b % c == 0)) {
641  changed = true;
642  return Const::get(0);
643  }
644  if ((a % c) == 0) {
645  changed = true;
646  return Binary::get(opMod, rightOfPlus, Const::get(c));
647  }
648  if ((b % c) == 0) {
649  changed = true;
650  return Binary::get(opMod, leftOfPlus, Const::get(c));
651  }
652  }
653  }
654  }

660 SharedExp ExpSimplifier::postModify(const std::shared_ptr<Ternary> &exp)
661 {
662  bool &changed = m_modified;
663 
664  // p ? 1 : 0 -> p != 0
665  // p ? 0 : 1 -> p == 0
666  if (exp->getOper() == opTern && exp->getSubExp2()->isIntConst() &&
667  exp->getSubExp3()->isIntConst()) {
668  const int val2 = exp->access<Const, 2>()->getInt();
669  const int val3 = exp->access<Const, 3>()->getInt();
670 
671  if (val2 == 1 && val3 == 0) {
672  changed = true;
673  return Binary::get(opNotEqual, exp->getSubExp1(), Const::get(0));
674  }
675  else if (val2 == 0 && val3 == 1) {
676  changed = true;
677  return Binary::get(opEquals, exp->getSubExp1(), Const::get(0));
678  }
679  }
680 
681  // Const ? x : y
682  if (exp->getOper() == opTern && exp->getSubExp1()->isIntConst()) {
683  const int val = exp->access<Const, 1>()->getInt();
684  if (val != 1 && val != 0) {
685  LOG_VERBOSE("Treating constant value %1 as true in Ternary '%2'", val, exp);
686  }
687 
688  changed = true;
689  return (val != 0) ? exp->getSubExp2() : exp->getSubExp3();
690  }
691 
692  // a ? x : x
693  if (exp->getOper() == opTern && *exp->getSubExp2() == *exp->getSubExp3()) {
694  changed = true;
695  return exp->getSubExp2();
696  }
697 
698  /// sign-extend constant value
699  if (exp->getOper() == opSgnEx && exp->getSubExp1()->isIntConst() &&
700  exp->getSubExp2()->isIntConst() && exp->getSubExp3()->isIntConst()) {
701  const int from = exp->access<Const, 1>()->getInt();
702  const int to = exp->access<Const, 2>()->getInt();
703 
704  if (from >= 0 && to >= 0) {
705  const int oldVal = exp->access<Const, 3>()->getInt();
706  if (to <= from) {
707  changed = true;
708  return exp->getSubExp3();
709  }
710  else if (from == 0) {
711  changed = true;
712  return Const::get(0);
713  }
714 
715  const bool sign = ((oldVal >> (from - 1)) & 1) == 1;
716  if (sign) {
717  changed = true;
718  if (to > 32) {
719  return Const::get((Util::getLowerBitMask(to - from) << from) |
720  (oldVal & Util::getLowerBitMask(from)));
721  }
722  else {
723  return Const::get((int)(Util::getLowerBitMask(to - from) << from) |
724  (int)(oldVal & Util::getLowerBitMask(from)));
725  }
726  }
727  else {
728  changed = true;
729  return Const::get((int)(oldVal & Util::getLowerBitMask(from)));
730  }
731  }
732  }
733 
734  if (exp->getOper() == opZfill && exp->getSubExp3()->isIntConst()) {
735  changed = true;
736  return exp->getSubExp3();
737  }
738 
739  if (exp->getOper() == opFsize && exp->getSubExp3()->getOper() == opItof &&
740  *exp->getSubExp1() == *exp->access<Exp, 3, 2>() &&
741  *exp->getSubExp2() == *exp->access<Exp, 3, 1>()) {
742  changed = true;
743  return exp->getSubExp3();
744  }
745 
746  if (exp->getOper() == opFsize && exp->getSubExp3()->isFltConst()) {
747  changed = true;
748  return exp->getSubExp3();
749  }
750 
751 
752  if (exp->getOper() == opItof && exp->getSubExp2()->isIntConst() &&
753  exp->getSubExp3()->isIntConst() && exp->access<Const, 2>()->getInt() == 32) {
754  changed = true;
755  unsigned int n = exp->access<Const, 3>()->getInt();
756  return Const::get(*reinterpret_cast<float *>(&n));
757  }
758 
759  if (exp->getOper() == opFsize && exp->getSubExp3()->isMemOf() &&
760  exp->access<Exp, 3, 1>()->isIntConst()) {
761  assert(exp->getSubExp3()->isLocation());
762  Address u = exp->access<Const, 3, 1>()->getAddr();
763  UserProc *p = exp->access<Location, 3>()->getProc();
764 
765  if (p) {
766  Prog *prog = p->getProg();
767  double d;
768  const bool ok = prog->getFloatConstant(u, d, exp->access<Const, 1>()->getInt());
769 
770  if (ok) {
771  changed = true;
772  LOG_VERBOSE("Replacing %1 with %2 in %3", exp->getSubExp3(), d,
773  exp->shared_from_this());
774  return Const::get(d);
775  }
776  }
777  }
778 
779  if (exp->getOper() == opTruncu && exp->getSubExp3()->isIntConst()) {
780  int from = exp->access<Const, 1>()->getInt();
781  int to = exp->access<Const, 2>()->getInt();
782  unsigned int val = exp->access<Const, 3>()->getInt();
783 
784  if (from > to) {
785  changed = true;
786  return Const::get(Address(val & (int)Util::getLowerBitMask(to)));
787  }
788  }
789 
790  if (exp->getOper() == opTruncs && exp->getSubExp3()->isIntConst()) {
791  int from = exp->access<Const, 1>()->getInt();
792  int to = exp->access<Const, 2>()->getInt();
793  int val = exp->access<Const, 3>()->getInt();
794 
795  if (from > to) {
796  changed = true;
797  return Const::get(val & (int)Util::getLowerBitMask(to));
798  }
799  }
800 
801  if (exp->getOper() == opAt && exp->getSubExp1()->isIntConst() &&
802  exp->getSubExp2()->isIntConst() && exp->getSubExp3()->isIntConst()) {
803  const int val = exp->access<Const, 1>()->getInt();
804  const int from = exp->access<Const, 2>()->getInt();
805  const int to = exp->access<Const, 3>()->getInt();
806  const unsigned int mask = Util::getLowerBitMask(to + 1) & ~Util::getLowerBitMask(from);
807 
808  changed = true;
809  return Const::get((int)(val & mask));
810  }
811 
812  return exp;

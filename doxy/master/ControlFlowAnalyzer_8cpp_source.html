61 void ControlFlowAnalyzer::updateImmedPDom()
62 {
63  // traverse the nodes in order (i.e from the bottom up)
64  for (int i = m_revPostOrdering.size() - 1; i >= 0; i--) {
65  const BasicBlock *bb = m_revPostOrdering[i];
66 
67  for (BasicBlock *succ : bb->getSuccessors()) {
68  if (getRevOrd(succ) > getRevOrd(bb)) {
69  setImmPDom(bb, findCommonPDom(getImmPDom(bb), succ));
70  }
71  }
72  }
73 
74  // make a second pass but consider the original CFG ordering this time
75  for (const BasicBlock *bb : m_postOrdering) {
76  if (bb->getNumSuccessors() <= 1) {
77  continue;
78  }
79 
80  for (auto &succ : bb->getSuccessors()) {
81  BasicBlock *succNode = succ;
82  setImmPDom(bb, findCommonPDom(getImmPDom(bb), succNode));
83  }
84  }
85 
86  // one final pass to fix up nodes involved in a loop
87  for (const BasicBlock *bb : m_postOrdering) {
88  if (bb->getNumSuccessors() > 1) {
89  for (auto &succ : bb->getSuccessors()) {
90  BasicBlock *succNode = succ;
91 
92  if (isBackEdge(bb, succNode) && (bb->getNumSuccessors() > 1) &&
93  getImmPDom(succNode) &&
94  (getPostOrdering(getImmPDom(succ)) < getPostOrdering(getImmPDom(bb)))) {
95  setImmPDom(bb, findCommonPDom(getImmPDom(succNode), getImmPDom(bb)));
96  }
97  else {
98  setImmPDom(bb, findCommonPDom(getImmPDom(bb), succNode));
99  }
100  }
101  }
102  }
103 }

106 const BasicBlock *ControlFlowAnalyzer::findCommonPDom(const BasicBlock *currImmPDom,
107  const BasicBlock *succImmPDom)
108 {
109  if (!currImmPDom) {
110  return succImmPDom;
111  }
112 
113  if (!succImmPDom) {
114  return currImmPDom;
115  }
116 
117  if (getRevOrd(currImmPDom) == getRevOrd(succImmPDom)) {
118  return currImmPDom; // ordering hasn't been done
119  }
120 
121  const BasicBlock *oldCurImmPDom = currImmPDom;
122  const BasicBlock *oldSuccImmPDom = succImmPDom;
123 
124  int giveup = 0;
125 #define GIVEUP 10000
126 
127  while (giveup < GIVEUP && currImmPDom && succImmPDom && (currImmPDom != succImmPDom)) {
128  if (getRevOrd(currImmPDom) > getRevOrd(succImmPDom)) {
129  succImmPDom = getImmPDom(succImmPDom);
130  }
131  else {
132  currImmPDom = getImmPDom(currImmPDom);
133  }
134 
135  giveup++;
136  }
137 
138  if (giveup >= GIVEUP) {
139  LOG_VERBOSE("Failed to find commonPDom for %1 and %2", oldCurImmPDom->getLowAddr(),
140  oldSuccImmPDom->getLowAddr());
141 
142  return oldCurImmPDom; // no change
143  }
144 
145  return currImmPDom;
146 }

149 void ControlFlowAnalyzer::structConds()
150 {
151  // Process the nodes in order
152  for (const BasicBlock *currNode : m_postOrdering) {
153  if (currNode->getNumSuccessors() <= 1) {
154  // not an if/case condition
155  continue;
156  }
157 
158  // if the current conditional header is a two way node and has a back edge, then it
159  // won't have a follow
160  if (hasBackEdge(currNode) && (currNode->getType() == BBType::Twoway)) {
161  setStructType(currNode, StructType::Cond);
162  continue;
163  }
164 
165  // set the follow of a node to be its immediate post dominator
166  setCondFollow(currNode, getImmPDom(currNode));
167 
168  // set the structured type of this node
169  setStructType(currNode, StructType::Cond);
170 
171  // if this is an nway header, then we have to tag each of the nodes within the body of
172  // the nway subgraph
173  if (getCondType(currNode) == CondType::Case) {
174  setCaseHead(currNode, currNode, getCondFollow(currNode));
175  }
176  }
177 }

180 void ControlFlowAnalyzer::determineLoopType(const BasicBlock *header, bool *&loopNodes)
181 {
182  assert(getLatchNode(header));
183 
184  // if the latch node is a two way node then this must be a post tested loop
185  if (getLatchNode(header)->getType() == BBType::Twoway) {
186  setLoopType(header, LoopType::PostTested);
187 
188  // if the head of the loop is a two way node and the loop spans more than one block then it
189  // must also be a conditional header
190  if ((header->getType() == BBType::Twoway) && (header != getLatchNode(header))) {
191  setStructType(header, StructType::LoopCond);
192  }
193  }
194  // otherwise it is either a pretested or endless loop
195  else if (header->getType() == BBType::Twoway) {
196  // if the header is a two way node then it must have a conditional follow (since it can't
197  // have any backedges leading from it). If this follow is within the loop then this must be
198  // an endless loop
199  if (getCondFollow(header) && loopNodes[getPostOrdering(getCondFollow(header))]) {
200  setLoopType(header, LoopType::Endless);
201 
202  // retain the fact that this is also a conditional header
203  setStructType(header, StructType::LoopCond);
204  }
205  else {
206  setLoopType(header, LoopType::PreTested);
207  }
208  }
209  else {
210  // both the header and latch node are one way nodes so this must be an endless loop
211  setLoopType(header, LoopType::Endless);
212  }
213 }

216 void ControlFlowAnalyzer::findLoopFollow(const BasicBlock *header, bool *&loopNodes)
217 {
218  assert(getStructType(header) == StructType::Loop ||
219  getStructType(header) == StructType::LoopCond);
220  const LoopType loopType = getLoopType(header);
221  const BasicBlock *latch = getLatchNode(header);
222 
223  if (loopType == LoopType::PreTested) {
224  // if the 'while' loop's true child is within the loop, then its false child is the loop
225  // follow
226  if (loopNodes[getPostOrdering(header->getSuccessor(BTHEN))]) {
227  setLoopFollow(header, header->getSuccessor(BELSE));
228  }
229  else {
230  setLoopFollow(header, header->getSuccessor(BTHEN));
231  }
232  }
233  else if (loopType == LoopType::PostTested) {
234  // the follow of a post tested ('repeat') loop is the node on the end of the non-back edge
235  // from the latch node
236  if (latch->getSuccessor(BELSE) == header) {
237  setLoopFollow(header, latch->getSuccessor(BTHEN));
238  }
239  else {
240  setLoopFollow(header, latch->getSuccessor(BELSE));
241  }
242  }
243  else {
244  // endless loop
245  const BasicBlock *follow = nullptr;
246 
247  // traverse the ordering array between the header and latch nodes.
248  // BasicBlock * latch = header->getLatchNode(); initialized at function start
249  for (int i = getPostOrdering(header) - 1; i > getPostOrdering(latch); i--) {
250  const BasicBlock *&desc = m_postOrdering[i];
251  // the follow for an endless loop will have the following
252  // properties:
253  // i) it will have a parent that is a conditional header inside the loop whose follow
254  // is outside the loop
255  // ii) it will be outside the loop according to its loop stamp pair
256  // iii) have the highest ordering of all suitable follows (i.e. highest in the graph)
257 
258  if ((getStructType(desc) == StructType::Cond) && getCondFollow(desc) &&
259  (getLoopHead(desc) == header)) {
260  if (loopNodes[getPostOrdering(getCondFollow(desc))]) {
261  // if the conditional's follow is in the same loop AND is lower in the loop,
262  // jump to this follow
263  if (getPostOrdering(desc) > getPostOrdering(getCondFollow(desc))) {
264  i = getPostOrdering(getCondFollow(desc));
265  }
266  else {
267  // otherwise there is a backward jump somewhere to a node earlier in this
268  // loop. We don't need to any nodes below this one as they will all have a
269  // conditional within the loop.
270  break;
271  }
272  }
273  else {
274  // otherwise find the child (if any) of the conditional header that isn't inside
275  // the same loop
276  const BasicBlock *succ = desc->getSuccessor(BTHEN);
277 
278  if (loopNodes[getPostOrdering(succ)]) {
279  if (!loopNodes[getPostOrdering(desc->getSuccessor(BELSE))]) {
280  succ = desc->getSuccessor(BELSE);
281  }
282  else {
283  succ = nullptr;
284  }
285  }
286 
287  // if a potential follow was found, compare its ordering with the currently
288  // found follow
289  if (succ && (!follow || (getPostOrdering(succ) > getPostOrdering(follow)))) {
290  follow = succ;
291  }
292  }
293  }
294  }
295 
296  // if a follow was found, assign it to be the follow of the loop under
297  // investigation
298  if (follow) {
299  setLoopFollow(header, follow);
300  }
301  }
302 }

305 void ControlFlowAnalyzer::tagNodesInLoop(const BasicBlock *header, bool *&loopNodes)
306 {
307  // Traverse the ordering structure from the header to the latch node tagging the nodes
308  // determined to be within the loop. These are nodes that satisfy the following:
309  // i) header.loopStamps encloses curNode.loopStamps and curNode.loopStamps encloses
310  // latch.loopStamps
311  // OR
312  // ii) latch.revLoopStamps encloses curNode.revLoopStamps and curNode.revLoopStamps encloses
313  // header.revLoopStamps
314  // OR
315  // iii) curNode is the latch node
316 
317  const BasicBlock *latch = getLatchNode(header);
318  assert(latch);
319 
320  for (int i = getPostOrdering(header) - 1; i >= getPostOrdering(latch); i--) {
321  if (isBBInLoop(m_postOrdering[i], header, latch)) {
322  // update the membership map to reflect that this node is within the loop
323  loopNodes[i] = true;
324 
325  setLoopHead(m_postOrdering[i], header);
326  }
327  }
328 }

331 void ControlFlowAnalyzer::structLoops()
332 {
333  for (int i = m_postOrdering.size() - 1; i >= 0; i--) {
334  const BasicBlock *currNode = m_postOrdering[i]; // the current node under investigation
335  const BasicBlock *latch = nullptr; // the latching node of the loop
336 
337  // If the current node has at least one back edge into it, it is a loop header. If there are
338  // numerous back edges into the header, determine which one comes form the proper latching
339  // node. The proper latching node is defined to have the following properties:
340  // i) has a back edge to the current node
341  // ii) has the same case head as the current node
342  // iii) has the same loop head as the current node
343  // iv) is not an nway node
344  // v) is not the latch node of an enclosing loop
345  // vi) has a lower ordering than all other suitable candiates
346  // If no nodes meet the above criteria, then the current node is not a loop header
347 
348  for (const BasicBlock *pred : currNode->getPredecessors()) {
349  if ((getCaseHead(pred) == getCaseHead(currNode)) && // ii)
350  (getLoopHead(pred) == getLoopHead(currNode)) && // iii)
351  (!latch || (getPostOrdering(latch) > getPostOrdering(pred))) && // vi)
352  !(getLoopHead(pred) && (getLatchNode(getLoopHead(pred)) == pred)) && // v)
353  isBackEdge(pred, currNode)) { // i)
354  latch = pred;
355  }
356  }
357 
358  // if a latching node was found for the current node then it is a loop header.
359  if (!latch) {
360  continue;
361  }
362 
363  // define the map that maps each node to whether or not it is within the current loop
364  bool *loopNodes = new bool[m_postOrdering.size()];
365 
366  for (unsigned int j = 0; j < m_postOrdering.size(); j++) {
367  loopNodes[j] = false;
368  }
369 
370  setLatchNode(currNode, latch);
371 
372  // the latching node may already have been structured as a conditional header. If it is
373  // not also the loop header (i.e. the loop is over more than one block) then reset it to
374  // be a sequential node otherwise it will be correctly set as a loop header only later
375  if ((latch != currNode) && (getStructType(latch) == StructType::Cond)) {
376  setStructType(latch, StructType::Seq);
377  }
378 
379  // set the structured type of this node
380  setStructType(currNode, StructType::Loop);
381 
382  // tag the members of this loop
383  tagNodesInLoop(currNode, loopNodes);
384 
385  // calculate the type of this loop
386  determineLoopType(currNode, loopNodes);
387 
388  // calculate the follow node of this loop
389  findLoopFollow(currNode, loopNodes);
390 
391  delete[] loopNodes;
392  }
393 }

396 void ControlFlowAnalyzer::checkConds()
397 {
398  for (const BasicBlock *currNode : m_postOrdering) {
399  // consider only conditional headers that have a follow and aren't case headers
400  if (((getStructType(currNode) == StructType::Cond) ||
401  (getStructType(currNode) == StructType::LoopCond)) &&
402  getCondFollow(currNode) && (getCondType(currNode) != CondType::Case)) {
403  // define convenient aliases for the relevant loop and case heads and the out edges
404  const BasicBlock *myLoopHead = (getStructType(currNode) == StructType::LoopCond)
405  ? currNode
406  : getLoopHead(currNode);
407  const BasicBlock *follLoopHead = getLoopHead(getCondFollow(currNode));
408  const BasicBlock *bbThen = currNode->getSuccessor(BTHEN);
409  const BasicBlock *bbElse = currNode->getSuccessor(BELSE);
410 
411  // analyse whether this is a jump into/outof a loop
412  if (myLoopHead != follLoopHead) {
413  // we want to find the branch that the latch node is on for a jump out of a loop
414  if (myLoopHead) {
415  // this is a jump out of a loop (break or return)
416  if (getLoopHead(bbThen) != nullptr) {
417  // the "else" branch jumps out of the loop. (e.g. "if (!foo) break;")
418  setUnstructType(currNode, UnstructType::JumpInOutLoop);
419  setCondType(currNode, CondType::IfElse);
420  }
421  else {
422  assert(getLoopHead(bbElse) != nullptr);
423  // the "then" branch jumps out of the loop
424  setUnstructType(currNode, UnstructType::JumpInOutLoop);
425  setCondType(currNode, CondType::IfThen);
426  }
427  }
428 
429  if ((getUnstructType(currNode) == UnstructType::Structured) && follLoopHead) {
430  // find the branch that the loop head is on for a jump into a loop body. If a
431  // branch has already been found, then it will match this one anyway
432 
433  // does the else branch goto the loop head?
434  if (isBackEdge(bbThen, follLoopHead)) {
435  setUnstructType(currNode, UnstructType::JumpInOutLoop);
436  setCondType(currNode, CondType::IfElse);
437  }
438  // does the else branch goto the loop head?
439  else if (isBackEdge(bbElse, follLoopHead)) {
440  setUnstructType(currNode, UnstructType::JumpInOutLoop);
441  setCondType(currNode, CondType::IfThen);
442  }
443  }
444  }
445 
446  // this is a jump into a case body if either of its children don't have the same same
447  // case header as itself
448  if ((getUnstructType(currNode) == UnstructType::Structured) &&
449  ((getCaseHead(currNode) != getCaseHead(bbThen)) ||
450  (getCaseHead(currNode) != getCaseHead(bbElse)))) {
451  const BasicBlock *myCaseHead = getCaseHead(currNode);
452  const BasicBlock *thenCaseHead = getCaseHead(bbThen);
453  const BasicBlock *elseCaseHead = getCaseHead(bbElse);
454 
455  if ((thenCaseHead == myCaseHead) &&
456  (!myCaseHead || (elseCaseHead != getCondFollow(myCaseHead)))) {
457  setUnstructType(currNode, UnstructType::JumpIntoCase);
458  setCondType(currNode, CondType::IfElse);
459  }
460  else if ((elseCaseHead == myCaseHead) &&
461  (!myCaseHead || (thenCaseHead != getCondFollow(myCaseHead)))) {
462  setUnstructType(currNode, UnstructType::JumpIntoCase);
463  setCondType(currNode, CondType::IfThen);
464  }
465  }
466  }
467 
468  // for 2 way conditional headers that don't have a follow (i.e. are the source of a back
469  // edge) and haven't been structured as latching nodes, set their follow to be the non-back
470  // edge child.
471  if ((getStructType(currNode) == StructType::Cond) && !getCondFollow(currNode) &&
472  (getCondType(currNode) != CondType::Case) &&
473  (getUnstructType(currNode) == UnstructType::Structured)) {
474  // latching nodes will already have been reset to Seq structured type
475  if (hasBackEdge(currNode)) {
476  if (isBackEdge(currNode, currNode->getSuccessor(BTHEN))) {
477  setCondType(currNode, CondType::IfThen);
478  setCondFollow(currNode, currNode->getSuccessor(BELSE));
479  }
480  else {
481  setCondType(currNode, CondType::IfElse);
482  setCondFollow(currNode, currNode->getSuccessor(BTHEN));
483  }
484  }
485  }
486  }
487 }

521 void ControlFlowAnalyzer::updateLoopStamps(const BasicBlock *bb, int &time)
522 {
523  // timestamp the current node with the current time
524  // and set its traversed flag
525  setTravType(bb, TravType::DFS_LNum);
526  m_info[bb].m_preOrderID = time;
527 
528  // recurse on unvisited children and set inedges for all children
529  for (const BasicBlock *succ : bb->getSuccessors()) {
530  // set the in edge from this child to its parent (the current node)
531  // (not done here, might be a problem)
532  // outEdges[i]->inEdges.Add(this);
533 
534  // recurse on this child if it hasn't already been visited
535  if (getTravType(succ) != TravType::DFS_LNum) {
536  updateLoopStamps(succ, ++time);
537  }
538  }
539 
540  // set the the second loopStamp value
541  m_info[bb].m_postOrderID = ++time;
542 
543  // add this node to the ordering structure as well as recording its position within the ordering
544  m_info[bb].m_postOrderIndex = static_cast<int>(m_postOrdering.size());
545  m_postOrdering.push_back(bb);
546 }

549 void ControlFlowAnalyzer::updateRevLoopStamps(const BasicBlock *bb, int &time)
550 {
551  // timestamp the current node with the current time and set its traversed flag
552  setTravType(bb, TravType::DFS_RNum);
553  m_info[bb].m_revPreOrderID = time;
554 
555  // recurse on the unvisited children in reverse order
556  for (int i = bb->getNumSuccessors() - 1; i >= 0; i--) {
557  // recurse on this child if it hasn't already been visited
558  if (getTravType(bb->getSuccessor(i)) != TravType::DFS_RNum) {
559  updateRevLoopStamps(bb->getSuccessor(i), ++time);
560  }
561  }
562 
563  m_info[bb].m_revPostOrderID = ++time;
564 }

567 void ControlFlowAnalyzer::updateRevOrder(const BasicBlock *bb)
568 {
569  // Set this node as having been traversed during the post domimator DFS ordering traversal
570  setTravType(bb, TravType::DFS_PDom);

573  for (const BasicBlock *pred : bb->getPredecessors()) {
574  if (getTravType(pred) != TravType::DFS_PDom) {
575  updateRevOrder(pred);
576  }
577  }
578 
579  // add this node to the ordering structure and record the post dom. order of this node as its
580  // index within this ordering structure
581  m_info[bb].m_revPostOrderIndex = static_cast<int>(m_revPostOrdering.size());
582  m_revPostOrdering.push_back(bb);

586 void ControlFlowAnalyzer::setCaseHead(const BasicBlock *bb, const BasicBlock *head,
587  const BasicBlock *follow)
588 {
589  assert(!getCaseHead(bb));
590 
591  setTravType(bb, TravType::DFS_Case);
592 
593  // don't tag this node if it is the case header under investigation
594  if (bb != head) {
595  m_info[bb].m_caseHead = head;
596  }
597 
598  // if this is a nested case header, then it's member nodes
599  // will already have been tagged so skip straight to its follow
600  if (bb->isType(BBType::Nway) && (bb != head)) {
601  if (getCondFollow(bb) && (getTravType(getCondFollow(bb)) != TravType::DFS_Case) &&
602  (getCondFollow(bb) != follow)) {
603  setCaseHead(bb, head, follow);
604  }
605  }
606  else {
607  // traverse each child of this node that:
608  // i) isn't on a back-edge,
609  // ii) hasn't already been traversed in a case tagging traversal and,
610  // iii) isn't the follow node.
611  for (BasicBlock *succ : bb->getSuccessors()) {
612  if (!isBackEdge(bb, succ) && (getTravType(succ) != TravType::DFS_Case) &&
613  (succ != follow)) {
614  setCaseHead(succ, head, follow);
615  }
616  }
617  }
618 }

621 void ControlFlowAnalyzer::setStructType(const BasicBlock *bb, StructType structType)
622 {
623  // if this is a conditional header, determine exactly which type of conditional header it is
624  // (i.e. switch, if-then, if-then-else etc.)
625  if (structType == StructType::Cond) {
626  if (bb->isType(BBType::Nway)) {
627  m_info[bb].m_conditionHeaderType = CondType::Case;
628  }
629  else if (getCondFollow(bb) == bb->getSuccessor(BELSE)) {
630  m_info[bb].m_conditionHeaderType = CondType::IfThen;
631  }
632  else if (getCondFollow(bb) == bb->getSuccessor(BTHEN)) {
633  m_info[bb].m_conditionHeaderType = CondType::IfElse;
634  }
635  else {
636  m_info[bb].m_conditionHeaderType = CondType::IfThenElse;
637  }
638  }
639 
640  m_info[bb].m_structuringType = structType;
641 }

699 bool ControlFlowAnalyzer::isBBInLoop(const BasicBlock *bb, const BasicBlock *header,
700  const BasicBlock *latch) const
701 {
702  assert(getLatchNode(header) == latch);
703  assert(header == latch || ((m_info[header].m_preOrderID > m_info[latch].m_preOrderID &&
704  m_info[latch].m_postOrderID > m_info[header].m_postOrderID) ||
705  (m_info[header].m_preOrderID < m_info[latch].m_preOrderID &&
706  m_info[latch].m_postOrderID < m_info[header].m_postOrderID)));
707 
708  // this node is in the loop if it is the latch node OR
709  // this node is within the header and the latch is within this when using the forward loop
710  // stamps OR this node is within the header and the latch is within this when using the reverse
711  // loop stamps
712  return bb == latch ||
713  (m_info[header].m_preOrderID < m_info[bb].m_preOrderID &&
714  m_info[bb].m_postOrderID < m_info[header].m_postOrderID &&
715  m_info[bb].m_preOrderID < m_info[latch].m_preOrderID &&
716  m_info[latch].m_postOrderID < m_info[bb].m_postOrderID) ||
717  (m_info[header].m_revPreOrderID < m_info[bb].m_revPreOrderID &&
718  m_info[bb].m_revPostOrderID < m_info[header].m_revPostOrderID &&
719  m_info[bb].m_revPreOrderID < m_info[latch].m_revPreOrderID &&
720  m_info[latch].m_revPostOrderID < m_info[bb].m_revPostOrderID);
721 }

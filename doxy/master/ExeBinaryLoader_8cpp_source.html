45 bool ExeBinaryLoader::loadFromMemory(QByteArray &data)
46 {
47  if (data.length() < (int)sizeof(ExeHeader)) {
48  LOG_ERROR("Cannot read exe file!");
49  return false;
50  }
51 
52  const Address loadBaseAddr = Address::ZERO;
53 
54  if (m_header) {
55  delete m_header;
56  }
57 
58  // Always just 3 sections
59  m_header = new ExeHeader;
60 
61  QBuffer fp(&data);
62  if (!fp.open(QBuffer::ReadOnly)) {
63  LOG_ERROR("Cannot read Exe file");
64  return false;
65  }
66 
67  // Read rest of m_header
68  if (fp.read(reinterpret_cast<char *>(m_header), sizeof(ExeHeader)) != sizeof(ExeHeader)) {
69  LOG_ERROR("Cannot read Exe file");
70  return false;
71  }
72 
73  // Calculate the load module size.
74  // This is:
75  // - the number of pages in the file
76  // - less the length of the m_header and reloc table
77  // - less the number of bytes unused on last page
78  const SWord numPages = Util::readWord(&m_header->numPages, Endian::Little);
79  const SWord numParaHeader = Util::readWord(&m_header->numParaHeader, Endian::Little);
80  const SWord lastPageSize = Util::readWord(&m_header->lastPageSize, Endian::Little);
81  const SWord headerSize = numParaHeader * DOS_PARA_SIZE;
82  int cbImageSize = numPages * DOS_PAGE_SIZE - headerSize;
83 
84  if (lastPageSize > 0) {
85  // if lastPageSize is 0, this indicates that the last page is full; if lastPageSize
86  // is not 0, the last page is not full so we have to subtract the empty space
87  // past the image boundary.
88  cbImageSize -= DOS_PAGE_SIZE - lastPageSize;
89  }
90 
91  // Seek to start of image
92  const SWord initialPtrOffset = Util::readWord(&m_header->numParaHeader, Endian::Little);
93  if (((initialPtrOffset & 0xF000) != 0) ||
94  !Util::inRange(initialPtrOffset * DOS_PARA_SIZE, 0, data.size())) {
95  LOG_ERROR("Cannot read Exe file: Invalid offset for initial SP/IP values");
96  return false;
97  }
98 
99  if (!fp.seek(initialPtrOffset * DOS_PARA_SIZE)) {
100  LOG_ERROR("Cannot load Exe file: Cannot seek to offset %1",
101  initialPtrOffset * DOS_PARA_SIZE);
102  return false;
103  }
104 
105  // Initial PC and SP. Note that we fake the seg:offset by putting
106  // the segment in the top half, and offset in the bottom half.
107  const DWord initCS = Util::readWord(&m_header->initCS, Endian::Little);
108  const DWord initIP = Util::readWord(&m_header->initIP, Endian::Little);
109  const DWord initSS = Util::readWord(&m_header->initSS, Endian::Little);
110  const DWord initSP = Util::readWord(&m_header->initSP, Endian::Little);
111 
112  m_uInitPC = Address((initCS * DOS_PARA_SIZE) + initIP);
113  m_uInitSP = Address((initSS * DOS_PARA_SIZE) + initSP);
114 
115  if (!Util::inRange(cbImageSize, 0, data.size())) {
116  LOG_ERROR("Cannot read Exe file: Invalid image size.");
117  return false;
118  }
119 
120  /* Allocate a block of memory for the image. */
121  m_imageSize = cbImageSize;
122  m_loadedImage = new Byte[m_imageSize];
123 
124  if (cbImageSize != fp.read(reinterpret_cast<char *>(m_loadedImage), cbImageSize)) {
125  LOG_ERROR("Cannot read Exe file: Failed to read loaded image");
126  return false;
127  }
128 
129  // Use the following section layout:
130  // baseAddr..baseAddr+m_imageSize: .text
131  //
132 
133  BinarySection *header = m_image->createSection("$HEADER", loadBaseAddr + m_imageSize,
134  loadBaseAddr + m_imageSize + sizeof(ExeHeader));
135  header->setHostAddr(HostAddress(m_header));
136  header->setEntrySize(1);
137 
138  // The text and data section
139  BinarySection *text = m_image->createSection(".text", loadBaseAddr, loadBaseAddr + m_imageSize);
140  text->setCode(true);
141  text->setData(true);
142  text->setHostAddr(HostAddress(m_loadedImage));
143 
144  // create relocations and relocation section
145  applyRelocations(fp, data, loadBaseAddr);
146 
147  fp.close();
148  return true;
149 }

152 bool ExeBinaryLoader::applyRelocations(QBuffer &fp, QByteArray &data, Address loadBaseAddr)
153 {
154  /* We quietly ignore minAlloc and maxAlloc since for our
155  * purposes it doesn't really matter where in real memory
156  * the m_am would end up. EXE m_ams can't really rely on
157  * their load location so setting the PSP segment to 0 is fine.
158  * Certainly m_ams that prod around in DOS or BIOS are going
159  * to have to load DS from a constant so it'll be pretty
160  * obvious.
161  */
162  const int numReloc = std::max(0, (int)Util::readWord(&m_header->numReloc, Endian::Little));
163  const SWord offset = Util::readWord(&m_header->relocTabOffset, Endian::Little);
164 
165  if (!Util::inRange(offset, 0, data.size()) ||
166  !Util::inRange(offset + numReloc * (int)sizeof(DWord), 0, data.size())) {
167  LOG_ERROR("Cannot load Exe file: Relocation table extends past file boundary");
168  return false;
169  }
170 
171  if (!fp.seek(offset)) {
172  LOG_ERROR("Cannot load Exe file: Cannot seek to offset %1", offset);
173  return false;
174  }
175 
176  /* Relocate segment constants */
177  m_relocations.resize(numReloc);
178 
179  for (int i = 0; i < numReloc; i++) {
180  ExeReloc relocEntry;
181  if (sizeof(ExeReloc) != fp.read(reinterpret_cast<char *>(&relocEntry), sizeof(ExeReloc))) {
182  LOG_ERROR("Cannot load Exe file: Cannot read relocation table");
183  return false;
184  }
185 
186  relocEntry.offset = Util::normEndian(relocEntry.offset, Endian::Little);
187  relocEntry.segment = Util::normEndian(relocEntry.segment, Endian::Little);
188 
189  const SWord imageOffset = (relocEntry.segment << 4) + relocEntry.offset;
190  if (!Util::inRange(imageOffset, 0, m_imageSize)) {
191  LOG_WARN("Cannot read Exe relocation entry %1: Offset %2 is not valid", i,
192  relocEntry.offset);
193  continue;
194  }
195 
196  Byte *p = &m_loadedImage[imageOffset];
197  const SWord relocValue = Util::readWord(p, Endian::Little);
198  Util::writeWord(p, loadBaseAddr.value() + relocValue, Endian::Little);
199  }
200 
201  Address relocStart = loadBaseAddr + m_imageSize + sizeof(ExeHeader);
202  BinarySection *reloc = m_image->createSection("$RELOC", relocStart,
203  relocStart + sizeof(ExeReloc) * numReloc);
204  reloc->setEntrySize(sizeof(ExeReloc));
205 
206  // as of C++11, std::vector is guaranteed to be contiguous (except for std::vector<bool>),
207  // so we can read the relocated values directly from m_relocTable
208  if (numReloc > 0) {
209  reloc->setHostAddr(HostAddress(&m_relocations[0]));
210  }
211 
212  return true;
213 }

258 int ExeBinaryLoader::canLoad(QIODevice &fl) const
259 {
260  Byte buf[4] = { 0 };
261  const bool ok = fl.read(reinterpret_cast<char *>(buf), sizeof(buf)) == sizeof(buf);
262 
263  if (!ok) {
264  // cannot read file / file is too small
265  return 0;
266  }
267 
268  if (!Util::testMagic(buf, { 'M', 'Z' })) {
269  // No MZ header
270  return 0;
271  }
272 
273  /* Check if this is a Windows NE executable
274  * It will have a DOS MZ header, but also another header later on
275  * Offset of the NE header is at 0x3C
276  * The NE header will start with the signature 'NE'
277  * Ref: https://www.fileformat.info/format/exe/corion-mz.htm
278  */
279  if (fl.seek(0x3C)) {
280  if (fl.read(reinterpret_cast<char *>(buf), 2) == 2) {
281  const SWord possibleWinHeaderOffset = Util::readWord(buf, Endian::Little);
282  if (fl.seek(possibleWinHeaderOffset)) {
283  if (fl.read(reinterpret_cast<char *>(buf), 2) == 2) {
284  if (Util::testMagic(buf, { 'N', 'E' })) {
285  /* This is a Windows NE Executable */
286  return 0;
287  }
288  }
289  }
290  }
291  }
292 
293  /* DOS-based file */
294  return 2;
295 }

51  {
52  if (interval.lower() >= interval.upper()) {
53  return end(); // Don't insert invalid intervals
54  }
55 
56  typename Data::iterator firstInRange, lastInRange;
57  std::tie(firstInRange, lastInRange) = equalRange(interval);
58 
59  T minLower = interval.lower();
60  T maxUpper = interval.upper();
61 
62  if (firstInRange != m_data.end()) {
63  minLower = std::min(minLower, firstInRange->lower());
64  }
65  if (lastInRange != m_data.end()) {
66  maxUpper = std::max(maxUpper, lastInRange->upper());
67  }
68

83  std::pair<iterator, iterator> equalRange(const Interval<T> &interval)
84  {
85  if (interval.lower() >= interval.upper()) {
86  return { end(), end() };
87  }
88 
89  typename Data::iterator itLower = end();
90  typename Data::iterator itUpper = end();
91 
92  // todo: speed up
93  for (iterator existingIt = begin(); existingIt != end(); ++existingIt) {
94  if (existingIt->lower() >= interval.upper()) {
95  return { end(), end() }; // no blocking intervals
96  }
97  else if (existingIt->upper() > interval.lower()) {
98  itLower = existingIt;
99  break;
100  }
101  }
102 
103  if (itLower == end()) {
104  return { end(), end() };
105  }
106 
107  for (iterator existingIt = std::next(itLower); existingIt != end(); ++existingIt) {
108  if (existingIt->lower() >= interval.upper()) {
109  itUpper = existingIt;
110  break;
111  }
112  }
113 
114  return std::make_pair(itLower, itUpper);
115  }

131  }
132  else {

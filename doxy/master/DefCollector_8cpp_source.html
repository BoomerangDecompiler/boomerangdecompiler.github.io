52 void DefCollector::print(OStream &os) const
53 {
54  if (m_defs.empty()) {
55  os << "<None>";
56  return;
57  }
58 
59  size_t col = 36;
60  bool first = true;

63  QString tgt;
64  OStream ost(&tgt);
65  def->getLeft()->print(ost);
66  ost << "=";
67  def->getRight()->print(ost);
68  size_t len = tgt.length();
69 
70  if (first) {
71  first = false;
72  }
73  else if (col + 4 + len >= DEFCOL_COLS) { // 4 for a comma and three spaces
74  if (col != DEFCOL_COLS - 1) {
75  os << ","; // Comma at end of line
76  }
77

79  col = 16;
80  }
81  else {
82  os << ", ";
83  col += 4;
84  }
85

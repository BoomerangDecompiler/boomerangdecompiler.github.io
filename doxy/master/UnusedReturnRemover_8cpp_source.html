26 bool lessUserProc(const UserProc *lhs, const UserProc *rhs)
27 {
28  if (lhs && rhs) {
29  return lhs->getEntryAddress() < rhs->getEntryAddress();
30  }
31  else {
32  return lhs < rhs;
33  }
34 }

43 bool UnusedReturnRemover::removeUnusedReturns()
44 {
45  for (const auto &module : m_prog->getModuleList()) {
46  for (Function *proc : *module) {
47  if (proc && !proc->isLib() && static_cast<UserProc *>(proc)->isDecoded()) {
48  m_removeRetSet.insert(static_cast<UserProc *>(proc));
49  }
50  // else e.g. use -sf file to just prototype the proc
51  }
52  }
53 
54  bool change = false;
55  // The workset is processed in order of entry address. This is to provide a consistent
56  // deterministic order of processing. Note that sometimes changes propagate down the call tree
57  // (no caller uses potential returns for child), and sometimes up the call tree
58  // (removal of returns and/or dead code removes parameters, which affects all callers).
59  while (!m_removeRetSet.empty()) {
60  auto it = std::min_element(m_removeRetSet.begin(), m_removeRetSet.end(), lessUserProc);
61  assert(*it != nullptr);
62  const bool removedReturns = removeUnusedParamsAndReturns(*it);
63 
64  if (removedReturns) {
65  // Removing returns changes the uses of the callee.
66  // So we have to do type analyis to update the use information.
67  PassManager::get()->executePass(PassID::LocalTypeAnalysis, *it);
68 
69  // type analysis might propagate statements that could not be propagated before
70  PassManager::get()->executePass(PassID::UnusedStatementRemoval, *it);
71  }
72  change |= removedReturns;
73 
74  // Note: removing the currently processed item here should prevent
75  // unnecessary reprocessing of self recursive procedures
76  m_removeRetSet.erase(it);
77  }
78 
79  return change;
80 }

83 bool UnusedReturnRemover::removeUnusedParamsAndReturns(UserProc *proc)
84 {
85  assert(m_removeRetSet.find(proc) != m_removeRetSet.end());
86 
87  m_prog->getProject()->alertDecompiling(proc);
88  m_prog->getProject()->alertDecompileDebugPoint(proc, "before removing unused returns");
89 
90  // First remove the unused parameters
91  bool removedParams = PassManager::get()->executePass(PassID::UnusedParamRemoval, proc);
92 
93  if (proc->getRetStmt() == nullptr) {
94  return removedParams;
95  }
96 
97  if (m_prog->getProject()->getSettings()->debugUnused) {
98  LOG_MSG("%%% removing unused returns for %1 %%%", proc->getName());
99  }
100 
101  if (proc->getSignature()->isForced()) {
102  // Respect the forced signature, but use it to remove returns if necessary
103  bool removedRets = false;
104 
105  for (auto retIt = proc->getRetStmt()->begin(); retIt != proc->getRetStmt()->end();) {
106  assert(*retIt != nullptr && (*retIt)->isAssign());
107  Assign *retDef = static_cast<Assign *>(*retIt);
108  SharedExp lhs = retDef->getLeft();
109 
110  // For each location in the returns, check if in the signature
111  bool found = false;
112 
113  for (int i = 0; i < proc->getSignature()->getNumReturns(); i++) {
114  if (*proc->getSignature()->getReturnExp(i) == *lhs) {
115  found = true;
116  break;
117  }
118  }
119 
120  if (found) {
121  ++retIt; // Yes, in signature; OK
122  }
123  else {
124  if (m_prog->getProject()->getSettings()->debugUnused) {
125  LOG_MSG("%%% removing unused return %1 from proc %2 (forced signature)",
126  retDef, proc->getName());
127  }
128 
129  // This return is not in the signature. Remove it
130  retIt = proc->getRetStmt()->erase(retIt);
131  removedRets = true;
132  }
133  }
134 
135  if (removedRets) {
136  // Still may have effects on calls or now unused statements
137  updateForUseChange(proc);
138  }
139 
140  return removedRets;
141  }
142 
143  // FIXME: this needs to be more sensible when we don't decompile down from main! Probably should
144  // assume just the first return is valid, for example (presently assume none are valid)
145  LocationSet unionOfCallerLiveLocs;
146 
147  if (proc->getName() == "main") { // Probably not needed: main is forced so handled above
148  // Just insert one return for main. Note: at present, the first parameter is still the stack
149  // pointer
150  if (proc->getSignature()->getNumReturns() <= 1) {
151  // handle the case of missing main() signature
152  LOG_WARN("main signature definition is missing; assuming void main()");
153  }
154  else {
155  unionOfCallerLiveLocs.insert(proc->getSignature()->getReturnExp(1));
156  }
157  }
158  else {
159  for (CallStatement *cc : proc->getCallers()) {
160  // Prevent function from blocking its own removals
161  // TODO This only handles self-recursion. More analysis and testing is necessary
162  // for mutual recursion. (-> cc->getProc()->doesRecurseTo(proc))
163  if (cc->getProc() == proc) {
164  continue;
165  }
166 
167  UseCollector *useCol = cc->getUseCollector();
168  unionOfCallerLiveLocs.makeUnion(useCol->getLocSet());
169  }
170  }
171 
172  // Intersect with the current returns
173  bool removedRets = false;
174 
175  for (auto retIt = proc->getRetStmt()->begin(); retIt != proc->getRetStmt()->end();) {
176  Assign *retDef = static_cast<Assign *>(*retIt);
177 
178  // Check if the location defined by the return is actually used by any callee.
179  if (unionOfCallerLiveLocs.contains(retDef->getLeft())) {
180  ++retIt;
181  continue;
182  }
183 
184  if (m_prog->getProject()->getSettings()->debugUnused) {
185  LOG_MSG("%%% removing unused return %1 from proc %2", retDef, proc->getName());
186  }
187 
188  // If a component of the RHS referenced a call statement, the liveness used to be killed
189  // here. This was wrong; you need to notice the liveness changing inside
190  // updateForUseChange() to correctly recurse to callee
191  retIt = proc->getRetStmt()->erase(retIt);
192  removedRets = true;
193  }
194 
195  if (m_prog->getProject()->getSettings()->debugUnused) {
196  QString tgt;
197  OStream ost(&tgt);
198  unionOfCallerLiveLocs.print(ost);
199  LOG_MSG("%%% union of caller live locations for %1: %2", proc->getName(), tgt);
200  LOG_MSG("%%% final returns for %1: %2", proc->getName(),
201  proc->getRetStmt()->getReturns().toString());
202  }
203 
204  // removing returns might result in params that can be removed, might as well do it now.
205  removedParams |= PassManager::get()->executePass(PassID::UnusedParamRemoval, proc);
206 
207  ProcSet updateSet; // Set of procs to update
208 
209  if (removedParams || removedRets) {
210  // Update the statements that call us
211  for (CallStatement *call : proc->getCallers()) {
212  PassManager::get()->executePass(PassID::CallArgumentUpdate, proc);
213  updateSet.insert(call->getProc()); // Make sure we redo the dataflow
214  m_removeRetSet.insert(call->getProc()); // Also schedule caller proc for more analysis
215  }
216 
217  // Now update myself
218  updateForUseChange(proc);
219 
220  // Update any other procs that need updating
221  updateSet.erase(proc); // Already done this proc
222 
223  while (!updateSet.empty()) {
224  UserProc *_proc = *updateSet.begin();
225  updateSet.erase(_proc);
226  updateForUseChange(_proc);
227  }
228  }
229 
230  m_prog->getProject()->alertDecompileDebugPoint(proc,
231  "after removing unused and redundant returns");
232  return removedRets || removedParams;
233 }

236 void UnusedReturnRemover::updateForUseChange(UserProc *proc)
237 {
238  // We need to remember the parameters, and all the livenesses for all the calls, to see if these
239  // are changed by removing returns
240  if (m_prog->getProject()->getSettings()->debugUnused) {
241  LOG_MSG("%%% updating %1 for changes to uses (returns or arguments)", proc->getName());
242  LOG_MSG("%%% updating dataflow:");
243  }
244 
245  // Save the old parameters and call liveness
246  const size_t oldNumParameters = proc->getParameters().size();
247  std::map<CallStatement *, UseCollector> callLiveness;
248 
249  for (BasicBlock *bb : *proc->getCFG()) {
250  BasicBlock::RTLRIterator rrit;
251  StatementList::reverse_iterator srit;
252  CallStatement *c = dynamic_cast<CallStatement *>(bb->getLastStmt(rrit, srit));
253 
254  // Note: we may have removed some statements, so there may no longer be a last statement!
255  if (c == nullptr) {
256  continue;
257  }
258 
259  UserProc *dest = dynamic_cast<UserProc *>(c->getDestProc());
260 
261  // Not interested in unanalysed indirect calls (not sure) or calls to lib procs
262  if (dest == nullptr) {
263  continue;
264  }
265 
266  callLiveness[c].makeCloneOf(*c->getUseCollector());
267  }
268 
269  // Have to redo dataflow to get the liveness at the calls correct
270  PassManager::get()->executePass(PassID::CallLivenessRemoval, proc);
271  PassManager::get()->executePass(PassID::BlockVarRename, proc);
272 
273  // Perform type analysis. If we are relying (as we are at present) on TA to perform ellipsis
274  // processing, do the local TA pass now. Ellipsis processing often reveals additional uses (e.g.
275  // additional parameters to printf/scanf), and removing unused statements is unsafe without full
276  // use information
277  if (proc->getStatus() < ProcStatus::FinalDone) {
278  PassManager::get()->executePass(PassID::LocalTypeAnalysis, proc);
279 
280  // Now that locals are identified, redo the dataflow
281  PassManager::get()->executePass(PassID::PhiPlacement, proc);
282 
283  PassManager::get()->executePass(PassID::BlockVarRename, proc); // Rename the locals
284  PassManager::get()->executePass(PassID::StatementPropagation, proc);
285 
286  if (m_prog->getProject()->getSettings()->verboseOutput) {
287  proc->debugPrintAll("after propagating locals");
288  }
289  }
290 
291  PassManager::get()->executePass(PassID::UnusedStatementRemoval, proc);
292  PassManager::get()->executePass(PassID::FinalParameterSearch, proc);
293 
294  if (m_prog->getProject()->getSettings()->nameParameters) {
295  // Replace the existing temporary parameters with the final ones:
296  // mapExpressionsToParameters();
297  PassManager::get()->executePass(PassID::ParameterSymbolMap, proc);
298  proc->debugPrintAll("after adding new parameters");
299  }
300 
301  // Or just CallArgumentUpdate?
302  PassManager::get()->executePass(PassID::CallDefineUpdate, proc);
303  PassManager::get()->executePass(PassID::CallArgumentUpdate, proc);
304 
305  // Have the parameters changed? If so, then all callers will need to update their arguments, and
306  // do similar analysis to the removal of returns
307  PassManager::get()->executePass(PassID::UnusedParamRemoval, proc);
308 
309  if (proc->getParameters().size() != oldNumParameters) {
310  if (m_prog->getProject()->getSettings()->debugUnused) {
311  LOG_MSG("%%% parameters changed for %1", proc->getName());
312  }
313 
314  std::set<CallStatement *> &callers = proc->getCallers();
315 
316  for (CallStatement *cc : callers) {
317  cc->updateArguments();
318  // Schedule the callers for analysis
319  m_removeRetSet.insert(cc->getProc());
320  }
321  }
322 
323  // Check if the liveness of any calls has changed
324  for (auto &[call, oldLiveness] : callLiveness) {
325  const UseCollector &newLiveness = *call->getUseCollector();
326 
327  if (newLiveness != oldLiveness) {
328  if (m_prog->getProject()->getSettings()->debugUnused) {
329  LOG_MSG("%%% Liveness for call to %1 in %2 changed",
330  call->getDestProc()->getName(), proc->getName());
331  }
332 
333  m_removeRetSet.insert(static_cast<UserProc *>(call->getDestProc()));
334  }
335  }
336 }

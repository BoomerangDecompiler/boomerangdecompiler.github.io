38 bool CFGCompressor::removeEmptyJumps(ProcCFG *cfg)
39 {
40  std::deque<BasicBlock *> bbsToRemove;
41 
42  for (BasicBlock *bb : *cfg) {
43  // Check if the BB can be removed
44  if (bb->getNumSuccessors() == 1 && bb != cfg->getEntryBB() &&
45  (bb->isEmpty() || bb->isEmptyJump())) {
46  bbsToRemove.push_back(bb);
47  }
48  }
49 
50  bool bbsRemoved = false;
51 
52  while (!bbsToRemove.empty()) {
53  BasicBlock *bb = bbsToRemove.front();
54  bbsToRemove.pop_front();
55 
56  assert(bb->getNumSuccessors() == 1);
57  BasicBlock *succ = bb->getSuccessor(0); // the one and only successor
58 
59  if (succ == bb) {
60  continue;
61  }
62 
63  succ->removePredecessor(bb);
64  bb->removeSuccessor(succ);
65 
66  for (BasicBlock *pred : bb->getPredecessors()) {
67  for (int i = 0; i < pred->getNumSuccessors(); i++) {
68  if (pred->getSuccessor(i) == bb) {
69  pred->setSuccessor(i, succ);
70  succ->addPredecessor(pred);
71  }
72  }
73  }
74 
75  bb->removeAllPredecessors();
76  cfg->removeBB(bb);
77  bbsRemoved = true;
78  }
79 
80  return bbsRemoved;
81 }

84 bool CFGCompressor::removeOrphanBBs(ProcCFG *cfg)
85 {
86  std::deque<BasicBlock *> orphans;
87 
88  for (BasicBlock *potentialOrphan : *cfg) {
89  if (potentialOrphan == cfg->getEntryBB()) {
90  // don't remove entry BasicBlock
91  continue;
92  }
93  else if (potentialOrphan->isType(BBType::Ret)) {
94  // Don't remove the ReturnStatement for noreturn functions
95  continue;
96  }
97 
98  if (potentialOrphan->getNumPredecessors() == 0) {
99  orphans.push_back(potentialOrphan);
100  }
101  }
102 
103  const bool bbsRemoved = !orphans.empty();
104 
105  while (!orphans.empty()) {
106  BasicBlock *b = orphans.front();
107  orphans.pop_front();
108 
109  for (BasicBlock *child : b->getSuccessors()) {
110  child->removePredecessor(b);
111  b->removeSuccessor(child);
112  }
113 
114  cfg->removeBB(b);
115  }
116 
117  return bbsRemoved;
118 }

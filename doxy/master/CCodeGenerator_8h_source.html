36 /// Operator precedence
37 /**
38  * Operator Name Associativity Operators
39  * Primary scope resolution LTR ::
40  * Primary LTR () [] . -> dynamic_cast typeid
41  * Unary RTL ++ -- + - ! ~ & * (type_name) sizeof new delete
42  * C++ Pointer to Member LTR .* ->*
43  * Multiplicative LTR * / %
44  * Additive LTR + -
45  * Bitwise Shift LTR << >>
46  * Relational LTR < > <= >=
47  * Equality LTR == !=
48  * Bitwise AND LTR &
49  * Bitwise Exclusive OR LTR ^
50  * Bitwise Inclusive OR LTR |
51  * Logical AND LTR &&
52  * Logical OR LTR ||
53  * Conditional RTL ? :
54  * Assignment RTL = += -= *= /= <<= >>= %= &= ^= |=
55  * Comma LTR ,
56  */
57 enum class OpPrec : uint8_t
58 {
59  Invalid = 0,
60  Scope = 1, ///< (LTR) Primary scope resolution
61  Prim = 2, ///< (LTR) Primary
62  Unary = 3, ///< (RTL) Unary
63  PtrMem = 4, ///< (LTR) C++ Pointer to Member
64  Mult = 5, ///< (LTR) Multiplicative
65  Add = 6, ///< (LTR) Additive
66  BitShift = 7, ///< (LTR) Bitwise Shift
67  Comp3Way = 8, ///< (LTR) 3-way comparison operator (C++20)
68  Rel = 9, ///< (LTR) Relational
69  Equal = 10, ///< (LTR) Equality
70  BitAnd = 11, ///< (LTR) Bitwise AND
71  BitXor = 12, ///< (LTR) Bitwise Exclusive OR
72  BitOr = 13, ///< (LTR) Bitwise Inclusive OR
73  LogAnd = 14, ///< (LTR) Logical AND
74  LogOr = 15, ///< (LTR) Logical OR
75  Cond = 16, ///< (RTL) Ternary conditional operator
76  Assign = 16, ///< (RTL) Assignment (same precedence as ?: operator)
77  Comma = 17, ///< (LTR) Comma
78 
79  None = 0xFF, ///< Outer level (no parens required)
80 };

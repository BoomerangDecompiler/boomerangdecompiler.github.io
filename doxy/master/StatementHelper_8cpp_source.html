26  OPER op = opWild;
27  // Special for PPC unsigned compares; may be other cases in the future
28  bool makeUns = condExp->access<Const, 1>()->getStr().startsWith("SUBFLAGSNL");
29

31  case BranchType::JE: op = opEquals; break;
32 
33  case BranchType::JNE: op = opNotEqual; break;
34 
35  case BranchType::JSL:
36 
37  if (makeUns) {
38  op = opLessUns;
39  }
40  else {
41  op = opLess;
42  }
43 
44  break;
45 
46  case BranchType::JSLE:
47 
48  if (makeUns) {
49  op = opLessEqUns;
50  }
51  else {
52  op = opLessEq;
53  }
54 
55  break;
56 
57  case BranchType::JSGE:
58 
59  if (makeUns) {
60  op = opGtrEqUns;
61  }
62  else {
63  op = opGtrEq;
64  }
65 
66  break;
67 
68  case BranchType::JSG:
69 
70  if (makeUns) {
71  op = opGtrUns;
72  }
73  else {
74  op = opGtr;
75  }
76 
77  break;
78 
79  case BranchType::JUL: op = opLessUns; break;
80 
81  case BranchType::JULE: op = opLessEqUns; break;
82 
83  case BranchType::JUGE: op = opGtrEqUns; break;
84 
85  case BranchType::JUG: op = opGtrUns; break;

87  case BranchType::JMI:
88  /*
89  * condExp
90  * / \
91  * Const opList
92  * "SUBFLAGS" / \
93  * P1 opList
94  * / \
95  * P2 opList
96  * / \
97  * P3 opNil
98  */

100  condExp->access<const Exp, 2, 2, 2, 1>()->clone(), Const::get(0));
101  break;
102 
103  case BranchType::JPOS:
104  condExp = Binary::get(opGtrEq, // P3 >= 0
105  condExp->access<const Exp, 2, 2, 2, 1>()->clone(), Const::get(0));
106  break;
107 
108  case BranchType::JOF:
109  case BranchType::JNOF:
110  case BranchType::JPAR:
111  case BranchType::JNPAR: break;
112 
113  case BranchType::INVALID: assert(false); break;
114  }
115 
116  if (op != opWild) {
117  condExp = Binary::get(op,
118  condExp->getSubExp2()->getSubExp1()->clone(), // P1
119  condExp->getSubExp2()->getSubExp2()->getSubExp1()->clone()); // P2
120  }
121  }
122  else if (condOp == opFlagCall &&
123  (condExp->access<Const, 1>()->getStr().startsWith("LOGICALFLAGS") ||
124  condExp->access<Const, 1>()->getStr().startsWith("INCDECFLAGS"))) {
125  OPER op = opWild;
126 
127  switch (jtCond) {
128  case BranchType::JE: op = opEquals; break;
129 
130  case BranchType::JNE: op = opNotEqual; break;
131 
132  case BranchType::JMI: op = opLess; break;
133 
134  case BranchType::JPOS: op = opGtrEq; break;
135 
136  // FIXME: This next set is quite shakey. Really, we should pull all the individual flag
137  // definitions out of the flag definitions, and substitute these into the equivalent
138  // conditions for the branches (a big, ugly job).
139  case BranchType::JSL: op = opLess; break;
140 
141  case BranchType::JSLE: op = opLessEq; break;
142 
143  case BranchType::JSGE: op = opGtrEq; break;
144 
145  case BranchType::JSG: op = opGtr; break;

147  // These next few seem to fluke working fine on architectures like X86 and SPARC which
148  // clear the carry on all logical operations.
149  case BranchType::JUL:
150  // NOTE: this is equivalent to never branching, since nothing
151  // can be unsigned less than zero
152  op = opLessUns;
153  break;
154 
155  case BranchType::JULE: op = opLessEqUns; break;
156 
157  case BranchType::JUGE:
158  op = opGtrEqUns;
159  break; // Similarly, this is equivalent to always branching

161  case BranchType::JUG: op = opGtrUns; break;

164  case BranchType::JNPAR: {

168  * condExp
169  * / \
170  * Const opList
171  * "LOGICALFLAGS8" / \
172  * opBitAnd opNil
173  * / \
174  * opFlagCall opIntConst

176  * Const opList
177  * "SETFFLAGS" / \
178  * P1 opList
179  * / \
180  * P2 opNil
181  */

184

186  test = test->getSubExp1();
187  }
188

190  return false; // Just not propagated yet
191  }

194 
195  if (flagsParam->getOper() == opBitAnd) {
196  SharedExp setFlagsParam = flagsParam->getSubExp2();
197 
198  if (setFlagsParam->isIntConst()) {
199  mask = setFlagsParam->access<Const>()->getInt();
200  }
201  }
202 
203  if (!flagsParam->getSubExp1() || !flagsParam->getSubExp2() || (mask & ~0x41) != 0) {
204  LOG_WARN("Unhandled pentium branch if parity with condExp = %1", condExp);
205  return false;
206  }
207 
208  const SharedExp at_opFlagsCall_List = flagsParam->getSubExp1()->getSubExp2();
209  if (!at_opFlagsCall_List || !at_opFlagsCall_List->getSubExp1() ||
210  !at_opFlagsCall_List->getSubExp2() ||
211  !at_opFlagsCall_List->getSubExp2()->getSubExp1()) {
212  LOG_WARN("Unhandled pentium branch if parity with condExp = %1", condExp);
213  return false;
214  }
215 
216  // Sometimes the mask includes the 0x4 bit, but we expect that to be off all the time.
217  // So effectively the branch is for any one of the (one or two) bits being on. For
218  // example, if the mask is 0x41, we are branching of less (0x1) or equal (0x41).
219  mask &= 0x41;
220  OPER _op;
221 
222  switch (mask) {
223  case 0:
224  LOG_WARN("Unhandled pentium branch if parity with condExp = %1", condExp);
225  return false;

228 
229  case 0x40: _op = opEquals; break;

231  case 0x41: _op = jtCond == BranchType::JPAR ? opLessEq : opGtr; break;

235  break;
236  }

238  condExp = Binary::get(_op, at_opFlagsCall_List->getSubExp1()->clone(),
239  at_opFlagsCall_List->getSubExp2()->getSubExp1()->clone());
240  return true; // This is a floating point comparison
241  }
242 
243  default: break;
244  }

247  condExp = Binary::get(op, condExp->getSubExp2()->getSubExp1()->clone(), Const::get(0));

279  // ICK! This is all PENTIUM SPECIFIC... needs to go somewhere else.
280  // Might be of the form (SETFFLAGS(...) & MASK) RELOP INTCONST where MASK could be a combination

288  SharedExp right = condExp->getSubExp2();
289  bool hasXor40 = false;
290 
291  if ((left->getOper() == opBitXor) && right->isIntConst()) {

293

296

312  int mask = left2->access<Const>()->getInt();

316

320  }
321  else {

323  case 1:
324

328  }
329  else {

331  }
332 
333  break;
334 
335  case 0x40:
336

338  ((condOp == opNotEqual) && (k == 0x40))) {

340  }

343  }
344

346 
347  case 0x41:

357  }

359  break;
360 
361  case 1:

364  op = opLess;
365  }
366  else {
367  op = opGtrEq;
368  }
369 
370  break;
371 
372  case 0x40:

392  }
393

395  condExp = Binary::get(op, left1->getSubExp2()->getSubExp1(),
396  left1->getSubExp2()->getSubExp2()->getSubExp1());
397  return true; // This is now a float comparison
398  }
399  }
400  }
401  }
402  else if (condExp->isFlagCall() && condExp->access<Const, 1>()->getStr() == "SAHFFLAGS") {
403  const SharedExp param = condExp->access<Exp, 2, 1>();
404  if (param->isFlagCall() && param->access<Const, 1>()->getStr() == "SETFFLAGS") {
405  // Can happen e.g. with the following assembly:
406  // fucompp
407  // fnstsw ax
408  // sahf
409  // jne <foo>

411  const SharedExp floatParam2 = param->access<Const, 2, 2, 1>();

414  // %ZF = floatParam1 == floatParam2
415  // %CF = floatParam1 < floatParam2
416  // %PF = 0 (if not unordered)
417  switch (jtCond) {
418  case BranchType::JNE: { // ~%ZF
419  condExp = Binary::get(opNotEqual, floatParam1, floatParam2);
420  return true;
421  }
422  case BranchType::JE: { // %ZF
423  condExp = Binary::get(opEquals, floatParam1, floatParam2);
424  return true;
425  }
426  case BranchType::JPAR: { // %PF
427  condExp = Terminal::get(opFalse);
428  return true;
429  }
430  case BranchType::JNPAR: { // ~%PF
431  condExp = Terminal::get(opTrue);
432  return true;
433  }
434  case BranchType::JUL: { // %CF
435  condExp = Binary::get(opLess, floatParam1, floatParam2);
436  return true;
437  }
438  case BranchType::JUGE: { // ~%CF
439  condExp = Binary::get(opGtrEq, floatParam1, floatParam2);
440  return true;
441  }
442  case BranchType::JULE: { // %CF | %ZF
443  condExp = Binary::get(opLessEq, floatParam1, floatParam2);
444  return true;
445  }
446  case BranchType::JUG: { // ~%CF & ~%ZF
447  condExp = Binary::get(opGtr, floatParam1, floatParam2);
448  return true;
449  }
450  default: break;
451  }
452  }
453  }

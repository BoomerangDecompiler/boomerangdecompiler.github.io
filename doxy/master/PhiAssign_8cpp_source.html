207 void PhiAssign::simplify()
208 {
209  m_lhs = m_lhs->simplify();
210 
211  if (m_defs.empty()) {
212  return;
213  }
214 
215  bool allSame = true;
216  Statement *firstDef = (*begin())->getDef();
217  UserProc *proc = this->getProc();
218 
219  for (auto &refExp : *this) {
220  if (refExp->getDef() != firstDef) {
221  allSame = false;
222  break;
223  }
224  }
225 
226  if (allSame) {
227  LOG_VERBOSE("all the same in %1", this);
228  proc->replacePhiByAssign(this, RefExp::get(m_lhs, firstDef));
229  return;
230  }
231 
232  bool onlyOneNotThis = true;
233  Statement *notthis = STMT_WILD;
234 
235  for (const std::shared_ptr<RefExp> &ref : *this) {
236  Statement *def = ref->getDef();
237  if (def == this) {
238  continue; // ok
239  }
240  else if (notthis == STMT_WILD) {
241  notthis = def;
242  }
243  else {
244  onlyOneNotThis = false;
245  break;
246  }
247  }
248 
249  if (onlyOneNotThis && (notthis != STMT_WILD)) {
250  LOG_VERBOSE("All but one not this in %1", this);
251 
252  proc->replacePhiByAssign(this, RefExp::get(m_lhs, notthis));
253  return;
254  }
255 }

288 void PhiAssign::removeAllReferences(const std::shared_ptr<RefExp> &refExp)
289 {
290  for (PhiDefs::iterator pi = m_defs.begin(); pi != m_defs.end();) {
291  std::shared_ptr<RefExp> &p = pi->second;
292  assert(p->getSubExp1());
293 
294  if (*p == *refExp) { // Will we ever see this?
295  pi = m_defs.erase(pi); // Erase this phi parameter
296  continue;
297  }
298 
299  // Chase the definition
300  Statement *def = p->getDef();
301  if (def && def->isAssign()) {
302  SharedExp rhs = static_cast<Assign *>(def)->getRight();
303 
304  if (*rhs == *refExp) { // Check if RHS is a single reference to this
305  pi = m_defs.erase(pi); // Yes, erase this phi parameter
306  continue;
307  }
308  }

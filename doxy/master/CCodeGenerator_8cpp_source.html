46 void CCodeGenerator::generateCode(const Prog *prog, Module *cluster, UserProc *proc,
47  bool /*intermixRTL*/)
48 {
49  const bool generate_all = cluster == nullptr || cluster == prog->getRootModule();
50  bool all_procedures = (proc == nullptr);
51 
52  // First declare prototypes
53  for (const auto &module : prog->getModuleList()) {
54  for (Function *func : *module) {
55  if (func->isLib()) {
56  continue;
57  }
58 
59  UserProc *_proc = static_cast<UserProc *>(func);
60  addPrototype(_proc); // May be the wrong signature if _proc has ellipsis
61  }
62  appendLine("");
63  }
64 
65  if (generate_all) {
66  if (proc == nullptr) {
67  const bool global = !prog->getGlobals().empty();
68 
69  for (auto &elem : prog->getGlobals()) {
70  // Check for an initial value
71  SharedExp e = elem->getInitialValue();
72  // if (e) {
73  addGlobal(elem->getName(), elem->getType(), e);
74  }
75 
76  if (global) {
77  print(prog->getRootModule());
78  }
79  }
80 
81  appendLine(""); // Separate prototype(s) from first proc
82  print(prog->getRootModule());
83  }
84 
85  for (const auto &module : prog->getModuleList()) {
86  if (!generate_all && (module.get() != cluster)) {
87  continue;
88  }
89 
90  for (Function *func : *module) {
91  if (func->isLib()) {
92  continue;
93  }
94 
95  UserProc *_proc = static_cast<UserProc *>(func);
96 
97  if (!_proc->isDecoded()) {
98  continue;
99  }
100 
101  if (!all_procedures && (proc != _proc)) {
102  continue;
103  }
104 
105  generateCode(_proc);
106  print(module.get());
107  }
108  }

112 void CCodeGenerator::addAssignmentStatement(const Assign *asgn)
113 {
114  // Gerard: shouldn't these 3 types of statements be removed earlier?
115  if (asgn->getLeft()->isPC()) {
116  return; // Never want to see assignments to %PC
117  }
118 
119  SharedExp result;
120 
121  if (asgn->getRight()->search(Terminal(opPC), result)) { // Gerard: what's this?
122  return;
123  }
124 
125  // ok I want this now
126  // if (asgn->getLeft()->isFlags())
127  // return;
128 
129  QString tgt;
130  OStream ost(&tgt);
131  indent(ost, m_indent);
132 
133  SharedType asgnType = asgn->getType();
134  SharedExp lhs = asgn->getLeft();
135  SharedExp rhs = asgn->getRight();
136 
137  if (*lhs == *rhs) {
138  return; // never want to see a = a;
139  }
140 
141  if (lhs->isMemOf() && asgnType && !asgnType->isVoid()) {
142  appendExp(ost, TypedExp::get(asgnType, lhs), OpPrec::Assign);
143  }
144  else if (lhs->isGlobal() && asgn->getType()->isArray()) {
145  appendExp(ost, Binary::get(opArrayIndex, lhs, Const::get(0)), OpPrec::Assign);
146  }
147  else if ((lhs->getOper() == opAt) && lhs->getSubExp2()->isIntConst() &&
148  lhs->getSubExp3()->isIntConst()) {
149  // exp1@[n:m] := rhs -> exp1 = exp1 & mask | rhs << m where mask = ~((1 << m-n+1)-1)
150  SharedExp exp1 = lhs->getSubExp1();
151  const int n = lhs->access<Const, 2>()->getInt();
152  const int m = lhs->access<Const, 3>()->getInt();
153  appendExp(ost, exp1, OpPrec::Assign);
154  ost << " = ";
155 
156  // MSVC winges without most of these parentheses
157  const int mask = ~(((1 << (m - n + 1)) - 1) << m);
158 
159  // clang-format off
160  rhs = Binary::get(opBitAnd,
161  exp1,
162  Binary::get(opBitOr,
163  Const::get(mask),
164  Binary::get(opShL, rhs, Const::get(m))));
165  // clang-format on
166  rhs = rhs->simplify();
167 
168  appendExp(ost, rhs, OpPrec::Assign);
169  ost << ";";
170  appendLine(tgt);
171  return;
172  }
173  else {
174  appendExp(ost, lhs, OpPrec::Assign); // Ordinary LHS
175  }
176 
177  // C has special syntax for this, eg += and ++
178  // however it's not always acceptable for assigns to m[] (?)
179  bool useIncrement = false; // use ++ / --
180  bool useShortForm = false; // use += / -=
181 
182  if ((rhs->getOper() == opPlus || rhs->getOper() == opMinus) && (*rhs->getSubExp1() == *lhs)) {
183  // we now have something like a = a + b -> shorten it
184  useShortForm = true;
185 
186  SharedConstExp b = rhs->getSubExp2();
187  if (b->isIntConst()) {
188  if (b->access<const Const>()->getInt() == 1) {
189  useIncrement = true;
190  }
191  else if (asgn->getType()->isPointer()) {
192  // add ptr, 4 in assembly for pointers to 32 bit data is ptr++ in C code
193  const Type::Size
194  ptrSize = asgn->getType()->as<PointerType>()->getPointsTo()->getSize();
195 
196  if (ptrSize == (Type::Size)rhs->access<const Const, 2>()->getInt() * 8) {
197  useIncrement = true;
198  }
199  }
200  }
201  }
202 
203  if (useIncrement) {
204  if (rhs->getOper() == opPlus) {
205  ost << "++;";
206  }
207  else {
208  ost << "--;";
209  }
210  }
211  else if (useShortForm) {
212  if (rhs->getOper() == opPlus) {
213  ost << " += ";
214  }
215  else {
216  ost << " -= ";
217  }
218 
219  appendExp(ost, rhs->getSubExp2(), OpPrec::Assign);
220  ost << ";";
221  }
222  else {
223  ost << " = ";
224  appendExp(ost, rhs, OpPrec::Assign);
225  ost << ";";
226  }
227 
228  appendLine(tgt);
229 }

233  const StatementList &args, const StatementList &results)
234 {
235  QString tgt;
236  OStream s(&tgt);
237 
238  indent(s, m_indent);

240  if (!results.empty()) {
241  // FIXME: Needs changing if more than one real result (return a struct)
242  SharedConstExp firstRet = (static_cast<const Assignment *>(*results.begin()))->getLeft();
243  appendExp(s, firstRet, OpPrec::Assign);

245  }
246 
247  s << name << "(";
248  bool first = true;
249  int n = 0;
250 
251  for (StatementList::const_iterator ss = args.begin(); ss != args.end(); ++ss, ++n) {
252  if (first) {
253  first = false;
254  }
255  else {
256  s << ", ";
257  }
258 
259  assert((*ss)->isAssignment() && dynamic_cast<Assignment *>(*ss) != nullptr);
260 
261  Assignment *arg_assign = static_cast<Assignment *>(*ss);
262  SharedType t = arg_assign->getType();
263  SharedExp as_arg = arg_assign->getRight();
264  bool ok = true;
265 
266  if (as_arg->isIntConst() && t && t->isPointer() &&
267  t->as<PointerType>()->getPointsTo()->isFunc()) {
268  Function *p = proc->getProg()->getFunctionByAddr(as_arg->access<Const>()->getAddr());
269 
270  if (p) {
271  s << p->getName();
272  ok = false;
273  }
274  }
275 
276  if (ok) {
277  appendExp(s, as_arg, OpPrec::Comma);
278  }
279  }
280 
281  s << ");";
282 
283  if (results.size() > 1) {
284  first = true;
285  s << " /* Warning: also results in ";
286 
287  for (auto ss = std::next(results.begin()); ss != results.end(); ++ss) {
288  if (first) {
289  first = false;
290  }
291  else {
292  s << ", ";
293  }
294 
295  assert(dynamic_cast<Assignment *>(*ss) != nullptr);
296  const Assignment *assign = static_cast<const Assignment *>(*ss);
297  appendExp(s, assign->getLeft(), OpPrec::Comma);
298  }
299 
300  s << " */";
301  }
302 
303  appendLine(tgt);
304 }

409 void CCodeGenerator::generateCode(UserProc *proc)
410 {
411  m_lines.clear();
412  m_proc = proc;
413 
414  if (!proc->getCFG() || !proc->getEntryBB()) {
415  return;
416  }
417 
418  m_analyzer.structureCFG(proc->getCFG());
419  PassManager::get()->executePass(PassID::UnusedLocalRemoval, proc);
420 
421  // Note: don't try to remove unused statements here; that requires the
422  // RefExps, which are all gone now (transformed out of SSA form)!
423 
424  if (m_proc->getProg()->getProject()->getSettings()->printRTLs) {
425  LOG_VERBOSE("%1", proc->toString());
426  }
427 
428  // Start generating code for this procedure.
429  this->addProcStart(proc);
430 
431  // Local variables; print everything in the locals map
432  std::map<QString, SharedType>::const_iterator last = proc->getLocals().end();
433 
434  if (!proc->getLocals().empty()) {
435  last = std::prev(last);
436  }
437 
438  for (auto it = proc->getLocals().begin(); it != proc->getLocals().end(); ++it) {
439  SharedType locType = it->second;
440 
441  if ((locType == nullptr) || locType->isVoid()) {
442  locType = IntegerType::get(STD_SIZE);
443  }
444 
445  addLocal(it->first, locType, it == last);
446  }
447 
448  // Start generating "real" code
449  std::list<const BasicBlock *> followSet, gotoSet;
450  generateCode(proc->getEntryBB(), nullptr, followSet, gotoSet, proc);
451 
452  addProcEnd();
453 
454  if (m_proc->getProg()->getProject()->getSettings()->removeLabels) {
455  removeUnusedLabels();
456  }
457 
458  proc->setStatus(ProcStatus::CodegenDone);
459 }

484 void CCodeGenerator::addFunctionSignature(UserProc *proc, bool open)
485 {
486  QString tgt;
487  OStream s(&tgt);
488  ReturnStatement *returns = proc->getRetStmt();
489  SharedType retType;
490 
491  if (proc->getSignature()->isForced()) {
492  if (proc->getSignature()->getNumReturns() == 0) {
493  s << "void ";
494  }
495  else {
496  int n = 0;
497  SharedExp e = proc->getSignature()->getReturnExp(0);
498 
499  if (e->isRegN(Util::getStackRegisterIndex(proc->getProg()))) {
500  n = 1;
501  }
502 
503  if (n < proc->getSignature()->getNumReturns()) {
504  retType = proc->getSignature()->getReturnType(n);
505  }
506 
507  if (retType == nullptr) {
508  s << "void ";
509  }
510  }
511  }
512  else if ((returns == nullptr) || (returns->getNumReturns() == 0)) {
513  s << "void ";
514  }
515  else {
516  Assign *firstRet = static_cast<Assign *>(*returns->begin());
517  retType = firstRet->getType();
518 
519  if ((retType == nullptr) || retType->isVoid()) {
520  // There is a real return; make it integer (Remove with AD HOC type analysis)
521  retType = IntegerType::get(STD_SIZE, Sign::Unknown);
522  }
523  }
524 
525  if (retType) {
526  appendType(s, retType);
527 
528  if (!retType->isPointer()) { // NOTE: assumes type *proc( style
529  s << " ";
530  }
531  }
532 
533  s << proc->getName() << "(";
534  StatementList &parameters = proc->getParameters();
535 
536  if ((parameters.size() > 10) && open) {
537  LOG_WARN("Proc %1 has %2 parameters", proc->getName(), parameters.size());
538  }
539 
540  bool first = true;
541 
542  for (auto &parameter : parameters) {
543  if (first) {
544  first = false;
545  }
546  else {
547  s << ", ";
548  }
549 
550  Assignment *as = static_cast<Assignment *>(parameter);
551  SharedExp left = as->getLeft();
552  SharedType ty = as->getType();
553 
554  if (ty == nullptr) {
555  if (proc->getProg()->getProject()->getSettings()->verboseOutput) {
556  LOG_ERROR("No type for parameter %1!", left);
557  }
558 
559  ty = IntegerType::get(STD_SIZE, Sign::Unknown);
560  }
561 
562  QString name;
563 
564  if (left->isParam()) {
565  name = left->access<Const, 1>()->getStr();
566  }
567  else {
568  LOG_ERROR("Parameter %1 is not opParam!", left);
569  name = "??";
570  }
571 
572  if (ty->isPointer() && ty->as<PointerType>()->getPointsTo()->isArray()) {
573  // C does this by default when you pass an array, i.e. you pass &array meaning array
574  // Replace all m[param] with foo, param with foo, then foo with param
575  ty = ty->as<PointerType>()->getPointsTo();
576 
577  SharedExp foo = Const::get("foo123412341234");
578  m_proc->searchAndReplace(*Location::memOf(left, nullptr), foo);
579  m_proc->searchAndReplace(*left, foo);
580  m_proc->searchAndReplace(*foo, left);
581  }
582 
583  appendTypeIdent(s, ty, name);
584  }
585 
586  s << ")";
587 
588  if (open) {
589  appendLine(tgt);
590  appendLine("{");
591  m_indent++;
592  }
593  else {
594  s << ";";
595  appendLine(tgt);
596  }
597 }

978 void CCodeGenerator::appendExp(OStream &str, const SharedConstExp &exp, OpPrec curPrec,
979  bool uns /* = false */)
980 {
981  const OPER op = exp->getOper();
982 
983  switch (op) {
984  case opIntConst: {
985  const Const &constExp = *exp->access<Const>();
986  int K = constExp.getInt();
987 
988  if (uns && (K < 0)) {
989  // An unsigned constant. Use some heuristics
990  unsigned rem = static_cast<unsigned int>(K) % 100;
991 
992  if ((rem == 0) || (rem == 99) || (K > -128)) {
993  // A multiple of 100, or one less; use 4000000000U style
994  char num[16];
995  sprintf(num, "%u", static_cast<unsigned int>(K));
996  str << num << "U";
997  }
998  else {
999  // Output it in 0xF0000000 style
1000  str << "0x" << QString::number(uint32_t(K), 16);
1001  }
1002  }
1003  else {
1004  if (constExp.getType() && constExp.getType()->isChar()) {
1005  switch (K) {
1006  case '\a': str << "'\\a'"; break;
1007  case '\b': str << "'\\b'"; break;
1008  case '\f': str << "'\\f'"; break;
1009  case '\n': str << "'\\n'"; break;
1010  case '\r': str << "'\\r'"; break;
1011  case '\t': str << "'\\t'"; break;
1012  case '\v': str << "'\\v'"; break;
1013  case '\\': str << "'\\\\'"; break;
1014  case '\?': str << "'\\?'"; break;
1015  case '\'': str << "'\\''"; break;
1016  case '\"': str << "'\\\"'"; break;
1017  default: str << "'" << static_cast<char>(K) << "'";
1018  }
1019  }
1020  else {
1021  // More heuristics
1022  if ((-2048 < K) && (K < 2048)) {
1023  str << K; // Just a plain vanilla int
1024  }
1025  else {
1026  str << "0x" << QString::number(uint32_t(K), 16); // 0x2000 style
1027  }
1028  }
1029  }
1030 
1031  break;
1032  }
1033 
1034  case opLongConst: {
1035  const Const &constExp = *exp->access<Const>();
1036  // str << std::dec << c->getLong() << "LL"; break;
1037  if ((static_cast<long long>(constExp.getLong()) < -1000LL) ||
1038  (constExp.getLong() > 1000ULL)) {
1039  str << "0x" << QString::number(constExp.getLong(), 16) << "LL";
1040  }
1041  else {
1042  str << constExp.getLong() << "LL";
1043  }
1044  break;
1045  }
1046 
1047  case opFltConst: {
1048  const Const &constExp = *exp->access<Const>();
1049  // str.precision(4); // What to do with precision here? Would be nice to avoid 1.00000
1050  // or 0.99999
1051  QString flt_val = QString::number(constExp.getFlt(), 'g', 8);
1052 
1053  if (!flt_val.contains('.')) {
1054  flt_val += '.';
1055  }
1056 
1057  str << flt_val;
1058  break;
1059  }
1060 
1061  case opStrConst: {
1062  const Const &constExp = *exp->access<Const>();
1063  // escape string:
1064  str << "\"" << Util::escapeStr(constExp.getRawStr()) << "\"";
1065  break;
1066  }
1067 
1068  case opFuncConst: {
1069  const Const &constExp = *exp->access<Const>();
1070  str << constExp.getFuncName();
1071  break;
1072  }
1073 
1074  case opAddrOf: {
1075  const Unary &unaryExp = *exp->access<Unary>();
1076  const SharedConstExp sub = unaryExp.getSubExp1();
1077 
1078  if (sub->isGlobal()) {
1079  Prog *prog = m_proc->getProg();
1080  SharedType gt = prog->getGlobalType(sub->access<const Const, 1>()->getStr());
1081 
1082  if (gt && (gt->resolvesToArray() || gt->isCString())) {
1083  // Special C requirement: don't emit "&" for address of an array or char*
1084  appendExp(str, sub, curPrec);
1085  break;
1086  }
1087  }
1088 
1089  if (sub->isMemOf()) {
1090  // Avoid &*(type*)sub, just emit sub
1091  appendExp(str, sub->getSubExp1(), OpPrec::Unary);
1092  }
1093  else {
1094  openParen(str, curPrec, OpPrec::Unary);
1095  str << "&";
1096  appendExp(str, sub, OpPrec::Unary);
1097  closeParen(str, curPrec, OpPrec::Unary);
1098  }
1099 
1100  break;

2759

2762  SharedExp caseVal;

77 /*
78  * The layout of the file depends on the filetype. For all but the MH_OBJECT
79  * file type the segments are padded out and aligned on a segment alignment
80  * boundary for efficient demand pageing. The MH_EXECUTE, MH_FVMLIB, MH_DYLIB,
81  * MH_DYLINKER and MH_BUNDLE file types also have the headers included as part
82  * of their first segment.
83  *
84  * The file type MH_OBJECT is a compact format intended as output of the
85  * assembler and input (and possibly output) of the link editor (the .o
86  * format). All sections are in one unnamed segment with no segment padding.
87  * This format is used as an executable format when the file is so small the
88  * segment padding greatly increases its size.
89  *
90  * The file type MH_PRELOAD is an executable format intended for things that
91  * are not executed under the kernel (proms, stand alones, kernels, etc). The
92  * format can be executed under the kernel but may demand paged it and not
93  * preload it before execution.
94  *
95  * A core file is in MH_CORE format and can be any in an arbritray legal
96  * Mach-O file.
97  *
98  * Constants for the filetype field of the mach_header
99  */
100 #define MH_OBJECT 0x1 /* relocatable object file */
101 #define MH_EXECUTE 0x2 /* demand paged executable file */
102 #define MH_FVMLIB 0x3 /* fixed VM shared library file */
103 #define MH_CORE 0x4 /* core file */
104 #define MH_PRELOAD 0x5 /* preloaded executable file */
105 #define MH_DYLIB 0x6 /* dynamically bound shared library */
106 #define MH_DYLINKER 0x7 /* dynamic link editor */
107 #define MH_BUNDLE 0x8 /* dynamically bound bundle file */
108 #define MH_DYLIB_STUB 0x9 /* shared library stub for static */
109 /* linking only, no section contents */

133 /*
134  * The load commands directly follow the mach_header. The total size of all
135  * of the commands is given by the sizeofcmds field in the mach_header. All
136  * load commands must have as their first two fields cmd and cmdsize. The cmd
137  * field is filled in with a constant for that command type. Each command type
138  * has a structure specifically for it. The cmdsize field is the size in bytes
139  * of the particular load command structure plus anything that follows it that
140  * is a part of the load command (i.e. section structures, strings, etc.). To
141  * advance to the next load command the cmdsize can be added to the offset or
142  * pointer of the current load command. The cmdsize MUST be a multiple of
143  * 4 bytes (this is forever the maximum alignment of any load commands).
144  * The padded bytes must be zero. All tables in the object file must also
145  * follow these rules so the file can be memory mapped. Otherwise the pointers
146  * to these tables will not work well or at all on some machines. With all
147  * padding zeroed like objects will compare byte for byte.
148  */
149 struct load_command
150 {
151  uint32_t cmd; /* type of load command */
152  uint32_t cmdsize; /* total size of command in bytes */
153 };

212 /*
213  * The segment load command indicates that a part of this file is to be
214  * mapped into the task's address space. The size of this segment in memory,
215  * vmsize, maybe equal to or larger than the amount to map from this file,
216  * filesize. The file is mapped starting at fileoff to the beginning of
217  * the segment in memory, vmaddr. The rest of the memory of the segment,
218  * if any, is allocated zero fill on demand. The segment's maximum virtual
219  * memory protection and initial virtual memory protection are specified
220  * by the maxprot and initprot fields. If the segment has sections then the
221  * section structures directly follow the segment command and their size is
222  * reflected in cmdsize.
223  */
224 struct segment_command
225 {
226  uint32_t cmd; /* LC_SEGMENT */
227  uint32_t cmdsize; /* includes sizeof section structs */
228  char segname[16]; /* segment name */
229  uint32_t vmaddr; /* memory address of this segment */
230  uint32_t vmsize; /* memory size of this segment */
231  uint32_t fileoff; /* file offset of this segment */
232  uint32_t filesize; /* amount to map from the file */
233  vm_prot_t maxprot; /* maximum VM protection */
234  vm_prot_t initprot; /* initial VM protection */
235  uint32_t nsects; /* number of sections in segment */
236  uint32_t flags; /* flags */
237 };

249 /*
250  * A segment is made up of zero or more sections. Non-MH_OBJECT files have
251  * all of their segments with the proper sections in each, and padded to the
252  * specified segment alignment when produced by the link editor. The first
253  * segment of a MH_EXECUTE and MH_FVMLIB format file contains the mach_header
254  * and load commands of the object file before its first section. The zero
255  * fill sections are always last in their segment (in all formats). This
256  * allows the zeroed segment padding to be mapped into memory where zero fill
257  * sections might be.
258  *
259  * The MH_OBJECT format has all of its sections in one segment for
260  * compactness. There is no padding to a specified segment boundary and the
261  * mach_header and load commands are not part of the segment.
262  *
263  * Sections with the same section name, sectname, going into the same segment,
264  * segname, are combined by the link editor. The resulting section is aligned
265  * to the maximum alignment of the combined sections and is the new section's
266  * alignment. The combined sections are aligned to their original alignment in
267  * the combined section. Any padded bytes to get the specified alignment are
268  * zeroed.
269  *
270  * The format of the relocation entries referenced by the reloff and nreloc
271  * fields of the section structure for mach object files is described in the
272  * header file <reloc.h>.
273  */
274 struct section
275 {
276  char sectname[16]; /* name of this section */
277  char segname[16]; /* segment this section goes in */
278  uint32_t addr; /* memory address of this section */
279  uint32_t size; /* size in bytes of this section */
280  uint32_t offset; /* file offset of this section */
281  uint32_t align; /* section alignment (power of 2) */
282  uint32_t reloff; /* file offset of relocation entries */
283  uint32_t nreloc; /* number of relocation entries */
284  uint32_t flags; /* flags (section type and attributes)*/
285  uint32_t reserved1; /* reserved */
286  uint32_t reserved2; /* reserved */
287 };

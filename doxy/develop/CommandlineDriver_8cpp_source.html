104 int CommandlineDriver::applyCommandline(const QStringList &args)
105 {
106  bool interactiveMode = false;
107 
108  if (args.size() < 2) {
109  help();
110  return 1;
111  }
112 
113  QString binaryPath;
114 
115  for (int i = 1; i < args.size(); ++i) {
116  const QString arg = args[i];
117 
118  if (!arg.startsWith('-') || arg == "-") {
119  // every argument but last must begin with '-'
120  if (i + 1 == args.size()) {
121  binaryPath = arg;

352  return 1;

363 
364  continue;
365  }
366  else if (arg == "-S") {
367  if (++i == args.size()) {
368  help();
369  return 1;
370  }
371 
372  bool converted = false;
373  minsToStopAfter = args[i].toInt(&converted, 0);
374 
375  if (!converted) {
376  std::cerr << "'-S': Bad argument '" << args[i].toStdString() << "' (try --help)."
377  << std::endl;
378  return 1;
379  }
380 
381  continue;
382  }
383  else if (arg == "--") {
384  if (i + 2 != args.size()) {
385  help();
386  return 1;
387  }
388 
389  binaryPath = args[i + 1];
390  break;
391  }
392  else if (arg.size() == 2) {
393  switch (arg[1].toLatin1()) {
394  case 'E':
395  m_project->getSettings()->decodeChildren = false;
396  // Fall through
397 
398  case 'e': {
399  Address addr;
400  m_project->getSettings()->decodeMain = false;
401 
402  if (++i == args.size()) {
403  help();
404  return 1;
405  }
406 
407  bool converted = false;
408  addr = Address(args[i].toLongLong(&converted, 0));
409 
410  if (!converted) {
411  std::cerr << "'" << arg.toStdString()
412  << "' Bad address: " << args[i].toStdString() << " (try --help)."
413  << std::endl;
414  return 1;
415  }
416 
417  m_project->getSettings()->m_entryPoints.push_back(addr);
418  continue;
419  }
420  default: help(); return 1;
421  }
422  }
423 
424 
425  if (i + 1 != args.size()) {
426  // switch not recognized
427  help();
428  return false;
429  }
430  else {
431  binaryPath = arg;
432  }
433  }
434 
435  if (interactiveMode) {
436  return interactiveMain();
437  }
438  else if (binaryPath == "") {
439  help();
440  return 1;
441  }
442 
443  if (minsToStopAfter > 0) {
444  LOG_MSG("Stopping decompile after %1 minutes", minsToStopAfter);
445  m_kill_timer.setSingleShot(true);
446  m_kill_timer.start(1000 * 60 * minsToStopAfter);
447  }
448 
449  m_pathToBinary = binaryPath;
450  return 0;

454 int CommandlineDriver::interactiveMain()
455 {
456  m_project->loadPlugins();
457  m_console.reset(new Console(m_project.get()));
458 
459  CommandStatus status = m_console->replayFile(m_project->getSettings()->replayFile);
460 
461  if (status == CommandStatus::ExitProgram) {
462  return 2;
463  }
464 
465  // now handle user commands
466  QTextStream strm(stdin);
467  QString line;
468 
469  while (true) {
470  std::cout << "boomerang: ";
471  std::cout.flush();
472 
473  if (strm.atEnd()) {
474  return 0;
475  }
476 
477  line = strm.readLine();
478  status = m_console->handleCommand(line);
479 
480  if (status == CommandStatus::ExitProgram) {
481  return 2;
482  }
483  }

525 int CommandlineDriver::decompile(const QString &fname, const QString &pname)
526 {
527  time_t start;
528  time(&start);
529 
530  if (!loadAndDecode(fname, pname)) {
531  return 1;
532  }
533 
534  if (m_project->getSettings()->stopBeforeDecompile) {
535  if (!m_project->getSettings()->dotFile.isEmpty()) {
536  CFGDotWriter().writeCFG(m_project->getProg(), m_project->getSettings()->dotFile);
537  }
538 
539  return 0;
540  }
541 
542  m_project->decompileBinaryFile();
543 
544  if (!m_project->getSettings()->dotFile.isEmpty()) {
545  CFGDotWriter().writeCFG(m_project->getProg(), m_project->getSettings()->dotFile);
546  }
547 
548  m_project->generateCode();
549 
550  QDir outDir = m_project->getSettings()->getOutputDirectory();
551  LOG_MSG("Output written to '%1'", outDir.absolutePath());
552 
553  time_t end;
554  time(&end);
555  const int hours = static_cast<int>((end - start) / 60 / 60);
556  const int mins = static_cast<int>((end - start) / 60 - hours * 60);
557  const int secs = static_cast<int>((end - start) - (hours * 60 * 60) - (mins * 60));
558 
559  LOG_MSG("Completed in %1 hours %2 minutes %3 seconds.", hours, mins, secs);
560  return 0;

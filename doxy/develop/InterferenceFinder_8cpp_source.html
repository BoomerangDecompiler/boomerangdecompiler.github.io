27 void InterferenceFinder::findInterferences(ConnectionGraph &ig)
28 {
29  if (m_cfg->getNumFragments() == 0) {
30  return;
31  }
32 
33  std::list<IRFragment *> workList; // List of fragments still to be processed
34  std::set<IRFragment *> workSet; // Set of the same; used for quick membership test
35  appendFrags(workList, workSet);
36 
37  int count = 0;
38  const bool debugLive = m_cfg->getProc()->getProg()->getProject()->getSettings()->debugLiveness;
39 
40  while (!workList.empty() && count++ < 1E5) {
41  IRFragment *currFrag = workList.back();
42  workList.erase(--workList.end());
43  workSet.erase(currFrag);
44 
45  // Calculate live locations and interferences
46  assert(currFrag->getProc() != nullptr);
47  bool change = m_livenessAna.calcLiveness(currFrag, ig, currFrag->getProc());
48 
49  if (!change) {
50  continue;
51  }
52 
53  if (debugLive) {
54  SharedStmt last = currFrag->getLastStmt();
55 
56  LOG_MSG("Revisiting BB ending with stmt %1 due to change",
57  last ? QString::number(last->getNumber(), 10) : "<none>");
58  }
59 
60  updateWorkListRev(currFrag, workList, workSet);
61  }
62 }

65 void InterferenceFinder::updateWorkListRev(IRFragment *currFrag, std::list<IRFragment *> &workList,
66  std::set<IRFragment *> &workSet)
67 {
68  // Insert inedges of currFrag into the worklist, unless already there
69  for (IRFragment *pred : currFrag->getPredecessors()) {
70  if (workSet.find(pred) == workSet.end()) {
71  workList.push_front(pred);
72  workSet.insert(pred);

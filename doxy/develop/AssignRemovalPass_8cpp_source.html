40 bool AssignRemovalPass::removeSpAssigns(UserProc *proc)
41 {
42  // if there are no uses of sp other than sp{-} in the whole procedure,
43  // we can safely remove all assignments to sp, this will make the output
44  // more readable for human eyes.
45 
46  SharedExp sp = Location::regOf(Util::getStackRegisterIndex(proc->getProg()));
47  bool foundone = false;
48 
49  StatementList stmts;
50  proc->getStatements(stmts);
51 
52  for (SharedStmt stmt : stmts) {
53  if (stmt->isAssign() && (*stmt->as<Assign>()->getLeft() == *sp)) {
54  foundone = true;
55  }
56 
57  LocationSet refs;
58  stmt->addUsedLocs(refs);
59 
60  for (const SharedExp &rr : refs) {
61  if (rr->isSubscript() && (*rr->getSubExp1() == *sp)) {
62  SharedStmt def = rr->access<RefExp>()->getDef();
63 
64  if (def && (def->getProc() == proc)) {
65  return false;
66  }
67  }
68  }
69  }
70 
71  if (!foundone) {
72  return false;
73  }
74 
75  proc->getProg()->getProject()->alertDecompileDebugPoint(
76  proc, "Before removing stack pointer assigns.");
77 
78  for (auto &stmt : stmts) {
79  if (stmt->isAssign()) {
80  std::shared_ptr<Assign> a = stmt->as<Assign>();
81 
82  if (*a->getLeft() == *sp) {
83  proc->removeStatement(a);
84  }
85  }
86  }
87 
88  proc->getProg()->getProject()->alertDecompileDebugPoint(
89  proc, "After removing stack pointer assigns.");
90  return true;
91 }

94 bool AssignRemovalPass::removeMatchingAssigns(UserProc *proc, SharedExp e)
95 {
96  // if there are no uses of %flags in the whole procedure,
97  // we can safely remove all assignments to %flags, this will make the output
98  // more readable for human eyes and makes short circuit analysis easier.
99 
100  bool foundone = false;
101 
102  StatementList stmts;
103  proc->getStatements(stmts);
104 
105  for (auto stmt : stmts) {
106  if (stmt->isAssign() && (*stmt->as<Assign>()->getLeft() == *e)) {
107  foundone = true;
108  }
109 
110  if (stmt->isPhi()) {
111  if (*stmt->as<PhiAssign>()->getLeft() == *e) {
112  foundone = true;
113  }
114 
115  continue;
116  }
117 
118  LocationSet refs;
119  stmt->addUsedLocs(refs);
120 
121  for (const SharedExp &rr : refs) {
122  if (rr->isSubscript() && (*rr->getSubExp1() == *e)) {
123  SharedStmt def = rr->access<RefExp>()->getDef();
124 
125  if (def && (def->getProc() == proc)) {
126  return false;
127  }
128  }
129  }
130  }
131 
132  if (!foundone) {
133  return false;
134  }
135 
136  QString msg;
137  OStream str(&msg);
138  str << "Before removing matching assigns (" << e << ").";
139 
140  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, qPrintable(msg));
141 
142  for (auto &stmt : stmts) {
143  if ((stmt)->isAssign()) {
144  std::shared_ptr<Assign> a = stmt->as<Assign>();
145 
146  if (*a->getLeft() == *e) {
147  proc->removeStatement(a);
148  }
149  }
150  else if ((stmt)->isPhi()) {
151  std::shared_ptr<PhiAssign> a = stmt->as<PhiAssign>();
152 
153  if (*a->getLeft() == *e) {
154  proc->removeStatement(a);
155  }
156  }
157  }
158 
159  msg.clear();
160  str << "After removing matching assigns (" << e << ").";
161  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, qPrintable(msg));

165 }

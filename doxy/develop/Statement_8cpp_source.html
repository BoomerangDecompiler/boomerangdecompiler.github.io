145 bool Statement::canPropagateToExp(const Exp &exp)
146 {
147  if (!exp.isSubscript()) {
148  return false;
149  }
150 
151  const RefExp &ref = static_cast<const RefExp &>(exp);
152 
153  if (ref.isImplicitDef()) {
154  // Can't propagate statement "-" or "0" (implicit assignments)
155  return false;
156  }
157 
158  const SharedConstStmt def = ref.getDef();
159  if (def->isNullStatement()) {
160  // Don't propagate a null statement! Can happen with %pc's (would have no effect, and would
161  // infinitely loop)
162  return false;
163  }
164 
165  if (!def->isAssign()) {
166  return false; // Only propagate ordinary assignments (so far)
167  }
168 
169  // Assigning to an array, don't propagate (Could be alias problems?)
170  return !def->as<const Assign>()->getType()->isArray();
171 }

174 bool Statement::propagateToThis(int propMaxDepth, const ExpIntMap *destCounts, bool force)
175 {
176  bool thisChange = false;
177  int changes = 0;
178 
179  do {
180  // addUsedLocs(..,true) -> true to also add uses from collectors. For example, want to
181  // propagate into the reaching definitions of calls. Third parameter is false to find
182  // all locations, not just those inside m[...]
183  LocationSet usedExps;
184  addUsedLocs(usedExps, true, false);
185  thisChange = false; // True if changed this iteration of the do/while loop
186 
187  // Example: m[r24{10}] := r25{20} + m[r26{30}]
188  // exps has r24{10}, r25{20}, m[r26{30}], r26{30}
189  for (SharedExp usedHere : usedExps) {
190  if (!Statement::canPropagateToExp(*usedHere)) {
191  continue;
192  }
193 
194  assert(usedHere->access<RefExp>()->getDef()->isAssignment());
195  std::shared_ptr<Assignment>
196  def = usedHere->access<RefExp>()->getDef()->as<Assignment>();
197  SharedExp rhs = def->getRight();
198 
199  // Must never propagate unsubscripted memofs, or memofs that don't yet have symbols.
200  // You could be propagating past a definition, thereby invalidating the IR.
201  // If force is true, ignore the fact that a memof should not be propagated
202  // (for switch analysis)
203  if (rhs->containsBadMemof() && !(force && rhs->isMemOf())) {
204  continue;
205  }
206 
207  const SharedExp lhs = def->getLeft();
208 
209  // Check if the -l flag (propMaxDepth) prevents this propagation,
210  // but always propagate to %flags
211  if (!destCounts || lhs->isFlags() || def->getRight()->containsFlags()) {
212  thisChange |= replaceRef(usedHere, def);
213  }
214  else {
215  ExpIntMap::const_iterator ff = destCounts->find(usedHere);
216 
217  if (ff == destCounts->end()) {
218  thisChange |= replaceRef(usedHere, def);
219  }
220  else if (ff->second <= 1) {
221  thisChange |= replaceRef(usedHere, def);
222  }
223  else if (rhs->getComplexityDepth(m_proc) < propMaxDepth) {
224  thisChange |= replaceRef(usedHere, def);
225  }
226  }
227  }
228  } while (thisChange && ++changes < 10);
229 
230  // Simplify is very costly, especially for calls.
231  // I hope that doing one simplify at the end will not affect any result...
232  simplify();
233 
234  return changes > 0;
235 }

238 bool Statement::propagateFlagsToThis()
239 {
240  bool thisChange = false;
241  int changes = 0;
242 
243  do {
244  LocationSet usedExps;
245  addUsedLocs(usedExps, true);
246 
247  for (SharedExp usedHere : usedExps) {
248  if (!usedHere->isSubscript()) {
249  continue; // e.g. %pc
250  }
251 
252  const std::shared_ptr<Assignment> def = std::dynamic_pointer_cast<Assignment>(
253  usedHere->access<RefExp>()->getDef());
254 
255  // process only if it has definition with rhs
256  if (!def || !def->getRight()) {
257  continue;
258  }
259 
260  const SharedExp base = usedHere->access<Exp, 1>(); // Either RefExp or Location ?
261  if (base->isFlags() || base->isMainFlag()) {
262  thisChange |= replaceRef(usedHere, def);
263  }
264  }
265  } while (thisChange && ++changes < 10);
266 
267  simplify();
268 
269  return changes > 0;
270 }

279 bool Statement::replaceRef(SharedExp e, const std::shared_ptr<Assignment> &def)
280 {
281  SharedExp rhs = def->getRight();
282  assert(rhs);
283 
284  SharedExp base = e->getSubExp1();
285  // Could be propagating %flags into %CF
286  SharedExp lhs = def->getLeft();
287 
288  /* When one of the main flags is used bare, and was defined via a flag function,
289  * apply the semantics for it. For example, the x86 'sub lhs, rhs' instruction effectively
290  * sets the CF flag to 'lhs <u rhs'.
291  * Extract the arguments of the flag function call, and apply the semantics manually.
292  * This should rather be done in the SSL file (not hard-coded), but doing this currently breaks
293  * Type Analysis (because Type Analysis is also done for subexpressions of dead definitions).
294  * Note: the flagcall is a binary, with a Const (the name), and a list of expressions. Example
295  * for the 'SUBFLAGS' flag call:
296  *
297  * defRhs
298  * / \
299  * Const opList
300  * "SUBFLAGS" / \
301  * P1 opList
302  * / \
303  * P2 opList
304  * / \
305  * P3 opNil
306  */
307  if (lhs && lhs->isFlags()) {
308  if (!rhs) {
309  return false;
310  }
311  else if (rhs->isIntConst() && *lhs != *rhs) {
312  // e.g. %flags := 0
313  searchAndReplace(*e, rhs, true);
314  return true;
315  }
316  else if (!rhs->isFlagCall()) {
317  return false;
318  }
319 
320  const QString flagFuncName = rhs->access<Const, 1>()->getStr();
321 
322  if (flagFuncName.startsWith("SUBFLAGSFL")) {
323  switch (base->getOper()) {
324  case opCF: {
325  // for float cf we'll replace the CF with (P1<P2)
326  SharedExp replacement = Binary::get(opLess, rhs->access<Exp, 2, 1>(),
327  rhs->access<Exp, 2, 2, 1>());
328  searchAndReplace(*RefExp::get(Terminal::get(opCF), def), replacement, true);
329  return true;
330  }
331  case opZF: {
332  // for float zf we'll replace the ZF with (P1==P2)
333  SharedExp replacement = Binary::get(opEquals, rhs->access<Exp, 2, 1>(),
334  rhs->access<Exp, 2, 2, 1>());
335  searchAndReplace(*RefExp::get(Terminal::get(opZF), def), replacement, true);
336  return true;
337  }
338 
339  default: break;
340  }
341  }
342  else if (flagFuncName.startsWith("SUBFLAGS")) {
343  const SharedExp subLhs = rhs->access<Exp, 2, 1>();
344  const SharedExp subRhs = rhs->access<Exp, 2, 2, 1>();
345  const SharedExp subResult = rhs->access<Exp, 2, 2, 2, 1>();
346 
347  switch (base->getOper()) {
348  case opCF: {
349  const SharedExp replacement = Binary::get(opLessUns, subLhs, subRhs);
350  searchAndReplace(*RefExp::get(Terminal::get(opCF), def), replacement, true);
351  return true;
352  }
353  case opZF: {
354  // for zf we only want to check if the result part of the subflags is equal to zero
355  const SharedExp replacement = Binary::get(opEquals, subResult, Const::get(0));
356  searchAndReplace(*RefExp::get(Terminal::get(opZF), def), replacement, true);
357  return true;
358  }
359  case opNF: {
360  // for sf we only want to check if the result part of the subflags is less than zero
361  const SharedExp replacement = Binary::get(opLess, subResult, Const::get(0));
362  searchAndReplace(*RefExp::get(Terminal::get(opNF), def), replacement, true);
363  return true;
364  }
365  case opOF: {
366  // (op1 < 0 && op2 >= 0 && result >= 0) || (op1 >= 0 && op2 < 0 && result < 0)
367  const SharedExp replacement = Binary::get(
368  opOr,
369  Binary::get(opAnd,
370  Binary::get(opAnd, Binary::get(opLess, subLhs, Const::get(0)),
371  Binary::get(opGtrEq, subRhs, Const::get(0))),
372  Binary::get(opGtrEq, subResult, Const::get(0))),
373  Binary::get(opAnd,
374  Binary::get(opAnd, Binary::get(opGtrEq, subLhs, Const::get(0)),
375  Binary::get(opLess, subRhs, Const::get(0))),
376  Binary::get(opLess, subResult, Const::get(0))));
377  searchAndReplace(*RefExp::get(Terminal::get(opOF), def), replacement, true);
378  return true;
379  }
380  default: break;
381  }
382  }
383  else if (flagFuncName.startsWith("LOGICALFLAGS")) {
384  const SharedExp param = rhs->access<Exp, 2, 1>();
385  switch (base->getOper()) {
386  case opNF: {
387  SharedExp replacement = Binary::get(opLess, param, Const::get(0));
388  searchAndReplace(*RefExp::get(Terminal::get(opNF), def), replacement, true);
389  return true;
390  }
391  case opZF: {
392  SharedExp replacement = Binary::get(opEquals, param, Const::get(0));
393  searchAndReplace(*RefExp::get(Terminal::get(opZF), def), replacement, true);
394  return true;
395  }
396  case opCF: {
397  SharedExp replacement = Const::get(0);
398  searchAndReplace(*RefExp::get(Terminal::get(opCF), def), replacement, true);
399  return true;
400  }
401  case opOF: {
402  const SharedExp replacement = Const::get(0);
403  searchAndReplace(*RefExp::get(Terminal::get(opOF), def), replacement, true);
404  return true;
405  }
406  default: break;
407  }
408  }
409  else if (flagFuncName.startsWith("INCDECFLAGS")) {
410  const SharedExp param = rhs->access<Exp, 2, 1>();
411  switch (base->getOper()) {
412  case opOF: {
413  const SharedExp replacement = Const::get(0);
414  searchAndReplace(*RefExp::get(Terminal::get(opOF), def), replacement, true);
415  return true;
416  }
417  case opZF: {
418  const SharedExp replacement = Binary::get(opEquals, param, Const::get(0));
419  searchAndReplace(*RefExp::get(Terminal::get(opZF), def), replacement, true);
420  return true;
421  }
422  case opNF: {
423  const SharedExp replacement = Binary::get(opLess, param, Const::get(0));
424  searchAndReplace(*RefExp::get(Terminal::get(opNF), def), replacement, true);
425  return true;
426  }
427  default: break;
428  }
429  }
430  }
431 
432  // do the replacement; last parameter true to also change collectors
433  return searchAndReplace(*e, rhs, true);
434 }

437 bool Statement::isNullStatement() const
438 {
439  if (!this->isAssign()) {
440  return false;
441  }
442 
443  SharedExp right = static_cast<const Assign *>(this)->getRight();
444 
445  if (right->isSubscript()) {
446  // Must refer to self to be null
447  return right->access<RefExp>()->getDef().get() == this;
448  }
449  else {
450  // Null if left == right
451  return *shared_from_this()->as<const Assign>()->getLeft() == *right;
452  }
453 }

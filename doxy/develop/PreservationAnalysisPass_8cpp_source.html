27 bool PreservationAnalysisPass::execute(UserProc *proc)
28 {
29  std::set<SharedExp> removes;
30 
31  if (proc->getRetStmt() == nullptr) {
32  if (proc->getProg()->getProject()->getSettings()->debugProof) {
33  LOG_MSG("Can't find preservations as there is no return statement!");
34  }
35 
36  return false;
37  }
38 
39  // prove preservation for all modifieds in the return statement
40  for (SharedStmt mod : proc->getRetStmt()->getModifieds()) {
41  SharedExp lhs = mod->as<Assignment>()->getLeft();
42  auto equation = Binary::get(opEquals, lhs, lhs);
43 
44  if (proc->getProg()->getProject()->getSettings()->debugProof) {
45  LOG_MSG("Attempting to prove %1 is preserved by %2", equation, proc->getName());
46  }
47 
48  if (proc->preservesExp(lhs)) {
49  removes.insert(equation);
50  }
51  }
52 
53  if (proc->getProg()->getProject()->getSettings()->debugProof) {
54  LOG_MSG("### Proven true for procedure %1:", proc->getName());
55 
56  for (auto &elem : proc->getProvenTrue()) {
57  LOG_MSG(" %1 = %2", elem.first, elem.second);
58  }
59 
60  LOG_MSG("### End proven true for procedure %1", proc->getName());
61  }
62 
63  // Remove the preserved locations from the modifieds and the returns
64  for (auto pp = proc->getProvenTrue().begin(); pp != proc->getProvenTrue().end(); ++pp) {
65  SharedExp lhs = pp->first;
66  SharedExp rhs = pp->second;
67 
68  // Has to be of the form loc = loc, not say loc+4, otherwise the bypass logic won't see the
69  // add of 4
70  if (!(*lhs == *rhs)) {
71  continue;
72  }
73 
74  proc->getRetStmt()->removeFromModifiedsAndReturns(lhs);
75  }
76 
77  return true;
78 }

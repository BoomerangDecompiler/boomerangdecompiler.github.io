49 bool CallDefineUpdatePass::updateCallDefines(UserProc *proc,
50  const std::shared_ptr<CallStatement> &callStmt)
51 {
52  assert(callStmt->getProc() == proc);
53  Function *callee = callStmt->getDestProc();
54 
55  std::shared_ptr<Signature> sig = callee ? callee->getSignature() : proc->getSignature();
56 
57  if (callee && callee->isLib()) {
58  StatementList defines;
59  sig->getLibraryDefines(defines); // Set the locations defined
60  callStmt->setDefines(defines);
61  return true;
62  }
63  else if (proc->getProg()->getProject()->getSettings()->assumeABI) {
64  // Risky: just assume the ABI caller save registers are defined
65  Signature::getABIDefines(proc->getProg()->getMachine(), callStmt->getDefines());
66  return true;
67  }
68 
69  // Move the defines to a temporary list. We must make sure that all defines
70  // that are not inserted into m_defines again are deleted.
71  StatementList newDefines(callStmt->getDefines());
72  callStmt->getDefines().clear();
73 
74  if (callee && callStmt->getCalleeReturn()) {
75  assert(!callee->isLib());
76  const StatementList
77  &modifieds = static_cast<UserProc *>(callee)->getRetStmt()->getModifieds();
78 
79  for (SharedStmt mm : modifieds) {
80  std::shared_ptr<Assignment> as = mm->as<Assignment>();
81  SharedExp loc = as->getLeft();
82 
83  if (!proc->canBeReturn(loc)) {
84  continue;
85  }
86 
87  SharedType ty = as->getType();
88 
89  if (!newDefines.existsOnLeft(loc)) {
90  newDefines.append(std::make_shared<ImplicitAssign>(ty, loc));
91  }
92  }
93  }
94  else {
95  // Ensure that everything in the UseCollector has an entry in oldDefines
96  for (SharedExp loc : *callStmt->getUseCollector()) {
97  if (!proc->canBeReturn(loc)) {
98  continue; // Filtered out
99  }
100 
101  if (!newDefines.existsOnLeft(loc)) {
102  std::shared_ptr<ImplicitAssign> as(new ImplicitAssign(loc->clone()));
103  as->setProc(proc);
104  as->setFragment(callStmt->getFragment());
105  newDefines.append(as);
106  }
107  }
108  }
109 
110  for (SharedStmt stmt : newDefines) {
111  // Make sure the LHS is still in the return or collector
112  std::shared_ptr<Assignment> as = stmt->as<Assignment>();
113  SharedExp lhs = as->getLeft();
114 
115  if (callStmt->getCalleeReturn()) {
116  if (!callStmt->getCalleeReturn()->definesLoc(lhs)) {
117  continue; // Not in callee returns -> delete it
118  }
119  }
120  else if (!callStmt->getUseCollector()->hasUse(lhs)) {
121  continue; // Not in collector: delete it (don't copy it)
122  }
123 
124  if (!proc->canBeReturn(lhs)) {
125  continue; // Filtered out: delete it
126  }
127 
128  callStmt->getDefines().append(stmt);
129  }
130 
131  callStmt->getDefines().sort([sig](SharedStmt left, SharedStmt right) {
132  return sig->returnCompare(*left->as<Assignment>(), *right->as<Assignment>());
133  });
134 
135  return true;
136 }

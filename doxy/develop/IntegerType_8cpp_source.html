54 bool IntegerType::operator==(const Type &other) const
55 {
56  if (!other.isInteger()) {
57  return false;
58  }
59 
60  const IntegerType &otherInt = static_cast<const IntegerType &>(other);
61 
62  return
63  // Note: zero size matches any other size (wild, or unknown, size)
64  (m_size == 0 || otherInt.m_size == 0 || m_size == otherInt.m_size) &&
65  // Note: actual value of signedness is disregarded, just whether less than, equal to, or
66  // greater than 0
67  ((isUnsigned() && otherInt.isUnsigned()) || (isSignUnknown() && otherInt.isSignUnknown()) ||
68  (isSigned() && otherInt.isSigned()));
69 }

72 bool IntegerType::operator<(const Type &other) const
73 {
74  if (m_id != other.getId()) {
75  return m_id < other.getId();
76  }
77 
78  const IntegerType &otherInt = static_cast<const IntegerType &>(other);
79 
80  if (m_size != otherInt.m_size) {
81  return m_size < otherInt.m_size;
82  }
83 
84  // note: We cannot compare the sign directly here, because otherwise e.g.
85  // IntegerType(32, Sign::Signed) == IntegerType(32, Sign::SignedStrong)
86  // would be true, while
87  // IntegerType(32, Sign::Signed) < IntegerType(32, Sign::SignedStrong)
88  // would also be true.
89  return (isUnsigned() && otherInt.isMaybeSigned()) || (isSignUnknown() && otherInt.isSigned());
90 }

130 SharedType IntegerType::meetWith(SharedType other, bool &changed, bool useHighestPtr) const
131 {
132  if (other->resolvesToVoid() || other->resolvesToChar()) {
133  return const_cast<IntegerType *>(this)->shared_from_this();
134  }
135 
136  if (other->resolvesToInteger()) {
137  std::shared_ptr<IntegerType> otherInt = other->as<IntegerType>();
138  std::shared_ptr<IntegerType> result = this->clone()->as<IntegerType>();
139 
140  // Signedness
141  if (otherInt->isSigned()) {
142  result->hintAsSigned();
143  }
144  else if (otherInt->isUnsigned()) {
145  result->hintAsUnsigned();
146  }
147 
148  // Changed from signed to not necessarily signed
149  changed |= result->isSigned() != isSigned();
150  // Changed from unsigned to not necessarily unsigned
151  changed |= result->isUnsigned() != isUnsigned();
152 
153  // Size. Assume 0 indicates unknown size
154  result->m_size = std::max(m_size, otherInt->m_size);
155  changed |= (result->m_size != m_size);
156 
157  return result;
158  }
159  else if (other->resolvesToSize()) {
160  std::shared_ptr<IntegerType> result = std::dynamic_pointer_cast<IntegerType>(this->clone());
161  std::shared_ptr<SizeType> other_sz = other->as<SizeType>();
162 
163  if (m_size == 0) { // Doubt this will ever happen
164  result->m_size = other_sz->getSize();
165  changed = true;
166  return result;
167  }
168 
169  if (m_size == other_sz->getSize()) {
170  return result;
171  }
172 
173  LOG_VERBOSE("Integer size %1 meet with SizeType size %2!", m_size, other_sz->getSize());
174 
175  result->m_size = std::max(m_size, other_sz->getSize());
176  changed = result->m_size != m_size;
177  return result;
178  }
179 
180  return createUnion(other, changed, useHighestPtr);
181 }

68 void ControlFlowAnalyzer::updateImmedPDom()
69 {
70  // traverse the nodes in order (i.e from the bottom up)
71  for (int i = m_revPostOrdering.size() - 1; i >= 0; i--) {
72  const IRFragment *frag = m_revPostOrdering[i];
73 
74  for (IRFragment *succ : frag->getSuccessors()) {
75  if (getRevOrd(succ) > getRevOrd(frag)) {
76  setImmPDom(frag, findCommonPDom(getImmPDom(frag), succ));
77  }
78  }
79  }
80 
81  // make a second pass but consider the original CFG ordering this time
82  for (const IRFragment *frag : m_postOrdering) {
83  if (frag->getNumSuccessors() <= 1) {
84  continue;
85  }
86 
87  for (auto &succ : frag->getSuccessors()) {
88  IRFragment *succNode = succ;
89  setImmPDom(frag, findCommonPDom(getImmPDom(frag), succNode));
90  }
91  }
92 
93  // one final pass to fix up nodes involved in a loop
94  for (const IRFragment *frag : m_postOrdering) {
95  if (frag->getNumSuccessors() > 1) {
96  for (const IRFragment *succ : frag->getSuccessors()) {
97  if (isBackEdge(frag, succ) && (frag->getNumSuccessors() > 1) && getImmPDom(succ) &&
98  (getPostOrdering(getImmPDom(succ)) < getPostOrdering(getImmPDom(frag)))) {
99  setImmPDom(frag, findCommonPDom(getImmPDom(succ), getImmPDom(frag)));
100  }
101  else {
102  setImmPDom(frag, findCommonPDom(getImmPDom(frag), succ));
103  }
104  }
105  }
106  }
107 }

110 const IRFragment *ControlFlowAnalyzer::findCommonPDom(const IRFragment *currImmPDom,
111  const IRFragment *succImmPDom)
112 {
113  if (!currImmPDom) {
114  return succImmPDom;
115  }
116 
117  if (!succImmPDom) {
118  return currImmPDom;
119  }
120 
121  if (getRevOrd(currImmPDom) == getRevOrd(succImmPDom)) {
122  return currImmPDom; // ordering hasn't been done
123  }
124 
125  const IRFragment *oldCurImmPDom = currImmPDom;
126  const IRFragment *oldSuccImmPDom = succImmPDom;
127 
128  int giveup = 0;
129 #define GIVEUP 10000
130 
131  while (giveup < GIVEUP && currImmPDom && succImmPDom && (currImmPDom != succImmPDom)) {
132  if (getRevOrd(currImmPDom) > getRevOrd(succImmPDom)) {
133  succImmPDom = getImmPDom(succImmPDom);
134  }
135  else {
136  currImmPDom = getImmPDom(currImmPDom);
137  }
138 
139  giveup++;
140  }
141 
142  if (giveup >= GIVEUP) {
143  LOG_VERBOSE("Failed to find commonPDom for %1 and %2", oldCurImmPDom->getLowAddr(),
144  oldSuccImmPDom->getLowAddr());
145 
146  return oldCurImmPDom; // no change
147  }
148 
149  return currImmPDom;
150 }

153 void ControlFlowAnalyzer::structConds()
154 {
155  // Process the nodes in order
156  for (const IRFragment *currNode : m_postOrdering) {
157  if (currNode->getNumSuccessors() <= 1) {
158  // not an if/case condition
159  continue;
160  }
161 
162  // if the current conditional header is a two way node and has a back edge,
163  // then it won't have a follow
164  if (hasBackEdge(currNode) && currNode->isType(FragType::Twoway)) {
165  setStructType(currNode, StructType::Cond);
166  continue;
167  }
168 
169  // set the follow of a node to be its immediate post dominator
170  setCondFollow(currNode, getImmPDom(currNode));
171 
172  // set the structured type of this node
173  setStructType(currNode, StructType::Cond);
174 
175  // if this is an nway header, then we have to tag each of the nodes within the body of
176  // the nway subgraph
177  if (getCondType(currNode) == CondType::Case) {
178  setCaseHead(currNode, currNode, getCondFollow(currNode));
179  }
180  }

184 void ControlFlowAnalyzer::determineLoopType(const IRFragment *header, bool *&loopNodes)
185 {
186  assert(getLatchNode(header));
187 
188  // if the latch node is a two way node then this must be a post tested loop
189  if (getLatchNode(header)->isType(FragType::Twoway)) {
190  setLoopType(header, LoopType::PostTested);
191 
192  // if the head of the loop is a two way node and the loop spans more than one block then it
193  // must also be a conditional header
194  if (header->isType(FragType::Twoway) && (header != getLatchNode(header))) {
195  setStructType(header, StructType::LoopCond);
196  }
197  }
198  // otherwise it is either a pretested or endless loop
199  else if (header->isType(FragType::Twoway)) {
200  // if the header is a two way node then it must have a conditional follow (since it can't
201  // have any backedges leading from it). If this follow is within the loop then this must be
202  // an endless loop
203  if (getCondFollow(header) && loopNodes[getPostOrdering(getCondFollow(header))]) {
204  setLoopType(header, LoopType::Endless);
205 
206  // retain the fact that this is also a conditional header
207  setStructType(header, StructType::LoopCond);
208  }
209  else {
210  setLoopType(header, LoopType::PreTested);
211  }
212  }
213  else {
214  // both the header and latch node are one way nodes so this must be an endless loop
215  setLoopType(header, LoopType::Endless);
216  }
217 }

220 void ControlFlowAnalyzer::findLoopFollow(const IRFragment *header, bool *&loopNodes)
221 {
222  assert(getStructType(header) == StructType::Loop ||
223  getStructType(header) == StructType::LoopCond);
224  const LoopType loopType = getLoopType(header);
225  const IRFragment *latch = getLatchNode(header);
226 
227  if (loopType == LoopType::PreTested) {
228  // if the 'while' loop's true child is within the loop, then its false child is the loop
229  // follow
230  if (loopNodes[getPostOrdering(header->getSuccessor(BTHEN))]) {
231  setLoopFollow(header, header->getSuccessor(BELSE));
232  }
233  else {
234  setLoopFollow(header, header->getSuccessor(BTHEN));
235  }
236  }
237  else if (loopType == LoopType::PostTested) {
238  // the follow of a post tested ('repeat') loop is the node on the end of the non-back edge
239  // from the latch node
240  if (latch->getSuccessor(BELSE) == header) {
241  setLoopFollow(header, latch->getSuccessor(BTHEN));
242  }
243  else {
244  setLoopFollow(header, latch->getSuccessor(BELSE));
245  }
246  }
247  else {
248  // endless loop
249  const IRFragment *follow = nullptr;
250 
251  // traverse the ordering array between the header and latch nodes.
252  // IRFragment * latch = header->getLatchNode(); initialized at function start
253  for (int i = getPostOrdering(header) - 1; i > getPostOrdering(latch); i--) {
254  const IRFragment *&desc = m_postOrdering[i];
255  // the follow for an endless loop will have the following
256  // properties:
257  // i) it will have a parent that is a conditional header inside the loop whose follow
258  // is outside the loop
259  // ii) it will be outside the loop according to its loop stamp pair
260  // iii) have the highest ordering of all suitable follows (i.e. highest in the graph)
261 
262  if ((getStructType(desc) == StructType::Cond) && getCondFollow(desc) &&
263  (getLoopHead(desc) == header)) {
264  if (loopNodes[getPostOrdering(getCondFollow(desc))]) {
265  // if the conditional's follow is in the same loop AND is lower in the loop,
266  // jump to this follow
267  if (getPostOrdering(desc) > getPostOrdering(getCondFollow(desc))) {
268  i = getPostOrdering(getCondFollow(desc));
269  }
270  else {
271  // otherwise there is a backward jump somewhere to a node earlier in this
272  // loop. We don't need to any nodes below this one as they will all have a
273  // conditional within the loop.
274  break;
275  }
276  }
277  else {
278  // otherwise find the child (if any) of the conditional header that isn't inside
279  // the same loop
280  const IRFragment *succ = desc->getSuccessor(BTHEN);
281 
282  if (loopNodes[getPostOrdering(succ)]) {
283  if (!loopNodes[getPostOrdering(desc->getSuccessor(BELSE))]) {
284  succ = desc->getSuccessor(BELSE);
285  }
286  else {
287  succ = nullptr;
288  }
289  }
290 
291  // if a potential follow was found, compare its ordering with the currently
292  // found follow
293  if (succ && (!follow || (getPostOrdering(succ) > getPostOrdering(follow)))) {
294  follow = succ;
295  }
296  }
297  }
298  }
299 
300  // if a follow was found, assign it to be the follow of the loop under
301  // investigation
302  if (follow) {
303  setLoopFollow(header, follow);
304  }
305  }
306 }

309 void ControlFlowAnalyzer::tagNodesInLoop(const IRFragment *header, bool *&loopNodes)
310 {
311  // Traverse the ordering structure from the header to the latch node tagging the nodes
312  // determined to be within the loop. These are nodes that satisfy the following:
313  // i) header.loopStamps encloses curNode.loopStamps and curNode.loopStamps encloses
314  // latch.loopStamps
315  // OR
316  // ii) latch.revLoopStamps encloses curNode.revLoopStamps and curNode.revLoopStamps encloses
317  // header.revLoopStamps
318  // OR
319  // iii) curNode is the latch node
320 
321  const IRFragment *latch = getLatchNode(header);
322  assert(latch);
323 
324  for (int i = getPostOrdering(header) - 1; i >= getPostOrdering(latch); i--) {
325  if (isFragInLoop(m_postOrdering[i], header, latch)) {
326  // update the membership map to reflect that this node is within the loop
327  loopNodes[i] = true;
328 
329  setLoopHead(m_postOrdering[i], header);
330  }
331  }
332 }

335 void ControlFlowAnalyzer::structLoops()
336 {
337  for (int i = m_postOrdering.size() - 1; i >= 0; i--) {
338  const IRFragment *currFrag = m_postOrdering[i]; // the current node under investigation
339  const IRFragment *latch = nullptr; // the latching node of the loop
340 
341  // If the current node has at least one back edge into it, it is a loop header. If there are
342  // numerous back edges into the header, determine which one comes form the proper latching
343  // node. The proper latching node is defined to have the following properties:
344  // i) has a back edge to the current node
345  // ii) has the same case head as the current node
346  // iii) has the same loop head as the current node
347  // iv) is not an nway node
348  // v) is not the latch node of an enclosing loop
349  // vi) has a lower ordering than all other suitable candiates
350  // If no nodes meet the above criteria, then the current node is not a loop header
351 
352  for (const IRFragment *pred : currFrag->getPredecessors()) {
353  if ((getCaseHead(pred) == getCaseHead(currFrag)) && // ii)
354  (getLoopHead(pred) == getLoopHead(currFrag)) && // iii)
355  (!latch || (getPostOrdering(latch) > getPostOrdering(pred))) && // vi)
356  !(getLoopHead(pred) && (getLatchNode(getLoopHead(pred)) == pred)) && // v)
357  isBackEdge(pred, currFrag)) { // i)
358  latch = pred;
359  }
360  }
361 
362  // if a latching node was found for the current node then it is a loop header.
363  if (!latch) {
364  continue;
365  }
366 
367  // define the map that maps each node to whether or not it is within the current loop
368  bool *loopNodes = new bool[m_postOrdering.size()];
369 
370  for (unsigned int j = 0; j < m_postOrdering.size(); j++) {
371  loopNodes[j] = false;
372  }
373 
374  setLatchNode(currFrag, latch);
375 
376  // the latching node may already have been structured as a conditional header. If it is
377  // not also the loop header (i.e. the loop is over more than one block) then reset it to
378  // be a sequential node otherwise it will be correctly set as a loop header only later
379  if ((latch != currFrag) && (getStructType(latch) == StructType::Cond)) {
380  setStructType(latch, StructType::Seq);
381  }
382 
383  // set the structured type of this node
384  setStructType(currFrag, StructType::Loop);
385 
386  // tag the members of this loop
387  tagNodesInLoop(currFrag, loopNodes);
388 
389  // calculate the type of this loop
390  determineLoopType(currFrag, loopNodes);
391 
392  // calculate the follow node of this loop
393  findLoopFollow(currFrag, loopNodes);
394 
395  delete[] loopNodes;
396  }
397 }

400 void ControlFlowAnalyzer::checkConds()
401 {
402  for (const IRFragment *currNode : m_postOrdering) {
403  // consider only conditional headers that have a follow and aren't case headers
404  if (((getStructType(currNode) == StructType::Cond) ||
405  (getStructType(currNode) == StructType::LoopCond)) &&
406  getCondFollow(currNode) && (getCondType(currNode) != CondType::Case)) {
407  // define convenient aliases for the relevant loop and case heads and the out edges
408  const IRFragment *myLoopHead = (getStructType(currNode) == StructType::LoopCond)
409  ? currNode
410  : getLoopHead(currNode);
411  const IRFragment *follLoopHead = getLoopHead(getCondFollow(currNode));
412  const IRFragment *fragThen = currNode->getSuccessor(BTHEN);
413  const IRFragment *fragElse = currNode->getSuccessor(BELSE);
414 
415  // analyse whether this is a jump into/outof a loop
416  if (myLoopHead != follLoopHead) {
417  // we want to find the branch that the latch node is on for a jump out of a loop
418  if (myLoopHead) {
419  // this is a jump out of a loop (break or return)
420  if (getLoopHead(fragThen) != nullptr) {
421  // the "else" branch jumps out of the loop. (e.g. "if (!foo) break;")
422  setUnstructType(currNode, UnstructType::JumpInOutLoop);
423  setCondType(currNode, CondType::IfElse);
424  }
425  else {
426  assert(getLoopHead(fragElse) != nullptr);
427  // the "then" branch jumps out of the loop
428  setUnstructType(currNode, UnstructType::JumpInOutLoop);
429  setCondType(currNode, CondType::IfThen);
430  }
431  }
432 
433  if ((getUnstructType(currNode) == UnstructType::Structured) && follLoopHead) {
434  // find the branch that the loop head is on for a jump into a loop body. If a
435  // branch has already been found, then it will match this one anyway
436 
437  // does the else branch goto the loop head?
438  if (isBackEdge(fragThen, follLoopHead)) {
439  setUnstructType(currNode, UnstructType::JumpInOutLoop);
440  setCondType(currNode, CondType::IfElse);
441  }
442  // does the else branch goto the loop head?
443  else if (isBackEdge(fragElse, follLoopHead)) {
444  setUnstructType(currNode, UnstructType::JumpInOutLoop);
445  setCondType(currNode, CondType::IfThen);
446  }
447  }
448  }
449 
450  // this is a jump into a case body if either of its children don't have the same same
451  // case header as itself
452  if ((getUnstructType(currNode) == UnstructType::Structured) &&
453  ((getCaseHead(currNode) != getCaseHead(fragThen)) ||
454  (getCaseHead(currNode) != getCaseHead(fragElse)))) {
455  const IRFragment *myCaseHead = getCaseHead(currNode);
456  const IRFragment *thenCaseHead = getCaseHead(fragThen);
457  const IRFragment *elseCaseHead = getCaseHead(fragElse);
458 
459  if ((thenCaseHead == myCaseHead) &&
460  (!myCaseHead || (elseCaseHead != getCondFollow(myCaseHead)))) {
461  setUnstructType(currNode, UnstructType::JumpIntoCase);
462  setCondType(currNode, CondType::IfElse);
463  }
464  else if ((elseCaseHead == myCaseHead) &&
465  (!myCaseHead || (thenCaseHead != getCondFollow(myCaseHead)))) {
466  setUnstructType(currNode, UnstructType::JumpIntoCase);
467  setCondType(currNode, CondType::IfThen);
468  }
469  }
470  }
471 
472  // for 2 way conditional headers that don't have a follow (i.e. are the source of a back
473  // edge) and haven't been structured as latching nodes, set their follow to be the non-back
474  // edge child.
475  if ((getStructType(currNode) == StructType::Cond) && !getCondFollow(currNode) &&
476  (getCondType(currNode) != CondType::Case) &&
477  (getUnstructType(currNode) == UnstructType::Structured)) {
478  // latching nodes will already have been reset to Seq structured type
479  if (hasBackEdge(currNode)) {
480  if (isBackEdge(currNode, currNode->getSuccessor(BTHEN))) {
481  setCondType(currNode, CondType::IfThen);
482  setCondFollow(currNode, currNode->getSuccessor(BELSE));
483  }
484  else {
485  setCondType(currNode, CondType::IfElse);
486  setCondFollow(currNode, currNode->getSuccessor(BTHEN));
487  }
488  }
489  }
490  }
491 }

525 void ControlFlowAnalyzer::updateLoopStamps(const IRFragment *frag, int &time)
526 {
527  // timestamp the current node with the current time
528  // and set its traversed flag
529  setTravType(frag, TravType::DFS_LNum);
530  m_info[frag].m_preOrderID = time;
531 
532  // recurse on unvisited children and set inedges for all children
533  for (const IRFragment *succ : frag->getSuccessors()) {
534  // set the in edge from this child to its parent (the current node)
535  // (not done here, might be a problem)
536  // outEdges[i]->inEdges.Add(this);
537 
538  // recurse on this child if it hasn't already been visited
539  if (getTravType(succ) != TravType::DFS_LNum) {
540  updateLoopStamps(succ, ++time);
541  }
542  }
543 
544  // set the the second loopStamp value
545  m_info[frag].m_postOrderID = ++time;
546 
547  // add this node to the ordering structure as well as recording its position within the ordering
548  m_info[frag].m_postOrderIndex = static_cast<int>(m_postOrdering.size());
549  m_postOrdering.push_back(frag);
550 }

553 void ControlFlowAnalyzer::updateRevLoopStamps(const IRFragment *frag, int &time)
554 {
555  // timestamp the current node with the current time and set its traversed flag
556  setTravType(frag, TravType::DFS_RNum);
557  m_info[frag].m_revPreOrderID = time;
558 
559  // recurse on the unvisited children in reverse order
560  for (int i = frag->getNumSuccessors() - 1; i >= 0; i--) {
561  // recurse on this child if it hasn't already been visited
562  if (getTravType(frag->getSuccessor(i)) != TravType::DFS_RNum) {
563  updateRevLoopStamps(frag->getSuccessor(i), ++time);
564  }
565  }
566 
567  m_info[frag].m_revPostOrderID = ++time;
568 }

571 void ControlFlowAnalyzer::updateRevOrder(const IRFragment *frag)
572 {
573  // Set this node as having been traversed during the post domimator DFS ordering traversal
574  setTravType(frag, TravType::DFS_PDom);
575 
576  // recurse on unvisited children
577  for (const IRFragment *pred : frag->getPredecessors()) {
578  if (getTravType(pred) != TravType::DFS_PDom) {
579  updateRevOrder(pred);
580  }
581  }
582 
583  // add this node to the ordering structure and record the post dom. order of this node as its
584  // index within this ordering structure
585  m_info[frag].m_revPostOrderIndex = static_cast<int>(m_revPostOrdering.size());
586  m_revPostOrdering.push_back(frag);

590 void ControlFlowAnalyzer::setCaseHead(const IRFragment *frag, const IRFragment *head,
591  const IRFragment *follow)
592 {
593  assert(!getCaseHead(frag));
594 
595  setTravType(frag, TravType::DFS_Case);
596 
597  // don't tag this node if it is the case header under investigation
598  if (frag != head) {
599  m_info[frag].m_caseHead = head;
600  }
601 
602  // if this is a nested case header, then it's member nodes
603  // will already have been tagged so skip straight to its follow
604  if (frag->isType(FragType::Nway) && (frag != head)) {
605  if (getCondFollow(frag) && (getTravType(getCondFollow(frag)) != TravType::DFS_Case) &&
606  (getCondFollow(frag) != follow)) {
607  setCaseHead(frag, head, follow);
608  }
609  }
610  else {
611  // traverse each child of this node that:
612  // i) isn't on a back-edge,
613  // ii) hasn't already been traversed in a case tagging traversal and,
614  // iii) isn't the follow node.
615  for (IRFragment *succ : frag->getSuccessors()) {
616  if (!isBackEdge(frag, succ) && (getTravType(succ) != TravType::DFS_Case) &&
617  (succ != follow)) {
618  setCaseHead(succ, head, follow);
619  }
620  }
621  }
622 }

625 void ControlFlowAnalyzer::setStructType(const IRFragment *frag, StructType structType)
626 {
627  // if this is a conditional header, determine exactly which type of conditional header it is
628  // (i.e. switch, if-then, if-then-else etc.)
629  if (structType == StructType::Cond) {
630  if (frag->isType(FragType::Nway)) {
631  m_info[frag].m_conditionHeaderType = CondType::Case;
632  }
633  else if (getCondFollow(frag) == frag->getSuccessor(BELSE)) {
634  m_info[frag].m_conditionHeaderType = CondType::IfThen;
635  }
636  else if (getCondFollow(frag) == frag->getSuccessor(BTHEN)) {
637  m_info[frag].m_conditionHeaderType = CondType::IfElse;
638  }
639  else {
640  m_info[frag].m_conditionHeaderType = CondType::IfThenElse;
641  }

703 bool ControlFlowAnalyzer::isFragInLoop(const IRFragment *frag, const IRFragment *header,
704  const IRFragment *latch) const
705 {
706  assert(getLatchNode(header) == latch);
707  assert(header == latch || ((m_info[header].m_preOrderID > m_info[latch].m_preOrderID &&
708  m_info[latch].m_postOrderID > m_info[header].m_postOrderID) ||
709  (m_info[header].m_preOrderID < m_info[latch].m_preOrderID &&
710  m_info[latch].m_postOrderID < m_info[header].m_postOrderID)));
711 
712  // this node is in the loop if it is the latch node OR
713  // this node is within the header and the latch is within this when using the forward loop
714  // stamps OR this node is within the header and the latch is within this when using the reverse
715  // loop stamps
716  return frag == latch ||
717  (m_info[header].m_preOrderID < m_info[frag].m_preOrderID &&
718  m_info[frag].m_postOrderID < m_info[header].m_postOrderID &&
719  m_info[frag].m_preOrderID < m_info[latch].m_preOrderID &&
720  m_info[latch].m_postOrderID < m_info[frag].m_postOrderID) ||
721  (m_info[header].m_revPreOrderID < m_info[frag].m_revPreOrderID &&
722  m_info[frag].m_revPostOrderID < m_info[header].m_revPostOrderID &&
723  m_info[frag].m_revPreOrderID < m_info[latch].m_revPreOrderID &&
724  m_info[latch].m_revPostOrderID < m_info[frag].m_revPostOrderID);
725 }

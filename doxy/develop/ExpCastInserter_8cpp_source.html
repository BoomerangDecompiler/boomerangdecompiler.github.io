22 static SharedExp checkSignedness(SharedExp e, Sign reqSignedness)
23 {
24  SharedConstType ty = e->ascendType();
25  const bool isInt = ty->resolvesToInteger();
26  Sign currSignedness = Sign::Unknown;
27 
28  if (isInt) {
29  currSignedness = ty->as<IntegerType>()->isMaybeSigned() ? Sign::Signed : Sign::Unsigned;
30  }
31 
32  // Don't want to cast e.g. floats to integer
33  if (isInt && (currSignedness != reqSignedness)) {
34  // Transfer size
35  std::shared_ptr<IntegerType> newtype = IntegerType::get(
36  ty->as<const IntegerType>()->getSize(), reqSignedness);
37 
38  newtype->setSignedness(reqSignedness);
39  return TypedExp::get(newtype, e);
40  }
41 
42  return e;
43 }

96 bool Const::operator<(const Exp &o) const
97 {
98  if (m_oper != o.getOper()) {
99  return m_oper < o.getOper();
100  }
101 
102  const Const &otherConst = static_cast<const Const &>(o);
103 
104  switch (m_oper) {
105  case opIntConst: return getInt() < otherConst.getInt();
106  case opLongConst: return getLong() < otherConst.getLong();
107  case opFltConst: return getFlt() < otherConst.getFlt();
108  case opStrConst: return getStr() < otherConst.getStr();
109 
110  default: LOG_FATAL("Invalid operator %1", operToString(m_oper));
111  }
112 
113  return false;
114 }

276 bool Const::operator==(const Exp &other) const
277 {
278  // Note: the casts of o to Const& are needed, else op is protected! Duh.
279  if (other.getOper() == opWild) {
280  return true;
281  }
282 
283  if ((other.getOper() == opWildIntConst) && (m_oper == opIntConst)) {
284  return true;
285  }
286 
287  if ((other.getOper() == opWildStrConst) && (m_oper == opStrConst)) {
288  return true;
289  }
290 
291  if (m_oper != other.getOper()) {
292  return false;
293  }
294 
295  const Const &otherConst = static_cast<const Const &>(other);
296 
297  switch (m_oper) {
298  case opIntConst: return getInt() == otherConst.getInt();
299  case opLongConst: return getLong() == otherConst.getLong();
300  case opFltConst: return getFlt() == otherConst.getFlt();
301  case opStrConst: return getStr() == otherConst.getStr();
302  case opFuncConst: {
303  const Function *const *myData = std::get_if<Function *>(&m_value);
304  const Function *const *otherData = std::get_if<Function *>(&otherConst.m_value);
305  const Function *const myFunc = myData ? *myData : nullptr;
306  const Function *const otherFunc = otherData ? *otherData : nullptr;
307 
308  return myFunc && otherFunc && myFunc == otherFunc;
309  }
310  default: LOG_FATAL("Invalid operator %1", operToString(m_oper));
311  }
312 
313  return false;
314 }

317 SharedType Const::ascendType()
318 {
319  if (m_type->resolvesToVoid()) {
320  switch (m_oper) {
321  // could be anything, Boolean, Character, we could be bit fiddling pointers for all we
322  // know - trentw
323  case opIntConst: return VoidType::get();
324  case opLongConst: return m_type = IntegerType::get(STD_SIZE * 2, Sign::Unknown);
325  case opFltConst: return m_type = FloatType::get(64);
326  case opStrConst: return m_type = PointerType::get(CharType::get());
327  case opFuncConst: return m_type = PointerType::get(FuncType::get());
328  default: assert(false); // Bad Const
329  }
330  }
331 
332  return m_type;
333 }

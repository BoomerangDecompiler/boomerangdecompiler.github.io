30 bool StatementPropagationPass::execute(UserProc *proc)
31 {
32  StatementList stmts;
33  proc->getStatements(stmts);
34 
35  // count the number of times each assignment LHS would be propagated somewhere
36  std::map<SharedExp, int, lessExpStar> destCounts;
37 
38  // Also maintain a set of locations which are used by phi statements
39  for (SharedStmt s : stmts) {
40  ExpDestCounter edc(destCounts);
41  StmtDestCounter sdc(&edc);
42  s->accept(&sdc);
43  }
44 
45  // A fourth pass to propagate only the flags
46  // (these must be propagated even if it results in extra locals)
47  bool change = false;
48 
49  for (SharedStmt s : stmts) {
50  if (!s->isPhi()) {
51  change |= s->propagateFlagsToThis();
52  }
53  }
54 
55  // Finally the actual propagation
56  const int propMaxDepth = proc->getProg()->getProject()->getSettings()->propMaxDepth;
57  for (SharedStmt s : stmts) {
58  if (!s->isPhi()) {
59  change |= s->propagateToThis(propMaxDepth, &destCounts);
60  }
61  }
62 
63  PassManager::get()->executePass(PassID::FragSimplify, proc);
64  propagateToCollector(&proc->getUseCollector());
65 
66  return change;
67 }

70 void StatementPropagationPass::propagateToCollector(UseCollector *collector)
71 {
72  // TODO propagateToCollector(proc->getUseCollector());
73  for (auto it = collector->begin(); it != collector->end();) {
74  if (!(*it)->isMemOf()) {
75  ++it;
76  continue;
77  }
78 
79  auto addr = (*it)->getSubExp1();
80  LocationSet used;
81  addr->addUsedLocs(used);
82 
83  for (const SharedExp &v : used) {
84  if (!v->isSubscript()) {
85  continue;
86  }
87 
88  auto r = v->access<RefExp>();
89  if (!r->getDef() || !r->getDef()->isAssign()) {
90  continue;
91  }
92 
93  std::shared_ptr<Assign> as = r->getDef()->as<Assign>();
94 
95  bool ch;
96  auto res = addr->clone()->searchReplaceAll(*r, as->getRight(), ch);
97 
98  if (!ch) {
99  continue; // No change
100  }
101 
102  auto memOfRes = Location::memOf(res)->simplify();
103 
104  // First check to see if memOfRes is already in the set
105  if (collector->hasUse(memOfRes)) {
106  it = collector->removeUse(it); // Already exists; just remove the old one
107  continue;
108  }
109  else {
110  LOG_VERBOSE("Propagating %1 to %2 in collector; result %3", r, as->getRight(),
111  memOfRes);
112  (*it)->setSubExp1(res); // Change the child of the memof
113  }
114  }
115 
116  ++it;
117  }
118 }

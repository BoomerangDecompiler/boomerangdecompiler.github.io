37 /**
38  * Translates Capstone register IDs to Boomerang internal register IDs.
39  * \returns RegNumSpecial if register not found.
40  */
41 RegNum fixRegNum(int csRegID)
42 {
43  // GPR
44  if (csRegID >= cs::PPC_REG_R0 && csRegID <= cs::PPC_REG_R31) {
45  return REG_PPC_G0 + (csRegID - cs::PPC_REG_R0);
46  }
47  // FPR
48  else if (csRegID >= cs::PPC_REG_F0 && csRegID <= cs::PPC_REG_F31) {
49  return REG_PPC_F0 + (csRegID - cs::PPC_REG_F0);
50  }
51  // Vector regs
52  else if (csRegID >= cs::PPC_REG_V0 && csRegID <= cs::PPC_REG_V31) {
53  return REG_PPC_VR0 + (csRegID - cs::PPC_REG_V0);
54  }
55  // CR fields
56  else if (csRegID >= cs::PPC_REG_CR0 && csRegID <= cs::PPC_REG_CR7) {
57  return REG_PPC_CR0 + (csRegID - cs::PPC_REG_CR0);
58  }
59 
60  auto it = oldRegMap.find((cs::ppc_reg)csRegID);
61  return (it != oldRegMap.end()) ? it->second : RegNumSpecial;
62 }

65 SharedExp operandToExp(const cs::cs_ppc_op &operand)
66 {
67  switch (operand.type) {
68  case cs::PPC_OP_IMM: {
69  return Const::get(Address(operand.imm));
70  }
71  case cs::PPC_OP_REG: {
72  return Location::regOf(fixRegNum(operand.reg));
73  }
74  case cs::PPC_OP_MEM: {
75  return Location::memOf(Binary::get(opPlus, Location::regOf(fixRegNum(operand.mem.base)),
76  Const::get(operand.mem.disp)))
77  ->simplifyArith();
78  }
79  default: LOG_ERROR("Unknown ppc instruction operand type %1", (int)operand.type); break;
80  }
81 
82  return nullptr;
83 }

93 bool CapstonePPCDecoder::disassembleInstruction(Address pc, ptrdiff_t delta,
94  MachineInstruction &result)
95 {
96  const Byte *instructionData = reinterpret_cast<const Byte *>((HostAddress(delta) + pc).value());
97 
98  cs::cs_insn *decodedInstruction;
99  size_t numInstructions = cs_disasm(m_handle, instructionData, PPC_INSN_LENGTH, pc.value(), 1,
100  &decodedInstruction);
101  const bool valid = numInstructions > 0;
102 
103  if (!valid) {
104  return false;
105  }
106 
107  // Adjust the operands of cr* instructions (e.g. crxor).
108  // This is to work around a bug in Capstone: The operands are disassembled as PPC_OP_REG
109  // instead of PPC_OP_IMM or PPC_OP_CRX. See https://github.com/aquynh/capstone/issues/971
110  // for details.
111  if (isCRManip(decodedInstruction)) {
112  for (std::size_t i = 0; i < decodedInstruction->detail->ppc.op_count; ++i) {
113  cs::cs_ppc_op &operand = decodedInstruction->detail->ppc.operands[i];
114 
115  const int bitNum = operand.reg - cs::PPC_REG_R0;
116  operand.type = cs::PPC_OP_IMM;
117  operand.imm = bitNum;
118  }
119  }
120 
121  result.m_addr = Address(decodedInstruction->address);
122  result.m_id = decodedInstruction->id;
123  result.m_size = decodedInstruction->size;
124 
125  std::strncpy(result.m_mnem.data(), decodedInstruction->mnemonic, MNEM_SIZE);
126  std::strncpy(result.m_opstr.data(), decodedInstruction->op_str, OPSTR_SIZE);
127  result.m_mnem[MNEM_SIZE - 1] = '\0';
128  result.m_opstr[OPSTR_SIZE - 1] = '\0';
129 
130  const std::size_t numOperands = decodedInstruction->detail->ppc.op_count;
131  result.m_operands.resize(numOperands);
132 
133  for (std::size_t i = 0; i < numOperands; ++i) {
134  result.m_operands[i] = operandToExp(decodedInstruction->detail->ppc.operands[i]);
135  }
136 
137  result.m_templateName = getTemplateName(decodedInstruction);
138 
139  result.setGroup(MIGroup::Call, isCall(decodedInstruction));
140  result.setGroup(MIGroup::Jump, isJump(decodedInstruction));
141  result.setGroup(MIGroup::Ret, isRet(decodedInstruction));
142 
143  cs_free(decodedInstruction, numInstructions);
144  return true;
145 }

168 std::unique_ptr<RTL> CapstonePPCDecoder::createRTLForInstruction(const MachineInstruction &insn)
169 {
170  std::unique_ptr<RTL> rtl = instantiateRTL(insn);
171 
172  if (rtl == nullptr) {
173  return nullptr;
174  }
175 
176  const QString insnID = insn.m_templateName;
177  const std::size_t numOperands = insn.getNumOperands();
178 
179  if (insnID == "BL" || insnID == "BLA") {
180  Address callDest = Address(insn.m_operands[0]->access<Const>()->getLong());
181  std::shared_ptr<CallStatement> callStmt(new CallStatement(callDest));
182  callStmt->setIsComputed(false);
183 
184  rtl->append(std::make_shared<Assign>(SizeType::get(32), Location::regOf(REG_PPC_LR),
185  Const::get(insn.m_addr + PPC_INSN_LENGTH)));
186  rtl->append(callStmt);
187 
188  if (m_prog) {
189  Function *callee = m_prog->getOrCreateFunction(callDest);
190  if (callee && callee != reinterpret_cast<Function *>(-1)) {
191  callStmt->setDestProc(callee);
192  }
193  }
194  }
195  else if (insnID == "BCTR") {
196  std::shared_ptr<CaseStatement> jump(new CaseStatement(Location::regOf(REG_PPC_CTR)));
197  jump->setIsComputed(true);
198  rtl->append(jump);
199  }
200  else if (insnID == "BCTRL") {
201  rtl->append(std::make_shared<Assign>(SizeType::get(32), Location::regOf(REG_PPC_LR),
202  Const::get(Address(insn.m_addr + PPC_INSN_LENGTH))));
203 
204  std::shared_ptr<CallStatement> call(new CallStatement(Location::regOf(REG_PPC_CTR)));
205  call->setIsComputed(true);
206  rtl->append(call);
207  }
208  else if (insnID == "BGT") {
209  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
210  jump->setCondType(BranchType::JSG);
211  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
212  jump->setDest(insn.m_addr);
213  }
214  else {
215  jump->setDest(insn.m_operands[numOperands - 1]);
216  }
217  }
218  else if (insnID == "BGE") {
219  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
220  jump->setCondType(BranchType::JSGE);
221  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
222  jump->setDest(insn.m_addr);
223  }
224  else {
225  jump->setDest(insn.m_operands[numOperands - 1]);
226  }
227  }
228  else if (insnID == "BLT") {
229  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
230  jump->setCondType(BranchType::JSL);
231  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
232  jump->setDest(insn.m_addr);
233  }
234  else {
235  jump->setDest(insn.m_operands[numOperands - 1]);
236  }
237  }
238  else if (insnID == "BLE") {
239  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
240  jump->setCondType(BranchType::JSLE);
241  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
242  jump->setDest(insn.m_addr);
243  }
244  else {
245  jump->setDest(insn.m_operands[numOperands - 1]);
246  }
247  }
248  else if (insnID == "BNE") {
249  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
250  jump->setCondType(BranchType::JNE);
251  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
252  jump->setDest(insn.m_addr);
253  }
254  else {
255  jump->setDest(insn.m_operands[numOperands - 1]);
256  }
257  }
258  else if (insnID == "BEQ") {
259  std::shared_ptr<BranchStatement> jump = rtl->back()->as<BranchStatement>();
260  jump->setCondType(BranchType::JE);
261  if (numOperands == 0 || !insn.m_operands[numOperands - 1]->isIntConst()) {
262  jump->setDest(insn.m_addr);
263  }
264  else {
265  jump->setDest(insn.m_operands[numOperands - 1]);
266  }
267  }
268  else if (insnID == "BDNZ" || insnID == "BDNZL") {
269  const Address dest = insn.m_operands[numOperands - 1]->access<Const>()->getAddr();
270  if (dest != insn.m_addr + PPC_INSN_LENGTH) {
271  std::shared_ptr<BranchStatement> jump(new BranchStatement(dest));
272  jump->setCondType(BranchType::JNE);
273  rtl->append(jump);
274  }
275  }
276  else if (insnID == "STMW") {
277  rtl->clear();
278  const RegNum startRegNum = insn.m_operands[0]->access<Const, 1>()->getInt();
279  const SharedConstExp startAddrExp = Unary::get(opAddrOf, insn.m_operands[1]->clone())
280  ->simplify();
281 
282  for (RegNum reg = startRegNum; reg <= REG_PPC_G31; ++reg) {
283  const int i = reg - startRegNum;
284  const SharedExp memExp = Location::memOf(
285  Binary::get(opPlus, startAddrExp->clone(), Const::get(4 * i)));
286 
287  std::shared_ptr<Assign> asgn(
288  new Assign(SizeType::get(STD_SIZE), memExp->simplify(), Location::regOf(reg)));
289 
290  rtl->append(asgn);
291  }
292  }
293  else if (insnID == "LMW") {
294  rtl->clear();
295  const RegNum startRegNum = insn.m_operands[0]->access<Const, 1>()->getInt();
296  const SharedConstExp startAddrExp = Unary::get(opAddrOf, insn.m_operands[1]->clone())
297  ->simplify();
298 
299  for (RegNum reg = startRegNum; reg <= REG_PPC_G31; ++reg) {
300  const int i = reg - startRegNum;
301  const SharedExp memExp = Location::memOf(
302  Binary::get(opPlus, startAddrExp->clone(), Const::get(4 * i)));
303 
304  std::shared_ptr<Assign> asgn(
305  new Assign(SizeType::get(STD_SIZE), Location::regOf(reg), memExp->simplify()));
306 
307  rtl->append(asgn);
308  }
309  }
310  else if (insnID == "LBZU" || insnID == "LHZU" || insnID == "LWZU" || insnID == "LFSU" ||
311  insnID == "LFDU" || insnID == "LHAU" || insnID == "STFSU" || insnID == "STFDU") {
312  const SharedExp srcBase = Location::regOf(
313  insn.m_operands[1]->access<Const, 1, 1, 1>()->getInt());
314  const SharedExp offset = Const::get(insn.m_operands[1]->access<Const, 1, 2>()->getInt());
315 
316  rtl->append(std::make_shared<Assign>(SizeType::get(32), srcBase,
317  Binary::get(opPlus, srcBase, offset)));
318  }
319 
320  return rtl;
321 }

365 QString CapstonePPCDecoder::getTemplateName(const cs::cs_insn *instruction) const
366 {
367  QString insnID = instruction->mnemonic; // cs::cs_insn_name(m_handle, instruction->id);
368  insnID = insnID.toUpper();
369 
370  // Chop off branch prediction hints
371  if (insnID.endsWith("+") || insnID.endsWith("-")) {
372  insnID = insnID.left(insnID.length() - 1);
373  }
374 
375  // . cannot be part of an identifier -> use q instead
376  insnID = insnID.replace('.', 'q');
377  return insnID;
378 }

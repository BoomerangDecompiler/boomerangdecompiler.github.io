76 void Type::addNamedType(const QString &name, SharedType type)
77 {
78  if (g_namedTypes.find(name) != g_namedTypes.end()) {
79  if (!(*type == *g_namedTypes[name])) {
80  LOG_WARN("Redefinition of type %1", name);
81  LOG_WARN(" type = %1", type->getCtype());
82  LOG_WARN(" previous = %1", g_namedTypes[name]->getCtype());
83  g_namedTypes[name] = type; // WARN: was *type==*namedTypes[name], verify !
84  }
85  }
86  else {
87  // check if it is:
88  // typedef int a;
89  // typedef a b;
90  // we then need to define b as int
91  // we create clones to keep the GC happy
92  if (g_namedTypes.find(type->getCtype()) != g_namedTypes.end()) {
93  g_namedTypes[name] = g_namedTypes[type->getCtype()]->clone();
94  }
95  else {
96  g_namedTypes[name] = type->clone();
97  }
98  }
99 }

181 // A crude shortcut representation of a type
182 OStream &operator<<(OStream &os, const Type &type)
183 {
184  switch (type.getId()) {
185  case TypeClass::Integer: {
186  int sg = (int)type.as<IntegerType>()->getSign();
187  // 'j' for either i or u, don't know which
188  os << (sg == 0 ? 'j' : sg > 0 ? 'i' : 'u');
189  os << type.as<IntegerType>()->getSize();
190  break;
191  }
192 
193  case TypeClass::Float:
194  os << 'f';
195  os << type.as<FloatType>()->getSize();
196  break;
197 
198  case TypeClass::Pointer: os << type.as<PointerType>()->getPointsTo() << '*'; break;
199  case TypeClass::Size: os << type.getSize(); break;
200  case TypeClass::Char: os << 'c'; break;
201  case TypeClass::Void: os << 'v'; break;
202  case TypeClass::Boolean: os << 'b'; break;
203  case TypeClass::Compound: os << "struct"; break;
204  case TypeClass::Union: os << "union"; break;
205  case TypeClass::Func: os << "func"; break;
206  case TypeClass::Named: os << type.as<NamedType>()->getName(); break;
207  case TypeClass::Array:
208  os << '[' << type.as<ArrayType>()->getBaseType();
209 
210  if (!type.as<ArrayType>()->isUnbounded()) {
211  os << ", " << type.as<ArrayType>()->getLength();
212  }
213 
214  os << ']';
215  break;
216  }
217 
218  return os;
219 }

245 SharedType Type::createUnion(SharedType other, bool &changed, bool useHighestPtr) const
246 {
247  // `this' should not be a UnionType
248  assert(!resolvesToUnion());
249 
250  // Put all the hard union logic in one place
251  if (other->resolvesToUnion()) {
252  SharedType result = const_cast<Type *>(this)->shared_from_this();
253  result = other->meetWith(result, changed, useHighestPtr)->clone();
254  changed = true;
255  return result;
256  }
257 
258  // Check for anytype meet compound with anytype as first element
259  if (other->resolvesToCompound()) {
260  auto otherComp = other->as<CompoundType>();
261  SharedType firstType = otherComp->getMemberTypeByIdx(0);
262 
263  if (firstType->isCompatibleWith(*this)) {
264  // struct meet first element = struct
265  return other->clone();
266  }
267  }
268 
269  // Check for anytype meet array of anytype
270  if (other->resolvesToArray()) {
271  auto otherArr = other->as<ArrayType>();
272  SharedType elemTy = otherArr->getBaseType();
273 
274  if (elemTy->isCompatibleWith(*this)) {
275  // x meet array[x] == array[x]
276  changed = true; // since 'this' type is not an array, but the returned type is
277  return other->clone();
278  }
279  }
280 
281  changed = true;
282  return std::make_shared<UnionType>(
283  std::initializer_list<SharedType>{ this->clone(), other->clone() });
284 }

39 ProcStatus ProcDecompiler::tryDecompileRecursive(UserProc *proc)
40 {
41  Project *project = proc->getProg()->getProject();
42 
43  if (proc->getStatus() < ProcStatus::Visited) {
44  LOG_MSG("Visiting procedure '%1'", proc->getName());
45  }
46  else {
47  project->alertDiscovered(proc);
48  LOG_MSG("Re-visiting procedure '%1'", proc->getName());
49  }
50 
51  // Prevent infinite loops when there are cycles in the call graph (should never happen now)
52  if (proc->isDecompiled()) {
53  LOG_WARN("Not decompiling '%1' because it is already decompiled.", proc->getName());
54  return ProcStatus::FinalDone;
55  }
56  else if (proc->getStatus() < ProcStatus::Decoded) {
57  // Can happen e.g. if a callee is visible only after analysing a switch statement
58  // Actually decoding for the first time, not REdecoding
59  if (!proc->getProg()->reDecode(proc)) {
60  return ProcStatus::Undecoded;
61  }
62  }
63 
64  if (proc->getStatus() < ProcStatus::Visited) {
65  // We have at least visited this proc "on the way down"
66  proc->setStatus(ProcStatus::Visited);
67  }
68 
69  m_callStack.push_back(proc);
70 
71  if (project->getSettings()->verboseOutput) {
72  printCallStack();
73  }
74 
75  PassManager::get()->executePass(PassID::StatementInit, proc);
76  project->alertDecompileDebugPoint(proc, "after lifting");
77 
78  proc->numberStatements();
79 
80  earlyDecompile(proc);
81 
82  if (project->getSettings()->decodeChildren) {
83  // Recurse to callees first, to perform a depth first search
84  for (IRFragment *frag : *proc->getCFG()) {
85  if (!frag->isType(FragType::Call)) {
86  continue;
87  }
88 
89  // The call statement will be in the last RTL in this fragment
90  if (!frag->getRTLs()) {
91  continue; // not lifted yet
92  }
93 
94  SharedStmt hl = frag->getRTLs()->back()->getHlStmt();
95 
96  if (!hl || !hl->isCall()) {
97  LOG_WARN("Fragment at address %1 is a CALL but last stmt is not a call: %2",
98  frag->getLowAddr(), hl);
99  continue;
100  }
101 
102  std::shared_ptr<CallStatement> call = hl->as<CallStatement>();
103  UserProc *callee = dynamic_cast<UserProc *>(call->getDestProc());
104 
105  if (callee == nullptr) { // not an user proc, or missing dest
106  continue;
107  }
108 
109  if (callee->isDecompiled()) {
110  // Already decompiled, but the return statement still needs to be set for this call
111  call->setCalleeReturn(callee->getRetStmt());
112  continue;
113  }
114 
115  decompileCallee(callee, proc);
116 
117  // Callee has at least done middleDecompile(), possibly more
118  call->setCalleeReturn(callee->getRetStmt());
119  }
120  }
121 
122  project->alertDecompileDebugPoint(proc, "after decompiling callees for the first time");
123 
124  // if no callee is involved in recursion
125  if (proc->getStatus() != ProcStatus::InCycle) {
126  project->alertDecompiling(proc);
127  LOG_MSG("Decompiling procedure '%1'", proc->getName());
128  middleDecompile(proc);
129 
130  if (project->getSettings()->verboseOutput) {
131  printCallStack();
132  }
133  }
134 
135  if (proc->getStatus() != ProcStatus::InCycle) {
136  lateDecompile(proc); // Do the whole works
137  proc->setStatus(ProcStatus::FinalDone);
138  project->alertEndDecompile(proc);
139  }
140  else if (m_recursionGroups.find(proc) != m_recursionGroups.end()) {
141  // This proc's callees, and hence this proc, is/are involved in recursion.
142  // Find first element f in the call stack that is also in our recursion group
143  ProcList::iterator f = std::find_if(
144  m_callStack.begin(), m_callStack.end(), [proc](UserProc *func) {
145  return proc->getRecursionGroup()->find(func) != proc->getRecursionGroup()->end();
146  });
147 
148  // The big test: have we found the whole strongly connected component (in the call graph)?
149  if (*f == proc) {
150  // Yes, process these procs as a group
151  recursionGroupAnalysis(proc->getRecursionGroup());
152  proc->setStatus(ProcStatus::FinalDone);
153  project->alertEndDecompile(proc);
154  }
155  }
156 
157  // Remove last element (= this) from path
158  assert(!m_callStack.empty());
159  assert(m_callStack.back() == proc);
160  m_callStack.pop_back();
161 
162  LOG_MSG("Finished decompile of '%1'", proc->getName());
163 
164  if (project->getSettings()->verboseOutput) {
165  printCallStack();
166  }
167 
168  return proc->getStatus();
169 }

172 void ProcDecompiler::createRecursionGoup(const std::shared_ptr<ProcSet> &newGroup)
173 {
174  LOG_VERBOSE("Creating recursion group:");
175  for (UserProc *proc : *newGroup) {
176  LOG_VERBOSE(" %1", proc->getName());
177  }
178 
179  // find all exisiting groups and union them with the new one
180  std::shared_ptr<ProcSet> unionGroup = newGroup;
181 
182  for (UserProc *proc : *newGroup) {
183  auto it = m_recursionGroups.find(proc);
184 
185  if (it != m_recursionGroups.end()) {
186  // proc is already part of a recursion group
187  if (unionGroup == newGroup) {
188  unionGroup.reset(new ProcSet);
189  }
190 
191  for (UserProc *existingProc : *newGroup) {
192  unionGroup->insert(existingProc);
193 
194  if (existingProc->getRecursionGroup()) {
195  unionGroup->insert(existingProc->getRecursionGroup()->begin(),
196  existingProc->getRecursionGroup()->end());
197  }
198  }
199  }
200  }
201 
202  for (UserProc *proc : *unionGroup) {
203  m_recursionGroups[proc] = unionGroup;
204  proc->setRecursionGroup(unionGroup);
205  proc->setStatus(ProcStatus::InCycle);
206  }
207 }

249 void ProcDecompiler::earlyDecompile(UserProc *proc)
250 {
251  Project *project = proc->getProg()->getProject();
252  project->alertStartDecompile(proc);
253  project->alertDecompileDebugPoint(proc, "before earlyDecompile");
254 
255  // Remove branches with false guards
256  PassManager::get()->executePass(PassID::FragSimplify, proc);
257  PassManager::get()->executePass(PassID::Dominators, proc);
258 
259  if (proc->getStatus() < ProcStatus::MiddleDone) {
260  // Update the defines in the calls. Will redo if involved in recursion
261  PassManager::get()->executePass(PassID::CallDefineUpdate, proc);
262  PassManager::get()->executePass(PassID::GlobalConstReplace, proc);
263 
264  // First placement of phi functions, renaming, and initial propagation.
265  // This is mostly for the stack pointer.
266  // TODO: Check if this makes sense. It seems to me that we only want to do one pass of
267  // propagation here, since the status == check had been knobbled below. Hopefully,
268  // one call to placing phi functions etc will be equivalent to depth 0 in the old scheme
269  PassManager::get()->executePass(PassID::PhiPlacement, proc);
270 
271  // Rename variables
272  PassManager::get()->executePass(PassID::BlockVarRename, proc);
273  PassManager::get()->executePass(PassID::StatementPropagation, proc);
274  }
275 
276  project->alertDecompileDebugPoint(proc, "after earlyDecompile");
277 }

280 void ProcDecompiler::middleDecompile(UserProc *proc)
281 {
282  assert(m_callStack.back() == proc);
283  Project *project = proc->getProg()->getProject();
284 
285  project->alertDecompileDebugPoint(proc, "before middleDecompile");
286 
287  // The call bypass logic should be staged as well. For example, consider m[r1{11}]{11} where 11
288  // is a call. The first stage bypass yields m[r1{2}]{11}, which needs another round of
289  // propagation to yield m[r1{-}-32]{11} (which can safely be processed at depth 1). Except that
290  // this is now inherent in the visitor nature of the latest algorithm.
291 
292  PassManager::get()->executePass(PassID::CallAndPhiFix, proc);
293 
294  if (proc->getStatus() != ProcStatus::InCycle) { // FIXME: need this test?
295  PassManager::get()->executePass(PassID::StatementPropagation, proc);
296  }
297 
298  // Oops - the idea of splitting the sp from the rest of the preservations was to allow correct
299  // naming of locals so you are alias conservative. But of course some locals are ebp (etc)
300  // based, and so these will never be correct until all the registers have preservation analysis
301  // done. So I may as well do them all together here.
302  PassManager::get()->executePass(PassID::SPPreservation, proc);
303  PassManager::get()->executePass(PassID::PreservationAnalysis, proc);
304  PassManager::get()->executePass(PassID::CallAndPhiFix, proc);
305 
306  project->alertDecompileDebugPoint(proc, "after preservation, bypass and propagation");
307 
308  if (project->getSettings()->usePromotion) {
309  // We want functions other than main to be promoted. Needed before mapExpressionsToLocals
310  proc->promoteSignature();
311  }
312 
313  // The problem with doing locals too early is that the symbol map ends up with some {-} and some
314  // {0} Also, once named as a local, it is tempting to propagate the memory location, but that
315  // might be unsafe if the address is taken.
316 
317  // Update the arguments for calls (mainly for the non recursion affected calls)
318  // We have only done limited propagation and collecting to this point. Need e.g. to put m[esp-K]
319  // into the collectors of calls, so when a stack parameter is created, it will be correctly
320  // localised Note that we'd like to limit propagation before this point, because we have not yet
321  // created any arguments, so it is possible to get "excessive propagation" to parameters. In
322  // fact, because uses vary so much throughout a program, it may end up better not limiting
323  // propagation until very late in the decompilation, and undoing some propagation just before
324  // removing unused statements. Or even later, if that is possible. For now, we create the
325  // initial arguments here (relatively early), and live with the fact that some apparently
326  // distinct memof argument expressions (e.g. m[eax{30}] and m[esp{40}-4]) will turn out to be
327  // duplicates, and so the duplicates must be eliminated.
328  PassManager::get()->executePass(PassID::PhiPlacement, proc);
329  PassManager::get()->executePass(PassID::BlockVarRename, proc);
330 
331  // Otherwise sometimes sp is not fully propagated
332  PassManager::get()->executePass(PassID::StatementPropagation, proc);
333  PassManager::get()->executePass(PassID::CallArgumentUpdate, proc);
334  PassManager::get()->executePass(PassID::StrengthReductionReversal, proc);
335 
336  project->alertDecompileDebugPoint(proc, "after updating call arguments");
337 
338  // Repeat until no change
339  int pass = 3;
340  bool change;
341 
342  do {
343  // Redo the renaming process to take into account the arguments
344  change = PassManager::get()->executePass(PassID::PhiPlacement, proc);
345  change |= PassManager::get()->executePass(PassID::BlockVarRename, proc);
346 
347  // Seed the return statement with reaching definitions
348  // FIXME: does this have to be in this loop?
349  if (proc->getRetStmt()) {
350  // Everything including new arguments reaching the exit
351  proc->getRetStmt()->updateModifieds();
352  proc->getRetStmt()->updateReturns();
353  }
354 
355  // (* Was: mapping expressions to Parameters as we go *)
356 
357  // FIXME: Check if this is needed any more. At least fib seems to need it at present.
358  if (project->getSettings()->changeSignatures) {
359  for (int i = 0; i < 3; i++) { // FIXME: should be iterate until no change
360  LOG_VERBOSE("### update returns loop iteration %1 ###", i);
361 
362  if (proc->getStatus() != ProcStatus::InCycle) {
363  PassManager::get()->executePass(PassID::BlockVarRename, proc);
364  }
365 
366  PassManager::get()->executePass(PassID::PreservationAnalysis, proc);
367 
368  // Returns have uses which affect call defines (if childless)
369  PassManager::get()->executePass(PassID::CallDefineUpdate, proc);
370  PassManager::get()->executePass(PassID::CallAndPhiFix, proc);
371 
372  // Preserveds subtract from returns
373  PassManager::get()->executePass(PassID::PreservationAnalysis, proc);
374  }
375  }
376 
377  change |= PassManager::get()->executePass(PassID::StatementPropagation, proc);
378  change |= PassManager::get()->executePass(PassID::BlockVarRename, proc);
379 
380  // this is just to make it readable, do NOT rely on these statements being removed
381  PassManager::get()->executePass(PassID::AssignRemoval, proc);
382  project->alertDecompileDebugPoint(proc,
383  "after updating returns pass " + QString::number(pass));
384  } while (change && ++pass < 12);
385 
386  // At this point, there will be some memofs that have still not been renamed. They have been
387  // prevented from getting renamed so that they didn't get renamed incorrectly (usually as {-}),
388  // when propagation and/or bypassing may have ended up changing the address expression. There is
389  // now no chance that this will happen, so we need to rename the existing memofs. Note that this
390  // can still link uses to definitions, e.g. 50 r26 := phi(...) 51 m[r26{50}] := 99;
391  // ... := m[r26{50}]{should be 51}
392 
393  project->alertDecompileDebugPoint(proc, "before renaming memofs");
394  proc->getDataFlow()->setRenameLocalsParams(true);
395 
396  PassManager::get()->executePass(PassID::PhiPlacement, proc);
397  PassManager::get()->executePass(PassID::BlockVarRename, proc);
398  PassManager::get()->executePass(PassID::StatementPropagation, proc);
399 
400  // Now that memofs are renamed, the bypassing for memofs can work
401  PassManager::get()->executePass(PassID::CallAndPhiFix, proc);
402 
403  project->alertDecompileDebugPoint(proc, "after renaming memofs");
404 
405  // Check for indirect jumps or calls not already removed by propagation of constants
406  bool changed = false;
407  IndirectJumpAnalyzer analyzer;
408 
409  for (IRFragment *frag : *proc->getCFG()) {
410  changed |= analyzer.decodeIndirectJmp(frag, proc);
411  }
412 
413  project->alertDecompileDebugPoint(proc, "after analyzing indirect jumps");
414 
415  if (changed) {
416  // There was at least one indirect jump or call found and decoded. That means that most of
417  // what has been done to this function so far is invalid. So redo everything. Very
418  // expensive!!
419  reDecompileRecursive(proc);
420  return;
421  }
422 
423  PassManager::get()->executePass(PassID::PreservationAnalysis, proc);
424  PassManager::get()->executePass(PassID::DuplicateArgsRemoval, proc);
425 
426  // Perform type analysis. If we are relying (as we are at present) on TA to perform ellipsis
427  // processing, do the local TA pass now. Ellipsis processing often reveals additional uses (e.g.
428  // additional parameters to printf/scanf), and removing unused statements is unsafe without full
429  // use information
430  if (!proc->isDecompiled()) {
431  PassManager::get()->executePass(PassID::LocalTypeAnalysis, proc);
432 
433  // Now that locals are identified, redo the dataflow
434  PassManager::get()->executePass(PassID::PhiPlacement, proc);
435  PassManager::get()->executePass(PassID::BlockVarRename, proc);
436  PassManager::get()->executePass(PassID::StatementPropagation, proc);
437 
438  proc->debugPrintAll("after propagating locals");
439  }
440 
441  tryConvertCallsToDirect(proc);
442  tryConvertFunctionPointerAssignments(proc);
443 
444  proc->setStatus(ProcStatus::MiddleDone);
445  project->alertDecompileDebugPoint(proc, "after middleDecompile");
446 }

449 bool ProcDecompiler::decompileProcInRecursionGroup(UserProc *proc, ProcSet &visited)
450 {
451  bool changed = false;
452  Project *project = proc->getProg()->getProject();
453 
454  visited.insert(proc);
455  m_callStack.push_back(proc);
456 
457  for (Function *c : proc->getCallees()) {
458  if (c->isLib()) {
459  continue;
460  }
461 
462  UserProc *callee = static_cast<UserProc *>(c);
463  if (visited.find(callee) != visited.end()) {
464  continue;
465  }
466  else if (proc->getRecursionGroup()->find(callee) == proc->getRecursionGroup()->end()) {
467  // not in recursion group any more
468  continue;
469  }
470 
471  // visit unvisited callees first
472  changed |= decompileProcInRecursionGroup(callee, visited);
473  }
474 
475  proc->setStatus(ProcStatus::InCycle); // So the calls are treated as childless
476  project->alertDecompiling(proc);
477  LOG_MSG("Decompiling proc '%1' in recursion group", proc->getName());
478  earlyDecompile(proc);
479 
480  // The standard preservation analysis should automatically perform conditional preservation.
481  middleDecompile(proc);
482  proc->setStatus(ProcStatus::Preserveds);
483 
484  // Mark all the relevant calls as non childless (will harmlessly get done again later)
485  // FIXME: why exactly do we do this?
486  proc->markAsNonChildless(proc->getRecursionGroup());
487 
488  // Need to propagate into the initial arguments, since arguments are uses,
489  // and we are about to remove unused statements.
490  changed |= PassManager::get()->executePass(PassID::LocalAndParamMap, proc);
491  changed |= PassManager::get()->executePass(PassID::CallArgumentUpdate, proc);
492  changed |= PassManager::get()->executePass(PassID::Dominators, proc);
493  changed |= PassManager::get()->executePass(PassID::StatementPropagation, proc);
494 
495  assert(m_callStack.back() == proc);
496  m_callStack.pop_back();
497  return changed;
498 }

501 void ProcDecompiler::recursionGroupAnalysis(const std::shared_ptr<ProcSet> &group)
502 {
503  /* Overall algorithm:
504  * for each proc in the group
505  * initialise
506  * earlyDecompile
507  * for each proc in the group
508  * middleDecompile
509  * mark all calls involved in cs as non-childless
510  * for each proc in cs
511  * update parameters and returns, redoing call bypass, until no change
512  * for each proc in cs
513  * remove unused statements
514  * for each proc in cs
515  * update parameters and returns, redoing call bypass, until no change
516  */
517  if (group->empty()) {
518  return;
519  }
520 
521  LOG_MSG("Performing recursion group analysis for %1 recursive procedures: ", group->size());
522  for (UserProc *proc : *group) {
523  LOG_MSG(" %1", proc->getName());
524  }
525 
526  UserProc *entry = *group->begin();
527  bool changed = false;
528  int numRepeats = 0;
529 
530  do {
531  ProcSet visited;
532  changed = decompileProcInRecursionGroup(entry, visited);
533  } while (changed && numRepeats++ < 2);
534 
535  // while no change
536  for (int i = 0; i < 2; i++) {
537  for (UserProc *proc : *group) {
538  lateDecompile(proc); // Also does final parameters and arguments at present
539  }
540  }
541 
542  LOG_MSG("=== End recursion group analysis ===");
543  for (UserProc *proc : *group) {
544  proc->getProg()->getProject()->alertEndDecompile(proc);
545  }
546 }

580 ProcStatus ProcDecompiler::reDecompileRecursive(UserProc *proc)
581 {
582  Project *project = proc->getProg()->getProject();
583 
584  LOG_MSG("Restarting decompilation of '%1'", proc->getName());
585  project->alertDecompileDebugPoint(proc, "before restarting decompilation");
586 
587  // decode from scratch
588  proc->removeRetStmt();
589  proc->getCFG()->clear();
590 
591  proc->getDataFlow()->setRenameLocalsParams(false); // Start again with memofs
592  proc->setStatus(ProcStatus::Visited); // Back to only visited progress
593 
594  assert(m_callStack.back() == proc);
595 
596  m_callStack.pop_back(); // Remove self from call stack
597  ProcStatus status = tryDecompileRecursive(proc); // Restart decompiling this proc
598  m_callStack.push_back(proc); // Restore self to call stack
599 
600  return status;
601 }

604 bool ProcDecompiler::tryConvertCallsToDirect(UserProc *proc)
605 {
606  bool change = false;
607  for (IRFragment *frag : *proc->getCFG()) {
608  if (frag->isType(FragType::CompCall)) {
609  std::shared_ptr<CallStatement> call = frag->getLastStmt()->as<CallStatement>();
610  const bool converted = call->tryConvertToDirect();
611  if (converted) {
612  Function *f = call->getDestProc();
613  if (f && !f->isLib()) {
614  decompileCallee(static_cast<UserProc *>(f), proc);
615  call->setCalleeReturn(static_cast<UserProc *>(f)->getRetStmt());
616  change = true;
617  }
618  }
619  }
620  }
621 
622  return change;
623 }

626 bool ProcDecompiler::tryConvertFunctionPointerAssignments(UserProc *proc)
627 {
628  bool changed = false;
629  StatementList statements;
630  proc->getStatements(statements);
631 
632  for (SharedStmt stmt : statements) {
633  if (stmt->isAssign()) {
634  std::shared_ptr<Assign> asgn = stmt->as<Assign>();
635  if (asgn->getType()->resolvesToFuncPtr()) {
636  if (asgn->getRight()->isIntConst()) {
637  std::shared_ptr<Const> rhs = asgn->getRight()->access<Const>();
638  Function *f = tryDecompileRecursive(rhs->getAddr(), proc->getProg(), proc);
639  asgn->setRight(Const::get(f));
640  changed = true;
641  }
642  else if (asgn->getRight()->getOper() == opTern &&
643  asgn->getRight()->getSubExp2()->isIntConst() &&
644  asgn->getRight()->getSubExp3()->isIntConst()) {
645  std::shared_ptr<Const> rhsLeft = asgn->getRight()->access<Const, 2>();

681 ProcStatus ProcDecompiler::decompileCallee(UserProc *callee, UserProc *proc)
682 {
683  Project *project = proc->getProg()->getProject();
684 
685  // check if the callee has already been visited but not done (apart from global
686  // analyses). This means that we have found a new cycle or a part of an existing cycle
687  if ((callee->getStatus() >= ProcStatus::Visited) &&
688  (callee->getStatus() <= ProcStatus::MiddleDone)) {
689  // if callee is in callStack
690  ProcList::iterator calleeIt = std::find(m_callStack.begin(), m_callStack.end(), callee);
691 
692  if (calleeIt != m_callStack.end()) {
693  // This is a completely new cycle
694  std::shared_ptr<ProcSet> newRecursionGroup(new ProcSet());
695  newRecursionGroup->insert(calleeIt, m_callStack.end());
696  createRecursionGoup(newRecursionGroup);
697  }
698  else if (callee->getRecursionGroup()) {
699  // This is a new branch of an existing cycle that was visited previously
700  std::shared_ptr<ProcSet> recursionGroup = callee->getRecursionGroup();
701 
702  // Find first element func of callStack that is in callee->recursionGroup
703  ProcList::iterator _pi = std::find_if(
704  m_callStack.begin(), m_callStack.end(), [callee](UserProc *func) {
705  return callee->getRecursionGroup()->find(func) !=
706  callee->getRecursionGroup()->end();
707  });
708 
709  // Insert every proc after func to the end of path into child
710  assert(_pi != m_callStack.end());
711  for (auto it = std::next(_pi); it != m_callStack.end(); ++it) {
712  addToRecursionGroup(*it, recursionGroup);
713  }
714  }
715 
716  proc->setStatus(ProcStatus::InCycle);
717  }
718  else {
719  // No new cycle
720  LOG_VERBOSE("Preparing to decompile callee '%1' of '%2'", callee->getName(),
721  proc->getName());
722 
723  if (project->getSettings()->usePromotion) {
724  callee->promoteSignature();
725  }
726 
727  tryDecompileRecursive(callee);
728 
729  if (proc->getStatus() != ProcStatus::InCycle &&
730  m_recursionGroups.find(proc) != m_recursionGroups.end()) {
731  proc->setStatus(ProcStatus::InCycle);
732  proc->setRecursionGroup(m_recursionGroups.find(proc)->second);
733  }
734  }
735 
736  return proc->getStatus();
737 }

43 bool BranchAnalysisPass::doBranchAnalysis(UserProc *proc)
44 {
45  std::set<IRFragment *, Util::ptrCompare<IRFragment>> fragsToRemove;
46 
47  for (IRFragment *a : *proc->getCFG()) {
48  if (!a->isType(FragType::Twoway)) {
49  continue;
50  }
51  else if (fragsToRemove.find(a) != fragsToRemove.end()) {
52  continue;
53  }
54 
55  if (a->isEmpty() || isOnlyBranch(a)) {
56  if (a->getSuccessor(BTHEN) == a) {
57  for (IRFragment *pred : a->getPredecessors()) {
58  proc->getCFG()->replaceEdge(pred, a, a->getSuccessor(BELSE));
59  }
60  fragsToRemove.insert(a);
61  continue;
62  }
63  else if (a->getSuccessor(BELSE) == a) {
64  for (IRFragment *pred : a->getPredecessors()) {
65  proc->getCFG()->replaceEdge(pred, a, a->getSuccessor(BTHEN));
66  }
67  fragsToRemove.insert(a);
68  continue;
69  }
70  }
71 
72  IRFragment *b = a->getSuccessor(BELSE);
73  if (!b || !b->isType(FragType::Twoway)) {
74  continue;
75  }
76  else if (!isOnlyBranch(b)) {
77  continue;
78  }
79 
80  assert(a->getLastStmt()->isBranch());
81  assert(b->getLastStmt()->isBranch());
82 
83  std::shared_ptr<BranchStatement> aBranch = a->getLastStmt()->as<BranchStatement>();
84  std::shared_ptr<BranchStatement> bBranch = b->getLastStmt()->as<BranchStatement>();
85 
86  // A: branch to D if cond1
87  // B: branch to D if cond2
88  // C: something
89  // D:
90  // ->
91  // A branch to D if cond1 || cond2
92  // C: something
93  // D:
94  if (b->getSuccessor(BTHEN) == a->getSuccessor(BTHEN) && b->getNumPredecessors() == 1) {
95  const SharedExp newCond = Binary::get(opOr, aBranch->getCondExpr(),
96  bBranch->getCondExpr());
97 
98  aBranch->setCondExpr(newCond->clone()->simplify());
99  aBranch->setFallFragment(bBranch->getFallFragment());
100 
101  assert(b->getNumPredecessors() == 0);
102  assert(b->getNumSuccessors() == 2);
103 
104  IRFragment *succ1 = b->getSuccessor(BTHEN);
105  IRFragment *succ2 = b->getSuccessor(BELSE);
106 
107  b->removeSuccessor(succ1);
108  b->removeSuccessor(succ2);
109 
110  succ1->removePredecessor(b);
111  succ2->removePredecessor(b);
112 
113  fragsToRemove.insert(b);
114  }
115 
116  // A: branch to C if cond1
117  // B: branch to D if cond2
118  // C: something
119  // D:
120  // ->
121  // A: branch to D if !cond1 && cond2
122  // C: something
123  // D:
124  else if (a->getSuccessor(BTHEN) == b->getSuccessor(BELSE) && b->getNumPredecessors() == 1) {
125  const SharedExp newCond = Binary::get(opAnd, Unary::get(opLNot, aBranch->getCondExpr()),
126  bBranch->getCondExpr());
127 
128  aBranch->setCondExpr(newCond->clone()->simplify());
129  aBranch->setDest(bBranch->getFixedDest());
130  aBranch->setTakenFragment(bBranch->getTakenFragment());
131  aBranch->setFallFragment(bBranch->getFallFragment());
132 
133  assert(b->getNumPredecessors() == 0);
134  assert(b->getNumSuccessors() == 2);
135 
136  IRFragment *succ1 = b->getSuccessor(BTHEN);
137  IRFragment *succ2 = b->getSuccessor(BELSE);
138 
139  b->removeSuccessor(succ1);
140  b->removeSuccessor(succ2);
141 
142  succ1->removePredecessor(b);
143  succ2->removePredecessor(b);
144 
145  fragsToRemove.insert(b);
146  }
147  }
148 
149  const bool removedFragments = !fragsToRemove.empty();
150  for (IRFragment *bb : fragsToRemove) {
151  proc->getCFG()->removeFragment(bb);
152  }
153 
154  return removedFragments;
155 }

158 void BranchAnalysisPass::fixUglyBranches(UserProc *proc)
159 {
160  StatementList stmts;
161  proc->getStatements(stmts);
162 
163  for (auto stmt : stmts) {
164  if (!stmt->isBranch()) {
165  continue;
166  }
167 
168  SharedExp hl = stmt->as<BranchStatement>()->getCondExpr();
169 
170  // of the form: x{n} - 1 >= 0
171  if (hl && (hl->getOper() == opGtrEq) && hl->getSubExp2()->isIntConst() &&
172  (hl->access<Const, 2>()->getInt() == 0) && (hl->getSubExp1()->getOper() == opMinus) &&
173  hl->getSubExp1()->getSubExp2()->isIntConst() &&
174  (hl->access<Const, 1, 2>()->getInt() == 1) &&
175  hl->getSubExp1()->getSubExp1()->isSubscript()) {
176  SharedStmt n = hl->access<RefExp, 1, 1>()->getDef();
177 
178  if (n && n->isPhi()) {
179  std::shared_ptr<PhiAssign> p = n->as<PhiAssign>();
180 
181  for (const auto &phi : *p) {
182  if (!phi->getDef()->isAssign()) {
183  continue;
184  }
185 
186  std::shared_ptr<Assign> a = phi->getDef()->as<Assign>();
187 
188  if (*a->getRight() == *hl->getSubExp1()) {
189  hl->setSubExp1(RefExp::get(a->getLeft(), a));
190  break;
191  }
192  }
193  }
194  }
195  }
196 }

199 bool BranchAnalysisPass::isOnlyBranch(IRFragment *frag) const
200 {
201  const RTLList *rtls = frag->getRTLs();
202  if (!rtls || rtls->empty()) {
203  return false;
204  }
205 
206  StatementList::reverse_iterator sIt;
207  IRFragment::RTLRIterator rIt;
208  bool last = true;
209 
210  for (SharedStmt s = frag->getLastStmt(rIt, sIt); s != nullptr;
211  s = frag->getPrevStmt(rIt, sIt)) {
212  if (!last) {
213  return false; // there are other statements beside the last branch
214  }
215  else if (!s->isBranch()) {
216  return false; // last stmt is not a branch, can't handle this
217  }
218  else {
219  last = false;
220  }
221  }
222 
223  return true;
224 }

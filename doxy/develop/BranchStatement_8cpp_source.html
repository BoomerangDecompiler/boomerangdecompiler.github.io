170 void BranchStatement::print(OStream &os) const
171 {
172  os << qSetFieldWidth(4) << m_number << qSetFieldWidth(0) << " ";
173  os << "BRANCH ";
174 
175  if (!m_dest->isIntConst()) {
176  os << m_dest;
177  }
178  else {
179  // Really we'd like to display the destination label here...
180  os << getFixedDest();
181  }
182 
183  os << ", condition ";
184 
185  switch (m_jumpType) {
186  case BranchType::JE: os << "equals"; break;
187  case BranchType::JNE: os << "not equals"; break;
188  case BranchType::JSL: os << "signed less"; break;
189  case BranchType::JSLE: os << "signed less or equals"; break;
190  case BranchType::JSGE: os << "signed greater or equals"; break;
191  case BranchType::JSG: os << "signed greater"; break;
192  case BranchType::JUL: os << "unsigned less"; break;
193  case BranchType::JULE: os << "unsigned less or equals"; break;
194  case BranchType::JUGE: os << "unsigned greater or equals"; break;
195  case BranchType::JUG: os << "unsigned greater"; break;
196  case BranchType::JMI: os << "minus"; break;
197  case BranchType::JPOS: os << "plus"; break;
198  case BranchType::JOF: os << "overflow"; break;
199  case BranchType::JNOF: os << "no overflow"; break;
200  case BranchType::JPAR: os << "parity"; break;
201  case BranchType::JNPAR: os << "no parity"; break;
202  case BranchType::INVALID: assert(false); break;
203  }
204 
205  if (m_isFloat) {
206  os << " float";
207  }
208 
209  os << '\n';
210 
211  if (m_cond) {
212  os << "High level: ";
213  m_cond->print(os);
214  }
215 }

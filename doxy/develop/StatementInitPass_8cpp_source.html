26 bool StatementInitPass::execute(UserProc *proc)
27 {
28  proc->getCFG()->clear();
29  proc->removeRetStmt();
30 
31  if (!proc->getProg()->getFrontEnd()->liftProc(proc)) {
32  return false;
33  }
34 
35  for (IRFragment *frag : *proc->getCFG()) {
36  IRFragment::RTLIterator rit;
37  StatementList::iterator sit;
38 
39  for (SharedStmt stmt = frag->getFirstStmt(rit, sit); stmt != nullptr;
40  stmt = frag->getNextStmt(rit, sit)) {
41  assert(stmt->getProc() == nullptr || stmt->getProc() == proc);
42 
43  // Remove out edges of fragments of noreturn calls (e.g. call fragments to abort())
44  if (!stmt->isCall()) {
45  continue;
46  }
47 
48  std::shared_ptr<CallStatement> call = stmt->as<CallStatement>();
49  call->setSigArguments();
50 
51  if ((frag->getNumSuccessors() != 1)) {
52  continue;
53  }
54 
55  Function *destProc = call->getDestProc();
56  if (!destProc) {
57  continue;
58  }
59 
60  if (!destProc->isLib() &&
61  static_cast<UserProc *>(destProc)->getStatus() != ProcStatus::Visited) {
62  continue; // Proc was not visited yet - We cannot know if it will return
63  }
64 
65  if (!destProc->isNoReturn()) {
66  continue;
67  }
68 
69 
70  IRFragment *nextFrag = frag->getSuccessor(0);
71 
72  // Do not remove the only predecessor of a return fragment
73  if ((nextFrag == proc->getCFG()->getExitFragment()) &&
74  proc->getCFG()->getExitFragment()->getNumPredecessors() == 1) {
75  continue;
76  }
77 
78  nextFrag->removePredecessor(frag);
79  frag->removeAllSuccessors();
80  }
81  }
82 
83  // Removing out edges of noreturn calls might sever paths between the entry fragment
84  // and other (now orphaned) fragments. We have to remove these fragments
85  // since all fragments must be reachable from the entry fragment for data-flow analysis
86  // to work.
87  CFGCompressor().compressCFG(proc->getCFG());
88  return true;
89 }

38 BasicBlock *LowLevelCFG::createBB(BBType bbType, const std::vector<MachineInstruction> &bbInsns)
39 {
40  assert(!bbInsns.empty());
41 
42  // First find the native address of the first instruction
43  const Address startAddr = bbInsns.front().m_addr;
44  assert(startAddr != Address::INVALID);
45 
46  BasicBlock *currentBB = nullptr;
47  BBStartMap::iterator mi = m_bbStartMap.find(startAddr);
48 
49  if ((mi != m_bbStartMap.end()) && mi->second) {
50  currentBB = mi->second;
51 
52  // It should be incomplete, or the BB there should be zero
53  // (we have called ensureBBExists() but not yet created the BB for it).
54  // Else we have duplicated BBs.
55  // Note: this can happen with forward jumps into the middle of a loop,
56  // so not error
57  if (currentBB->isComplete()) {
58  LOG_VERBOSE("Not creating a BB at address %1 because a BB already exists",
59  currentBB->getLowAddr());
60  return nullptr;
61  }
62  else {
63  // Fill in the details, and return it
64  currentBB->completeBB(bbInsns);
65  currentBB->setType(bbType);
66  }
67  }
68 
69  if (currentBB == nullptr) {
70  currentBB = new BasicBlock(bbType, bbInsns);
71 
72  // Note that currentBB->getLowAddr() == startAddr
73  if (startAddr == Address::INVALID) {
74  LOG_FATAL("Cannot add BB with invalid lowAddr %1", startAddr);
75  }
76 
77  insertBB(currentBB);
78  mi = m_bbStartMap.find(startAddr);
79  }
80 
81  if (mi != m_bbStartMap.end()) {
82  //
83  // Existing New +---+ "low" part of new
84  // +---+ +---+
85  // | | | Fall through
86  // +---+ | | ==> +---+
87  // | | | | | | Existing; rest of new discarded
88  // +---+ +---+ +---+
89  //
90  // Check for overlap of the just added BB with the next BB (address wise).
91  // If there is an overlap, truncate the RTLList for the new BB to not overlap,
92  // and make this a fall through BB.
93  // We still want to do this even if the new BB overlaps with an incomplete BB,
94  // though in this case, splitBB needs to fill in the details for the "high"
95  // BB of the split.
96  // Also, in this case, we return a pointer to the newly completed BB,
97  // so it will get out edges added (if required). In the other case
98  // (i.e. we overlap with an existing, completed BB), we want to return 0, since
99  // the out edges are already created.
100  //
101  mi = std::next(mi);
102 
103  if (mi != m_bbStartMap.end()) {
104  BasicBlock *nextBB = (*mi).second;
105  Address nextAddr = (*mi).first;
106  bool nextIsIncomplete = !nextBB->isComplete();
107 
108  if (nextAddr < currentBB->getHiAddr()) {
109  // Need to truncate the current BB. We use splitBB(), but pass it nextBB so it
110  // doesn't create a new BB for the "bottom" BB of the split pair
111  splitBB(currentBB, nextAddr, nextBB);
112 
113  // If the overlapped BB was incomplete, return the "bottom" part of the BB, so
114  // adding out edges will work properly.
115  if (nextIsIncomplete) {
116  assert(nextBB);
117  return nextBB;
118  }
119 
120  LOG_VERBOSE("Not creating a BB at address %1 because a BB already exists",
121  currentBB->getLowAddr());
122  return nullptr;
123  }
124  }
125 
126  // Existing New +---+ Top of existing
127  // +---+ +---+
128  // | | +---+ +---+ Fall through
129  // | | | | => | |
130  // | | | | | | New; rest of existing discarded
131  // +---+ +---+ +---+
132  //
133  // Note: no need to check the other way around, because in this case,
134  // we will have called ensureBBExists(), which will have split
135  // the existing BB already.
136  }
137 
138  assert(currentBB);
139  return currentBB;
140 }

158 bool LowLevelCFG::ensureBBExists(Address addr, BasicBlock *&currBB)
159 {
160  // check for overlapping incomplete or complete BBs.
161  BBStartMap::iterator itExistingBB = m_bbStartMap.lower_bound(addr);
162 
163  BasicBlock *overlappingBB = nullptr;
164  if (itExistingBB != m_bbStartMap.end() && itExistingBB->second->getLowAddr() == addr) {
165  overlappingBB = itExistingBB->second;
166  }
167  else if (itExistingBB != m_bbStartMap.begin()) {
168  --itExistingBB;
169  if (itExistingBB->second && itExistingBB->second->getLowAddr() <= addr &&
170  itExistingBB->second->getHiAddr() > addr) {
171  overlappingBB = itExistingBB->second;
172  }
173  }
174 
175  if (!overlappingBB) {
176  // no BB at addr -> create a new incomplete BB
177  createIncompleteBB(addr);
178  return false;
179  }
180  else if (!overlappingBB->isComplete()) {
181  return false;
182  }
183  else if (overlappingBB && overlappingBB->getLowAddr() < addr) {
184  splitBB(overlappingBB, addr);
185  BasicBlock *highBB = getBBStartingAt(addr);
186 
187  if (currBB == overlappingBB) {
188  // This means that the BB that we are expecting to use, usually to add
189  // out edges, has changed. We must change this pointer so that the right
190  // BB gets the out edges. However, if the new BB is not the BB of
191  // interest, we mustn't change currBB
192  currBB = highBB;
193  }
194  return true;
195  }
196  else {
197  // addr is the start of a complete BB
198  return true;
199  }
200 }

216 void LowLevelCFG::removeBB(BasicBlock *bb)
217 {
218  if (bb == nullptr) {
219  return;
220  }
221 
222  BBStartMap::iterator firstIt, lastIt;
223  std::tie(firstIt, lastIt) = m_bbStartMap.equal_range(bb->getLowAddr());
224 
225  for (auto it = firstIt; it != lastIt; ++it) {
226  if (it->second == bb) {
227  m_bbStartMap.erase(it);
228  delete bb;
229  return;
230  }
231  }
232 
233  LOG_WARN("Tried to remove BB at address %1; does not exist in CFG", bb->getLowAddr());
234  delete bb;
235 }

238 void LowLevelCFG::addEdge(BasicBlock *sourceBB, BasicBlock *destBB)
239 {
240  if (!sourceBB || !destBB) {
241  return;
242  }
243 
244  // Wire up edges
245  sourceBB->addSuccessor(destBB);
246  destBB->addPredecessor(sourceBB);
247 
248  // special handling for upgrading oneway BBs to twoway BBs
249  if ((sourceBB->getType() == BBType::Oneway) && (sourceBB->getNumSuccessors() > 1)) {
250  sourceBB->setType(BBType::Twoway);
251  }

255 void LowLevelCFG::addEdge(BasicBlock *sourceBB, Address addr)
256 {
257  // If we already have a BB for this address, add the edge to it.
258  // If not, create a new incomplete BB at the destination address.
259  BasicBlock *destBB = getBBStartingAt(addr);
260 
261  if (!destBB) {
262  destBB = createIncompleteBB(addr);
263  }
264 
265  this->addEdge(sourceBB, destBB);

269 bool LowLevelCFG::isWellFormed() const
270 {
271  for (const BasicBlock *bb : *this) {
272  if (!bb->isComplete()) {
273  LOG_ERROR("CFG is not well formed: BB at address %1 is incomplete", bb->getLowAddr());
274  return false;
275  }
276 
277  for (const BasicBlock *pred : bb->getPredecessors()) {
278  if (!pred->isPredecessorOf(bb)) {
279  LOG_ERROR("CFG is not well formed: Edge from BB at %1 to BB at %2 is malformed.",
280  pred->getLowAddr(), bb->getLowAddr());
281  return false;
282  }
283  else if (pred->getProc() != bb->getProc()) {
284  const UserProc *myFunc = bb->getProc();
285  const UserProc *predFunc = pred->getProc();
286  LOG_ERROR("CFG is not well formed: Interprocedural edge from '%1' to '%2' found",
287  myFunc ? myFunc->getName() : "<invalid>",
288  predFunc ? predFunc->getName() : "<invalid>");
289  return false;
290  }
291  }
292 
293  for (const BasicBlock *succ : bb->getSuccessors()) {
294  if (!succ->isSuccessorOf(bb)) {
295  LOG_ERROR("CFG is not well formed: Edge from BB at %1 to BB at %2 is malformed.",
296  bb->getLowAddr(), succ->getLowAddr());
297  return false;
298  }
299  else if (succ->getProc() != bb->getProc()) {
300  const UserProc *myFunc = bb->getProc();
301  const UserProc *succFunc = succ->getProc();
302  LOG_ERROR("CFG is not well formed: Interprocedural edge from '%1' to '%2' found",
303  myFunc ? myFunc->getName() : "<invalid>",
304  succFunc ? succFunc->getName() : "<invalid>");
305  return false;
306  }
307  }
308  }
309 
310  return true;
311 }

326 BasicBlock *LowLevelCFG::splitBB(BasicBlock *bb, Address splitAddr, BasicBlock *_newBB /* = 0 */)
327 {
328  std::vector<MachineInstruction>::iterator splitIt;
329 
330  // First find which RTL has the split address; note that this could fail
331  // (e.g. jump into the middle of an instruction)
332  for (splitIt = bb->getInsns().begin(); splitIt != bb->getInsns().end(); ++splitIt) {
333  if (splitIt->m_addr == splitAddr) {
334  break;
335  }
336  }
337 
338  if (splitIt == bb->getInsns().end()) {
339  LOG_WARN("Cannot split BB at address %1 at split address %2", bb->getLowAddr(), splitAddr);
340  return bb;
341  }
342 
343  if (_newBB && _newBB->isComplete()) {
344  // we already have a BB for the high part. Delete overlapping RTLs and adjust edges.
345 
346  while (splitIt != bb->getInsns().end()) {
347  splitIt = bb->getInsns().erase(splitIt); // deletes RTLs
348  }
349 
350  _newBB->removeAllPredecessors();
351  for (BasicBlock *succ : bb->getSuccessors()) {
352  succ->removePredecessor(bb);
353  }
354 
355  bb->removeAllSuccessors();
356  addEdge(bb, _newBB);
357  bb->setType(BBType::Fall);
358  insertBB(_newBB);
359  return _newBB;
360  }
361  else if (!_newBB) {
362  // create a new incomplete BasicBlock.
363  _newBB = createIncompleteBB(splitAddr);
364  }
365 
366  // Now we have an incomplete BB at splitAddr;
367  // just complete it with the "high" RTLs from the original BB.
368  // We don't want to "deep copy" the RTLs themselves,
369  // because we want to transfer ownership from the original BB to the "high" part
370  std::vector<MachineInstruction> highInsns(splitIt, bb->getInsns().end());
371  bb->getInsns().erase(splitIt, bb->getInsns().end());
372 
373  _newBB->completeBB(highInsns);
374 
375  assert(_newBB->getNumPredecessors() == 0);
376  assert(_newBB->getNumSuccessors() == 0);
377 
378  const std::vector<BasicBlock *> &successors = bb->getSuccessors();
379  for (BasicBlock *succ : successors) {
380  succ->removePredecessor(bb);
381  succ->addPredecessor(_newBB);
382  _newBB->addSuccessor(succ);
383  }
384 
385  bb->removeAllSuccessors();
386  addEdge(bb, _newBB);
387  _newBB->setType(bb->getType());
388  bb->setType(BBType::Fall);
389  return _newBB;
390 }

25 SharedExp ExpArithSimplifier::postModify(const std::shared_ptr<Binary> &exp)
26 {
27  if (exp->getOper() != opPlus && exp->getOper() != opMinus) {
28  return exp->shared_from_this();
29  }
30 
31  // Partition this expression into positive non-integer terms, negative
32  // non-integer terms and integer terms.
33  std::list<SharedExp> positives;
34  std::list<SharedExp> negatives;
35  std::vector<int> integers;
36  exp->partitionTerms(positives, negatives, integers, false);
37 
38  // Now reduce these lists by cancelling pairs
39  cancelDuplicates(positives, negatives);
40 
41  // Summarise the set of integers to a single number.
42  int sum = std::accumulate(integers.begin(), integers.end(), 0);
43 
44  // Now put all these elements back together and return the result
45  if (positives.empty()) {
46  if (negatives.empty()) {
47  return Const::get(sum);
48  }
49  else {
50  // No positives, some negatives. sum - Acc
51  return Binary::get(opMinus, Const::get(sum), Exp::accumulate(negatives));
52  }
53  }
54 
55  if (negatives.empty()) {
56  // Positives + sum
57  if (sum == 0) {
58  // Just positives
59  return Exp::accumulate(positives);
60  }
61  else {
62  OPER _op = opPlus;
63 
64  if (sum < 0) {
65  _op = opMinus;
66  sum = -sum;
67  }
68 
69  return Binary::get(_op, Exp::accumulate(positives), Const::get(Address(sum)));
70  }
71  }
72 
73  // Some positives, some negatives
74  if (sum == 0) {
75  // positives - negatives
76  return Binary::get(opMinus, Exp::accumulate(positives), Exp::accumulate(negatives));
77  }
78 
79  // General case: some positives, some negatives, a sum
80  OPER _op = opPlus;
81 
82  if (sum < 0) {
83  _op = opMinus; // Return (pos - negs) - sum
84  sum = -sum;
85  }
86 
87  return Binary::get(_op,
88  Binary::get(opMinus, Exp::accumulate(positives), Exp::accumulate(negatives)),
89  Const::get(sum));
90 }

92 void ExpArithSimplifier::cancelDuplicates(std::list<SharedExp> &left, std::list<SharedExp> &right)
93 {
94  // Note: can't improve this algorithm using multisets, since can't instantiate multisets of type
95  // Exp (only Exp*). The Exp* in the multisets would be sorted by address, not by value of the
96  // expression. So they would be unsorted, same as lists!
97  auto itLeft = left.begin();
98 
99  while (itLeft != left.end()) {
100  auto itRight = right.begin();
101  bool removedOne = false;
102 
103  while (itRight != right.end()) {
104  if (**itLeft == **itRight) {
105  removedOne = true;
106  itLeft = left.erase(itLeft);
107  itRight = right.erase(itRight);
108  break;
109  }
110  else {
111  itRight++;
112  }
113  }
114 
115  if (!removedOne) {
116  itLeft++;
117  }
118  }

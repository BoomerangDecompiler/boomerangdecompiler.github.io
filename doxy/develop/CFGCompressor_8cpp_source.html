23 /**
24  * Stack where each element is present at most once
25  */
26 template<typename T>
27 class UniqueStack
28 {
29 public:
30  bool empty() const { return m_stack.empty(); }
31 
32  T pop()
33  {
34  T val = m_stack.back();
35  m_stack.pop_back();
36  return val;
37  }
38 
39  // Push a new item onto the stack. If the item already exists, nothing happens
40  void push(T val)
41  {
42  if (std::find(m_stack.begin(), m_stack.end(), val) == m_stack.end()) {
43  m_stack.push_back(val);
44  }
45  }
46 
47  void erase(T val)
48  {
49  auto it = std::find(m_stack.begin(), m_stack.end(), val);
50  if (it != m_stack.end()) {
51  m_stack.erase(it);
52  }
53  }
54 
55 private:
56  std::deque<T> m_stack;
57 };

78 bool CFGCompressor::removeEmptyJumps(ProcCFG *cfg)
79 {
80  std::deque<IRFragment *> fragsToRemove;
81 
82  for (IRFragment *frag : *cfg) {
83  // Check if the fragment can be removed
84  if (frag->getNumSuccessors() == 1 && frag != cfg->getEntryFragment() &&
85  (frag->isEmpty() || frag->isEmptyJump())) {
86  fragsToRemove.push_back(frag);
87  }
88  }
89 
90  bool fragsRemoved = false;
91 
92  while (!fragsToRemove.empty()) {
93  IRFragment *frag = fragsToRemove.front();
94  fragsToRemove.pop_front();
95 
96  assert(frag->getNumSuccessors() == 1);
97  IRFragment *succ = frag->getSuccessor(0); // the one and only successor
98 
99  if (succ == frag) {
100  continue;
101  }
102 
103  succ->removePredecessor(frag);
104  frag->removeSuccessor(succ);
105 
106  for (IRFragment *pred : frag->getPredecessors()) {
107  for (int i = 0; i < pred->getNumSuccessors(); i++) {
108  if (pred->getSuccessor(i) == frag) {
109  pred->setSuccessor(i, succ);
110  succ->addPredecessor(pred);
111  }
112  }
113  }
114 
115  frag->removeAllPredecessors();
116  cfg->removeFragment(frag);
117  fragsRemoved = true;
118  }
119 
120  return fragsRemoved;
121 }

124 bool CFGCompressor::removeOrphanFragments(ProcCFG *cfg)
125 {
126  std::deque<IRFragment *> orphans;
127 
128  for (IRFragment *potentialOrphan : *cfg) {
129  if (potentialOrphan == cfg->getEntryFragment()) {
130  // don't remove entry fragment
131  continue;
132  }
133  else if (potentialOrphan->isType(FragType::Ret)) {
134  // Don't remove the ReturnStatement for noreturn functions
135  continue;
136  }
137 
138  if (potentialOrphan->getNumPredecessors() == 0) {
139  orphans.push_back(potentialOrphan);
140  }
141  }
142 
143  const bool fragsRemoved = !orphans.empty();
144 
145  while (!orphans.empty()) {
146  IRFragment *b = orphans.front();
147  orphans.pop_front();
148 
149  for (IRFragment *child : b->getSuccessors()) {
150  child->removePredecessor(b);
151  b->removeSuccessor(child);
152  }
153 
154  cfg->removeFragment(b);
155  }
156 
157  return fragsRemoved;
158 }

161 bool CFGCompressor::compressFallthroughs(ProcCFG *cfg)
162 {
163  std::unordered_set<IRFragment *> visited;
164  UniqueStack<IRFragment *> toVisit;
165 
166  IRFragment *entry = cfg->getEntryFragment();
167  if (!entry) {
168  return false;
169  }
170 
171  bool change = false;
172  toVisit.push(entry);
173 
174  while (!toVisit.empty()) {
175  IRFragment *current = toVisit.pop();
176 
177  if (visited.find(current) != visited.end()) {
178  continue;
179  }
180 
181  visited.insert(current);
182 
183  if (current->getNumSuccessors() != 1) {
184  for (IRFragment *succ : current->getSuccessors()) {
185  toVisit.push(succ);
186  }
187  continue;
188  }
189 
190  IRFragment *succ = current->getSuccessor(0);
191  if (succ->getNumPredecessors() != 1) {
192  toVisit.push(succ);
193  continue;
194  }
195  else if (!current->isEmpty() && !current->getLastStmt()->isAssignment()) {
196  toVisit.push(succ);
197  continue;
198  }
199 
200  SharedStmt succFirst = succ->getFirstStmt();
201  if (succFirst->isPhi() || succFirst->isImplicit()) {
202  toVisit.push(succ);
203  continue;
204  }
205  else if (succ->getBB() != current->getBB()) {
206  toVisit.push(succ);
207  continue;
208  }
209 
210  std::unique_ptr<RTLList> combined(new RTLList);
211  for (auto &rtl : *current->getRTLs()) {
212  combined->push_back(std::move(rtl));
213  }
214  current->getRTLs()->clear();
215 
216  for (auto &rtl : *succ->getRTLs()) {
217  combined->push_back(std::move(rtl));
218  }
219  succ->getRTLs()->clear();
220 
221  IRFragment *combinedFrag = cfg->createFragment(succ->getType(), std::move(combined),
222  succ->getBB());
223 
224  for (IRFragment *pred : current->getPredecessors()) {
225  for (int i = 0; i < pred->getNumSuccessors(); ++i) {
226  if (pred->getSuccessor(i) == current) {
227  pred->setSuccessor(i, combinedFrag);
228  combinedFrag->addPredecessor(pred);
229  }
230  }
231  }
232 
233  for (IRFragment *succ2 : succ->getSuccessors()) {
234  for (int i = 0; i < succ2->getNumPredecessors(); ++i) {
235  if (succ2->getPredecessor(i) == succ) {
236  succ2->setPredecessor(i, combinedFrag);
237  combinedFrag->addSuccessor(succ2);
238  }
239  }
240  }
241 
242  IRFragment::RTLIterator rit;
243  RTL::iterator sit;
244 
245  for (SharedStmt s = combinedFrag->getFirstStmt(rit, sit); s != nullptr;
246  s = combinedFrag->getNextStmt(rit, sit)) {
247  s->setFragment(combinedFrag);
248  }
249 
250  if (current == cfg->getEntryFragment()) {
251  cfg->setEntryAndExitFragment(combinedFrag);
252  }
253 
254  const bool exitFragNeedsUpdate = (succ == cfg->getExitFragment());
255 
256  visited.erase(current);
257  visited.erase(succ);
258  toVisit.erase(current);
259  toVisit.erase(succ);
260  toVisit.push(combinedFrag);
261 
262  cfg->removeFragment(current);
263  cfg->removeFragment(succ);
264 
265  if (exitFragNeedsUpdate) {
266  cfg->setEntryAndExitFragment(cfg->getEntryFragment());
267  }
268 
269  change = true;
270  }
271 
272  return change;
273 }

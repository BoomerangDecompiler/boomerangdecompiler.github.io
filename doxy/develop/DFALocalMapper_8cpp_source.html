29 bool DfaLocalMapper::processExp(const SharedExp &exp)
30 {
31  // Check if this is an appropriate pattern for local variables
32  if (m_proc->isLocalOrParamPattern(exp)) {
33  const int spIndex = Util::getStackRegisterIndex(m_prog);
34  if (m_sig->isStackLocal(spIndex, exp)) {
35  change = true; // We've made a mapping
36  // We have probably not even run TA yet, so doing a full descendtype here would be silly
37  // Note also that void is compatible with all types, so the symbol effectively covers
38  // all types
39  m_proc->getSymbolExp(exp, VoidType::get(), true);
40  }
41 
42  return false; // set recur false: Don't dig inside m[x] to make m[a[m[x]]] !
43  }
44 
45  return true;
46 }

40 {
41  const BinarySection *section = getSectionByAddr(addr);
42 
43  if (section == nullptr || section->getHostAddr() == HostAddress::INVALID) {
44  LOG_WARN("Invalid read at address %1: Address is not mapped to a section", addr);
45  return false;
46  }
47

190 bool BinaryImage::writeNative4(Address addr, uint32_t value)
191 {
192  BinarySection *si = getSectionByAddr(addr);
193 
194  if (si == nullptr || si->getHostAddr() == HostAddress::INVALID) {
195  LOG_WARN("Ignoring write at address %1: Address is outside any writable section");
196  return false;
197  }
198  else if (addr + 4 > si->getSourceAddr() + si->getSize()) {
199  LOG_WARN("Invalid write at address %1: Write extends past section boundary", addr);
200  return false;
201  }
202 
203  si->addDefinedArea(addr, addr + 4);
204 
205  HostAddress host = si->getHostAddr() - si->getSourceAddr() + addr;
206  Util::writeDWord(reinterpret_cast<void *>(host.value()), value, si->getEndian());
207  return true;
208 }

211 void BinaryImage::updateTextLimits()
212 {
213  m_limitTextLow = Address::INVALID;
214  m_limitTextHigh = Address::INVALID;
215  m_textDelta = 0;
216 
217  for (BinarySection *section : m_sections) {
218  if (!section->isCode()) {
219  continue;
220  }
221 
222  // The .plt section is an anomaly. It's code, but we never want to
223  // decode it. For now, we make this ugly exception
224  if (section->getName() == ".plt") {
225  continue;
226  }
227 
228  if (m_limitTextLow == Address::INVALID || section->getSourceAddr() < m_limitTextLow) {
229  m_limitTextLow = section->getSourceAddr();
230  }
231 
232  const Address highAddress = section->getSourceAddr() + section->getSize();
233 
234  if (m_limitTextHigh == Address::INVALID || highAddress > m_limitTextHigh) {
235  m_limitTextHigh = highAddress;
236  }
237 
238  const ptrdiff_t hostNativeDiff = (section->getHostAddr() - section->getSourceAddr())
239  .value();
240 
241  if (m_textDelta == 0) {
242  m_textDelta = hostNativeDiff;
243  }
244  else if (m_textDelta != hostNativeDiff) {
245  LOG_WARN("TextDelta different for section %1 (ignoring).", section->getName());
246  }
247  }
248 }

285 BinarySection *BinaryImage::createSection(const QString &name, Address from, Address to)
286 {
287  if (from == Address::INVALID || to == Address::INVALID || to < from) {
288  LOG_ERROR("Could not create section '%1' with invalid extent [%2, %3)", name, from, to);
289  return nullptr;
290  }
291  else if (getSectionByName(name) != nullptr) {
292  LOG_ERROR("Could not create section '%1': A section with the same name already exists",
293  name);
294  return nullptr;
295  }
296 
297  if (from == to) {
298  to += 1; // open interval, so -> [from,to+1) is right
299  }
300 
301 #if DEBUG
302  // see
303  // https://stackoverflow.com/questions/25501044/gcc-ld-overlapping-sections-tbss-init-array-in-statically-linked-elf-bin
304  // Basically, the .tbss section is of type SHT_NOBITS, so there is no data associated to the
305  // section. It can therefore overlap other sections containing data. This is a quirk of ELF
306  // programs linked statically with glibc
307  if (name != ".tbss") {
308  IntervalMap<Address, std::unique_ptr<BinarySection>>::iterator itFrom, itTo;
309  std::tie(itFrom, itTo) = m_sectionMap.equalRange(from, to);
310 
311  for (auto clash_with = itFrom; clash_with != itTo; ++clash_with) {
312  if ((*clash_with->second).getName() != ".tbss") {
313  LOG_WARN("Segment %1 would intersect existing segment %2", name,
314  (*clash_with->second).getName());
315  return nullptr;
316  }
317  }
318  }
319 #endif
320 
321  BinarySection *sect = new BinarySection(from, (to - from).value(), name);
322 
323  if (m_sectionMap.insert(from, to, std::unique_ptr<BinarySection>(sect)) == m_sectionMap.end()) {
324  // section already existed
325  BinarySection *existing = getSectionByAddr(from);
326  LOG_ERROR("Could not create section '%1' from address %2 to %3: Section extent matches "
327  "existing section '%4'",
328  name, from, to, existing ? existing->getName() : "<invalid>");
329  return nullptr;
330  }
331  else {
332  m_sections.push_back(sect);
333  return sect;
334  }
335 }

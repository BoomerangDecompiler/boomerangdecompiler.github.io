69 bool DefaultFrontEnd::disassembleEntryPoints()
70 {
71  BinaryImage *image = m_program->getBinaryFile()->getImage();
72  const Address lowAddr = image->getLimitTextLow();
73  const int numBytes = (image->getLimitTextHigh() - lowAddr).value();
74 
75  m_program->getProject()->alertStartDecode(lowAddr, numBytes);
76 
77  bool gotMain;
78  const Address mainAddr = findMainEntryPoint(gotMain);
79 
80  if (gotMain) {
81  LOG_MSG("Found main at address %1", mainAddr);
82  }
83  else {
84  LOG_WARN("Could not find main, falling back to entry point(s)");
85  }
86 
87  if (!gotMain) {
88  std::vector<Address> entrypoints = findEntryPoints();
89 
90  return std::all_of(entrypoints.begin(), entrypoints.end(),
91  [this](Address entry) { return disassembleFunctionAtAddr(entry); });
92  }
93 
94  if (!disassembleFunctionAtAddr(mainAddr)) {
95  return false;
96  }
97 
98  m_program->addEntryPoint(mainAddr);
99 
100  if (!gotMain) {
101  return true; // Decoded successfully, but patterns don't match a known main() pattern
102  }
103 
104  static const char *mainName[] = { "main", "WinMain", "DriverEntry" };
105  QString name = m_program->getSymbolNameByAddr(mainAddr);
106 
107  if (name == nullptr) {
108  name = mainName[0];
109  }
110 
111  for (auto &elem : mainName) {
112  if (name != elem) {
113  continue;
114  }
115 
116  Function *proc = m_program->getFunctionByAddr(mainAddr);
117 
118  if (proc == nullptr) {
119  LOG_WARN("No proc found for address %1", mainAddr);
120  return false;
121  }
122 
123  auto fty = std::dynamic_pointer_cast<FuncType>(Type::getNamedType(name));
124 
125  if (!fty) {
126  LOG_WARN("Unable to find signature for known entrypoint %1", name);
127  }
128  else {
129  proc->setSignature(fty->getSignature()->clone());
130  proc->getSignature()->setName(name);
131  proc->getSignature()->setForced(true); // Don't add or remove parameters
132  }
133 
134  break;
135  }
136 
137  return true;
138 }

141 bool DefaultFrontEnd::disassembleAll()
142 {
143  bool change = true;
144  LOG_MSG("Looking for functions to disassemble...");
145 
146  while (change) {
147  change = false;
148 
149  for (const auto &m : m_program->getModuleList()) {
150  for (Function *function : *m) {
151  if (function->isLib()) {
152  continue;
153  }
154 
155  UserProc *userProc = static_cast<UserProc *>(function);
156  if (userProc->isDecoded()) {
157  continue;
158  }
159 
160  // Not yet disassembled - do it now
161  if (!disassembleProc(userProc, userProc->getEntryAddress())) {
162  return false;
163  }
164 
165  userProc->setDecoded();
166  change = true;
167 
168  // Break out of the loops if not decoding children
169  if (!m_program->getProject()->getSettings()->decodeChildren) {
170  break;
171  }
172  }
173  }
174 
175  if (!m_program->getProject()->getSettings()->decodeChildren) {
176  break;
177  }
178  }
179 
180  return m_program->isWellFormed();
181 }

184 bool DefaultFrontEnd::disassembleFunctionAtAddr(Address addr)
185 {
186  assert(addr != Address::INVALID);
187 
188  Function *newProc = m_program->getOrCreateFunction(addr);
189 
190  // Sometimes, we have to adjust the entry address since
191  // the instruction at addr is just a jump to another address.
192  addr = newProc->getEntryAddress();
193  LOG_MSG("Starting disassembly at address %1", addr);
194  UserProc *proc = static_cast<UserProc *>(m_program->getFunctionByAddr(addr));
195 
196  if (proc == nullptr) {
197  LOG_MSG("No proc found at address %1", addr);
198  return false;
199  }
200  else if (proc->isLib()) {
201  LOG_MSG("NOT decoding library proc at address %1", addr);
202  return false;
203  }
204 
205  if (disassembleProc(proc, addr)) {
206  proc->setDecoded();
207  }
208 
209  return m_program->isWellFormed();
210 }

213 bool DefaultFrontEnd::disassembleProc(UserProc *proc, Address addr)
214 {
215  LOG_VERBOSE("### Disassembing proc '%1' at address %2 ###", proc->getName(), addr);
216 
217  LowLevelCFG *cfg = proc->getProg()->getCFG();
218  assert(cfg);
219 
220  m_targetQueue.initial(addr);
221 
222  int numBytesDecoded = 0;
223  Address startAddr = addr;
224  Address lastAddr = addr;
225  MachineInstruction insn;
226 
227  while ((addr = m_targetQueue.popAddress(*cfg)) != Address::INVALID) {
228  std::list<MachineInstruction> bbInsns;
229 
230  // Indicates whether or not the next instruction to be decoded is the lexical successor of
231  // the current one. Will be true for all NCTs and for CTIs with a fall through branch.
232  bool sequentialDecode = true;
233 
234  while (sequentialDecode) {
235  BasicBlock *existingBB = cfg->getBBStartingAt(addr);
236  if (existingBB) {
237  if (!bbInsns.empty()) {
238  // if bbInsns is not empty, the previous instruction was not a CTI.
239  // Complete the BB as a fallthrough
240  BasicBlock *newBB = cfg->createBB(BBType::Fall, bbInsns);
241  bbInsns.clear();
242  cfg->addEdge(newBB, existingBB);
243  }
244 
245  if (existingBB->isComplete()) {
246  break; // do not disassemble BB twice
247  }
248  }
249 
250  if (!disassembleInstruction(addr, insn)) {
251  // We might have disassembled a valid instruction, but the disassembler
252  // does not recognize it. Do not throw away previous instructions;
253  // instead, create a new BB from them
254  if (!bbInsns.empty()) {
255  cfg->createBB(BBType::Fall, bbInsns);
256  }
257 
258  LOG_ERROR("Encountered invalid instruction");
259  sequentialDecode = false;
260  break; // try next instruction in queue
261  }
262 
263  if (m_program->getProject()->getSettings()->traceDecoder) {
264  LOG_MSG("*%1 %2 %3", addr, insn.m_mnem.data(), insn.m_opstr.data());
265  }
266 
267  // alert the watchers that we have decoded an instruction
268  numBytesDecoded += insn.m_size;
269  m_program->getProject()->alertInstructionDecoded(addr, insn.m_size);
270 
271  // classify the current instruction. If it is not a CTI,
272  // continue disassembling sequentially
273  const bool isCTI = insn.isInGroup(MIGroup::Call) || insn.isInGroup(MIGroup::Jump) ||
274  insn.isInGroup(MIGroup::Ret);
275 
276  if (!isCTI) {
277  addr += insn.m_size;
278  bbInsns.push_back(insn);
279 
280  lastAddr = std::max(lastAddr, addr);
281  continue;
282  }
283 
284  // this is a CTI. Lift the instruction to gain access to call/jump semantics
285  LiftedInstruction lifted;
286  if (!liftInstruction(insn, lifted)) {
287  LOG_ERROR("Cannot lift instruction '%1 %2 %3'", insn.m_addr, insn.m_mnem.data(),
288  insn.m_opstr.data());
289 
290  // try next insruction in queue
291  sequentialDecode = false;
292  break;
293  }
294 
295  bbInsns.push_back(insn);
296  const RTL::StmtList &sl = lifted.getFirstRTL()->getStatements();
297 
298  for (auto ss = sl.begin(); ss != sl.end(); ++ss) {
299  SharedStmt s = *ss;
300  s->setProc(proc); // let's do this really early!
301 
302  if (m_refHints.find(lifted.getFirstRTL()->getAddress()) != m_refHints.end()) {
303  const QString &name(m_refHints[lifted.getFirstRTL()->getAddress()]);
304  Address globAddr = m_program->getGlobalAddrByName(name);
305 
306  if (globAddr != Address::INVALID) {
307  s->searchAndReplace(Const(globAddr),
308  Unary::get(opAddrOf, Location::global(name, proc)));
309  }
310  }
311 
312  s->simplify();
313  assert(s->isAssignment() || (std::next(ss) == sl.end()));
314  }
315 
316  if (sl.empty()) {
317  addr += insn.m_size;
318  lastAddr = std::max(lastAddr, addr);
319  sequentialDecode = true;
320  continue;
321  }
322 
323  SharedStmt s = sl.back();
324 
325  switch (s->getKind()) {
326  case StmtType::Goto: {
327  std::shared_ptr<GotoStatement> jump = s->as<GotoStatement>();
328  assert(jump != nullptr);
329  const Address jumpDest = jump->getFixedDest();
330  sequentialDecode = false;
331 
332  // computed unconditional jumps have CaseStatement as last statement
333  // and not Goto
334  if (jumpDest == Address::INVALID) {
335  break;
336  }
337 
338  // Static unconditional jump
339  BasicBlock *currentBB = cfg->createBB(BBType::Oneway, bbInsns);
340 
341  // Exit the switch now if the basic block already existed
342  if (currentBB == nullptr) {
343  break;
344  }
345 
346  // Check if this is a jump to an already existing function. If so, this is
347  // actually a call that immediately returns afterwards
348  Function *destProc = m_program->getFunctionByAddr(jumpDest);
349  if (destProc && destProc != reinterpret_cast<Function *>(-1)) {
350  sequentialDecode = false;
351  break;
352  }
353 
354  BinarySymbol *sym = m_binaryFile->getSymbols()->findSymbolByAddress(jumpDest);
355  if (sym && sym->isFunction()) {
356  sequentialDecode = false;
357  break;
358  }
359  else if (jumpDest < m_program->getBinaryFile()->getImage()->getLimitTextHigh()) {
360  // Add the out edge if it is to a destination within the procedure
361  m_targetQueue.pushAddress(cfg, jumpDest, currentBB);
362  cfg->addEdge(currentBB, jumpDest);
363  }
364  else {
365  LOG_WARN("Goto instruction at address %1 branches beyond end of "
366  "section, to %2",
367  addr, jumpDest);
368  }
369  } break;
370 
371  case StmtType::Case: {
372  // We create the BB as a COMPJUMP type, then change to an NWAY if it turns out
373  // to be a switch stmt
374  cfg->createBB(BBType::CompJump, bbInsns);
375  sequentialDecode = false;
376  } break;
377 
378  case StmtType::Branch: {
379  std::shared_ptr<GotoStatement> jump = s->as<GotoStatement>();
380  BasicBlock *currentBB = cfg->createBB(BBType::Twoway, bbInsns);
381 
382  // Stop decoding sequentially if the basic block already existed otherwise
383  // complete the basic block
384  if (currentBB == nullptr) {
385  sequentialDecode = false;
386  break;
387  }
388 
389  // Add the out edge if it is to a destination within the section
390  const Address jumpDest = jump->getFixedDest();
391 
392  if (jumpDest < m_program->getBinaryFile()->getImage()->getLimitTextHigh()) {
393  m_targetQueue.pushAddress(cfg, jumpDest, currentBB);
394  cfg->addEdge(currentBB, jumpDest);
395  }
396  else {
397  LOG_WARN("Branch instruction at address %1 branches beyond end of "
398  "section, to %2",
399  addr, jumpDest);
400  currentBB->setType(BBType::Oneway);
401  }
402 
403  // Add the fall-through outedge
404  cfg->addEdge(currentBB, addr + insn.m_size);
405  } break;
406 
407  case StmtType::Call: {
408  std::shared_ptr<CallStatement> call = s->as<CallStatement>();
409 
410  // Check for a dynamic linked library function
411  if (refersToImportedFunction(call->getDest())) {
412  // Dynamic linked proc pointers are treated as static.
413  const Address linkedAddr = call->getDest()->access<Const, 1>()->getAddr();
414  const QString name = m_program->getBinaryFile()
415  ->getSymbols()
416  ->findSymbolByAddress(linkedAddr)
417  ->getName();
418 
419  Function *function = proc->getProg()->getOrCreateLibraryProc(name);
420  call->setDestProc(function);
421  call->setIsComputed(false);
422 
423  if (function->isNoReturn() || isNoReturnCallDest(function->getName())) {
424  sequentialDecode = false;
425  }
426  }
427 
428  const Address functionAddr = getAddrOfLibraryThunk(call, proc);
429  if (functionAddr != Address::INVALID) {
430  // Yes, it's a library function. Look up its name.
431  QString name = m_program->getBinaryFile()
432  ->getSymbols()
433  ->findSymbolByAddress(functionAddr)
434  ->getName();
435 
436  // Assign the proc to the call
437  Function *p = m_program->getOrCreateLibraryProc(name);
438 
439  if (call->getDestProc()) {
440  // prevent unnecessary __imp procs
441  m_program->removeFunction(call->getDestProc()->getName());
442  }
443 
444  call->setDestProc(p);
445  call->setIsComputed(false);
446  call->setDest(Location::memOf(Const::get(functionAddr)));
447 
448  if (p->isNoReturn() || isNoReturnCallDest(p->getName())) {
449  sequentialDecode = false;
450  }
451  }
452 
453  // Treat computed and static calls separately
454  if (call->isComputed()) {
455  BasicBlock *currentBB = cfg->createBB(BBType::CompCall, bbInsns);
456 
457  // Stop decoding sequentially if the basic block already
458  // existed otherwise complete the basic block
459  if (currentBB == nullptr) {
460  sequentialDecode = false;
461  }
462  else {
463  cfg->addEdge(currentBB, addr + insn.m_size);
464  bbInsns.clear(); // start a new BB
465  sequentialDecode = true;
466  }
467  }
468  else {
469  // Static call
470  const Address callAddr = call->getFixedDest();
471 
472  // Calls with 0 offset (i.e. call the next instruction) are simply
473  // pushing the PC to the stack. Treat these as non-control flow
474  // instructions and continue.
475  if (callAddr == addr + insn.m_size) {
476  break;
477  }
478 
479  // Record the called address as the start of a new procedure if it
480  // didn't already exist.
481  if (!callAddr.isZero() && (callAddr != Address::INVALID) &&
482  (m_program->getFunctionByAddr(callAddr) == nullptr)) {
483  if (m_program->getProject()->getSettings()->traceDecoder) {
484  LOG_MSG("p%1", callAddr);
485  }
486  }
487 
488  // Check if this is the _exit or exit function. May prevent us from
489  // attempting to decode invalid instructions, and getting invalid stack
490  // height errors
491  QString procName = m_program->getSymbolNameByAddr(callAddr);
492 
493  if (procName.isEmpty() && refersToImportedFunction(call->getDest())) {
494  Address a = call->getDest()->access<Const, 1>()->getAddr();
495  procName = m_program->getBinaryFile()
496  ->getSymbols()
497  ->findSymbolByAddress(a)
498  ->getName();
499  }
500 
501  if (!procName.isEmpty() && isNoReturnCallDest(procName)) {
502  // Make sure it has a return appended (so there is only one exit
503  // from the function)
504  cfg->createBB(BBType::Call, bbInsns);
505  sequentialDecode = false;
506  }
507  else {
508  // Create the new basic block
509  BasicBlock *currentBB = cfg->createBB(BBType::Call, bbInsns);
510 
511  // Add the fall through edge if the block didn't
512  // already exist
513  if (currentBB != nullptr) {
514  cfg->addEdge(currentBB, addr + insn.m_size);
515  }
516 
517  // start a new bb
518  bbInsns.clear();
519  sequentialDecode = true;
520  }
521  }
522  } break;
523 
524  case StmtType::Ret: {
525  cfg->createBB(BBType::Ret, bbInsns);
526  sequentialDecode = false;
527  } break;
528 
529  case StmtType::BoolAssign:
530  // This is just an ordinary instruction; no control transfer
531  // Fall through
532  // FIXME: Do we need to do anything here?
533  case StmtType::Assign:
534  case StmtType::PhiAssign:
535  case StmtType::ImpAssign:
536  // Do nothing
537  break;
538  case StmtType::INVALID: assert(false); break;
539  }
540 
541  addr += insn.m_size;
542  lastAddr = std::max(lastAddr, addr);
543  } // while sequentialDecode
544  } // while getNextAddress() != Address::INVALID
545 
546  tagFunctionBBs(proc);
547  proc->setStatus(ProcStatus::Decoded);
548  m_program->getProject()->alertFunctionDecoded(proc, startAddr, lastAddr, numBytesDecoded);
549 
550  LOG_VERBOSE("### Finished disassembling proc '%1' ###", proc->getName());
551  return true;
552 }

567 bool DefaultFrontEnd::liftProcImpl(UserProc *proc)
568 {
569  std::list<std::shared_ptr<CallStatement>> callList;
570 
571  LowLevelCFG *cfg = proc->getProg()->getCFG();
572  ProcCFG *procCFG = proc->getCFG();
573 
574  for (BasicBlock *bb : *cfg) {
575  liftBB(bb, proc, callList);
576  }
577 
578  for (const std::shared_ptr<CallStatement> &callStmt : callList) {
579  // Don't visit the destination of a register call
580  const Address dest = callStmt->getFixedDest();
581  Function *np = callStmt->getDestProc();
582 
583  if ((np == nullptr) && (dest != Address::INVALID)) {
584  np = proc->getProg()->getOrCreateFunction(dest);
585  }
586 
587  if (np != nullptr) {
588  proc->addCallee(np);
589  }
590  }
591 
592  // add edges for fragments
593  while (!m_needSuccessors.empty()) {
594  IRFragment *frag = m_needSuccessors.front();
595  m_needSuccessors.pop_front();
596 
597  const BasicBlock *bb = frag->getBB();
598 
599  // clang-format off
600  auto it = std::find_if(bb->getInsns().begin(), bb->getInsns().end(),
601  [frag, this](const MachineInstruction &insn) {
602  return m_lastFragment[&insn] == frag;
603  });
604  // clang-format on
605 
606  assert(it != bb->getInsns().end());
607  std::advance(it, 1);
608 
609  if (it == bb->getInsns().end()) {
610  for (BasicBlock *succ : bb->getSuccessors()) {
611  IRFragment *succFrag = m_firstFragment[&succ->getInsns().front()];
612  procCFG->addEdge(frag, succFrag);
613  }
614  }
615  else {
616  // this is within a BB
617  IRFragment *succFrag = m_firstFragment[&(*it)];
618  procCFG->addEdge(frag, succFrag);
619  }
620  }
621 
622  procCFG->setEntryAndExitFragment(procCFG->getFragmentByAddr(proc->getEntryAddress()));
623 
624  IRFragment::RTLIterator rit;
625  StatementList::iterator sit;
626 
627  for (IRFragment *frag : *procCFG) {
628  for (SharedStmt stmt = frag->getFirstStmt(rit, sit); stmt != nullptr;
629  stmt = frag->getNextStmt(rit, sit)) {
630  assert(stmt->getProc() == nullptr || stmt->getProc() == proc);
631  stmt->setProc(proc);
632  stmt->setFragment(frag);
633  }
634  }
635 
636  return true;
637 }

647 std::vector<Address> DefaultFrontEnd::findEntryPoints()
648 {
649  std::vector<Address> entrypoints;
650 
651  bool gotMain;
652  const Address a = findMainEntryPoint(gotMain);
653 
654  // TODO: find exported functions and add them too ?
655  if (gotMain) {
656  return { a };
657  }
658 
659  // try some other tricks
660  const QString fname; // = m_program->getProject()->getSettings()->getFilename();
661  const BinarySymbolTable *syms = m_program->getBinaryFile()->getSymbols();
662 
663  // X11 Module
664  if (fname.endsWith("_drv.o")) {
665  const int seploc = fname.lastIndexOf(QDir::separator());
666  const QString p = fname.mid(seploc + 1); // part after the last path separator
667 
668  if (p == fname) {
669  return {};
670  }
671 
672  const QString name = p.mid(0, p.length() - 6) + "ModuleData";
673  const BinarySymbol *sym = syms->findSymbolByName(name);
674  if (!sym) {
675  return {};
676  }
677 
678  const Address tmpaddr = sym->getLocation();
679 
680  const BinaryImage *image = m_program->getBinaryFile()->getImage();
681  DWord vers = 0, setup = 0, teardown = 0;
682  bool ok = true;
683  ok &= image->readNative4(tmpaddr, vers);
684  ok &= image->readNative4(tmpaddr, setup);
685  ok &= image->readNative4(tmpaddr, teardown);
686 
687  if (!ok) {
688  return {};
689  }
690 
691  // TODO: find use for vers ?
692  if (setup != 0 && createFunctionForEntryPoint(Address(setup), "ModuleSetupProc")) {
693  entrypoints.push_back(Address(setup));
694  }
695 
696  if (teardown != 0 && createFunctionForEntryPoint(Address(teardown), "ModuleTearDownProc")) {
697  entrypoints.push_back(Address(teardown));
698  }
699  }
700  // Linux kernel module
701  else if (fname.endsWith(".ko")) {
702  const BinarySymbol *sym = syms->findSymbolByName("init_module");
703  if (sym) {
704  entrypoints.push_back(sym->getLocation());
705  }
706 
707  sym = syms->findSymbolByName("cleanup_module");
708  if (sym) {
709  entrypoints.push_back(sym->getLocation());
710  }
711  }
712 
713  return entrypoints;
714 }

726 void DefaultFrontEnd::addRefHint(Address addr, const QString &name)

791  LiftedInstruction lifted;
792  if (!m_decoder->liftInstruction(insn, lifted)) {
793  LOG_ERROR("Cannot lift instruction '%1 %2 %3'", insn.m_addr, insn.m_mnem.data(),
794  insn.m_opstr.data());
795  return false;
796  }
797

799  // this is bsf/bsr/rep* etc.
800  std::list<LiftedInstructionPart> parts = lifted.use();
801  std::list<IRFragment *> frags;
802

804  std::unique_ptr<RTLList> bbRTLs(new RTLList);
805  bbRTLs->push_back(std::move(std::next(parts.begin(), i)->m_rtl));
806 
807  FragType fragType = FragType::Fall;
808  if (!bbRTLs->back()->empty()) {
809  switch (bbRTLs->back()->back()->getKind()) {
810  case StmtType::Goto: fragType = FragType::Oneway; break;
811  case StmtType::Branch: fragType = FragType::Twoway; break;
812  default: fragType = FragType::Fall; break;
813  }
814  }
815 
816  IRFragment *frag = procCFG->createFragment(fragType, std::move(bbRTLs), currentBB);
817  frags.push_back(frag);
818  }
819 
820  assert(parts.size() == frags.size());

823  LiftedInstructionPart &part = *std::next(parts.begin(), i);
824  IRFragment *frag = *std::next(frags.begin(), i);

827  const std::size_t idx = std::distance(

837  m_lastFragment[&insn] = frags.back();

966  ->getName();

990  m_needSuccessors.push_back(callFrag);

993 
994  // Add this call to the list of calls to analyse. We won't
995  // be able to analyse it's callee(s), of course.
996  callList.push_back(call);
997  }
998  else {
999  // Static call
1000  const Address callAddr = call->getFixedDest();
1001 
1002  // Call the virtual helper function. If implemented, will check for
1003  // machine specific funcion calls
1004  std::unique_ptr<RTLList> bbRTLs(new RTLList);
1005  if (isHelperFunc(callAddr, insn.m_addr, *bbRTLs)) {
1006  IRFragment *frag = procCFG->createFragment(FragType::Call, std::move(bbRTLs),
1007  currentBB);
1008  m_firstFragment[&insn] = frag;
1009  m_lastFragment[&insn] = frag;
1010  m_needSuccessors.push_back(frag);
1011 
1012  rtl.reset(); // Discard the call semantics
1013  break;
1014  }
1015 
1016  bbRTLs->push_back(std::move(rtl));
1017 
1018  // Add this non computed call site to the set of call sites which need
1019  // to be analysed later.
1020  callList.push_back(call);
1021

1023  // didn't already exist.
1024  if (!callAddr.isZero() && (callAddr != Address::INVALID) &&

1026  callList.push_back(call);

1029  LOG_MSG("p%1", callAddr);
1030  }
1031  }

1034  // attempting to decode invalid instructions, and getting invalid stack
1035  // height errors

1037 
1038  if (procName.isEmpty() && refersToImportedFunction(call->getDest())) {
1039  Address a = call->getDest()->access<Const, 1>()->getAddr();
1040  procName = m_program->getBinaryFile()
1041  ->getSymbols()
1042  ->findSymbolByAddress(a)
1043  ->getName();
1044  }
1045

1047  currentBB);
1048 
1049  if (!procName.isEmpty() && isNoReturnCallDest(procName)) {
1050  // Make sure it has a return appended (so there is only one exit
1051  // from the function)
1052  appendSyntheticReturn(callFrag);
1053  extraProcessCall(callFrag);
1054  }
1055  else {
1056  if (call->isReturnAfterCall()) {
1057  appendSyntheticReturn(callFrag);
1058  }
1059  else {
1060  m_needSuccessors.push_back(callFrag);
1061  }
1062 
1063  extraProcessCall(callFrag);
1064  }
1065 
1066  m_firstFragment[&insn] = callFrag;
1067  m_lastFragment[&insn] = callFrag;
1068  }
1069  } break;
1070 
1071  case StmtType::Ret: {
1072  // Create the list of RTLs for the next basic block and
1073  // continue with the next instruction.
1074  std::unique_ptr<RTLList> bbRTLs(new RTLList);
1075  bbRTLs->push_back(std::move(rtl));
1076  createReturnBlock(std::move(bbRTLs), currentBB);
1077  } break;
1078 
1079  case StmtType::Goto:
1080  case StmtType::Branch:
1081  case StmtType::Assign:
1082  case StmtType::BoolAssign: break;
1083  case StmtType::INVALID:
1084  default: assert(false); break;
1085  }
1086 
1087  if (rtl) {
1088  // we have not put the RTL into a fragment as yet
1089  std::unique_ptr<RTLList> bbRTLs(new RTLList);
1090  bbRTLs->push_back(std::move(rtl));
1091 
1092  FragType fragType;
1093  switch (bbRTLs->back()->back()->getKind()) {
1094  case StmtType::Goto: fragType = FragType::Oneway; break;
1095  case StmtType::Branch: fragType = FragType::Twoway; break;
1096  default: fragType = FragType::Fall; break;
1097  }
1098 
1099  IRFragment *frag = procCFG->createFragment(fragType, std::move(bbRTLs), currentBB);
1100 
1101  m_firstFragment[&insn] = frag;
1102  m_lastFragment[&insn] = frag;
1103  m_needSuccessors.push_back(frag);
1104  }
1105  }
1106 
1107  return true;
1108 }

1111 IRFragment *DefaultFrontEnd::createReturnBlock(std::unique_ptr<RTLList> newRTLs, BasicBlock *retBB)
1112 {
1113  UserProc *proc = retBB->getProc();
1114  ProcCFG *cfg = proc->getCFG();
1115 
1116  RTL *retRTL = newRTLs->back().get();
1117  Address retAddr = proc->getRetAddr();
1118  IRFragment *newFrag = nullptr;
1119 
1120  if (retAddr == Address::INVALID) {
1121  // Create the one and only return statement
1122  newFrag = cfg->createFragment(FragType::Ret, std::move(newRTLs), retBB);
1123  if (newFrag) {
1124  SharedStmt s = retRTL->back(); // The last statement should be the ReturnStatement
1125  proc->setRetStmt(s->as<ReturnStatement>(), retRTL->getAddress());
1126  }
1127 
1128  m_firstFragment[&retBB->getInsns().back()] = newFrag;
1129  m_lastFragment[&retBB->getInsns().back()] = newFrag;
1130  }
1131  else {
1132  // We want to replace the *whole* RTL with a branch to THE first return's RTL. There can
1133  // sometimes be extra semantics associated with a return (e.g. x86 ret instruction adds to
1134  // the stack pointer before setting %pc and branching). Other semantics are assumed
1135  // to appear in a previous RTL. It is assumed that THE return statement will have
1136  // the same semantics (NOTE: may not always be valid). To avoid this assumption,
1137  // we need branches to statements, not just to native addresses (RTLs).
1138  IRFragment *origRetFrag = proc->getCFG()->findRetFragment();
1139  assert(origRetFrag);
1140 
1141  if (origRetFrag->getFirstStmt()->isReturn()) {
1142  // ret node has no semantics, clearly we need to keep ours
1143  assert(!retRTL->empty());
1144  retRTL->pop_back();
1145  }
1146  else {
1147  retRTL->clear();
1148  }
1149 
1150  retRTL->append(std::make_shared<GotoStatement>(retAddr));
1151  newFrag = cfg->createFragment(FragType::Oneway, std::move(newRTLs), retBB);
1152 
1153  if (newFrag) {
1154  // make sure the return fragment only consists of a single RTL
1155  origRetFrag = cfg->splitFragment(origRetFrag, retAddr);
1156  cfg->addEdge(newFrag, origRetFrag);
1157 
1158  m_firstFragment[&retBB->getInsns().back()] = newFrag;
1159  m_lastFragment[&retBB->getInsns().back()] = newFrag;
1160  }
1161  }
1162 
1163  return newFrag;
1164 }

93 bool ST20Decoder::disassembleInstruction(Address pc, ptrdiff_t delta, MachineInstruction &result)
94 {
95  bool valid = false; //< Is this a valid instruction?
96  int total = 0; // Total value from all prefixes
97  result.m_size = 0;
98 
99  while (true) {
100  const Byte instructionData = Util::readByte(
101  (const void *)(pc.value() + delta + result.m_size));
102  const Byte functionCode = (instructionData >> 4) & 0xF;
103  const Byte oper = instructionData & 0xF;
104 
105  result.m_size++;
106 
107  switch (functionCode) {
108  case ST20_FUNC_J: { // unconditional jump
109  total += oper;
110  const Address jumpDest = pc + result.m_size + total;
111 
112  result.m_addr = pc;
113  result.m_id = ST20_FUNC_J;
114 
115  std::strcpy(result.m_mnem.data(), "j");
116  std::snprintf(result.m_opstr.data(), result.m_opstr.size(), "%s",
117  qPrintable(jumpDest.toString()));
118  result.m_operands.push_back(Const::get(jumpDest));
119  result.m_templateName = "J";
120 
121  valid = true;
122  } break;
123 
124  case ST20_FUNC_LDLP:
125  case ST20_FUNC_LDNL:
126  case ST20_FUNC_LDC:
127  case ST20_FUNC_LDNLP:
128  case ST20_FUNC_LDL:
129  case ST20_FUNC_ADC:
130  case ST20_FUNC_AJW:
131  case ST20_FUNC_EQC:
132  case ST20_FUNC_STL:
133  case ST20_FUNC_STNL: {
134  total += oper;
135 
136  result.m_addr = pc;
137  result.m_id = functionCode;
138 
139  std::strcpy(result.m_mnem.data(), functionNames[functionCode]);
140  std::snprintf(result.m_opstr.data(), result.m_opstr.size(), "0x%x", total);
141 
142  result.m_operands.push_back(Const::get(total));
143  result.m_templateName = QString(functionNames[functionCode]).toUpper();
144 
145  valid = true;
146  } break;
147 
148  case ST20_FUNC_PFIX: { // prefix
149  total = (total + oper) << 4;
150  continue;
151  }
152  case ST20_FUNC_NFIX: { // negative prefix
153  total = (total + ~oper) << 4;
154  continue;
155  }
156 
157  case ST20_FUNC_CALL: { // call
158  total += oper;
159  const Address callDest = Address(pc + result.m_size + total);
160 
161  result.m_addr = pc;
162  result.m_id = ST20_FUNC_CALL;
163 
164  std::strcpy(result.m_mnem.data(), "call");
165  std::snprintf(result.m_opstr.data(), result.m_opstr.size(), "%s",
166  qPrintable(callDest.toString()));
167 
168  result.m_operands.push_back(Const::get(callDest));
169  result.m_templateName = "CALL";
170 
171  valid = true;
172  } break;
173 
174  case ST20_FUNC_CJ: { // cond jump
175  total += oper;
176  const Address jumpDest = pc + result.m_size + total;
177 
178  result.m_addr = pc;
179  result.m_id = ST20_FUNC_CJ;
180 
181  std::strcpy(result.m_mnem.data(), "cj");
182  std::snprintf(result.m_opstr.data(), result.m_opstr.size(), "%s",
183  qPrintable(jumpDest.toString()));
184 
185  result.m_operands.push_back(Const::get(jumpDest));
186  result.m_templateName = "CJ";
187 
188  valid = true;
189  } break;
190 
191  case ST20_FUNC_OPR: { // operate
192  total += oper;
193  const char *insnName = getInstructionName(total);
194  if (!insnName) {
195  // invalid or unknown instruction
196  return false;
197  }
198 
199  result.m_addr = pc;
200  result.m_id = OPR_MASK |
201  (total > 0 ? total : ((~total & ~0xF) | (total & 0xF) | OPR_SIGN));
202 
203  std::strcpy(result.m_mnem.data(), insnName);
204  std::strcpy(result.m_opstr.data(), "");
205  result.m_templateName = QString(insnName).toUpper();
206 
207  valid = true;
208  } break;
209 
210  default: return false;
211  }
212 
213  break;
214  }
215 
216  return valid;
217 }

406 std::unique_ptr<RTL> ST20Decoder::instantiateRTL(const MachineInstruction &insn)
407 {
408  // Take the argument, convert it to upper case and remove any .'s
409  const QString sanitizedName = QString(insn.m_templateName).remove(".").toUpper();
410 
411  // Display a disassembly of this instruction if requested
412  if (m_prog && m_prog->getProject()->getSettings()->debugDecoder) {
413  QString msg{ insn.m_addr.toString() + " " + insn.m_templateName + " " };
414 
415  for (const SharedExp &itd : insn.m_operands) {
416  if (itd->isIntConst()) {
417  const int val = itd->access<Const>()->getInt();
418 
419  if ((val > 100) || (val < -100)) {
420  msg += "0x" + QString::number(val, 16);
421  }
422  else {
423  msg += QString::number(val);
424  }
425  }
426  else {
427  msg += itd->toString();
428  }
429 
430  msg += " ";
431  }
432 
433  LOG_MSG("%1", msg);
434  }
435 
436  return m_rtlDict.instantiateRTL(sanitizedName, insn.m_addr, insn.m_operands);
437 }

81 bool ReturnStatement::definesLoc(SharedExp loc) const
82 {
83  /// Does a ReturnStatement define anything?
84  /// Not really, the locations are already defined earlier in the procedure.
85  /// However, nothing comes after the return statement, so it doesn't hurt
86  /// to pretend it does, and this is a place to store the return type(s) for example.

387 void ReturnStatement::updateModifieds()
388 {
389  assert(m_proc != nullptr);
390  assert(m_fragment != nullptr);
391 
392  StatementList oldMods(m_modifieds); // Copy the old modifieds
393  m_modifieds.clear();
394 
395  if ((m_fragment->getNumPredecessors() == 1) &&
396  m_fragment->getPredecessor(0)->getLastStmt()->isCall()) {
397  std::shared_ptr<CallStatement>
398  call = m_fragment->getPredecessor(0)->getLastStmt()->as<CallStatement>();
399 
400  IFrontEnd *fe = m_proc->getProg()->getFrontEnd();
401  if (call->getDestProc() && fe->isNoReturnCallDest(call->getDestProc()->getName())) {
402  return;
403  }
404  }
405 
406  // For each location in the collector, make sure that there is an assignment in the old
407  // modifieds, which will be filtered and sorted to become the new modifieds Ick... O(N*M)
408  // (N existing modifeds, M collected locations)
409 
410  for (SharedStmt stmt : m_col) {
411  bool found = false;
412  std::shared_ptr<Assign> asgn = stmt->as<Assign>();
413  SharedExp colLhs = asgn->getLeft();
414 
415  if (!m_proc->canBeReturn(colLhs)) {
416  continue; // Filtered out
417  }
418 
419  for (SharedStmt s : oldMods) {
420  SharedExp lhs = s->as<Assignment>()->getLeft();
421 
422  if (*lhs == *colLhs) {
423  found = true;
424  break;
425  }
426  }
427 
428  if (!found) {
429  std::shared_ptr<ImplicitAssign> ias(
430  new ImplicitAssign(asgn->getType()->clone(), asgn->getLeft()->clone()));
431  ias->setProc(m_proc); // Comes from the Collector
432  ias->setFragment(m_fragment);
433  oldMods.append(ias);
434  }
435  }
436 
437  // Mostly the old modifications will be in the correct order,
438  // and inserting will be fastest near the start of the
439  // new list. So read the old modifications in reverse order
440  for (auto rit = oldMods.rbegin(); rit != oldMods.rend(); ++rit) {
441  // Make sure the LHS is still in the collector
442  std::shared_ptr<Assignment> asgn = (*rit)->as<Assignment>();
443  SharedExp lhs = asgn->getLeft();
444 
445  if (!m_col.hasDefOf(lhs)) {
446  continue; // Not in collector: delete it (don't copy it)
447  }
448 
449  // can happen e.g. when we can prove a function preserves an exp
450  // when we could not do this before, e.g. after propagation
451  if (!m_proc->canBeReturn(lhs)) {
452  continue;
453  }
454 
455  m_modifieds.append(asgn);
456  }
457 
458  auto sig = m_proc->getSignature();
459 
460  m_modifieds.sort([&sig](const SharedConstStmt &mod1, const SharedConstStmt &mod2) {
461  return sig->returnCompare(*mod1->as<const Assignment>(), *mod2->as<const Assignment>());
462  });
463 }

466 void ReturnStatement::updateReturns()
467 {
468  assert(m_proc != nullptr);
469 
470  auto sig = m_proc->getSignature();
471  const int sp = sig->getStackRegister();
472 
473  StatementList oldRets(m_returns); // Copy the old returns
474  m_returns.clear();
475 
476  // For each location in the modifieds, make sure that there is an assignment in the old returns,
477  // which will be filtered and sorted to become the new returns
478  // Ick... O(N*M) (N existing returns, M modifieds locations)
479  for (SharedStmt stmt : m_modifieds) {
480  bool found = false;
481  SharedExp loc = stmt->as<Assignment>()->getLeft();
482 
483  if (!m_proc->canBeReturn(loc)) {
484  continue; // Filtered out
485  }
486 
487  // Special case for the stack pointer: it has to be a modified (otherwise, the changes will
488  // bypass the calls), but it is not wanted as a return
489  if (loc->isRegN(sp)) {
490  continue;
491  }
492 
493  for (SharedStmt oldRet : oldRets) {
494  SharedExp lhs = oldRet->as<Assign>()->getLeft();
495 
496  if (*lhs == *loc) {
497  found = true;
498  break;
499  }
500  }
501 
502  if (!found) {
503  // Find the definition that reaches the return statement's collector
504  SharedExp rhs = m_col.findDefFor(loc);
505  std::shared_ptr<Assign> asgn(new Assign(loc->clone(), rhs->clone()));
506  asgn->setProc(m_proc);
507  asgn->setFragment(m_fragment);
508  oldRets.append(asgn);
509  }
510  }
511 
512  for (SharedStmt stmt : oldRets) {
513  // Make sure the LHS is still in the modifieds
514  assert(stmt->isAssign());
515  std::shared_ptr<Assign> asgn = stmt->as<Assign>();
516  SharedExp lhs = asgn->getLeft();
517 
518  if (!m_modifieds.existsOnLeft(lhs)) {
519  continue; // Not in modifieds: delete it (don't copy it)
520  }
521 
522  if (!m_proc->canBeReturn(lhs)) {
523  continue; // Filtered out: delete it
524  }
525 
526  // Preserveds are NOT returns (nothing changes, so what are we returning?)
527  // Check if it is a preserved location, e.g. r29 := r29{-}
528  SharedExp rhs = asgn->getRight();
529 
530  if (rhs->isSubscript() && rhs->access<RefExp>()->isImplicitDef() &&
531  (*rhs->getSubExp1() == *lhs)) {
532  continue;
533  }
534 
535  m_returns.append(asgn);
536  }
537 
538  m_returns.sort([&sig](const SharedConstStmt &ret1, const SharedConstStmt &ret2) {
539  return sig->returnCompare(*ret1->as<const Assignment>(), *ret2->as<const Assignment>());
540  });
541 }

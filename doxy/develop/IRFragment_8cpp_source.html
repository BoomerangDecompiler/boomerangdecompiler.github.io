96 SharedStmt IRFragment::getNextStmt(RTLIterator &rit, RTL::iterator &sit)
97 {
98  if (++sit != (*rit)->end()) {
99  return *sit; // End of current RTL not reached, so return next
100  }
101 
102  // Else, find next non-empty RTL & return its first statement
103  do {
104  if (++rit == m_listOfRTLs->end()) {
105  return nullptr; // End of all RTLs reached, return null Statement
106  }
107  } while ((*rit)->empty()); // Ignore all RTLs with no statements
108 
109  sit = (*rit)->begin(); // Point to 1st statement at start of next RTL
110  return *sit; // Return first statement
111 }

114 SharedStmt IRFragment::getPrevStmt(RTLRIterator &rit, RTL::reverse_iterator &sit)
115 {
116  if (++sit != (*rit)->rend()) {
117  return *sit; // Beginning of current RTL not reached, so return next
118  }
119 
120  // Else, find prev non-empty RTL & return its last statement
121  do {
122  if (++rit == m_listOfRTLs->rend()) {
123  return nullptr; // End of all RTLs reached, return null Statement
124  }
125  } while ((*rit)->empty()); // Ignore all RTLs with no statements
126 
127  sit = (*rit)->rbegin(); // Point to last statement at end of prev RTL
128  return *sit; // Return last statement
129 }

243 std::shared_ptr<ImplicitAssign> IRFragment::addImplicitAssign(const SharedExp &lhs)
244 {
245  assert(m_listOfRTLs);
246 
247  if (m_listOfRTLs->empty() || m_listOfRTLs->front()->getAddress() != Address::ZERO) {
248  m_listOfRTLs->push_front(std::unique_ptr<RTL>(new RTL(Address::ZERO)));
249  }
250 
251  // do not allow BB with 2 zero address RTLs
252  assert(m_listOfRTLs->size() < 2 ||
253  (*std::next(m_listOfRTLs->begin()))->getAddress() != Address::ZERO);
254 
255  for (const SharedStmt &s : *m_listOfRTLs->front()) {
256  if (s->isPhi() && *s->as<PhiAssign>()->getLeft() == *lhs) {
257  // phis kill implicits; don't add an implict assign
258  // if we already have a phi assigning to the LHS
259  return nullptr;
260  }
261  else if (s->isImplicit() && *s->as<ImplicitAssign>()->getLeft() == *lhs) {
262  // already present
263  return s->as<ImplicitAssign>();
264  }
265  }
266 
267  // no phi or implicit assigning to the LHS already
268  std::shared_ptr<ImplicitAssign> newImplicit(new ImplicitAssign(lhs));
269  newImplicit->setFragment(this);
270 
271  if (m_bb) {
272  newImplicit->setProc(m_bb->getProc());
273  }
274 
275  m_listOfRTLs->front()->append(newImplicit);
276  return newImplicit;
277 }

280 std::shared_ptr<PhiAssign> IRFragment::addPhi(const SharedExp &usedExp)
281 {
282  assert(m_listOfRTLs);
283 
284  if (m_listOfRTLs->empty() || m_listOfRTLs->front()->getAddress() != Address::ZERO) {
285  m_listOfRTLs->push_front(std::unique_ptr<RTL>(new RTL(Address::ZERO)));
286  }
287 
288  // do not allow BB with 2 zero address RTLs
289  assert(m_listOfRTLs->size() < 2 ||
290  (*std::next(m_listOfRTLs->begin()))->getAddress() != Address::ZERO);
291 
292  for (auto existingIt = m_listOfRTLs->front()->begin();
293  existingIt != m_listOfRTLs->front()->end();) {
294  SharedStmt s = *existingIt;
295  if (s->isPhi() && *s->as<PhiAssign>()->getLeft() == *usedExp) {
296  // already present
297  return s->as<PhiAssign>();
298  }
299  else if (s->isAssignment() && *s->as<Assignment>()->getLeft() == *usedExp) {
300  // the LHS is already assigned to properly, don't create a second assignment
301  return nullptr;
302  }
303 
304  ++existingIt;
305  }
306 
307  std::shared_ptr<PhiAssign> phi(new PhiAssign(usedExp));
308  phi->setFragment(this);
309 
310  if (m_bb) {
311  phi->setProc(m_bb->getProc());
312  }
313 
314  m_listOfRTLs->front()->append(phi);
315  return phi;
316 }

380 void IRFragment::updateAddresses()
381 {
382  if ((m_listOfRTLs == nullptr) || m_listOfRTLs->empty()) {
383  m_highAddr = Address::INVALID;
384  return;
385  }
386 
387  Address a = m_listOfRTLs->front()->getAddress();
388 
389  if (a.isZero() && (m_listOfRTLs->size() > 1)) {
390  RTLList::iterator it = m_listOfRTLs->begin();
391  const Address add2 = (*++it)->getAddress();
392 
393  // This is a bit of a hack for 286 programs, whose main actually starts at offset 0. A
394  // better solution would be to change orphan fragments' addresses to Address::INVALID, but I
395  // suspect that this will cause many problems. MVE
396  if (add2 < Address(0x10)) {
397  // Assume that 0 is the real address
398  m_lowAddr = Address::ZERO;
399  }
400  else {
401  m_lowAddr = add2;
402  }
403  }
404  else {
405  m_lowAddr = a;
406  }
407 
408  assert(m_listOfRTLs != nullptr);
409  m_highAddr = m_listOfRTLs->back()->getAddress();
410 }

487 SharedExp IRFragment::getDest() const
488 {
489  // The destianation will be in the last rtl
490  if (!m_listOfRTLs || m_listOfRTLs->empty()) {
491  return nullptr;
492  }
493 
494  const RTL *lastRTL = getLastRTL();
495 
496  // It should contain a GotoStatement or derived class
497  SharedStmt lastStmt = lastRTL->getHlStmt();
498  if (!lastStmt) {
499  if (getNumSuccessors() > 0) {
500  return Const::get(m_bb->getSuccessor(BTHEN)->getLowAddr());
501  }
502  else {
503  return nullptr;
504  }
505  }
506 
507  if (lastStmt->isCase()) {
508  // Get the expression from the switch info
509  const SwitchInfo *si = lastStmt->as<CaseStatement>()->getSwitchInfo();
510 
511  if (si) {
512  return si->switchExp;
513  }
514  }
515  else if (lastStmt->isGoto()) {
516  return lastStmt->as<GotoStatement>()->getDest();
517  }
518 
519  LOG_ERROR("Last statement of fragment at address %1 is not a goto!", m_bb->getLowAddr());
520  return nullptr;
521 }

543 void IRFragment::simplify()
544 {
545  if (m_listOfRTLs) {
546  for (auto &rtl : *m_listOfRTLs) {
547  rtl->simplify();
548  }
549  }
550 
551  if (isType(FragType::Twoway)) {
552  assert(getNumSuccessors() > 1);
553 
554  if ((m_listOfRTLs == nullptr) || m_listOfRTLs->empty()) {
555  setType(FragType::Fall);
556  }
557  else {
558  RTL *last = m_listOfRTLs->back().get();
559 
560  if (last->size() == 0) {
561  setType(FragType::Fall);
562  }
563  else if (last->back()->isGoto()) {
564  setType(FragType::Oneway);
565  }
566  else if (!last->back()->isBranch()) {
567  setType(FragType::Fall);
568  }
569  else if (getNumSuccessors() == 2 && getSuccessor(BTHEN) == getSuccessor(BELSE)) {
570  setType(FragType::Oneway);
571  }
572  }
573 
574  if (isType(FragType::Fall)) {
575  IRFragment *redundant = getSuccessor(BTHEN);
576  removeSuccessor(redundant);
577  redundant->removePredecessor(this);
578  }
579  else if (isType(FragType::Oneway)) {
580  IRFragment *redundant = getSuccessor(BELSE);
581  removeSuccessor(redundant);
582  redundant->removePredecessor(this);
583  }
584 
585  assert(m_bb->getProc()->getCFG()->isWellFormed());
586  }
587 }

590 void IRFragment::print(OStream &os) const
591 {
592  switch (getType()) {
593  case FragType::Oneway: os << "Oneway"; break;
594  case FragType::Twoway: os << "Twoway"; break;
595  case FragType::Nway: os << "Nway"; break;
596  case FragType::Call: os << "Call"; break;
597  case FragType::Ret: os << "Ret"; break;
598  case FragType::Fall: os << "Fall"; break;
599  case FragType::CompJump: os << "Computed Jump"; break;
600  case FragType::CompCall: os << "Computed Call"; break;
601  case FragType::Invalid: os << "Invalid"; break;
602  }
603 
604  os << " Fragment:\n";
605  os << " in edges: ";
606 
607  for (IRFragment *frag : getPredecessors()) {
608  os << frag->getHiAddr() << "(" << frag->getLowAddr() << ") ";
609  }
610 
611  os << "\n";
612  os << " out edges: ";
613 
614  for (IRFragment *frag : getSuccessors()) {
615  os << frag->getLowAddr() << " ";
616  }
617 
618  os << "\n";
619 
620  if (m_listOfRTLs) { // Can be null if e.g. INVALID
621  for (auto &rtl : *m_listOfRTLs) {
622  rtl->print(os);
623  }
624  }
625 }

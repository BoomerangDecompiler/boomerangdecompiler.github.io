133 void ExpPrinter::printPlain(OStream &os, const SharedConstExp &exp) const
134 {
135  const OPER oper = exp->getOper();
136 
137  // operators that need special attention
138  switch (oper) {
139  case opList:
140  print(os, exp->getSubExp1());
141  if (!exp->getSubExp2()->isNil()) {
142  os << ", ";
143  }
144  print(os, exp->getSubExp2());
145  return;
146 
147  case opFlagCall:
148  // The name of the flag function (e.g. ADDFLAGS) should be enough
149  exp->access<const Const, 1>()->printNoQuotes(os);
150  os << "( ";
151  print(os, exp->getSubExp2());
152  os << " )";
153  return;
154 
155  case opTemp:
156  if (exp->getSubExp1()->getOper() == opWildStrConst) {
157  os << "t[";
158  exp->access<const Terminal, 1>()->print(os);
159  os << "]";
160  return;
161  }
162  // fallthrough
163 
164  case opGlobal:
165  case opLocal:
166  case opParam:
167  // Print a more concise form than param["foo"] (just foo)
168  exp->access<const Const, 1>()->printNoQuotes(os);
169  return;
170 
171  case opMemberAccess:
172  print(os, exp->getSubExp1());
173  os << ".";
174  exp->access<const Const, 2>()->printNoQuotes(os);
175  return;
176 
177  case opIntConst:
178  if (exp->access<const Const>()->getInt() < -1000 ||
179  exp->access<const Const>()->getInt() > +1000) {
180  os << "0x" << QString::number(exp->access<const Const>()->getInt(), 16);
181  }
182  else {
183  os << exp->access<const Const>()->getInt();
184  }
185 
186  return;
187 
188  case opLongConst:
189  if ((static_cast<long long>(exp->access<const Const>()->getLong()) < -1000LL) ||
190  (static_cast<long long>(exp->access<const Const>()->getLong()) > +1000LL)) {
191  os << "0x" << QString::number(exp->access<const Const>()->getLong(), 16) << "LL";
192  }
193  else {
194  os << exp->access<const Const>()->getLong() << "LL";
195  }
196  return;
197 
198  case opFltConst:
199  os << QString("%1").arg(exp->access<const Const>()->getFlt()); // respects English locale
200  return;
201 
202  case opStrConst: os << "\"" << exp->access<const Const>()->getStr() << "\""; return;
203 
204  case opFuncConst: os << exp->access<const Const>()->getFuncName(); return;
205 
206  case opRegOf:
207  if (exp->getSubExp1()->isIntConst()) {
208  os << "r" << exp->access<const Const, 1>()->getInt();
209  return;
210  }
211  else if (exp->getSubExp1()->isTemp()) {
212  // Just print the temp { // balance }s
213  print(os, exp->getSubExp1());
214  return;
215  }
216  else {
217  os << "r[";
218  print(os, exp->getSubExp1());
219  os << "]";
220  return;
221  }
222 
223  case opSubscript:
224  print(os, exp->getSubExp1());
225  if (exp->access<const RefExp>()->getDef() == STMT_WILD) {
226  os << "{WILD}";
227  }
228  else if (exp->access<const RefExp>()->getDef()) {
229  os << "{" << exp->access<const RefExp>()->getDef()->getNumber() << "}";
230  }
231  else {
232  os << "{-}"; // So you can tell the difference with {0}
233  }
234  return;
235 
236  case opTypedExp: {
237  SharedConstType ty = exp->access<const TypedExp>()->getType();
238  os << "(" << *ty << ")(";
239  print(os, exp->getSubExp1());
240  os << ")";
241  return;
242  }
243 
244  default: break;
245  }
246 
247  assert(g_syntaxTable.find(oper) != g_syntaxTable.end());
248  os << g_syntaxTable[oper].m_prefix;
249 
250  if (exp->getArity() >= 1) {
251  if (childNeedsParentheses(exp, exp->getSubExp1()))
252  os << "(";
253  assert(exp->getSubExp1());
254  print(os, exp->getSubExp1());
255  if (childNeedsParentheses(exp, exp->getSubExp1()))
256  os << ")";
257 
258  if (exp->getArity() >= 2) {
259  os << g_syntaxTable[oper].m_infix1;
260 
261  if (childNeedsParentheses(exp, exp->getSubExp2()))
262  os << "(";
263  assert(exp->getSubExp2());
264  print(os, exp->getSubExp2());
265  if (childNeedsParentheses(exp, exp->getSubExp2()))
266  os << ")";
267 
268  if (exp->getArity() >= 3) {
269  os << g_syntaxTable[oper].m_infix2;
270 
271  if (childNeedsParentheses(exp, exp->getSubExp3()))
272  os << "(";
273  assert(exp->getSubExp3());
274  print(os, exp->getSubExp3());
275  if (childNeedsParentheses(exp, exp->getSubExp3()))
276  os << ")";
277  }
278  }
279  }
280 
281  os << g_syntaxTable[oper].m_postfix;
282 }

293 bool ExpPrinter::childNeedsParentheses(const SharedConstExp &exp, const SharedConstExp &child) const
294 {
295  // never parenthesize things like m[...] or foo{-} or constants
296  if (child->getArity() < 2) {
297  return false;
298  }
299  else if (child->getArity() == 3) {
300  switch (child->getOper()) {
301  case opTruncu:
302  case opTruncs:
303  case opZfill:
304  case opSgnEx:
305  case opFsize:
306  case opItof:
307  case opFtoi:
308  case opFround:
309  case opFtrunc: return false;
310  default: return true;
311  }
312  }
313 
314  if (exp->getArity() == 3) {
315  switch (exp->getOper()) {
316  case opTruncu:
317  case opTruncs:
318  case opZfill:
319  case opSgnEx:
320  case opFsize:
321  case opItof:
322  case opFtoi:
323  case opFround:
324  case opFtrunc: return false;
325  default: return true;
326  }
327  }
328  else if (exp->getArity() == 2) {
329  // parenthesize foo == (bar < baz)
330  if (exp->isComparison() && child->isComparison()) {
331  return true;
332  }
333 
334  return exp->getOper() != opList;
335  }
336  else if (exp->getArity() == 1) {
337  switch (exp->getOper()) {
338  case opBitNot: return true;
339  case opLNot: return true;
340  default: return false;
341  }
342  }
343 
344  return false;
345 }

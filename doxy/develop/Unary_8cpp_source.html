207 bool Unary::descendType(SharedType newType)
208 {
209  std::vector<SharedExp> matches;
210  bool changed = false;
211 
212  switch (m_oper) {
213  case opMemOf: {
214  std::shared_ptr<Binary> as_bin = std::dynamic_pointer_cast<Binary>(access<Exp, 1>());
215 
216  // Check for m[x*K1 + K2]: array with base K2 and stride K1
217  if (as_bin && (as_bin->getOper() == opPlus) &&
218  (as_bin->getSubExp1()->getOper() == opMult) && as_bin->getSubExp2()->isIntConst() &&
219  as_bin->getSubExp1()->getSubExp2()->isIntConst()) {
220  SharedExp leftOfPlus = as_bin->getSubExp1();
221  // We would expect the stride to be the same size as the base type
222  size_t stride = leftOfPlus->access<Const, 2>()->getInt();
223 
224  if (stride * 8 != newType->getSize()) {
225  LOG_VERBOSE("Type WARNING: apparent array reference at %1 has stride %2 bits, but "
226  "parent type %3 has size %4",
227  shared_from_this(), stride * 8, newType->getCtype(),
228  newType->getSize());
229  }
230 
231  // The index is integer type
232  SharedExp x = leftOfPlus->getSubExp1();
233  changed |= x->descendType(IntegerType::get(newType->getSize(), Sign::Unknown));
234 
235  // K2 is of type <array of parentType>
236  std::shared_ptr<Const> K2 = access<Const, 1, 2>();
237  Prog *prog = this->access<Location>()->getProc()->getProg();
238  changed |= K2->descendType(prog->makeArrayType(K2->getAddr(), newType));
239  }
240  else {
241  changed |= m_subExp1->descendType(PointerType::get(newType));
242  }
243 
244  break;
245  }
246 
247  case opAddrOf:
248  if (newType->resolvesToPointer()) {
249  changed |= m_subExp1->descendType(newType->as<PointerType>()->getPointsTo());
250  }
251 
252  break;
253 
254  case opGlobal: {
255  Prog *_prog = access<Location>()->getProc()->getProg();
256  QString name = access<Const, 1>()->getStr();
257  SharedType ty = _prog->getGlobalType(name);
258 
259  if (ty) {
260  ty = ty->meetWith(newType, changed);
261 
262  if (changed) {
263  _prog->setGlobalType(name, ty);
264  }
265  }
266 
267  break;
268  }
269 
270  default: break;
271  }
272 
273  return changed;
274 }

30 bool CallAndPhiFixPass::execute(UserProc *proc)
31 {
32  /* Algorithm:
33  * for each statement s in this proc
34  * if s is a phi statement ps
35  * let r be a ref made up of lhs and s
36  * for each parameter p of ps
37  * if p == r // e.g. test/x86/fromssa2 r28{56}
38  * remove p from ps
39  * let lhs be left hand side of ps
40  * allSame = true
41  * let first be a ref built from first p
42  * do bypass but not propagation on first
43  * if result is of the form lhs{x}
44  * replace first with x
45  * for each parameter p of ps after the first
46  * let current be a ref built from p
47  * do bypass but not propagation on current
48  * if result is of form lhs{x}
49  * replace cur with x
50  * if first != current
51  * allSame = false
52  * if allSame
53  * let best be ref built from the "best" parameter p in ps ({-} better than
54  * {assign} better than {call}) replace ps with an assignment lhs := best else (ordinary
55  * statement) do bypass and propagation for s
56  */
57  std::map<SharedExp, int, lessExpStar> destCounts;
58  StatementList stmts;
59  proc->getStatements(stmts);
60 
61  // a[m[]] hack, aint nothing better.
62  bool found = true;
63 
64  for (SharedStmt s : stmts) {
65  if (!s->isCall()) {
66  continue;
67  }
68 
69  std::shared_ptr<CallStatement> call = s->as<CallStatement>();
70 
71  for (auto &elem : call->getArguments()) {
72  std::shared_ptr<Assign> a = elem->as<Assign>();
73 
74  if (!a->getType()->resolvesToPointer()) {
75  continue;
76  }
77 
78  SharedExp e = a->getRight();
79 
80  if ((e->getOper() == opPlus) || (e->getOper() == opMinus)) {
81  if (e->getSubExp2()->isIntConst()) {
82  if (e->getSubExp1()->isSubscript() &&
83  e->getSubExp1()->getSubExp1()->isRegN(
84  proc->getSignature()->getStackRegister()) &&
85  (((e->access<RefExp, 1>())->getDef() == nullptr) ||
86  (e->access<RefExp, 1>())->getDef()->isImplicit())) {
87  a->setRight(Unary::get(opAddrOf, Location::memOf(e->clone())));
88  found = true;
89  }
90  }
91  }
92  }
93  }
94 
95  if (found) {
96  PassManager::get()->executePass(PassID::BlockVarRename, proc);
97  }
98 
99  // Scan for situations like this:
100  // 56 r28 := phi{6, 26}
101  // ...
102  // 26 r28 := r28{56}
103  // So we can remove the second parameter,
104  // then reduce the phi to an assignment, then propagate it
105  for (SharedStmt s : stmts) {
106  if (!s->isPhi()) {
107  continue;
108  }
109 
110  std::shared_ptr<PhiAssign> phi = s->as<PhiAssign>();
111  std::shared_ptr<RefExp> refExp = RefExp::get(phi->getLeft(), phi);
112 
113  phi->removeAllReferences(refExp);
114  }
115 
116  // Second pass
117  for (SharedStmt s : stmts) {
118  if (!s->isPhi()) { // Ordinary statement
119  s->bypass();
120  continue;
121  }
122 
123  std::shared_ptr<PhiAssign> phi = s->as<PhiAssign>();
124 
125  if (phi->getNumDefs() == 0) {
126  // Can happen e.g. for m[...] := phi {} when this proc is involved in a recursion group
127  continue;
128  }
129 
130  auto lhs = phi->getLeft();
131  bool allSame = true;
132  // Let first be a reference built from the first parameter
133  PhiAssign::iterator phi_iter = phi->begin();
134 
135  while (phi_iter != phi->end() && (*phi_iter)->getSubExp1() == nullptr) {
136  ++phi_iter; // Skip any null parameters
137  }
138 
139  assert(phi_iter != phi->end()); // Should have been deleted
140  const std::shared_ptr<RefExp> &phi_inf = *phi_iter;
141  SharedExp first = RefExp::get(phi_inf->getSubExp1(), phi_inf->getDef());
142 
143  // bypass to first
144  CallBypasser cb(phi);
145  first = first->acceptModifier(&cb);
146 
147  if (cb.isTopChanged()) {
148  first = first->simplify();
149  }
150 
151  first = first->propagateAll(); // Propagate everything repeatedly
152 
153  if (cb.isModified()) { // Modified?
154  // if first is of the form lhs{x}
155  if (first->isSubscript() && (*first->getSubExp1() == *lhs)) {
156  // replace first with x
157  phi_inf->setDef(first->access<RefExp>()->getDef());
158  }
159  }
160 
161  // For each parameter p of ps after the first
162  for (++phi_iter; phi_iter != phi->end(); ++phi_iter) {
163  assert((*phi_iter)->getSubExp1());
164  const std::shared_ptr<RefExp> &phi_inf2 = *phi_iter;
165  SharedExp current = RefExp::get(phi_inf2->getSubExp1(), phi_inf2->getDef());
166  CallBypasser cb2(phi);
167  current = current->acceptModifier(&cb2);
168 
169  if (cb2.isTopChanged()) {
170  current = current->simplify();
171  }
172 
173  current = current->propagateAll();
174 
175  if (cb2.isModified()) {
176  // if current is of the form lhs{x}
177  if (current->isSubscript() && (*current->getSubExp1() == *lhs)) {
178  // replace current with x
179  phi_inf2->setDef(current->access<RefExp>()->getDef());
180  }
181  }
182 
183  if (!(*first == *current)) {
184  allSame = false;
185  }
186  }
187 
188  if (allSame) {
189  // let best be ref built from the "best" parameter p in ps ({-} better than {assign}
190  // better than {call})
191  phi_iter = phi->begin();
192 
193  while (phi_iter != phi->end() && (*phi_iter)->getSubExp1() == nullptr) {
194  ++phi_iter; // Skip any null parameters
195  }
196 
197  assert(phi_iter != phi->end()); // Should have been deleted
198  auto best = RefExp::get((*phi_iter)->getSubExp1(), (*phi_iter)->getDef());
199 
200  for (++phi_iter; phi_iter != phi->end(); ++phi_iter) {
201  assert((*phi_iter)->getSubExp1());
202  auto current = RefExp::get((*phi_iter)->getSubExp1(), (*phi_iter)->getDef());
203 
204  if (current->isImplicitDef()) {
205  best = current;
206  break;
207  }
208 
209  if ((*phi_iter)->getDef()->isAssign()) {
210  best = current;
211  }
212 
213  // If phi_iter->second.def is a call, this is the worst case; keep only (via first)
214  // if all parameters are calls
215  }
216 
217  SharedStmt asgn = proc->replacePhiByAssign(phi, best);
218  LOG_VERBOSE2("Redundant phi replaced with copy assign; now %1", asgn);
219  }
220  }
221 
222  // Also do xxx in m[xxx] in the use collector
223  for (const SharedExp &cc : proc->getUseCollector()) {
224  if (!cc->isMemOf()) {
225  continue;
226  }
227 
228  auto addr = cc->getSubExp1();
229  CallBypasser cb(nullptr);
230  addr = addr->acceptModifier(&cb);
231 
232  if (cb.isModified()) {
233  cc->setSubExp1(addr);
234  }
235  }
236 
237  return true;
238 }

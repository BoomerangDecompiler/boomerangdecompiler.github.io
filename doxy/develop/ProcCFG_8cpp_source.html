75 IRFragment *ProcCFG::splitFragment(IRFragment *frag, Address splitAddr)
76 {
77  assert(hasFragment(frag));
78 
79  auto it = std::find_if(frag->getRTLs()->begin(), frag->getRTLs()->end(),
80  [splitAddr](const auto &rtl) { return splitAddr == rtl->getAddress(); });
81 
82  if (it == frag->getRTLs()->end()) {
83  // cannot split
84  return frag;
85  }
86  else if (it == frag->getRTLs()->begin()) {
87  // no need to split
88  return frag;
89  }
90 
91  // move RTLs with addr >= splitAddr to new list
92  std::unique_ptr<RTLList> newRTLs(new RTLList);
93  std::for_each(it, frag->getRTLs()->end(),
94  [&newRTLs](std::unique_ptr<RTL> &rtl) { newRTLs->push_back(std::move(rtl)); });
95  frag->getRTLs()->erase(it, frag->getRTLs()->end());
96 
97  IRFragment *newFrag = createFragment(frag->getType(), std::move(newRTLs), frag->getBB());
98  frag->setType(FragType::Fall);
99  addEdge(frag, newFrag);
100 
101  frag->updateAddresses();
102  newFrag->updateAddresses();
103 
104  assert(frag->getHiAddr() < splitAddr);
105  return newFrag;
106 }

109 void ProcCFG::removeFragment(IRFragment *frag)
110 {
111  assert(frag != nullptr);
112  assert(this->hasFragment(frag));
113 
114  RTLList::iterator rit;
115  StatementList::iterator sit;
116 
117  for (SharedStmt s = frag->getFirstStmt(rit, sit); s; s = frag->getNextStmt(rit, sit)) {
118  if (s->isCall()) {
119  std::shared_ptr<CallStatement> call = s->as<CallStatement>();
120  if (call->getDestProc() && !call->getDestProc()->isLib()) {
121  UserProc *callee = static_cast<UserProc *>(call->getDestProc());
122  callee->removeCaller(call);
123  }
124  }
125  }
126 
127  auto it = findFragment(frag);
128 
129  if (it == m_fragmentSet.end()) {
130  LOG_WARN("Tried to remove fragment at address %1; does not exist in CFG",
131  frag->getLowAddr());
132 
133  delete frag;
134  return;
135  }
136 
137  for (IRFragment *pred : frag->getPredecessors()) {
138  pred->removeSuccessor(frag);
139  }
140 
141  for (IRFragment *succ : frag->getSuccessors()) {
142  succ->removePredecessor(frag);
143  }
144 
145  frag->removeAllPredecessors();
146  frag->removeAllSuccessors();
147 
148  frag->clearPhis();
149 
150  assert(*it == frag);
151  m_fragmentSet.erase(it);
152  delete frag;
153 }

173 void ProcCFG::addEdge(IRFragment *sourceFrag, IRFragment *destFrag)
174 {
175  if (!sourceFrag || !destFrag) {
176  return;
177  }
178 
179  // Wire up edges
180  sourceFrag->addSuccessor(destFrag);
181  destFrag->addPredecessor(sourceFrag);
182 
183  // special handling for upgrading oneway BBs to twoway BBs
184  if (sourceFrag->isType(FragType::Oneway) && (sourceFrag->getNumSuccessors() > 1)) {
185  sourceFrag->setType(FragType::Twoway);
186  }

190 void ProcCFG::replaceEdge(IRFragment *src, IRFragment *oldDest, IRFragment *newDest)
191 {
192  assert(oldDest->isSuccessorOf(src));
193 
194  for (int i = 0; i < src->getNumSuccessors(); ++i) {
195  if (src->getSuccessor(i) == oldDest) {
196  oldDest->removePredecessor(src);
197  newDest->addPredecessor(src);
198  src->setSuccessor(i, newDest);
199  return;
200  }
201  }

205 bool ProcCFG::isWellFormed() const
206 {
207  for (const IRFragment *frag : *this) {
208  if (frag->getProc() != m_myProc) {
209  LOG_ERROR("CFG is not well formed: Fragment at address %1 does not belong to proc '%2'",
210  frag->getLowAddr(), m_myProc ? m_myProc->getName() : QString("<Unknown>"));
211  return false;
212  }
213 
214  for (const IRFragment *pred : frag->getPredecessors()) {
215  if (!pred->isPredecessorOf(frag)) {
216  LOG_ERROR("CFG is not well formed: Edge from fragment at %1 to fragment at %2 "
217  "is malformed.",
218  pred->getLowAddr(), frag->getLowAddr());
219  return false;
220  }
221  }
222 
223  for (const IRFragment *succ : frag->getSuccessors()) {
224  if (!succ->isSuccessorOf(frag)) {
225  LOG_ERROR("CFG is not well formed: Edge from fragment at %1 to fragment at %2 "
226  "is malformed.",
227  frag->getLowAddr(), succ->getLowAddr());
228  return false;
229  }
230  }
231  }
232 
233  return true;
234 }

257 SharedStmt ProcCFG::findOrCreateImplicitAssign(SharedExp exp)
258 {
259  ExpStatementMap::iterator it = m_implicitMap.find(exp);
260  if (it != m_implicitMap.end()) {
261  // implicit already present, use it
262  assert(it->second);
263  return it->second;
264  }
265 
266  if (!m_entryFrag) {
267  return nullptr;
268  }
269 
270  // In case the original gets changed
271  exp = exp->clone();
272 
273  // A use with no explicit definition. Create a new implicit assignment
274  std::shared_ptr<ImplicitAssign> def = m_entryFrag->addImplicitAssign(exp);
275 
276  // Remember it for later so we don't insert more than one implicit assignment for any one
277  // location We don't clone the copy in the map. So if the location is a m[...], the same type
278  // information is available in the definition as at all uses
279  m_implicitMap[exp] = def;
280  return def;
281 }

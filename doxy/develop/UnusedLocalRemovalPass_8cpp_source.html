33 bool UnusedLocalRemovalPass::execute(UserProc *proc)
34 {
35  QSet<QString> usedLocals;
36  StatementList stmts;
37  proc->getStatements(stmts);
38 
39  // First count any uses of the locals
40  bool all = false;
41 
42  for (SharedStmt s : stmts) {
43  LocationSet locs;
44  all |= addUsedLocalsForStmt(s, locs);
45 
46  for (SharedExp u : locs) {
47  // Must be a real symbol, and not defined in this statement, unless it is a return
48  // statement (in which case it is used outside this procedure), or a call statement.
49  // Consider local7 = local7+1 and return local7 = local7+1 and local7 = call(local7+1),
50  // where in all cases, local7 is not used elsewhere outside this procedure.
51  // With the assign, it can be deleted, but with the return or call statements, it can't.
52  if ((s->isReturn() || s->isCall() || !s->definesLoc(u))) {
53  if (!u->isLocal()) {
54  continue;
55  }
56 
57  QString name(u->access<Const, 1>()->getStr());
58  usedLocals.insert(name);
59 
60  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
61  LOG_MSG("Counted local %1 in %2", name, s);
62  }
63  }
64  }
65 
66  if (s->isAssignment() && !s->isImplicit() && s->as<Assignment>()->getLeft()->isLocal()) {
67  std::shared_ptr<Assignment> as = s->as<Assignment>();
68  auto c = as->getLeft()->access<Const, 1>();
69 
70  usedLocals.insert(c->getStr());
71 
72  if (proc->getProg()->getProject()->getSettings()->debugUnused) {
73  LOG_MSG("Counted local %1 on left of %2", c->getStr(), s);
74  }
75  }
76  }
77 
78  // Now record the unused ones in set removes
79 
80  QSet<QString> removes;
81 
82  for (auto it = proc->getLocals().begin(); it != proc->getLocals().end(); ++it) {
83  const QString &name(it->first);
84 
85  if (all && removes.size()) {
86  LOG_VERBOSE("WARNING: defineall seen in procedure %1, so not removing %2 locals", name,
87  removes.size());
88  }
89 
90  if ((usedLocals.find(name) == usedLocals.end()) && !all) {
91  if (proc->getProg()->getProject()->getSettings()->verboseOutput) {
92  LOG_VERBOSE("Removed unused local %1", name);
93  }
94 
95  removes.insert(name);
96  }
97  }
98 
99  // Remove any definitions of the removed locals
100  const bool assumeABICompliance = proc->getProg()->getProject()->getSettings()->assumeABI;
101 
102  for (SharedStmt s : stmts) {
103  LocationSet ls;
104  s->getDefinitions(ls, assumeABICompliance);
105 
106  for (const SharedExp &loc : ls) {
107  SharedType ty = s->getTypeForExp(loc);
108  QString name = proc->findLocal(loc, ty);
109 
110  if (name.isEmpty()) {
111  continue;
112  }
113 
114  if (removes.find(name) != removes.end()) {
115  // Remove it. If an assign, delete it; otherwise (call), remove the define
116  if (s->isAssignment()) {
117  proc->removeStatement(s);
118  break; // Break to next statement
119  }
120  else if (s->isCall()) {
121  // Remove just this define. May end up removing several defines from this call.
122  s->as<CallStatement>()->removeDefine(loc);
123  }
124 
125  // else if a ReturnStatement, don't attempt to remove it. The definition is used
126  // *outside* this proc.
127  }
128  }
129  }
130 
131  // Finally, remove them from locals, so they don't get declared
132  for (QString localName : removes) {
133  proc->getLocals().erase(localName);
134  }
135 
136  // Also remove them from the symbols, since symbols are a superset of locals at present
137  for (UserProc::SymbolMap::iterator sm = proc->getSymbolMap().begin();
138  sm != proc->getSymbolMap().end();) {
139  SharedExp mapsTo = sm->second;
140 
141  if (mapsTo->isLocal()) {
142  QString tmpName = mapsTo->access<Const, 1>()->getStr();
143 
144  if (removes.find(tmpName) != removes.end()) {
145  sm = proc->getSymbolMap().erase(sm);
146  continue;
147  }
148  }
149 
150  ++sm;
151  }
152 
153  proc->getProg()->getProject()->alertDecompileDebugPoint(proc, "After removing unused locals");
154  return true;
155 }

43 void DataFlow::dfs(FragIndex myIdx, FragIndex parentIdx)
44 {
45  assert(myIdx != INDEX_INVALID);
46  if (m_dfnum[myIdx] >= 0) {
47  // already visited
48  return;
49  }
50 
51  m_dfnum[myIdx] = N;
52  m_vertex[N] = myIdx;
53  m_parent[myIdx] = parentIdx;
54 
55  N++;
56 
57  // Recurse to successors
58  IRFragment *frag = m_frags[myIdx];
59 
60  for (IRFragment *succ : frag->getSuccessors()) {
61  dfs(m_indices[succ], myIdx);
62  }
63 }

66 bool DataFlow::calculateDominators()
67 {
68  ProcCFG *cfg = m_proc->getCFG();
69  IRFragment *entryFrag = cfg->getEntryFragment();
70  const std::size_t numFrags = cfg->getNumFragments();
71 
72  if (!entryFrag || numFrags == 0) {
73  return false; // nothing to do
74  }
75 
76  allocateData();
77  recalcSpanningTree();
78 
79  assert(N >= 1);
80 
81  // Process fragments in reverse pre-traversal order (i.e. return blocks first)
82  for (std::size_t i = N - 1; i >= 1; i--) {
83  FragIndex n = m_vertex[i];
84  FragIndex p = m_parent[n];
85  FragIndex s = p;
86 
87  // These lines calculate the semi-dominator of n, based on the Semidominator Theorem
88  for (IRFragment *pred : m_frags[n]->getPredecessors()) {
89  if (m_indices.find(pred) == m_indices.end()) {
90  LOG_ERROR("Fragment not in indices: ", pred->toString());
91  return false;
92  }
93 
94  const FragIndex v = m_indices[pred];
95  FragIndex sdash = v;
96 
97  if (isAncestorOf(v, n)) {
98  sdash = m_semi[getAncestorWithLowestSemi(v)];
99  }
100 
101  if (isAncestorOf(s, sdash)) {
102  s = sdash;
103  }
104  }
105 
106  m_semi[n] = s;
107 
108  // Calculation of n's dominator is deferred until the path from s to n
109  // has been linked into the forest
110  m_bucket[s].insert(n);
111  link(p, n);
112 
113  // for each v in bucket[p]
114  for (FragIndex v : m_bucket[p]) {
115  // Now that the path from p to v has been linked into the spanning forest,
116  // these lines calculate the dominator of v, based on the first clause of the
117  // Dominator Theorem, or else defer the calculation until y's dominator is known.
118  const FragIndex y = getAncestorWithLowestSemi(v);
119 
120  if (m_semi[y] == m_semi[v]) {
121  m_idom[v] = p; // Success!
122  }
123  else {
124  m_samedom[v] = y; // Defer
125  }
126  }
127 
128  m_bucket[p].clear();
129  }
130 
131  for (std::size_t i = 1; i < N - 1; i++) {
132  // Now all the deferred dominator calculations, based on the second clause of the Dominator
133  // Theorem, are performed.
134  FragIndex n = m_vertex[i];
135 
136  if (m_samedom[n] != INDEX_INVALID) {
137  m_idom[n] = m_idom[m_samedom[n]]; // Deferred success!
138  }
139  }
140 
141  const FragIndex entryIndex = fragToIdx(entryFrag);
142  assert(entryIndex != INDEX_INVALID);
143 
144  // the entry fragment is always executed.
145  m_idom[entryIndex] = entryIndex;
146  m_semi[entryIndex] = entryIndex;
147 
148  computeDF(entryIndex); // Finally, compute the dominance frontiers
149  return true;
150 }

153 FragIndex DataFlow::getAncestorWithLowestSemi(FragIndex v)
154 {
155  assert(v != INDEX_INVALID);
156 
157  const FragIndex a = m_ancestor[v];
158  if (a != INDEX_INVALID && m_ancestor[a] != INDEX_INVALID) {
159  const FragIndex b = getAncestorWithLowestSemi(a);
160  m_ancestor[v] = m_ancestor[a];
161 
162  if (isAncestorOf(m_semi[m_best[v]], m_semi[b])) {
163  m_best[v] = b;
164  }
165  }
166 
167  return m_best[v];
168 }

180 bool DataFlow::doesDominate(FragIndex n, FragIndex w)
181 {
182  assert(n != INDEX_INVALID);
183  assert(w != INDEX_INVALID);
184 
185  while (w != INDEX_INVALID && m_idom[w] != w) {
186  if (m_idom[w] == n) {
187  return true;
188  }
189 
190  w = m_idom[w]; // Move up the dominator tree
191  }
192 
193  return false;
194 }

197 void DataFlow::computeDF(FragIndex n)
198 {
199  assert(n != INDEX_INVALID);
200 
201  std::set<FragIndex> S;
202  // This loop computes DF_local[n]
203  // for each node y in succ(n)
204  IRFragment *frag = m_frags[n];
205 
206  for (IRFragment *succ : frag->getSuccessors()) {
207  FragIndex y = m_indices[succ];
208 
209  if (m_idom[y] != n) {
210  S.insert(y);
211  }
212  }
213 
214  // for each child c of n in the dominator tree
215  // Note: this is a linear search!
216  const size_t sz = m_idom.size(); // ? Was ancestor.size()
217 
218  for (FragIndex c = FragIndex(0); c < sz; ++c) {
219  if (m_idom[c] != n) {
220  continue;
221  }
222  else if (c != n) { // do not calculate DF for entry fragment again
223  computeDF(c);
224  }
225 
226  /* This loop computes DF_up[c] */
227  // for each element w of DF[c]
228  std::set<FragIndex> &s = m_DF[c];
229 
230  for (FragIndex w : s) {
231  if (n == w || !doesDominate(n, w)) {
232  S.insert(w);
233  }
234  }
235  }
236 
237  m_DF[n] = S;
238 }

241 bool DataFlow::canRename(SharedConstExp exp) const
242 {
243  if (exp->isSubscript()) {
244  exp = exp->getSubExp1(); // Look inside refs
245  }
246 
247  if (exp->isRegOf() || // Always rename registers
248  exp->isTemp() || // Always rename temps (always want to propagate away)
249  exp->isFlags() || // Always rename flags
250  exp->isMainFlag() || // Always rename individual flags like %CF
251  exp->isLocal()) { // Rename hard locals in the post fromSSA pass
252  return true;
253  }
254 
255  if (!exp->isMemOf()) {
256  return false; // Can't rename %pc or other junk
257  }
258 
259  // I used to check here if there was a symbol for the memory expression, and if so allow it to
260  // be renamed. However, even named locals and parameters could have their addresses escape the
261  // local function, so we need another test anyway. So locals and parameters should not be
262  // renamed (and hence propagated) until escape analysis is done (and hence renaleLocalsAndParams
263  // is set) Besides, before we have types and references, it is not easy to find a type for the
264  // location, so we can't tell if e.g. m[esp{-}+12] is evnp or a separate local. It certainly
265  // needs to have the local/parameter pattern
266  if (!m_proc->isLocalOrParamPattern(exp)) {
267  return false;
268  }
269 
270  // e is a local or parameter; allow it to be propagated iff we've done escape analysis and the
271  // address has not
272  return renameLocalsAndParams;
273 }

276 bool DataFlow::placePhiFunctions()
277 {
278  // First free some memory no longer needed
279  m_dfnum.resize(0);
280  m_semi.resize(0);
281  m_ancestor.resize(0);
282  m_samedom.resize(0);
283  m_vertex.resize(0);
284  m_parent.resize(0);
285  m_best.resize(0);
286  m_bucket.resize(0);
287  m_defsites.clear();
288  m_defallsites.clear();
289 
290  for (ExSet &exps : m_definedAt) {
291  for (auto iter = exps.begin(); iter != exps.end();) {
292  if (m_A_phi.find(*iter) == m_A_phi.end()) {
293  iter = exps.erase(iter);
294  }
295  else {
296  ++iter;
297  }
298  }
299  }
300 
301  m_definedAt.clear(); // and A_orig,
302  m_defStmts.clear(); // and the map from variable to defining Stmt
303 
304  for (IRFragment *frag : *m_proc->getCFG()) {
305  frag->clearPhis();
306  }
307 
308  // Set the sizes of needed vectors
309  const std::size_t numIndices = m_indices.size();
310  const std::size_t numFrags = m_proc->getCFG()->getNumFragments();
311  assert(numIndices == numFrags);
312  Q_UNUSED(numIndices);
313 
314  m_definedAt.resize(numFrags);
315 
316  const bool assumeABICompliance = m_proc->getProg()->getProject()->getSettings()->assumeABI;
317 
318  // We need to create m_definedAt[n] for all n
319  // Recreate each call because propagation and other changes make old data invalid
320  for (FragIndex n{ 0 }; n < numFrags; ++n) {
321  IRFragment::RTLIterator rit;
322  StatementList::iterator sit;
323  IRFragment *frag = m_frags[n];
324 
325  for (SharedStmt stmt = frag->getFirstStmt(rit, sit); stmt;
326  stmt = frag->getNextStmt(rit, sit)) {
327  LocationSet locationSet;
328  stmt->getDefinitions(locationSet, assumeABICompliance);
329 
330  // If this is a childless call, then this block defines every variable
331  if (stmt->isCall() && stmt->as<CallStatement>()->isChildless()) {
332  m_defallsites.insert(n);
333  }
334 
335  for (const SharedExp &exp : locationSet) {
336  if (canRename(exp)) {
337  m_definedAt[n].insert(exp->clone());
338  m_defStmts[exp] = stmt;
339  }
340  }
341  }
342  }
343 
344  for (FragIndex n{ 0 }; n < numFrags; ++n) {
345  for (const SharedExp &a : m_definedAt[n]) {
346  m_defsites[a].insert(n);
347  }
348  }
349 
350  bool change = false;
351 
352  // Those variables that are defined everywhere (i.e. in defallsites)
353  // need to be defined at every defsite, too
354  for (FragIndex defallsite : m_defallsites) {
355  for (auto &[exp, defsites] : m_defsites) {
356  Q_UNUSED(exp);
357  defsites.insert(defallsite);
358  }
359  }
360 
361  // For each variable a defined anywhere
362  for (auto &[a, defsites] : m_defsites) {
363  std::set<FragIndex> W = defsites;
364 
365  while (!W.empty()) {
366  // Pop first node from W
367  const FragIndex n = *W.begin();
368  W.erase(W.begin());
369 
370  for (FragIndex y : m_DF[n]) {
371  // phi function already created for y?
372  if (m_A_phi[a].find(y) != m_A_phi[a].end()) {
373  continue;
374  }
375 
376  // Insert trivial phi function for a at top of block y: a := phi()
377  change = true;
378  m_frags[y]->addPhi(a->clone());
379 
380  // A_phi[a] <- A_phi[a] U {y}
381  m_A_phi[a].insert(y);
382 
383  // if a !elementof A_orig[y]
384  if (!m_definedAt[y].contains(a)) {
385  // W <- W U {y}
386  W.insert(y);
387  }
388  }
389  }
390  }
391 
392  return change;
393 }

396 void DataFlow::convertImplicits()
397 {
398  ProcCFG *cfg = m_proc->getCFG();
399 
400  // Convert statements in A_phi from m[...]{-} to m[...]{0}
401  std::map<SharedExp, std::set<FragIndex>, lessExpStar> A_phi_copy = m_A_phi; // Object copy
402  ImplicitConverter ic(cfg);
403  m_A_phi.clear();
404 
405  for (auto &[exp, set] : A_phi_copy) {
406  SharedExp e = exp->clone()->acceptModifier(&ic);
407  m_A_phi[e] = set; // Copy the set (doesn't have to be deep)
408  }
409 
410  std::map<SharedExp, std::set<FragIndex>, lessExpStar> defsites_copy = m_defsites; // Object copy
411  m_defsites.clear();
412 
413  for (auto &[exp, set] : defsites_copy) {
414  SharedExp e = exp->clone()->acceptModifier(&ic);
415  m_defsites[e] = set; // Copy the set (doesn't have to be deep)
416  }
417 
418  std::vector<ExSet> definedAtCopy = m_definedAt;
419  m_definedAt.clear();
420 
421  for (ExSet &se : definedAtCopy) {
422  ExSet se_new;
423 
424  for (const SharedExp &ee : se) {
425  SharedExp e = ee->clone()->acceptModifier(&ic);
426  se_new.insert(e);
427  }
428 
429  // Copy the set (doesn't have to be a deep copy)
430  m_definedAt.insert(m_definedAt.end(), se_new);
431  }
432 }

450  m_best.assign(numFrags, INDEX_INVALID);

453  m_definedAt.assign(numFrags, {});
454 
455  m_A_phi.clear();

457  m_defallsites.clear();
458  m_defStmts.clear();
459 
460  // Set up the fragment and indices vectors.
461  // Do this here because sometimes a fragment can be unreachable
462  // (so relying on in-edges doesn't work)
463  std::size_t i = 0;
464  for (IRFragment *frag : *cfg) {
465  m_frags[i++] = frag;
466  }
467 
468  for (std::size_t j = 0; j < numFrags; j++) {
469  m_indices[m_frags[j]] = j;
